35  subroutine model_jp1d(myrank,x,rho,vp,vs,Qkappa,Qmu,idoubling, &
36  check_doubling_flag,ricb,rcmb,rtopddoubleprime, &
37  r670,r220,r771,r400,r80,rmoho,rmiddle_crust)
38 
39  use constants
40 
41  implicit none
42 
43  ! given a normalized radius x, gives the non-dimensionalized density rho,
44  ! speeds vp and vs, and the quality factors Qkappa and Qmu
45 
46  logical check_doubling_flag
47  integer idoubling,myrank
48 
49  double precision x,rho,vp,vs,Qkappa,Qmu,RICB,RCMB,RTOPDDOUBLEPRIME, &
50  R670,R220,R771,R400,R80,RMOHO,RMIDDLE_CRUST
51 
52  double precision r
53  double precision scaleval
54 
55  ! compute real physical radius in meters
56  r = x * r_earth
57 
58  ! check flags to make sure we correctly honor the discontinuities
59  ! we use strict inequalities since r has been slightly changed in mesher
60 
61  if (check_doubling_flag) then
62  !--- inner core
63  !
64  if (r >= 0.d0 .and. r < ricb) then
65  if (idoubling /= iflag_inner_core_normal .and. &
66  idoubling /= iflag_middle_central_cube .and. &
67  idoubling /= iflag_bottom_central_cube .and. &
68  idoubling /= iflag_top_central_cube .and. &
69  idoubling /= iflag_in_fictitious_cube) &
70  call exit_mpi(myrank,'wrong doubling flag for inner core point')
71  !
72  !--- outer core
73  !
74  else if (r > ricb .and. r < rcmb) then
75  if (idoubling /= iflag_outer_core_normal) &
76  call exit_mpi(myrank,'wrong doubling flag for outer core point')
77  !
78  !--- D" at the base of the mantle
79  !
80  else if (r > rcmb .and. r < rtopddoubleprime) then
81  if (idoubling /= iflag_mantle_normal) &
82  call exit_mpi(myrank,'wrong doubling flag for D" point')
83  !
84  !--- mantle: from top of D" to d670
85  !
86  else if (r > rtopddoubleprime .and. r < r670) then
87  if (idoubling /= iflag_mantle_normal) &
88  call exit_mpi(myrank,'wrong doubling flag for top D" to d670 point')
89  !
90  !--- mantle: from d670 to d220
91  !
92  else if (r > r670 .and. r < r220) then
93  if (idoubling /= iflag_670_220) &
94  call exit_mpi(myrank,'wrong doubling flag for d670 to d220 point')
95  !
96  !--- mantle and crust: from d220 to MOHO and then to surface
97  !
98  else if (r > r220) then
99  if (idoubling /= iflag_220_80 .and. idoubling /= iflag_80_moho .and. idoubling /= iflag_crust) &
100  call exit_mpi(myrank,'wrong doubling flag for d220 to Moho to surface point')
101  endif
102 
103  endif
104 
105 
106 !
107 !--- inner core
108 !
109  if (r >= 0.d0 .and. r <= ricb) then
110  rho=13.0885d0-8.8381d0*x*x
111  vp=11.24094-4.09689*x**2
112  vs=3.56454-3.45241*x**2
113  qmu=84.6d0
114  qkappa=1327.7d0
115 !
116 !--- outer core
117 !
118  else if (r > ricb .and. r <= rcmb) then
119  rho=12.5815d0-1.2638d0*x-3.6426d0*x*x-5.5281d0*x*x*x
120  vp=10.03904+3.75665*x-13.67046*x**2
121  vs=0.0d0
122  qmu=0.0d0
123  qkappa=57827.0d0
124 !
125 !--- D" at the base of the mantle
126 !
127  else if (r > rcmb .and. r <= rtopddoubleprime) then
128  rho=7.9565d0-6.4761d0*x+5.5283d0*x*x-3.0807d0*x*x*x
129  vp=14.49470-1.47089*x
130  vs=8.16616-1.58206*x
131  qmu=312.0d0
132  qkappa=57827.0d0
133 !
134 !--- mantle: from top of D" to d670
135 !
136  else if (r > rtopddoubleprime .and. r <= r771) then
137  rho=7.9565d0-6.4761d0*x+5.5283d0*x*x-3.0807d0*x*x*x
138  vp=-355.58324*x**4 + 1002.03178*x**3 - 1057.3873425*x**2 + 487.0891011*x - 68.520645
139  vs=-243.33862*x**4 + 668.06411*x**3 - 685.20113*x**2 + 308.04893*x - 43.737642
140  qmu=312.0d0
141  qkappa=57827.0d0
142  else if (r > r771 .and. r <= r670) then
143  rho=7.9565d0-6.4761d0*x+5.5283d0*x*x-3.0807d0*x*x*x
144  vp=-174.468866*x**2 + 286.37769*x - 106.034798
145  vs=-81.0865*x*x + 129.67095*x - 45.268933
146  qmu=312.0d0
147  qkappa=57827.0d0
148 !
149 !--- mantle: above d670
150 !
151  else if (r > r670 .and. r <= 5871000.d0) then
152  vp=-300.510146*x*x + 511.17372648*x - 206.265832
153  vs=-139.78275*x*x + 233.3097462*x - 91.0129372
154  rho=3.3d0 + (vs-4.4d0)*0.7d0
155  qmu=143.0d0
156  qkappa=57827.0d0
157 
158  else if (r > 5871000.d0 .and. r <= r400) then
159  vp=-601.0202917*x*x + 1063.3823*x - 459.9388738
160  vs=-145.2465705*x*x + 243.2807524*x - 95.561877
161  rho=3.3d0 + (vs - 4.4d0)*0.7d0
162  qmu=143.0d0
163  qkappa=57827.0d0
164 
165  else if (r > r400 .and. r <= r220) then
166  vp=25.042512155*x*x - 68.8367583*x + 51.4120272
167  vs=15.540158021*x*x - 40.2087657*x + 28.9578929
168  rho=3.3d0 + (vs - 4.4d0)*0.7d0
169  qmu=143.0d0
170  qkappa=57827.0d0
171 
172  else if (r > r220 .and. r <= r80) then
173  vp=27.0989608 - 19.473338*x
174  vs=13.920596 - 9.6309917*x
175  rho=3.3d0 + (vs - 4.4d0)*0.7d0
176  qmu=80.0d0
177  qkappa=57827.0d0
178 
179  else if (r > r80 .and. r <= rmoho) then
180  vp=26.7663028 - 19.13645*x
181  vs=13.4601434 - 9.164683*x
182  rho=3.3d0 + (vs - 4.4d0)*0.7d0
183  qmu=600.0d0
184  qkappa=57827.0d0
185 
186  else if (r > rmoho .and. r <= rmiddle_crust) then
187  rho=2.9d0
188  vp = 6.7d0
189  vs = 3.8d0
190  qmu=600.0d0
191  qkappa=57827.0d0
192  else
193  rho=2.6d0
194  vp = 6.0d0
195  vs = 3.5d0
196  qmu=600.0d0
197  qkappa=57827.0d0
198  endif
199 
200 
201 ! non-dimensionalize
202 ! time scaling (s^{-1}) is done with scaleval
203  scaleval=dsqrt(pi*grav*rhoav)
204  rho=rho*1000.0d0/rhoav
205  vp=vp*1000.0d0/(r_earth*scaleval)
206  vs=vs*1000.0d0/(r_earth*scaleval)
207 
208  end subroutine model_jp1d
