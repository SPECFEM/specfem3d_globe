133 
134  iglob = ibool_crust_mantle(i, j, k, ispec)

136 
137  enddo
138  enddo
139  enddo
140  enddo
141 
142  ! Create the parallel NetCDF file.
143  call checknc (nf90_create(fname, ior(nf90_netcdf4, nf90_mpiio), ncid, comm = comm, info = info_null))
144
