28 ! fix the non blocking arrays to assemble the slices inside each chunk: elements

60 ! mark all the points that are in the MPI buffers to assemble inside each chunk

72

74  mask_ibool(iboolright_eta(ipoin)) = .true.
75  enddo
76 
77 ! now label all the elements that have at least one corner belonging
78 ! to any of these buffers as elements that must contribute to the
79 ! first step of the calculations (performed on the edges before starting
80 ! the non blocking communications); there is no need to examine the inside
81 ! of the elements, checking their eight corners is sufficient
82  do ispec = 1,nspec
83  do k = 1,ngllz,ngllz-1
84  do j = 1,nglly,nglly-1
85  do i = 1,ngllx,ngllx-1
86  if (mask_ibool(ibool(i,j,k,ispec))) then
87  is_on_a_slice_edge(ispec) = .true.
88  goto 888
89  endif
90  enddo
91  enddo
92  enddo
93  888 continue
94  enddo
95 
96  deallocate(mask_ibool)
97 
98  end subroutine fix_non_blocking_slices
