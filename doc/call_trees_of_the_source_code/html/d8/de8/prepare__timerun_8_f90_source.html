227  subroutine prepare_timerun_mass_matrices()
228 
229  use specfem_par
230  use specfem_par_crustmantle
231  use specfem_par_innercore
232  use specfem_par_outercore
233  implicit none
234 
235  if (myrank == 0) then
236  write(imain,*) "preparing mass matrices"
237  call flush_imain()
238  endif
239 
240  ! mass matrices
241  !
242  ! in the case of Stacey boundary conditions, add C*deltat/2 contribution to the mass matrix
243  ! on Stacey edges for the crust_mantle and outer_core regions but not for the inner_core region
244  ! thus the mass matrix must be replaced by three mass matrices including the "C" damping matrix
245  !
246  ! if absorbing_conditions are not set or if NCHUNKS=6, only one mass matrix is needed
247  ! for the sake of performance, only "rmassz" array will be filled and "rmassx" & "rmassy" will be obsolete
248 
249 
250  ! mass matrices need to be assembled with MPI here once and for all
251  call prepare_timerun_rmass_assembly()
252 
253  ! checks that all the mass matrices are positive
254  ! ocean load
255  if (oceans_val) then
256  if (minval(rmass_ocean_load) <= 0._custom_real) &
257  call exit_mpi(myrank,'negative mass matrix term for the oceans')
258  endif
259 
260  ! checks mass matrices
261 
262  ! crust/mantle
263  if (minval(rmassx_crust_mantle) <= 0._custom_real) &
264  call exit_mpi(myrank,'negative mass matrix term for the crust_mantle rmassx')
265  if (minval(rmassy_crust_mantle) <= 0._custom_real) &
266  call exit_mpi(myrank,'negative mass matrix term for the crust_mantle rmassy')
267  if (minval(rmassz_crust_mantle) <= 0._custom_real) &
268  call exit_mpi(myrank,'negative mass matrix term for the crust_mantle rmassz')
269  ! kernel simulations
270  if (simulation_type == 3) then
271  if (minval(b_rmassx_crust_mantle) <= 0._custom_real) &
272  call exit_mpi(myrank,'negative mass matrix term for the b_crust_mantle b_rmassx')
273  if (minval(b_rmassy_crust_mantle) <= 0._custom_real) &
274  call exit_mpi(myrank,'negative mass matrix term for the b_crust_mantle b_rmassy')
275  if (minval(b_rmassz_crust_mantle) <= 0._custom_real) &
276  call exit_mpi(myrank,'negative mass matrix term for the b_crust_mantle b_rmassz')
277  endif
278 
279  ! inner core
280  ! checks mass matrices for rotation
281  if (minval(rmassx_inner_core) <= 0._custom_real) &
282  call exit_mpi(myrank,'negative mass matrix term for the inner core rmassx')
283  if (minval(rmassy_inner_core) <= 0._custom_real) &
284  call exit_mpi(myrank,'negative mass matrix term for the inner core rmassy')
285  if (minval(rmassz_inner_core) <= 0._custom_real) &
286  call exit_mpi(myrank,'negative mass matrix term for the inner core rmassz')
287  ! kernel simulations
288  if (simulation_type == 3) then
289  if (minval(b_rmassx_inner_core) <= 0._custom_real) &
290  call exit_mpi(myrank,'negative mass matrix term for the b_rmassx_inner_core')
291  if (minval(b_rmassy_inner_core) <= 0._custom_real) &
292  call exit_mpi(myrank,'negative mass matrix term for the b_rmassy_inner_core')
293  if (minval(b_rmassz_inner_core) <= 0._custom_real) &
294  call exit_mpi(myrank,'negative mass matrix term for the b_rmassz_inner_core')
295  endif
296 
297  ! outer core
298  if (minval(rmass_outer_core) <= 0._custom_real) &
299  call exit_mpi(myrank,'negative mass matrix term for the outer core')
300  if (simulation_type == 3) then
301  if (minval(b_rmass_outer_core) <= 0._custom_real) &
302  call exit_mpi(myrank,'negative mass matrix term for the outer core b_rmass')
303  endif
304 
305  ! mass matrix inversions
306  ! for efficiency, invert final mass matrix once and for all on each slice
307  ! ocean load
308  if (oceans_val) rmass_ocean_load = 1._custom_real / rmass_ocean_load
309 
310  ! mass matrices on Stacey edges
311  ! crust/mantle
312  if (((nchunks_val /= 6 .and. absorbing_conditions) .or. &
313  (rotation_val .and. exact_mass_matrix_for_rotation_val))) then
314  rmassx_crust_mantle = 1._custom_real / rmassx_crust_mantle
315  rmassy_crust_mantle = 1._custom_real / rmassy_crust_mantle
316  endif
317  if (simulation_type == 3) then
318  if (rotation_val .and. exact_mass_matrix_for_rotation_val) then
319  b_rmassx_crust_mantle = 1._custom_real / b_rmassx_crust_mantle
320  b_rmassy_crust_mantle = 1._custom_real / b_rmassy_crust_mantle
321  endif
322  endif
323  rmassz_crust_mantle = 1._custom_real / rmassz_crust_mantle
324  ! inner core
325  if (rotation_val .and. exact_mass_matrix_for_rotation_val) then
326  rmassx_inner_core = 1._custom_real / rmassx_inner_core
327  rmassy_inner_core = 1._custom_real / rmassy_inner_core
328  if (simulation_type == 3) then
329  b_rmassx_inner_core = 1._custom_real / b_rmassx_inner_core
330  b_rmassy_inner_core = 1._custom_real / b_rmassy_inner_core
331  endif
332  endif
333  rmassz_inner_core = 1._custom_real / rmassz_inner_core
334  ! outer core
335  rmass_outer_core = 1._custom_real / rmass_outer_core
336 
337  ! synchronizes processes
338  call synchronize_all()
339 
340  end subroutine prepare_timerun_mass_matrices

467  subroutine prepare_timerun_convert_coord()
468 
469 ! converts x/y/z into r/theta/phi spherical coordinates
470 
471  use specfem_par
472  use specfem_par_crustmantle
473  use specfem_par_innercore
474  use specfem_par_outercore
475  implicit none
476 
477  ! local parameters
478  integer :: i,ier
479  real(kind=CUSTOM_REAL) :: rval,thetaval,phival
480 
481  ! change x, y, z to r, theta and phi once and for all
482  !
483  ! note: we merged all 3 components into a single array rstore(:,:).
484  ! this makes it more suitable for performance improvements by compilers (better prefetch, more efficient memory access)
485 
486  ! convert in the crust and mantle
487  allocate(rstore_crust_mantle(ndim,nglob_crust_mantle),stat=ier)
488  if (ier /= 0) stop 'Error allocating rstore for crust/mantle'
489 
490 #ifdef DANIEL_TEST_OMP_RSTORE
491 !$OMP PARALLEL DEFAULT(NONE) &
492 !$OMP SHARED(xstore_crust_mantle,ystore_crust_mantle,zstore_crust_mantle,rstore_crust_mantle) &
493 !$OMP PRIVATE(i,rval,thetaval,phival)
494 !$OMP DO
495 #endif
496  do i = 1,nglob_crust_mantle
497  call xyz_2_rthetaphi(xstore_crust_mantle(i),ystore_crust_mantle(i),zstore_crust_mantle(i),rval,thetaval,phival)
498  rstore_crust_mantle(1,i) = rval
499  rstore_crust_mantle(2,i) = thetaval
500  rstore_crust_mantle(3,i) = phival
501  enddo
502 #ifdef DANIEL_TEST_OMP_RSTORE
503 !$OMP enddo
504 !$OMP END PARALLEL
505 #endif
506 
507  ! convert in the outer core
508  allocate(rstore_outer_core(ndim,nglob_outer_core),stat=ier)
509  if (ier /= 0) stop 'Error allocating rstore for outer core'
510 
511 #ifdef DANIEL_TEST_OMP_RSTORE
512 !$OMP PARALLEL DEFAULT(NONE) &
513 !$OMP SHARED(xstore_outer_core,ystore_outer_core,zstore_outer_core,rstore_outer_core) &
514 !$OMP PRIVATE(i,rval,thetaval,phival)
515 !$OMP DO
516 #endif
517  do i = 1,nglob_outer_core
518  call xyz_2_rthetaphi(xstore_outer_core(i),ystore_outer_core(i),zstore_outer_core(i),rval,thetaval,phival)
519  rstore_outer_core(1,i) = rval
520  rstore_outer_core(2,i) = thetaval
521  rstore_outer_core(3,i) = phival
522  enddo
523 #ifdef DANIEL_TEST_OMP_RSTORE
524 !$OMP enddo
525 !$OMP END PARALLEL
526 #endif
527 
528  ! convert in the inner core
529  allocate(rstore_inner_core(ndim,nglob_inner_core),stat=ier)
530  if (ier /= 0) stop 'Error allocating rstore for inner core'
531 
532 #ifdef DANIEL_TEST_OMP_RSTORE
533 !$OMP PARALLEL DEFAULT(NONE) &
534 !$OMP SHARED(xstore_inner_core,ystore_inner_core,zstore_inner_core,rstore_inner_core) &
535 !$OMP PRIVATE(i,rval,thetaval,phival)
536 !$OMP DO
537 #endif
538  do i = 1,nglob_inner_core
539  call xyz_2_rthetaphi(xstore_inner_core(i),ystore_inner_core(i),zstore_inner_core(i),rval,thetaval,phival)
540  rstore_inner_core(1,i) = rval
541  rstore_inner_core(2,i) = thetaval
542  rstore_inner_core(3,i) = phival
543  enddo
544 #ifdef DANIEL_TEST_OMP_RSTORE
545 !$OMP enddo
546 !$OMP END PARALLEL
547 #endif
548 
549  ! old x/y/z array not needed anymore
550  deallocate(xstore_crust_mantle,ystore_crust_mantle,zstore_crust_mantle)
551  deallocate(xstore_outer_core,ystore_outer_core,zstore_outer_core)
552  deallocate(xstore_inner_core,ystore_inner_core,zstore_inner_core)
553 
554  end subroutine prepare_timerun_convert_coord

560  subroutine prepare_timerun_constants()
561 
562 ! precomputes constants for time integration
563 
564  use specfem_par
565  implicit none
566 
567  if (myrank == 0) then
568  write(imain,*) "preparing constants"
569  call flush_imain()
570  endif
571 
572  ! define constants for the time integration
573  ! scaling to make displacement in meters and velocity in meters per second
574  scale_t = one/dsqrt(pi*grav*rhoav)
575  scale_t_inv = dsqrt(pi*grav*rhoav)
576 
577  scale_displ = r_earth
578  scale_displ_inv = one / scale_displ
579 
580  scale_veloc = scale_displ * scale_t_inv
581 
582  ! distinguish between single and double precision for reals
583  deltat = real(dt*scale_t_inv, kind=custom_real)
584  deltatover2 = 0.5d0*deltat
585  deltatsqover2 = 0.5d0*deltat*deltat
586
