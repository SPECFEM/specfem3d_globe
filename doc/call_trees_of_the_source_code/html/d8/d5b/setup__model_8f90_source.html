28  subroutine setup_model()

30  use meshfem3d_par

33  ! user output

35 
36  ! dynamic allocation of mesh arrays

38  allocate(ichunk_slice(0:nproctot-1))
39  allocate(iproc_xi_slice(0:nproctot-1))
40  allocate(iproc_eta_slice(0:nproctot-1))
41 
42  ! creates global slice addressing for solver
43  call create_addressing(nchunks,nproc,nproc_eta,nproc_xi,nproctot, &
44  addressing,ichunk_slice,iproc_xi_slice,iproc_eta_slice, &
45  output_files)
46 
47  ! this for the different counters (which are now different if the superbrick is cut in the outer core)
48  call setup_counters(nspec1d_radial,nspec2d_xi,nspec2d_eta,nglob1d_radial, &

52  nspec1d_radial_corner,nspec2d_xi_face, &
53  nspec2d_eta_face,nglob1d_radial_corner)
54 
55  ! distributes 3D models
56  call meshfem3d_models_broadcast(nspec,min_attenuation_period,max_attenuation_period, &
57  r80,r220,r670,rcmb,ricb, &
58  local_path)
59 
60  ! infos about additional mesh optimizations
61  if (myrank == 0) then
62  write(imain,*)
63  write(imain,*) 'additional mesh optimizations'
64  write(imain,*)
65 
66  ! moho stretching
67  write(imain,*) 'moho:'
68  if (crustal .and. case_3d) then
69  if (regional_moho_mesh) then
70  write(imain,*) ' enforcing regional 3-layer crust'
71  if (suppress_moho_stretching) then
72  write(imain,*) ' no element stretching for 3-D moho surface'
73  else
74  if (honor_deep_moho) then
75  write(imain,*) ' incorporating element stretching for 3-D moho surface with deep moho'
76  else
77  write(imain,*) ' incorporating element stretching for 3-D moho surface'
78  endif
79  endif
80  else
81  write(imain,'(a,i1,a)') ' default ',ner_crust,'-layer crust'
82  if (suppress_moho_stretching .or. (.not. topography)) then
83  write(imain,*) ' no element stretching for 3-D moho surface'
84  else
85  write(imain,*) ' incorporating element stretching for 3-D moho surface'
86  endif
87  endif
88  else
89  write(imain,*) ' no element stretching for 3-D moho surface'
90  endif
91  write(imain,*)
92 
93  ! internal topography
94  write(imain,*) 'internal topography 410/660:'
95  if ((.not. suppress_internal_topography) .and. &
96  (three_d_model == three_d_model_s362ani .or. three_d_model == three_d_model_s362wmani &
97  .or. three_d_model == three_d_model_s362ani_prem .or. three_d_model == three_d_model_s29ea &
98  .or. three_d_model == three_d_model_mantle_sh)) then
99  write(imain,*) ' incorporating element stretching for 3-D internal surfaces'
100  else
101  write(imain,*) ' no element stretching for 3-D internal surfaces'
102  endif
103  call flush_imain()
104  endif
105 
106  ! user output
107  if (myrank == 0) then
108  write(imain,*)
109  write(imain,*)
110  endif
111  call synchronize_all()
112 
113  end subroutine setup_model
