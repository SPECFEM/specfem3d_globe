163 subroutine write_1d_global_array_adios_dims(adios_handle, myrank, local_dim, sizeprocs, path)
164 
165  implicit none
166 
167  integer(kind=8), intent(in) :: adios_handle
168  integer, intent(in) :: sizeprocs, local_dim, myrank
169  character(len=*), intent(in) :: path
170 
171  integer :: adios_err
172  integer :: offset
173  integer(kind=8) :: global_dim ! should be long for large cases.
174 
175  ! global dimension
176  global_dim = int(local_dim, 8) * int(sizeprocs, 8)
177 
178  ! process offset
179  ! note: assumes that myrank starts is within [0,sizeprocs-1]
180  offset = local_dim * myrank
181 
182  call adios_write(adios_handle, path // "/local_dim", local_dim, adios_err)
183  call check_adios_err(myrank,adios_err)
184 
185  call adios_write(adios_handle, path // "/global_dim", global_dim, adios_err)
186  call check_adios_err(myrank,adios_err)
187 
188  call adios_write(adios_handle, path // "/offset", offset, adios_err)
189  call check_adios_err(myrank,adios_err)
190 
191 end subroutine write_1d_global_array_adios_dims
