275 extern EXTERN_LANG
276 void FC_FUNC_(transfer_adj_to_device,
277  TRANSFER_ADJ_TO_DEVICE)(long* Mesh_pointer_f,
278  int* h_nrec,
279  realw* h_adj_sourcearrays,
280  int* h_islice_selected_rec) {
281 
282  // transfers adjoint source arrays synchronously to GPU
283 
284  TRACE("transfer_adj_to_device");
285 
286  //get mesh pointer out of Fortran integer container
287  Mesh *mp = (Mesh *) *Mesh_pointer_f;
288 
289  // check if anything to do
290  if (mp->nadj_rec_local == 0) return;
291 
292  // total number of receivers/adjoint sources
293  int nrec = *h_nrec;
294 
295  // build slice of adj_sourcearrays because full array is *very *large.
296  //
297  // note: this copies array values for local adjoint sources at given time step "iadj_vec(it)"
298  // from large adj_sourcearrays array into h_adj_sourcearrays_slice
299  //
300  // dimension of global array version
301  // adj_sourcearrays is (NDIM,NGLLX,NGLLY,NGLLZ,nadj_rec_local,NTSTEP_BETWEEN_READ_ADJSRC)
302  // passed as function argument here is pointer to slice at time iadj_vec(it)
303  // which has dimension (NDIM,NGLLX,NGLLY,NGLLZ,nadj_rec_local)
304  int i,j,k,irec,irec_local;
305 
306  irec_local = 0;
307  for (irec = 0; irec < nrec; irec++) {
308  if (mp->myrank == h_islice_selected_rec[irec]) {
309  // takes only local sources
310  for (k = 0; k < NGLLX; k++) {
311  for (j = 0; j < NGLLX; j++) {
312  for (i = 0; i < NGLLX; i++) {
313  mp->h_adj_sourcearrays_slice[INDEX5(NDIM,NGLLX,NGLLX,NGLLX,0,i,j,k,irec_local)]
314  = h_adj_sourcearrays[INDEX5(NDIM,NGLLX,NGLLX,NGLLX,0,i,j,k,irec_local)];
315 
316  mp->h_adj_sourcearrays_slice[INDEX5(NDIM,NGLLX,NGLLX,NGLLX,1,i,j,k,irec_local)]
317  = h_adj_sourcearrays[INDEX5(NDIM,NGLLX,NGLLX,NGLLX,1,i,j,k,irec_local)];
318 
319  mp->h_adj_sourcearrays_slice[INDEX5(NDIM,NGLLX,NGLLX,NGLLX,2,i,j,k,irec_local)]
320  = h_adj_sourcearrays[INDEX5(NDIM,NGLLX,NGLLX,NGLLX,2,i,j,k,irec_local)];
321  }
322  }
323  }
324  // increases local receivers counter
325  irec_local++;
326  }
327  }
328  // check all local sources were added
329  if (irec_local != mp->nadj_rec_local) {
330  exit_on_error ("irec_local not equal to nadj_rec_local\n");
331  }
332 
333  // copies extracted array values onto GPU
334  gpuCopy_todevice_realw (&mp->d_adj_sourcearrays, mp->h_adj_sourcearrays_slice, mp->nadj_rec_local * NDIM * NGLL3);
335 
336  GPU_ERROR_CHECKING ("compute_add_sources_adjoint_gpu");
337 }
