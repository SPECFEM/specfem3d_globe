428  subroutine assemble_mpi_scalar(NPROC,NGLOB_AB,array_val, &
429  num_interfaces,max_nibool_interfaces, &
430  nibool_interfaces,ibool_interfaces, &
431  my_neighbours)
432 
433 ! blocking send/receive
434 
435  use constants
436 
437  implicit none
438 
439  integer :: NPROC
440  integer :: NGLOB_AB
441 
442  ! array to assemble
443  real(kind=CUSTOM_REAL), dimension(NGLOB_AB) :: array_val
444 
445  integer :: num_interfaces,max_nibool_interfaces
446  integer, dimension(num_interfaces) :: nibool_interfaces,my_neighbours
447  integer, dimension(max_nibool_interfaces,num_interfaces) :: ibool_interfaces
448 
449  ! local parameters
450  real(kind=CUSTOM_REAL), dimension(:,:), allocatable :: buffer_send_scalar
451  real(kind=CUSTOM_REAL), dimension(:,:), allocatable :: buffer_recv_scalar
452  integer, dimension(:), allocatable :: request_send_scalar
453  integer, dimension(:), allocatable :: request_recv_scalar
454 
455 
456  integer ipoin,iinterface,ier
457 
458 ! here we have to assemble all the contributions between partitions using MPI
459 
460  ! assemble only if more than one partition
461  if (nproc > 1) then
462 
463  allocate(buffer_send_scalar(max_nibool_interfaces,num_interfaces),stat=ier)
464  if (ier /= 0 ) stop 'Error allocating array buffer_send_scalar'
465  allocate(buffer_recv_scalar(max_nibool_interfaces,num_interfaces),stat=ier)
466  if (ier /= 0 ) stop 'Error allocating array buffer_recv_scalar'
467  allocate(request_send_scalar(num_interfaces),stat=ier)
468  if (ier /= 0 ) stop 'Error allocating array request_send_scalar'
469  allocate(request_recv_scalar(num_interfaces),stat=ier)
470  if (ier /= 0 ) stop 'Error allocating array request_recv_scalar'
471 
472  ! partition border copy into the buffer
473  do iinterface = 1, num_interfaces
474  do ipoin = 1, nibool_interfaces(iinterface)
475  buffer_send_scalar(ipoin,iinterface) = array_val(ibool_interfaces(ipoin,iinterface))
476  enddo
477  enddo
478 
479  ! send messages
480  do iinterface = 1, num_interfaces
481  ! non-blocking synchronous send request
482  call isend_cr(buffer_send_scalar(1:nibool_interfaces(iinterface),iinterface), &
483  nibool_interfaces(iinterface), &
484  my_neighbours(iinterface), &
485  itag, &
486  request_send_scalar(iinterface) &
487  )
488  ! receive request
489  call irecv_cr(buffer_recv_scalar(1:nibool_interfaces(iinterface),iinterface), &
490  nibool_interfaces(iinterface), &
491  my_neighbours(iinterface), &
492  itag, &
493  request_recv_scalar(iinterface) &
494  )
495  enddo
496 
497  ! wait for communications completion (recv)
498  do iinterface = 1, num_interfaces
499  call wait_req(request_recv_scalar(iinterface))
500  enddo
501 
502  ! adding contributions of neighbours
503  do iinterface = 1, num_interfaces
504  do ipoin = 1, nibool_interfaces(iinterface)
505  array_val(ibool_interfaces(ipoin,iinterface)) = &
506  array_val(ibool_interfaces(ipoin,iinterface)) + buffer_recv_scalar(ipoin,iinterface)
507  enddo
508  enddo
509 
510  ! wait for communications completion (send)
511  do iinterface = 1, num_interfaces
512  call wait_req(request_send_scalar(iinterface))
513  enddo
514 
515  deallocate(buffer_send_scalar)
516  deallocate(buffer_recv_scalar)
517  deallocate(request_send_scalar)
518  deallocate(request_recv_scalar)
519 
520  endif
521 
522  end subroutine assemble_mpi_scalar
