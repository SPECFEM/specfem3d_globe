29  subroutine get_attenuation_model_3d(iregion_code, &
30  one_minus_sum_beta, &

37

40  integer,intent(in) :: iregion_code
41 
42  ! note: factor_common,one_minus_sum_beta and factor_scale are real custom.
43  ! this is better, it works fine and these arrays are really huge
44  ! in the crust_mantle region, thus let us not double their size
45  integer,intent(in) :: vnspec
46  real(kind=CUSTOM_REAL), dimension(ATT1_VAL,ATT2_VAL,ATT3_VAL,vnspec),intent(out) :: one_minus_sum_beta, factor_scale

51  ! local parameters

82  factor_common(:,:,:,:,:) = factor_common(:,:,:,:,:) * scale_t_inv ! This is really tau_e, not factor_common

124  subroutine get_attenuation_property_values(tau_s, tau_e, factor_common, one_minus_sum_beta)
125 
126  use constants
127 
128  implicit none
129 
130  double precision, dimension(N_SLS),intent(in) :: tau_s, tau_e
131  double precision, dimension(N_SLS),intent(out) :: factor_common
132  double precision,intent(out) :: one_minus_sum_beta
133 
134  ! local parameters
135  double precision, dimension(N_SLS) :: tauinv,beta
136  integer :: i
137 
138  tauinv(:) = -1.0d0 / tau_s(:)
139 
140  beta(:) = 1.0d0 - tau_e(:) / tau_s(:)
141  one_minus_sum_beta = 1.0d0
142 
143  ! factor to scale from relaxed to unrelaxed moduli: see e.g. Komatitsch & Tromp 1999, eq. (7)
144  do i = 1,n_sls
145  one_minus_sum_beta = one_minus_sum_beta - beta(i)
146  enddo
147 
148 !ZN beware, here the expression differs from the strain used in memory variable equation (6) in D. Komatitsch and J. Tromp 1999,
149 !ZN here Brian Savage uses the engineering strain which are epsilon = 1/2*(grad U + (grad U)^T),
150 !ZN where U is the displacement vector and grad the gradient operator, i.e. there is a 1/2 factor difference between the two.
151 !ZN Both expressions are fine, but we need to keep in mind that if we have put the 1/2 factor there we need to remove it
152 !ZN from the expression in which we use the strain here in the code.
153 !ZN This is why here Brian Savage multiplies beta(:) * tauinv(:) by 2.0 to compensate for the 1/2 factor used before
154  factor_common(:) = 2.0d0 * beta(:) * tauinv(:)
155 
156  end subroutine get_attenuation_property_values

162  subroutine get_attenuation_scale_factor(T_c_source, tau_mu, tau_sigma, Q_mu, scale_factor)
163 
164  use constants, only: zero,one,two,pi,grav,rhoav,two_pi,n_sls,myrank
165  use specfem_par, only: scale_t
166 
167  implicit none
168 
169  double precision,intent(in) :: T_c_source,Q_mu
170  double precision, dimension(N_SLS),intent(in) :: tau_mu, tau_sigma
171 
172  double precision,intent(out) :: scale_factor
173 
174  ! local parameters
175  double precision :: f_c_source, w_c_source, f_0_prem
176  double precision :: factor_scale_mu0, factor_scale_mu
177  double precision :: a_val, b_val
178  double precision :: big_omega
179  integer :: i
180 
181  !--- compute central angular frequency of source (non dimensionalized)
182  f_c_source = one / t_c_source
183  w_c_source = two_pi * f_c_source
184 
185  !--- non dimensionalize PREM reference of 1 second
186  f_0_prem = one / ( one / scale_t)
187 
188 !--- quantity by which to scale mu_0 to get mu
189 ! this formula can be found for instance in
190 ! Liu, H. P., Anderson, D. L. and Kanamori, H., Velocity dispersion due to
191 ! anelasticity: implications for seismology and mantle composition,
192 ! Geophys. J. R. Astron. Soc., vol. 47, pp. 41-58 (1976)
193 ! and in Aki, K. and Richards, P. G., Quantitative seismology, theory and methods,
194 ! W. H. Freeman, (1980), second edition, sections 5.5 and 5.5.2, eq. (5.81) p. 170
195  factor_scale_mu0 = one + two * log(f_c_source / f_0_prem) / (pi * q_mu)
196 
197  !--- compute a, b and Omega parameters, also compute one minus sum of betas
198  a_val = one
199  b_val = zero
200 
201  do i = 1,n_sls
202  a_val = a_val - w_c_source * w_c_source * tau_mu(i) * &
203  (tau_mu(i) - tau_sigma(i)) / (1.d0 + w_c_source * w_c_source * tau_mu(i) * tau_mu(i))
204  b_val = b_val + w_c_source * (tau_mu(i) - tau_sigma(i)) / &
205  (1.d0 + w_c_source * w_c_source * tau_mu(i) * tau_mu(i))
206  enddo
207 
208  big_omega = a_val*(sqrt(1.d0 + b_val*b_val/(a_val*a_val))-1.d0)
209 
210  !--- quantity by which to scale mu to get mu_relaxed
211  factor_scale_mu = b_val * b_val / (two * big_omega)
212 
213  !--- total factor by which to scale mu0
214  scale_factor = factor_scale_mu * factor_scale_mu0
215 
216  !--- check that the correction factor is close to one
217  if (scale_factor < 0.8d0 .or. scale_factor > 1.2d0) then
218  print *,'Error: incorrect scale factor: ', scale_factor
219  call exit_mpi(myrank,'incorrect correction factor in attenuation model')
220  endif
221 
222  end subroutine get_attenuation_scale_factor
