29 ! SEA 99 model
30 !
31 ! contains relative Vs anomalies dVs/Vs from
32 ! SV-velocity model for SE Asia - W Pacific.
33 !
34 ! defined for:
35 ! -20.00 45.00 -- min, max latitude
36 ! 95.00 160.00 -- min, max longitude
37 ! and depths between 6 km to 860 km
38 !
39 ! computed by Lebedev and Nolet in 1999, to come out in JGR in 2003.
40 ! reference period: 50 s.
41 !--------------------------------------------------------------------------------------------------

149  subroutine model_sea99_s(radius,theta,phi,dvs)
150 
151 ! returns Vs perturbation (dvs) for given position r/theta/phi
152 
153  use constants
154  use model_sea99_s_par
155 
156  implicit none
157 
158  integer :: id1,i,ilat,ilon
159  double precision :: alat1,alon1,radius,theta,phi,dvs
160  double precision :: xxx,yyy,dep,pla,plo,xd1,dd1,dd2,ddd(2)
161 
162  ! initializes
163  dvs = 0.d0
164 
165  id1 = 0
166  xd1 = 0
167 
168  !----------------------- depth in the model ------------------
169  dep=r_earth_km*(r_unit_sphere - radius)
170  if (dep <= sea99_depth(1)) then
171  id1 = 1
172  xd1 = 0
173  else if (dep >= sea99_depth(sea99_ndep)) then
174  id1 = sea99_ndep
175  xd1 = 0
176  else
177  do i = 2, sea99_ndep
178  if (dep <= sea99_depth(i)) then
179  id1 = i-1
180  xd1 = (dep-sea99_depth(i-1)) / (sea99_depth(i) - sea99_depth(i-1))
181  exit
182  endif
183  enddo
184  endif
185 
186  !----------------------- value at a point ---------------------
187  !----- approximate interpolation, OK for the (dense) 1-degree sampling ------
188 
189  ! latitude / longitude in degree
190  pla = 90.0d0 - theta/degrees_to_radians
191  plo = phi/degrees_to_radians
192 
193  ! model defined for:
194  ! -20.00 45.00 -- min, max latitude
195  ! 95.00 160.00 -- min, max longitude
196  ! checks range
197  if (pla < alatmin .or. pla > alatmax &
198  .or. plo < alonmin .or. plo > alonmax ) return
199 
200  ! array indices
201  ilat = int((pla - alatmin)/sea99_ddeg) + 1
202  ilon = int((plo - alonmin)/sea99_ddeg) + 1
203  alat1 = alatmin + (ilat-1)*sea99_ddeg
204  alon1 = alonmin + (ilon-1)*sea99_ddeg
205 
206  do i = 1, 2
207  xxx = (pla-alat1)/sea99_ddeg
208  yyy = sea99_vs(ilat+1,ilon,id1+i-1)-sea99_vs(ilat,ilon,id1+i-1)
209  dd1 = sea99_vs(ilat,ilon,id1+i-1) + yyy*xxx
210  yyy = sea99_vs(ilat+1,ilon+1,id1+i-1)-sea99_vs(ilat,ilon+1,id1+i-1)
211  dd2 = sea99_vs(ilat,ilon+1,id1+i-1) + yyy*xxx
212  xxx = (plo-alon1)/sea99_ddeg
213  yyy = dd2 - dd1
214  ddd(i) = dd1 + yyy*xxx
215  enddo
216  dvs = ddd(1) + (ddd(2)-ddd(1)) * xd1
217 
218  ! checks perturbation
219  if (dvs > 1.d0) dvs = 0.0d0
220 
221  end subroutine model_sea99_s
