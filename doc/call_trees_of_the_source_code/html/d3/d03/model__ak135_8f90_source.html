30 ! Modified AK135 model:
31 !
32 ! Spherically symmetric isotropic AK135 model (Kennett et al., 1995).
33 ! modified to use the density and Q attenuation models of Montagner and Kennett (1995).
34 ! That modified model is traditionally called AK135-F,
35 ! see http://rses.anu.edu.au/seismology/ak135/ak135f.html for more details.
36 ! As we do not want to use the 300 m-thick mud layer from that model nor the ocean layer,
37 ! above the d120 discontinuity we switch back to the classical AK135 model of Kennett et al. (1995),
38 ! i.e., we use AK135-F below and AK135 above.
39 
40 ! B. L. N. Kennett, E. R. Engdahl and R. Buland,
41 ! Constraints on seismic velocities in the Earth from traveltimes,
42 ! Geophysical Journal International, volume 122, issue 1, pages 108-124 (1995),
43 ! DOI: 10.1111/j.1365-246X.1995.tb03540.x
44 !--------------------------------------------------------------------------------------------------
45 
46 ! J. P. Montagner and B. L. N. Kennett,
47 ! How to reconcile body-wave and normal-mode reference Earth models?,
48 ! Geophysical Journal International, volume 122, issue 1, pages 229-248 (1995)
49 
50 !! DK DK values below entirely checked and fixed by Dimitri Komatitsch in December 2012.
51

105  subroutine model_ak135(x,rho,vp,vs,Qkappa,Qmu,iregion_code)
106 
107  use constants
108  use model_ak135_par
109 
110  implicit none
111 
112 ! input:
113 ! radius r: meters
114 
115 ! output:
116 ! density rho: kg/m^3
117 ! compressional wave speed vp: km/s
118 ! shear wave speed vs: km/s
119 
120  double precision :: x,rho,vp,vs,Qmu,Qkappa
121  integer :: iregion_code
122 
123  ! local parameters
124  double precision :: r,frac,scaleval
125  integer :: i
126 
127  ! compute real physical radius in meters
128  r = x * r_earth
129 
130  i = 1
131  do while(r >= mak135_v_radius_ak135(i) .and. i /= nr_ak135f_no_mud)
132  i = i + 1
133  enddo
134 
135 ! make sure we stay in the right region and never take a point above
136 ! and a point below the ICB or the CMB and interpolate between them,
137 ! which would lead to a wrong value (keeping in mind that we interpolate
138 ! between points i-1 and i below)
139  if (iregion_code == iregion_inner_core .and. i > 24) i = 24
140 
141  if (iregion_code == iregion_outer_core .and. i < 26) i = 26
142  if (iregion_code == iregion_outer_core .and. i > 69) i = 69
143 
144  if (iregion_code == iregion_crust_mantle .and. i < 71) i = 71
145 
146  if (i == 1) then
147  rho = mak135_v_density_ak135(i)
148  vp = mak135_v_vp_ak135(i)
149  vs = mak135_v_vs_ak135(i)
150  qmu = mak135_v_qmu_ak135(i)
151  qkappa = mak135_v_qkappa_ak135(i)
152  else
153 
154 ! interpolate from radius_ak135(i-1) to r using the values at i-1 and i
155  frac = (r-mak135_v_radius_ak135(i-1))/(mak135_v_radius_ak135(i)-mak135_v_radius_ak135(i-1))
156 
157  rho = mak135_v_density_ak135(i-1) + frac * (mak135_v_density_ak135(i)-mak135_v_density_ak135(i-1))
158  vp = mak135_v_vp_ak135(i-1) + frac * (mak135_v_vp_ak135(i)-mak135_v_vp_ak135(i-1))
159  vs = mak135_v_vs_ak135(i-1) + frac * (mak135_v_vs_ak135(i)-mak135_v_vs_ak135(i-1))
160  qmu = mak135_v_qmu_ak135(i-1) + frac * (mak135_v_qmu_ak135(i)-mak135_v_qmu_ak135(i-1))
161  qkappa = mak135_v_qkappa_ak135(i-1) + frac * (mak135_v_qkappa_ak135(i)-mak135_v_qkappa_ak135(i-1))
162 
163  endif
164 
165 ! make sure Vs is zero in the outer core even if roundoff errors on depth
166 ! also set fictitious attenuation to a very high value (attenuation is not used in the fluid)
167  if (iregion_code == iregion_outer_core) then
168  vs = 0.d0
169  qkappa = 3000.d0
170  qmu = 3000.d0
171  endif
172 
173 ! non-dimensionalize
174 ! time scaling (s^{-1}) is done with scaleval
175  scaleval=dsqrt(pi*grav*rhoav)
176  rho=rho*1000.0d0/rhoav
177  vp=vp*1000.0d0/(r_earth*scaleval)
178  vs=vs*1000.0d0/(r_earth*scaleval)
179 
180  end subroutine model_ak135
