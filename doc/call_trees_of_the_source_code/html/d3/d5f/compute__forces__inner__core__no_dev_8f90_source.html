267  ! 0 0 0 c44 0 0

327  sigma_zz = lambdalplus2mul*duzdzl + lambdal*duxdxl_plus_duydyl

329  sigma_xy = mul*duxdyl_plus_duydxl

331  sigma_yz = mul*duzdyl_plus_duydzl

333  endif

429  rho_s_h(i,j,k,3) = real(factor * (sx_l * Hxzl + sy_l * Hyzl + sz_l * Hzzl), kind=custom_real)

436  tempz1(i,j,k) = jacobianl * (sigma_xz*xixl + sigma_yz*xiyl + sigma_zz*xizl)

501  enddo
502  enddo

504  ! sum contributions from each element to the global mesh and add gravity terms
505  do k = 1,ngllz
506  do j = 1,nglly
507  do i = 1,ngllx
508  iglob = ibool(i,j,k,ispec)
509  accel_inner_core(1,iglob) = accel_inner_core(1,iglob) + sum_terms(i,j,k,1)
510  accel_inner_core(2,iglob) = accel_inner_core(2,iglob) + sum_terms(i,j,k,2)
511  accel_inner_core(3,iglob) = accel_inner_core(3,iglob) + sum_terms(i,j,k,3)
512  enddo
513  enddo
514  enddo
515 
516  ! use Runge-Kutta scheme to march memory variables in time
517  ! convention for attenuation
518  ! term in xx = 1
519  ! term in yy = 2
520  ! term in xy = 3
521  ! term in xz = 4
522  ! term in yz = 5
523  ! term in zz not computed since zero trace
524  ! This is because we only implement Q_\mu attenuation and not Q_\kappa.
525  ! Note that this does *NOT* imply that there is no attenuation for P waves
526  ! because for Q_\kappa = infinity one gets (see for instance Dahlen and Tromp (1998)
527  ! equation (9.59) page 350): Q_\alpha = Q_\mu * 3 * (V_p/V_s)^2 / 4
528  ! therefore Q_\alpha is not zero; for instance for V_p / V_s = sqrt(3)
529  ! we get Q_\alpha = (9 / 4) * Q_\mu = 2.25 * Q_\mu
530

532  ! updates R_memory

542  att1_val,att2_val,att3_val,vnspec,factor_common, &

544  c44store,muvstore, &
545  epsilondev_xx,epsilondev_yy,epsilondev_xy, &
546  epsilondev_xz,epsilondev_yz, &
547  epsilondev_loc)
548  endif
549  endif
550 
551  ! save deviatoric strain for Runge-Kutta scheme
552  if (compute_and_store_strain) then
553  do k = 1,ngllz
554  do j = 1,nglly
555  do i = 1,ngllx
556  epsilondev_xx(i,j,k,ispec) = epsilondev_loc(i,j,k,1)
557  epsilondev_yy(i,j,k,ispec) = epsilondev_loc(i,j,k,2)
558  epsilondev_xy(i,j,k,ispec) = epsilondev_loc(i,j,k,3)
559  epsilondev_xz(i,j,k,ispec) = epsilondev_loc(i,j,k,4)
560  epsilondev_yz(i,j,k,ispec) = epsilondev_loc(i,j,k,5)
561  enddo
562  enddo
563  enddo
564  endif
565 
566  endif ! end test to exclude fictitious elements in central cube
567 
568  enddo ! spectral element loop
