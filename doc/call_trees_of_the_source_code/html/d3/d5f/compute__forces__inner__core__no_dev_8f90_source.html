444  tempx3l = tempx3l + tempx3(i,j,l)*fac3
445  tempy3l = tempy3l + tempy3(i,j,l)*fac3
446  tempz3l = tempz3l + tempz3(i,j,l)*fac3
447  enddo

451  fac3 = wgllwgll_xy(i,j)

454  sum_terms(i,j,k,2) = - (fac1*tempy1l + fac2*tempy2l + fac3*tempy3l)
455  sum_terms(i,j,k,3) = - (fac1*tempz1l + fac2*tempz2l + fac3*tempz3l)

460  sum_terms(i,j,k,3) = sum_terms(i,j,k,3) + rho_s_h(3,i,j,k)
461  endif
462  enddo
463  enddo
464  enddo
465 
466  ! sum contributions from each element to the global mesh and add gravity terms
467  do k = 1,ngllz
468  do j = 1,nglly
469  do i = 1,ngllx
470  iglob = ibool(i,j,k,ispec)
471  accel_inner_core(1,iglob) = accel_inner_core(1,iglob) + sum_terms(i,j,k,1)
472  accel_inner_core(2,iglob) = accel_inner_core(2,iglob) + sum_terms(i,j,k,2)
473  accel_inner_core(3,iglob) = accel_inner_core(3,iglob) + sum_terms(i,j,k,3)
474  enddo
475  enddo
476  enddo
477 
478  ! use Runge-Kutta scheme to march memory variables in time
479  ! convention for attenuation
480  ! term in xx = 1
481  ! term in yy = 2
482  ! term in xy = 3
483  ! term in xz = 4
484  ! term in yz = 5
485  ! term in zz not computed since zero trace
486  ! This is because we only implement Q_\mu attenuation and not Q_\kappa.
487  ! Note that this does *NOT* imply that there is no attenuation for P waves
488  ! because for Q_\kappa = infinity one gets (see for instance Dahlen and Tromp (1998)
489  ! equation (9.59) page 350): Q_\alpha = Q_\mu * 3 * (V_p/V_s)^2 / 4
490  ! therefore Q_\alpha is not zero; for instance for V_p / V_s = sqrt(3)
491  ! we get Q_\alpha = (9 / 4) * Q_\mu = 2.25 * Q_\mu
492 
493  if (attenuation_val .and. .not. partial_phys_dispersion_only_val) then
494  ! updates R_memory
495  if (use_lddrk) then
496  call compute_element_att_memory_ic_lddrk(ispec,r_xx,r_yy,r_xy,r_xz,r_yz, &
497  r_xx_lddrk,r_yy_lddrk,r_xy_lddrk,r_xz_lddrk,r_yz_lddrk, &
498  att1_val,att2_val,att3_val,vnspec,factor_common, &
499  c44store,muvstore, &

502  else
503  call compute_element_att_memory_ic(ispec,r_xx,r_yy,r_xy,r_xz,r_yz, &
504  att1_val,att2_val,att3_val,vnspec,factor_common, &
505  alphaval,betaval,gammaval, &
506  c44store,muvstore, &
507  epsilondev_xx,epsilondev_yy,epsilondev_xy, &
508  epsilondev_xz,epsilondev_yz, &
509  epsilondev_loc)
510  endif
511  endif
512 
513  ! save deviatoric strain for Runge-Kutta scheme
514  if (compute_and_store_strain) then
515  do k = 1,ngllz
516  do j = 1,nglly
517  do i = 1,ngllx
518  epsilondev_xx(i,j,k,ispec) = epsilondev_loc(i,j,k,1)
519  epsilondev_yy(i,j,k,ispec) = epsilondev_loc(i,j,k,2)
520  epsilondev_xy(i,j,k,ispec) = epsilondev_loc(i,j,k,3)
521  epsilondev_xz(i,j,k,ispec) = epsilondev_loc(i,j,k,4)
522  epsilondev_yz(i,j,k,ispec) = epsilondev_loc(i,j,k,5)
523  enddo
524  enddo
525  enddo
526  endif
527 
528  endif ! end test to exclude fictitious elements in central cube
529 
530  enddo ! spectral element loop
