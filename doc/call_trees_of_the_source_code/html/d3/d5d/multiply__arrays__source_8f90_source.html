49 ! multiplies acceleration with inverse of mass matrices in crust/mantle,solid inner core region
50 
51  use constants_solver, only: custom_real,ndim,rotation_val
52 
53  implicit none
54 
55  real(kind=CUSTOM_REAL),intent(in) :: two_omega_earth
56 
57  ! crust/mantle
58  integer,intent(in) :: NGLOB_CM
59  ! velocity & acceleration
60  real(kind=CUSTOM_REAL), dimension(NDIM,NGLOB_CM),intent(in) :: veloc_cm
61  real(kind=CUSTOM_REAL), dimension(NDIM,NGLOB_CM),intent(inout) :: accel_cm
62  ! mass matrices
63  real(kind=CUSTOM_REAL), dimension(NGLOB_CM),intent(in) :: rmassx_cm,rmassy_cm,rmassz_cm
64 
65  ! inner core
66  integer,intent(in) :: NGLOB_IC
67  ! velocity & acceleration
68  real(kind=CUSTOM_REAL), dimension(NDIM,NGLOB_IC),intent(in) :: veloc_ic
69  real(kind=CUSTOM_REAL), dimension(NDIM,NGLOB_IC),intent(inout) :: accel_ic
70  ! mass matrices
71  real(kind=CUSTOM_REAL), dimension(NGLOB_IC),intent(in) :: rmassx_ic,rmassy_ic,rmassz_ic
72 
73  ! local parameters
74  integer :: i
75 
76  ! note: mass matrices
77  !
78  ! in the case of Stacey boundary conditions, add C*deltat/2 contribution to the mass matrix
79  ! on Stacey edges for the crust_mantle and outer_core regions but not for the inner_core region
80  ! thus the mass matrix must be replaced by three mass matrices including the "C" damping matrix
81  !
82  ! if absorbing_conditions are not set or if NCHUNKS=6, only one mass matrix is needed
83  ! for the sake of performance, only "rmassz" array will be filled and "rmassx" & "rmassy" will be obsolete
84 
85  ! updates acceleration w/ rotation in elastic region
86 
87  ! see input call, differs for corrected mass matrices for rmassx,rmassy,rmassz
88 
89  ! divides by mass matrix
90  ! (rmassx holds inverted mass matrix; numerically multiplication is faster than division)
91  if (rotation_val) then
92  ! adds contributions due to rotation
93 !$OMP PARALLEL DEFAULT(NONE) &
94 !$OMP SHARED(two_omega_earth, &
95 !$OMP NGLOB_CM, accel_cm, veloc_cm, rmassx_cm, rmassy_cm, rmassz_cm, &
96 !$OMP NGLOB_IC, accel_ic, veloc_ic, rmassx_ic, rmassy_ic, rmassz_ic) &
97 !$OMP PRIVATE(i)
98 
99  ! crust/mantle
100 !$OMP DO
101  do i = 1,nglob_cm
102  accel_cm(1,i) = accel_cm(1,i)*rmassx_cm(i) + two_omega_earth * veloc_cm(2,i)
103  accel_cm(2,i) = accel_cm(2,i)*rmassy_cm(i) - two_omega_earth * veloc_cm(1,i)
104  accel_cm(3,i) = accel_cm(3,i)*rmassz_cm(i)
105  enddo
106 !$OMP enddo NOWAIT
107 
108  ! inner core
109 !$OMP DO
110  do i = 1,nglob_ic
111  accel_ic(1,i) = accel_ic(1,i)*rmassx_ic(i) + two_omega_earth * veloc_ic(2,i)
112  accel_ic(2,i) = accel_ic(2,i)*rmassy_ic(i) - two_omega_earth * veloc_ic(1,i)
113  accel_ic(3,i) = accel_ic(3,i)*rmassz_ic(i)
114  enddo
115 !$OMP enddo
116 !$OMP END PARALLEL
117 
118  else
119  ! no rotation
120 !$OMP PARALLEL DEFAULT(NONE) &
121 !$OMP SHARED(two_omega_earth, &
122 !$OMP NGLOB_CM, accel_cm, veloc_cm, rmassx_cm, rmassy_cm, rmassz_cm, &
123 !$OMP NGLOB_IC, accel_ic, veloc_ic, rmassx_ic, rmassy_ic, rmassz_ic) &
124 !$OMP PRIVATE(i)
125 
126  ! crust/mantle
127 !$OMP DO
128  do i = 1,nglob_cm
129  accel_cm(1,i) = accel_cm(1,i)*rmassx_cm(i)
130  accel_cm(2,i) = accel_cm(2,i)*rmassy_cm(i)
131  accel_cm(3,i) = accel_cm(3,i)*rmassz_cm(i)
132  enddo
133 !$OMP enddo NOWAIT
134 
135  ! inner core
136 !$OMP DO
137  do i = 1,nglob_ic
138  accel_ic(1,i) = accel_ic(1,i)*rmassx_ic(i)
139  accel_ic(2,i) = accel_ic(2,i)*rmassy_ic(i)
140  accel_ic(3,i) = accel_ic(3,i)*rmassz_ic(i)
141  enddo
142 !$OMP enddo
143 !$OMP END PARALLEL
144 
145  endif
146 
147  end subroutine multiply_accel_elastic
