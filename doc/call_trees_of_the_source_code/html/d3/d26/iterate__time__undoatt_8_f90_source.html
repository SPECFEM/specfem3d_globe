134  !! (purposely) huge and designed to use almost all the memory available (by carefully optimizing the
135  !! value of NT_DUMP_ATTENUATION); when writing to these buffers, it will then be OK to use non-blocking writes
136  !! from the device to the host, since we do not reuse the content of these buffers until much later, in a second part

138  !! because we then use the values read immediately (one value at a time, but to reduce the total number of reads

142  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating b_displ_cm_store_buffer')
143  allocate(b_displ_oc_store_buffer(nglob_outer_core_adjoint,nt_dump_attenuation),stat=ier)
144  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating b_displ_oc_store_buffer')
145  allocate(b_accel_oc_store_buffer(nglob_outer_core_adjoint,nt_dump_attenuation),stat=ier)
146  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating b_accel_oc_store_buffer')
147  allocate(b_displ_ic_store_buffer(ndim,nglob_inner_core_adjoint,nt_dump_attenuation),stat=ier)
148  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating b_displ_ic_store_buffer')
149 
150  ! noise kernel for source strength (sigma_kernel) needs buffer for reconstructed noise_surface_movie array,
151  ! otherwise we need file i/o which will considerably slow down performance
152  if (noise_tomography == 3) then
153  allocate(b_noise_surface_movie_buffer(ndim,ngllx,nglly,nspec_top,nt_dump_attenuation),stat=ier)
154  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating b_noise_surface_movie_buffer')
155  endif
156  endif
157 
158  ! synchronize all processes to make sure everybody is ready to start time loop
159  call synchronize_all()
160  if (myrank == 0) write(imain,*) 'All processes are synchronized before time loop'
161 
162  if (myrank == 0) then
163  write(imain,*)
164  write(imain,*) 'Starting time iteration loop (undoatt)...'
165  write(imain,*)
166  call flush_imain()
167  endif
168 
169  ! create an empty file to monitor the start of the simulation
170  if (myrank == 0) then
171  open(unit=iout,file=trim(output_files)//'/starttimeloop.txt',status='unknown',action='write')
172  write(iout,*) 'hello, starting time loop'
173  close(iout)
174  endif
175 
176  ! initializes variables for writing seismograms
177  seismo_offset = it_begin-1
178  seismo_current = 0
179 
180  ! initializes time increments
181  it = 0
182 
183  ! get MPI starting time
184  time_start = wtime()
185 
186  ! *********************************************************
187  ! ************* MAIN LOOP OVER THE TIME STEPS *************
188  ! *********************************************************
189 
190  if (exact_undoing_to_disk) call setup_exact_undoing_to_disk()
191 
192  ! loops over time subsets
193  do iteration_on_subset = 1, nsubset_iterations
194 
195  ! wavefield storage
196  if (simulation_type == 1 .and. save_forward) then
197  ! saves forward wavefields
198  call save_forward_arrays_undoatt()
199 
200  else if (simulation_type == 3) then
201  ! reads in last stored forward wavefield
202  call read_forward_arrays_undoatt()
203 
204  ! note: after reading the restart files of displacement back from disk, recompute the strain from displacement;
205  ! this is better than storing the strain to disk as well, which would drastically increase I/O volume
206  ! computes strain based on current backward/reconstructed wavefield
207  if (compute_and_store_strain) call it_compute_strain_att_backward()
208  endif
209 
210  ! time loop within this iteration subset
211  select case (simulation_type)
212  case (1, 2)
213  ! forward and adjoint simulations
214 
215  ! increment end of this subset
216  if (iteration_on_subset < nsubset_iterations) then
217  ! takes full length of subset
218  it_subset_end = nt_dump_attenuation
219  else
220  ! loops over remaining steps in last subset
221  it_subset_end = nstep - (iteration_on_subset-1)*nt_dump_attenuation
222  endif
223  ! checks end index
224  if (it_subset_end > nt_dump_attenuation) &
225  call exit_mpi(myrank,'Error invalid buffer index for undoing attenuation')
226 
227  ! subset loop
228  do it_of_this_subset = 1, it_subset_end
229 
230  it = it + 1
231 
232  ! simulation status output and stability check
233  if (mod(it,ntstep_between_output_info) == 0 .or. it == it_begin + 4 .or. it == it_end) then
234  call check_stability()
235  if (i_am_running_on_a_slow_node) goto 100
236  endif
237 
238  do istage = 1, nstage_time_scheme ! is equal to 1 if Newmark because only one stage then
239 
240  if (use_lddrk) then
241  ! update displacement using Runge-Kutta time scheme
242  call update_displ_lddrk()
243  else
244  ! update displacement using Newmark time scheme
245  call update_displ_newmark()
246  endif
247 
248  ! acoustic solver for outer core
249  ! (needs to be done first, before elastic one)
250  call compute_forces_acoustic()
251 
252  ! elastic solver for crust/mantle and inner core
253  call compute_forces_viscoelastic()
254 
255  enddo ! istage
256 
257  ! write the seismograms with time shift
258  call write_seismograms()
259 
260  ! outputs movie files
261  if (movie_surface .or. movie_volume) call write_movie_output()
262 
263  ! first step of noise tomography, i.e., save a surface movie at every time step
264  ! modified from the subroutine 'write_movie_surface'
265  if (noise_tomography == 1) then
266  call noise_save_surface_movie()
267  endif
268 
269  ! updates VTK window
270  if (vtk_mode) then
271  call it_update_vtkwindow()
272  endif
273 
274  enddo ! subset loop
275 
276  case (3)
277  ! kernel simulations
278 
279  ! intermediate storage of it and seismo_current positions
280  it_temp = it
281  seismo_current_temp = seismo_current
282 
283  ! increment end of this subset
284  if (iteration_on_subset == 1) then
285  ! loops over remaining steps in last forward subset
286  it_subset_end = nstep - (nsubset_iterations-1)*nt_dump_attenuation
287  else
288  ! takes full length of subset
289  it_subset_end = nt_dump_attenuation
290  endif
291  ! checks end index
292  if (it_subset_end > nt_dump_attenuation) &
293  call exit_mpi(myrank,'Error invalid buffer index for undoing attenuation')
294 
295  ! reconstructs forward wavefields based on last stored wavefield data
296 
297  ! note: we step forward in time here, starting from last snapshot.
298  ! the newly computed, reconstructed forward wavefields (b_displ_..) get stored in buffers.
299 
300  ! subset loop
301  do it_of_this_subset = 1, it_subset_end
302 
303  it = it + 1
304 
305  ! simulation status output and stability check
306  if (mod(it,ntstep_between_output_info) == 0 .or. it == it_begin + 4 .or. it == it_end) then
307  call check_stability_backward()
308  endif
309 
310  do istage = 1, nstage_time_scheme ! is equal to 1 if Newmark because only one stage then
311 
312  if (use_lddrk) then
313  ! update displacement using Runge-Kutta time scheme
314  call update_displ_lddrk_backward()
315  else
316  ! update displacement using Newmark time scheme
317  call update_displ_newmark_backward()
318  endif
319 
320  ! acoustic solver for outer core
321  ! (needs to be done first, before elastic one)
322  call compute_forces_acoustic_backward()
323 
324  ! elastic solver for crust/mantle and inner core
325  call compute_forces_viscoelastic_backward()
326 
327  enddo ! istage
328 
329  ! transfers wavefields from GPU to CPU for buffering
330  if (gpu_mode) then
331  ! daniel debug: check if these transfers could be made async to overlap
332  call transfer_b_displ_cm_from_device(ndim*nglob_crust_mantle,b_displ_crust_mantle,mesh_pointer)
333  call transfer_b_displ_ic_from_device(ndim*nglob_inner_core,b_displ_inner_core,mesh_pointer)
334  call transfer_b_displ_oc_from_device(nglob_outer_core,b_displ_outer_core,mesh_pointer)
335  call transfer_b_accel_oc_from_device(nglob_outer_core,b_accel_outer_core,mesh_pointer)
336  endif
337 
338  ! stores wavefield in buffers
339  b_displ_cm_store_buffer(:,:,it_of_this_subset) = b_displ_crust_mantle(:,:)
340  b_displ_oc_store_buffer(:,it_of_this_subset) = b_displ_outer_core(:)
341  b_accel_oc_store_buffer(:,it_of_this_subset) = b_accel_outer_core(:)
342  b_displ_ic_store_buffer(:,:,it_of_this_subset) = b_displ_inner_core(:,:)
343 
344  ! for noise kernel
345  if (noise_tomography == 3) then
346  b_noise_surface_movie_buffer(:,:,:,:,it_of_this_subset) = noise_surface_movie(:,:,:,:)
347  endif
348  enddo ! subset loop
349 
350  ! resets current it and seismo_current positions
351  it = it_temp
352  seismo_current = seismo_current_temp
353 
354  ! computes strain based on current adjoint wavefield
355  if (compute_and_store_strain) call it_compute_strain_att()
356 
357  ! adjoint wavefield simulation
358  do it_of_this_subset = 1, it_subset_end
359 
360  ! reads backward/reconstructed wavefield from buffers
361  ! note: uses wavefield at corresponding time (NSTEP - it + 1 ), i.e. we have now time-reversed wavefields
362  ! crust/mantle
363  do j = 1,nglob_crust_mantle_adjoint
364  do i = 1, ndim
365  b_displ_crust_mantle(i,j) = b_displ_cm_store_buffer(i,j,it_subset_end-it_of_this_subset+1)
366  enddo
367  enddo
368  ! outer core
369  do j = 1,nglob_outer_core_adjoint
370  b_displ_outer_core(j) = b_displ_oc_store_buffer(j,it_subset_end-it_of_this_subset+1)
371  b_accel_outer_core(j) = b_accel_oc_store_buffer(j,it_subset_end-it_of_this_subset+1)
372  enddo
373  ! inner core
374  do j = 1,nglob_inner_core_adjoint
375  do i = 1, ndim
376  b_displ_inner_core(i,j) = b_displ_ic_store_buffer(i,j,it_subset_end-it_of_this_subset+1)
377  enddo
378  enddo
379 
380  ! for noise kernel
381  if (noise_tomography == 3) then
382  noise_surface_movie(:,:,:,:) = b_noise_surface_movie_buffer(:,:,:,:,it_subset_end-it_of_this_subset+1)
383  endif
384 
385  ! transfers wavefields from CPU to GPU
386  if (gpu_mode) then
387  ! daniel debug: check if these transfers could be made async to overlap
388  call transfer_b_displ_cm_to_device(ndim*nglob_crust_mantle,b_displ_crust_mantle,mesh_pointer)
389  call transfer_b_displ_ic_to_device(ndim*nglob_inner_core,b_displ_inner_core,mesh_pointer)
390  call transfer_b_displ_oc_to_device(nglob_outer_core,b_displ_outer_core,mesh_pointer)
391  call transfer_b_accel_oc_to_device(nglob_outer_core,b_accel_outer_core,mesh_pointer)
392  endif
393 
394  it = it + 1
395 
396  ! simulation status output and stability check
397  if (mod(it,ntstep_between_output_info) == 0 .or. it == it_begin + 4 .or. it == it_end) then
398  call check_stability()
399  if (i_am_running_on_a_slow_node) goto 100
400  endif
401 
402  ! computes adjoint wavefield
403  do istage = 1, nstage_time_scheme ! is equal to 1 if Newmark because only one stage then
404 
405  if (use_lddrk) then
406  ! update displacement using Runge-Kutta time scheme
407  call update_displ_lddrk()
408  else
409  ! update displacement using Newmark time scheme
410  call update_displ_newmark()
411  endif
412 
413  ! acoustic solver for outer core
414  ! (needs to be done first, before elastic one)
415  call compute_forces_acoustic()
416 
417  ! elastic solver for crust/mantle and inner core
418  call compute_forces_viscoelastic()
419 
420  enddo ! istage
421 
422  ! write the seismograms with time shift
423  call write_seismograms()
424 
425  ! kernel computation
426  ! adjoint simulations: kernels
427  call compute_kernels()
428 
429  enddo ! subset loop
430 
431  end select ! SIMULATION_TYPE
432 
433  !
434  !---- end of time iteration loop
435  !
436  enddo ! end of main time loop
437 
438  100 continue
439 
440  ! frees undo_attenuation buffers
441  if (simulation_type == 3) then
442  deallocate(b_displ_cm_store_buffer, &
443  b_displ_oc_store_buffer, &
444  b_accel_oc_store_buffer, &
445  b_displ_ic_store_buffer)
446  ! noise simulations
447  if (noise_tomography == 3) then
448  deallocate(b_noise_surface_movie_buffer)
449  endif
450  endif
451 
452  ! close the huge file that contains a dump of all the time steps to disk
453  if (exact_undoing_to_disk) call finish_exact_undoing_to_disk()
454 
455  ! user output of runtime
456  call print_elapsed_time()
457 
458  ! Transfer fields from GPU card to host for further analysis
459  if (gpu_mode) call it_transfer_from_gpu()
460 
461 !---- close energy file
462  if (output_energy .and. myrank == 0) close(iout_energy)
463 
464  ! safety check of last time loop increment
465  if (it /= it_end) then
466  print *,'Error time increments: it_end = ',it_end,' and last it = ',it,' do not match!'
467  call exit_mpi(myrank,'Error invalid time increment ending')
468  endif
469 
470  end subroutine iterate_time_undoatt
