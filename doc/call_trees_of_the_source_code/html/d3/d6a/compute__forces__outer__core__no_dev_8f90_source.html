132  !!! can merge these loops because NGLLX = NGLLY = NGLLZ enddo

139  xixl = xix(i,j,k,ispec)
140  xiyl = xiy(i,j,k,ispec)
141  xizl = xiz(i,j,k,ispec)

143  etayl = etay(i,j,k,ispec)

147  gammazl = gammaz(i,j,k,ispec)

151  - xiyl*(etaxl*gammazl-etazl*gammaxl) &
152  + xizl*(etaxl*gammayl-etayl*gammaxl))

155  dpotentialdyl = xiyl*tempx1l + etayl*tempx2l + gammayl*tempx3l

190  ! add (chi/rho)grad(rho) term in no gravity case
191  if (.not. gravity_val) then
192  ! With regards to the non-gravitating case: we cannot set N^2 = 0 *and* let g = 0.
193  ! We can *either* assume N^2 = 0 but keep gravity g, *or* we can assume that gravity
194  ! is negligible to begin with, as in our GJI 2002a, in which case N does not arise.
195  ! We get:
196  !
197  ! \ddot\chi = \rho^{-1}\kappa\bdel\cdot(\bdel\chi+\chi\bdel\ln\rho)
198  !
199  ! Then the displacement is
200  !
201  ! \bu = \bdel\chi+\chi\bdel\ln\rho = \rho^{-1}\bdel(\rho\chi)
202  !
203  ! and the pressure is
204  !
205  ! p = -\rho\ddot{\chi}
206  !
207  ! Thus in our 2002b GJI paper eqn (21) is wrong, and equation (41)
208  ! in our AGU monograph is incorrect; these equations should be replaced by
209  !
210  ! \ddot\chi = \rho^{-1}\kappa\bdel\cdot(\bdel\chi+\chi\bdel\ln\rho)
211  !
212  ! Note that the fluid potential we use in GJI 2002a differs from the one used here:
213  !
214  ! \chi_GJI2002a = \rho\partial\t\chi
215  !
216  ! such that
217  !
218  ! \bv = \partial_t\bu=\rho^{-1}\bdel\chi_GJI2002a (GJI 2002a eqn 20)
219  !
220  ! p = - \partial_t\chi_GJI2002a (GJI 2002a eqn 19)
221

225

228  phi = dble(rstore(3,iglob))
229 
230  cos_theta = dcos(theta)

232  cos_phi = dcos(phi)
233  sin_phi = dsin(phi)
234

236

240  grad_z_ln_rho = cos_theta * d_ln_density_dr_table(int_radius)

245  dpotentialdzl = dpotentialdzl + displfluid(iglob) * grad_z_ln_rho
246 
247 
248  else

253

271

273  ! integrate and multiply by rho / Kappa

275  gyl = sin_theta*sin_phi

278  ! distinguish between single and double precision for reals
279  gravity_term(i,j,k) = &
280  real(minus_rho_g_over_kappa_fluid(int_radius) * dble(jacobianl) * wgll_cube(i,j,k) & * (dble(dpotentialdx_with_rot) * gxl & + dble(dpotentialdy_with_rot) * gyl & + dble(dpotentialdzl) * gzl), &
281  kind=custom_real)
282 
283  ! divergence of displacement field with gravity on
284  ! note: these calculations are only considered for SIMULATION_TYPE == 1 .and. SAVE_FORWARD
285  ! and one has set MOVIE_VOLUME_TYPE == 4 when MOVIE_VOLUME is .true.;

291  endif
292 
293  endif
294 
295  tempx1(i,j,k) = jacobianl*(xixl*dpotentialdx_with_rot + xiyl*dpotentialdy_with_rot + xizl*dpotentialdzl)
296  tempx2(i,j,k) = jacobianl*(etaxl*dpotentialdx_with_rot + etayl*dpotentialdy_with_rot + etazl*dpotentialdzl)
297  tempx3(i,j,k) = jacobianl*(gammaxl*dpotentialdx_with_rot + gammayl*dpotentialdy_with_rot + gammazl*dpotentialdzl)

299  enddo

301  enddo

308  tempx2l = 0._custom_real
309  tempx3l = 0._custom_real

321 
322  accelfluid(ibool(i,j,k,ispec)) = accelfluid(ibool(i,j,k,ispec)) + sum_terms
323 
324  enddo
325  enddo
326  enddo
327 
328  ! update rotation term with Euler scheme
329  if (rotation_val) then
330  if (use_lddrk) then
331  ! use the source saved above
332  a_array_rotation_lddrk(:,:,:,ispec) = alpha_lddrk(istage) * a_array_rotation_lddrk(:,:,:,ispec) + source_euler_a(:,:,:)

338  ! use the source saved above
339  a_array_rotation(:,:,:,ispec) = a_array_rotation(:,:,:,ispec) + source_euler_a(:,:,:)
340  b_array_rotation(:,:,:,ispec) = b_array_rotation(:,:,:,ispec) + source_euler_b(:,:,:)
341  endif
342  endif
343 
344  enddo ! spectral element loop
