93  integer :: num_elements,ispec_p

95 ! ****************************************************
96 ! big loop over all spectral elements in the fluid
97 ! ****************************************************
98 
99  if (movie_volume .and. nspec_outer_core_3dmovie /= 1 .and. (iphase == 1)) then
100  div_displfluid(:,:,:,:) = 0._custom_real
101  endif
102 
103 ! computed_elements = 0
104  if (iphase == 1) then
105  ! outer elements
106  num_elements = nspec_outer
107  else
108  ! inner elements
109  num_elements = nspec_inner
110  endif
111 
112  do ispec_p = 1,num_elements
113 
114  ispec = phase_ispec_inner(ispec_p,iphase)
115 
116  ! only compute element which belong to current phase (inner or outer elements)
117 
118  do k = 1,ngllz
119  do j = 1,nglly
120  do i = 1,ngllx
121 
122  tempx1l = 0._custom_real
123  tempx2l = 0._custom_real
124  tempx3l = 0._custom_real
125 
126  do l = 1,ngllx
127  !!! can merge these loops because NGLLX = NGLLY = NGLLZ enddo
128  tempx1l = tempx1l + displfluid(ibool(l,j,k,ispec)) * hprime_xx(i,l)
129  tempx2l = tempx2l + displfluid(ibool(i,l,k,ispec)) * hprime_yy(j,l)
130  tempx3l = tempx3l + displfluid(ibool(i,j,l,ispec)) * hprime_zz(k,l)
131  enddo
132 
133  ! get derivatives of velocity potential with respect to x, y and z
134  xixl = xix(i,j,k,ispec)
135  xiyl = xiy(i,j,k,ispec)
136  xizl = xiz(i,j,k,ispec)
137  etaxl = etax(i,j,k,ispec)
138  etayl = etay(i,j,k,ispec)
139  etazl = etaz(i,j,k,ispec)
140  gammaxl = gammax(i,j,k,ispec)
141  gammayl = gammay(i,j,k,ispec)
142  gammazl = gammaz(i,j,k,ispec)
143 
144  ! compute the Jacobian
145  jacobianl = 1._custom_real / (xixl*(etayl*gammazl-etazl*gammayl) &
146  - xiyl*(etaxl*gammazl-etazl*gammaxl) &
147  + xizl*(etaxl*gammayl-etayl*gammaxl))
148 
149  dpotentialdxl = xixl*tempx1l + etaxl*tempx2l + gammaxl*tempx3l
150  dpotentialdyl = xiyl*tempx1l + etayl*tempx2l + gammayl*tempx3l
151  dpotentialdzl = xizl*tempx1l + etazl*tempx2l + gammazl*tempx3l
152 
153  ! compute contribution of rotation and add to gradient of potential
154  ! this term has no Z component
155  if (rotation_val) then
156 
157  ! store the source for the Euler scheme for A_rotation and B_rotation
158  two_omega_deltat = deltat * two_omega_earth
159 
160  cos_two_omega_t = cos(two_omega_earth*timeval)
161  sin_two_omega_t = sin(two_omega_earth*timeval)
162 
163  ! time step deltat of Euler scheme is included in the source
164  source_euler_a(i,j,k) = two_omega_deltat &
165  * (cos_two_omega_t * dpotentialdyl + sin_two_omega_t * dpotentialdxl)
166  source_euler_b(i,j,k) = two_omega_deltat &
167  * (sin_two_omega_t * dpotentialdyl - cos_two_omega_t * dpotentialdxl)
168 
169  a_rotation = a_array_rotation(i,j,k,ispec)
170  b_rotation = b_array_rotation(i,j,k,ispec)
171 
172  ux_rotation = a_rotation*cos_two_omega_t + b_rotation*sin_two_omega_t
173  uy_rotation = - a_rotation*sin_two_omega_t + b_rotation*cos_two_omega_t
174 
175  dpotentialdx_with_rot = dpotentialdxl + ux_rotation
176  dpotentialdy_with_rot = dpotentialdyl + uy_rotation
177 
178  else
179 
180  dpotentialdx_with_rot = dpotentialdxl
181  dpotentialdy_with_rot = dpotentialdyl
182 
183  endif ! end of section with rotation
184 
185  ! add (chi/rho)grad(rho) term in no gravity case
186  if (.not. gravity_val) then
187  ! With regards to the non-gravitating case: we cannot set N^2 = 0 *and* let g = 0.
188  ! We can *either* assume N^2 = 0 but keep gravity g, *or* we can assume that gravity
189  ! is negligible to begin with, as in our GJI 2002a, in which case N does not arise.
190  ! We get:
191  !
192  ! \ddot\chi = \rho^{-1}\kappa\bdel\cdot(\bdel\chi+\chi\bdel\ln\rho)
193  !
194  ! Then the displacement is
195  !
196  ! \bu = \bdel\chi+\chi\bdel\ln\rho = \rho^{-1}\bdel(\rho\chi)
197  !
198  ! and the pressure is
199  !
200  ! p = -\rho\ddot{\chi}
201  !
202  ! Thus in our 2002b GJI paper eqn (21) is wrong, and equation (41)
203  ! in our AGU monograph is incorrect; these equations should be replaced by
204  !
205  ! \ddot\chi = \rho^{-1}\kappa\bdel\cdot(\bdel\chi+\chi\bdel\ln\rho)
206  !
207  ! Note that the fluid potential we use in GJI 2002a differs from the one used here:
208  !
209  ! \chi_GJI2002a = \rho\partial\t\chi
210  !
211  ! such that
212  !
213  ! \bv = \partial_t\bu=\rho^{-1}\bdel\chi_GJI2002a (GJI 2002a eqn 20)
214  !
215  ! p = - \partial_t\chi_GJI2002a (GJI 2002a eqn 19)
216 
217  ! use mesh coordinates to get theta and phi
218  ! x y z contain r theta phi
219  iglob = ibool(i,j,k,ispec)
220 
221  vec_x = gravity_pre_store(1,iglob)
222  vec_y = gravity_pre_store(2,iglob)
223  vec_z = gravity_pre_store(3,iglob)
224 
225  ! adding (chi/rho)grad(rho)
226  dpotentialdx_with_rot = dpotentialdx_with_rot + displfluid(iglob) * vec_x
227  dpotentialdy_with_rot = dpotentialdy_with_rot + displfluid(iglob) * vec_y
228  dpotentialdzl = dpotentialdzl + displfluid(iglob) * vec_z
229 
230 
231  else
232  ! if gravity is turned on
233 
234  ! compute divergence of displacement
235  ! precompute and store gravity term
236 
237  ! use mesh coordinates to get theta and phi
238  ! x y z contain r theta phi
239  iglob = ibool(i,j,k,ispec)
240 
241  ! get g, rho and dg/dr=dg
242  ! Cartesian components of the gravitational acceleration
243  ! integrate and multiply by rho / Kappa
244  vec_x = gravity_pre_store(1,iglob)
245  vec_y = gravity_pre_store(2,iglob)
246  vec_z = gravity_pre_store(3,iglob)
247 
248  gravity_term(i,j,k) = jacobianl * wgll_cube(i,j,k) &
249  * (dpotentialdx_with_rot * vec_x &
250  + dpotentialdy_with_rot * vec_y &
251  + dpotentialdzl * vec_z)
252 
253  ! divergence of displacement field with gravity on
254  ! note: these calculations are only considered for SIMULATION_TYPE == 1 .and. SAVE_FORWARD
255  ! and one has set MOVIE_VOLUME_TYPE == 4 when MOVIE_VOLUME is .true.;
256  ! in case of SIMULATION_TYPE == 3, it gets overwritten by compute_kernels_outer_core()
257  if (movie_volume .and. nspec_outer_core_3dmovie /= 1) then
258  div_displfluid(i,j,k,ispec) = dpotentialdx_with_rot * vec_x &
259  + dpotentialdy_with_rot * vec_y &
260  + dpotentialdzl * vec_z
261  endif
262 
263  endif
264 
265  tempx1(i,j,k) = jacobianl*(xixl*dpotentialdx_with_rot + xiyl*dpotentialdy_with_rot + xizl*dpotentialdzl)
266  tempx2(i,j,k) = jacobianl*(etaxl*dpotentialdx_with_rot + etayl*dpotentialdy_with_rot + etazl*dpotentialdzl)
267  tempx3(i,j,k) = jacobianl*(gammaxl*dpotentialdx_with_rot + gammayl*dpotentialdy_with_rot + gammazl*dpotentialdzl)
268 
269  enddo
270  enddo
271  enddo
272 
273  do k = 1,ngllz
274  do j = 1,nglly
275  do i = 1,ngllx
276 
277  tempx1l = 0._custom_real
278  tempx2l = 0._custom_real
279  tempx3l = 0._custom_real
280 
281  do l = 1,ngllx
282  !!! can merge these loops because NGLLX = NGLLY = NGLLZ enddo
283  tempx1l = tempx1l + tempx1(l,j,k) * hprimewgll_xx(l,i)
284  tempx2l = tempx2l + tempx2(i,l,k) * hprimewgll_yy(l,j)
285  tempx3l = tempx3l + tempx3(i,j,l) * hprimewgll_zz(l,k)
286  enddo
287 
288  ! sum contributions from each element to the global mesh and add gravity term
289  sum_terms = - (wgllwgll_yz(j,k)*tempx1l + wgllwgll_xz(i,k)*tempx2l + wgllwgll_xy(i,j)*tempx3l)
290  if (gravity_val) sum_terms = sum_terms + gravity_term(i,j,k)
291 
292  accelfluid(ibool(i,j,k,ispec)) = accelfluid(ibool(i,j,k,ispec)) + sum_terms
293 
294  enddo
295  enddo
296  enddo
297 
298  ! update rotation term with Euler scheme
299  if (rotation_val) then
300  if (use_lddrk) then
301  ! use the source saved above
302  a_array_rotation_lddrk(:,:,:,ispec) = alpha_lddrk(istage) * a_array_rotation_lddrk(:,:,:,ispec) + source_euler_a(:,:,:)
303  a_array_rotation(:,:,:,ispec) = a_array_rotation(:,:,:,ispec) + beta_lddrk(istage) * a_array_rotation_lddrk(:,:,:,ispec)
304 
305  b_array_rotation_lddrk(:,:,:,ispec) = alpha_lddrk(istage) * b_array_rotation_lddrk(:,:,:,ispec) + source_euler_b(:,:,:)
306  b_array_rotation(:,:,:,ispec) = b_array_rotation(:,:,:,ispec) + beta_lddrk(istage) * b_array_rotation_lddrk(:,:,:,ispec)
307  else
308  ! use the source saved above
309  a_array_rotation(:,:,:,ispec) = a_array_rotation(:,:,:,ispec) + source_euler_a(:,:,:)
310  b_array_rotation(:,:,:,ispec) = b_array_rotation(:,:,:,ispec) + source_euler_b(:,:,:)
311  endif
312  endif
313 
314  enddo ! spectral element loop
315 
316  end subroutine compute_forces_outer_core_nodev
