146  b_eps_trace_over_3_loc_matrix(:,:,:) = b_eps_trace_over_3_crust_mantle(:,:,:,ispec)

152  endif

154  ! computes fully anisotropic kernel cijkl_kl
155  do_loop_ijk
156 
157  iglob = ibool_crust_mantle(index_ijk,ispec)
158 
159  ! density kernel: see e.g. Tromp et al.(2005), equation (14)
160  ! b_displ_crust_mantle is the backward/reconstructed wavefield, that is s(x,t) in eq. (14),
161  ! accel_crust_mantle is the adjoint wavefield, that corresponds to s_dagger(x,T-t)
162  !
163  ! note with respect to eq. (14) the second time derivative is applied to the
164  ! adjoint wavefield here rather than the backward/reconstructed wavefield.
165  ! this is a valid operation and the resultant kernel identical to the eq. (14).
166  !
167  ! reason for this is that the adjoint wavefield is in general smoother
168  ! since the adjoint sources normally are obtained for filtered traces.
169  ! numerically, the time derivative by a finite-difference scheme should
170  ! behave better for smoother wavefields, thus containing less numerical artifacts.
171  rho_kl_crust_mantle(index_ijk,ispec) = rho_kl_crust_mantle(index_ijk,ispec) &
172  + deltat * (accel_crust_mantle(1,iglob) * b_displ_crust_mantle(1,iglob) &
173  + accel_crust_mantle(2,iglob) * b_displ_crust_mantle(2,iglob) &
174  + accel_crust_mantle(3,iglob) * b_displ_crust_mantle(3,iglob) )
175 
176  ! fully anisotropic kernel
177  ! temporary arrays
178  epsilondev_loc(1) = epsilondev_xx_crust_mantle(index_ijk,ispec)
179  epsilondev_loc(2) = epsilondev_yy_crust_mantle(index_ijk,ispec)
180  epsilondev_loc(3) = epsilondev_xy_crust_mantle(index_ijk,ispec)
181  epsilondev_loc(4) = epsilondev_xz_crust_mantle(index_ijk,ispec)
182  epsilondev_loc(5) = epsilondev_yz_crust_mantle(index_ijk,ispec)
183 
184  b_epsilondev_loc(:) = b_epsilondev_loc_matrix(:,index_ijk)
185 
186  call compute_strain_product(prod,eps_trace_over_3_crust_mantle(index_ijk,ispec),epsilondev_loc, &
187  b_eps_trace_over_3_loc_matrix(index_ijk),b_epsilondev_loc)
188 
189  cijkl_kl_crust_mantle(:,index_ijk,ispec) = cijkl_kl_crust_mantle(:,index_ijk,ispec) + deltat * prod(:)
190 
191  ENDdo_loop_ijk
192  enddo
193 !$OMP enddo
194 
195  else
196

198

200  do ispec = 1, nspec_crust_mantle
201 
202  ! simulations with UNDO_ATTENUATION save as much memory as possible;

205  if (use_deville_products_val) then

214  b_displ_crust_mantle, &

218  gammax_crust_mantle,gammay_crust_mantle,gammaz_crust_mantle, &
219  b_epsilondev_loc_matrix,b_eps_trace_over_3_loc_matrix)
220  endif
221  else
222  ! backward/reconstructed strain arrays
223  b_eps_trace_over_3_loc_matrix(:,:,:) = b_eps_trace_over_3_crust_mantle(:,:,:,ispec)
224  b_epsilondev_loc_matrix(1,:,:,:) = b_epsilondev_xx_crust_mantle(:,:,:,ispec)
225  b_epsilondev_loc_matrix(2,:,:,:) = b_epsilondev_yy_crust_mantle(:,:,:,ispec)
226  b_epsilondev_loc_matrix(3,:,:,:) = b_epsilondev_xy_crust_mantle(:,:,:,ispec)
227  b_epsilondev_loc_matrix(4,:,:,:) = b_epsilondev_xz_crust_mantle(:,:,:,ispec)
228  b_epsilondev_loc_matrix(5,:,:,:) = b_epsilondev_yz_crust_mantle(:,:,:,ispec)
229  endif
230 
231  ! if EXACT_UNDOING_TO_DISK is set, currently the rho and beta kernels below will be computed but will be equal
232  ! to zero everywhere because the backward field is then not computed, and only the alpha kernel
233  ! will be computed correctly based on the values of the forward run saved to disk in the first run
234  ! and read back at the beginning of this routine
235  do_loop_ijk
236 
237  iglob = ibool_crust_mantle(index_ijk,ispec)
238 
239  ! density kernel: see e.g. Tromp et al.(2005), equation (14)
240  ! b_displ_crust_mantle is the backward/reconstructed wavefield, that is s(x,t) in eq. (14),
241  ! accel_crust_mantle is the adjoint wavefield, that corresponds to s_dagger(x,T-t)
242  !
243  ! note with respect to eq. (14) the second time derivative is applied to the
244  ! adjoint wavefield here rather than the backward/reconstructed wavefield.
245  ! this is a valid operation and the resultant kernel identical to the eq. (14).
246  !
247  ! reason for this is that the adjoint wavefield is in general smoother
248  ! since the adjoint sources normally are obtained for filtered traces.
249  ! numerically, the time derivative by a finite-difference scheme should
250  ! behave better for smoother wavefields, thus containing less numerical artifacts.
251  rho_kl_crust_mantle(index_ijk,ispec) = rho_kl_crust_mantle(index_ijk,ispec) &
252  + deltat * (accel_crust_mantle(1,iglob) * b_displ_crust_mantle(1,iglob) &
253  + accel_crust_mantle(2,iglob) * b_displ_crust_mantle(2,iglob) &
254  + accel_crust_mantle(3,iglob) * b_displ_crust_mantle(3,iglob) )
255 
256  ! isotropic kernels
257  ! temporary arrays
258  epsilondev_loc(1) = epsilondev_xx_crust_mantle(index_ijk,ispec)
259  epsilondev_loc(2) = epsilondev_yy_crust_mantle(index_ijk,ispec)
260  epsilondev_loc(3) = epsilondev_xy_crust_mantle(index_ijk,ispec)
261  epsilondev_loc(4) = epsilondev_xz_crust_mantle(index_ijk,ispec)
262  epsilondev_loc(5) = epsilondev_yz_crust_mantle(index_ijk,ispec)
263 
264  b_epsilondev_loc(:) = b_epsilondev_loc_matrix(:,index_ijk)
265 
266  ! kernel for shear modulus, see e.g. Tromp et al. (2005), equation (17)
267  ! note: multiplication with 2*mu(x) will be done after the time loop
268  beta_kl_crust_mantle(index_ijk,ispec) = beta_kl_crust_mantle(index_ijk,ispec) &
269  + deltat * &
270  (epsilondev_loc(1)*b_epsilondev_loc(1) + epsilondev_loc(2)*b_epsilondev_loc(2) &
271  + (epsilondev_loc(1)+epsilondev_loc(2)) * (b_epsilondev_loc(1)+b_epsilondev_loc(2)) &
272  + 2 * (epsilondev_loc(3)*b_epsilondev_loc(3) + epsilondev_loc(4)*b_epsilondev_loc(4) + &
273  epsilondev_loc(5)*b_epsilondev_loc(5)) )
274 
275  ! kernel for bulk modulus, see e.g. Tromp et al. (2005), equation (18)
276  ! note: multiplication with kappa(x) will be done after the time loop
277  alpha_kl_crust_mantle(index_ijk,ispec) = alpha_kl_crust_mantle(index_ijk,ispec) &
278  + deltat * (9 * eps_trace_over_3_crust_mantle(index_ijk,ispec) &
279  * b_eps_trace_over_3_loc_matrix(index_ijk))
280 
281  ENDdo_loop_ijk
282 
283  enddo
284 !$OMP enddo
285 
286  endif ! ANISOTROPIC_KL
287 
288 !$OMP END PARALLEL
289

292

295 
296  endif

809 
810  subroutine compute_strain_product(prod,eps_trace_over_3,epsdev, &
811  b_eps_trace_over_3,b_epsdev)
812 
813  ! Purpose: compute the 21 strain products at a grid point
814  ! (ispec,i,j,k fixed) and at a time t to compute then the kernels cij_kl (Voigt notation)
815  ! (eq. 15 of Tromp et al., 2005)
816  ! prod(1)=eps11*eps11 -> c11, prod(2)=eps11eps22 -> c12, prod(3)=eps11eps33 -> c13, ...
817  ! prod(7)=eps22*eps22 -> c22, prod(8)=eps22eps33 -> c23, prod(9)=eps22eps23 -> c24, ...
818  ! prod(19)=eps13*eps13 -> c55, prod(20)=eps13eps12 -> c56, prod(21)=eps12eps12 -> c66
819  ! This then gives how the 21 kernels are organized
820  ! For crust_mantle
821 
822  use constants_solver
823 
824  implicit none
825 
826  real(kind=CUSTOM_REAL),dimension(21) :: prod
827  real(kind=CUSTOM_REAL) :: eps_trace_over_3,b_eps_trace_over_3
828  real(kind=CUSTOM_REAL),dimension(5) :: epsdev,b_epsdev
829 
830  real(kind=CUSTOM_REAL), dimension(6) :: eps,b_eps
831  integer :: p,i,j
832 
833  ! Building of the local matrix of the strain tensor
834  ! for the adjoint field and the regular backward field
835  eps(1) = epsdev(1) + eps_trace_over_3 !eps11 et eps22
836  eps(2) = epsdev(2) + eps_trace_over_3 !eps11 et eps22
837  eps(3) = -(eps(1) + eps(2)) + 3._custom_real * eps_trace_over_3 !eps33
838  eps(4) = epsdev(5) !eps23
839  eps(5) = epsdev(4) !eps13
840  eps(6) = epsdev(3) !eps12
841 
842  b_eps(1) = b_epsdev(1) + b_eps_trace_over_3
843  b_eps(2) = b_epsdev(2) + b_eps_trace_over_3
844  b_eps(3) = -(b_eps(1) + b_eps(2)) + 3._custom_real * b_eps_trace_over_3
845  b_eps(4) = b_epsdev(5)
846  b_eps(5) = b_epsdev(4)
847  b_eps(6) = b_epsdev(3)
848 
849  ! Computing the 21 strain products without assuming eps(i)*b_eps(j) = eps(j)*b_eps(i)
850  p = 1
851  do i = 1,6
852  do j = i,6
853  prod(p) = eps(i)*b_eps(j)
854  if (j > i) then
855  prod(p) = prod(p)+eps(j)*b_eps(i)
856  if (j > 3 .and. i < 4) prod(p) = prod(p) * 2.0_custom_real
857  endif
858  if (i > 3) prod(p) = prod(p) * 4.0_custom_real
859  p = p+1
860  enddo
861  enddo
862 
863  end subroutine compute_strain_product
