287  subroutine write_movie_vtk_snapshot()
288 
289  use specfem_par, only: it,myrank,mesh_pointer, &
290  gpu_mode,simulation_type, &
291  ntstep_between_frames,movie_start,movie_stop, &
292  max_string_len,iflag_crust,ndim, &
293  nglob_crust_mantle,nspec_crust_mantle
294 
295  use specfem_par_crustmantle, only: displ_crust_mantle,b_displ_crust_mantle, &
296  ibool_crust_mantle,rstore_crust_mantle
297 
298  implicit none
299 
300  character(len=MAX_STRING_LEN) :: filename
301  integer,dimension(:),allocatable :: dummy_i
302  integer :: ier
303 
304  ! checks if anything to do
305  if (.not. (mod(it-movie_start,ntstep_between_frames) == 0 .and. it >= movie_start .and. it <= movie_stop)) return
306 
307  ! crust mantle
308  allocate(dummy_i(nspec_crust_mantle),stat=ier)
309  if (ier /= 0) stop 'Error allocating dummy array for snapshot'
310  dummy_i(:) = iflag_crust
311 
312  !output displacement
313  if (gpu_mode) then
314  call transfer_displ_cm_from_device(ndim*nglob_crust_mantle,displ_crust_mantle,mesh_pointer)
315  endif
316 
317  ! VTK file output
318  ! one file per process
319  write(filename,'(a,i6.6,a,i6.6)') 'OUTPUT_FILES/snapshot_proc',myrank,'_reg_1_displ_',it
320  call write_vtk_data_cr(dummy_i,nspec_crust_mantle,nglob_crust_mantle, &
321  rstore_crust_mantle, &
322  ibool_crust_mantle,displ_crust_mantle,filename)
323 
324  ! backward/reconstructed field
325  if (simulation_type == 3) then
326  if (gpu_mode) then
327  call transfer_b_displ_cm_from_device(ndim*nglob_crust_mantle,b_displ_crust_mantle,mesh_pointer)
328  endif
329 
330  write(filename,'(a,i6.6,a,i6.6)') 'OUTPUT_FILES/snapshot_proc',myrank,'_reg_1_b_displ_',it
331  call write_vtk_data_cr(dummy_i,nspec_crust_mantle,nglob_crust_mantle, &
332  rstore_crust_mantle, &
333  ibool_crust_mantle,b_displ_crust_mantle,filename)
334  endif
335 
336  deallocate(dummy_i)
337 
338  end subroutine write_movie_vtk_snapshot
