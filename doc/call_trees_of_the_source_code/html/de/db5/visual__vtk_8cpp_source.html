274 void set_color_scale(int icolor) {
275  if (icolor == 0 ) {
276  // sets (default) rainbow color scale
277  //# Set the hue range: from low to high the
278  //# table passes through blue, green, yellow,
279  //# orange, and red
280  gcolor_min = 0.0;
281  fs.vtk.lut->SetTableRange( gcolor_min, gcolor_max );
282  fs.vtk.lut->SetScaleToLinear();
283 
284  fs.vtk.lut->SetValueRange( 0.6, 1.0 );
285  fs.vtk.lut->SetHueRange( 0.66667, 0.0 );
286  fs.vtk.lut->SetSaturationRange( 1.0, 1.0 );
287  }else if (icolor == 1 ) {
288  // red-blue color scale
289  // Since the default lut is
290  // a rainbow lut, we only have
291  // to worry about the hue range.
292  gcolor_min = 0.0;
293  fs.vtk.lut->SetTableRange( gcolor_min, gcolor_max );
294  fs.vtk.lut->SetScaleToLinear();
295 
296  fs.vtk.lut->SetValueRange( 0.6, 1.0 );
297  fs.vtk.lut->SetHueRange( 0.0, 0.667 );
298  fs.vtk.lut->SetSaturationRange( 1.0, 1.0 );
299  }else if (icolor == 2 ) {
300  // red color scale
301  gcolor_min = 1.e-3*gcolor_max;
302  fs.vtk.lut->SetTableRange( gcolor_min, gcolor_max );
303  fs.vtk.lut->SetScaleToLog10();
304 
305  fs.vtk.lut->SetValueRange( 0.4, 1.0 );
306  fs.vtk.lut->SetHueRange( 0.0, 0.4 );
307  fs.vtk.lut->SetSaturationRange( 0.5, 0.0 );
308  }else if (icolor == 3 ) {
309  // blue color scale
310  gcolor_min = 1.e-3*gcolor_max;
311  fs.vtk.lut->SetTableRange( gcolor_min, gcolor_max );
312  fs.vtk.lut->SetScaleToLog10();
313 
314  fs.vtk.lut->SetValueRange( 0.4, 1.0 );
315  fs.vtk.lut->SetHueRange( 0.6, 0.6 );
316  fs.vtk.lut->SetSaturationRange( 0.5, 0.0 );
317  }else if (icolor == 4 ) {
318  // black color scale
319  gcolor_min = 1.e-3*gcolor_max;
320  fs.vtk.lut->SetTableRange( gcolor_min, gcolor_max );
321  fs.vtk.lut->SetScaleToLog10();
322 
323  fs.vtk.lut->SetValueRange( 0.4, 1.0 ); // from black to white
324  fs.vtk.lut->SetHueRange( 0.0, 1.0 );
325  fs.vtk.lut->SetSaturationRange( 0.0, 0.0 ); // no color saturation
326  }else{
327  // reset
328  fs.vtk.icolor = 0;
329  // sets (default) rainbow color scale
330  //# Set the hue range: from low to high the
331  //# table passes through blue, green, yellow,
332  //# orange, and red
333  gcolor_min = 0.0;
334  fs.vtk.lut->SetTableRange( gcolor_min, gcolor_max );
335  fs.vtk.lut->SetScaleToLinear();
336 
337  fs.vtk.lut->SetValueRange( 0.6, 1.0 );
338  fs.vtk.lut->SetHueRange( 0.66667, 0.0 );
339  fs.vtk.lut->SetSaturationRange( 1.0, 1.0 );
340  }
341  fs.vtk.lut->Build();
342 }

823  printf(" model: zmin/zmax = %f / %f\n",zmin,zmax);

880  // clip box
881  //
882  // bounds

885  // normalized source location
886  double norm = sqrt( fs.vtk.pos_source[0]*fs.vtk.pos_source[0]
887  + fs.vtk.pos_source[1]*fs.vtk.pos_source[1]
888  + fs.vtk.pos_source[2]*fs.vtk.pos_source[2]);
889  // normalized vector to source
890  double v[3];
891  if (norm > 0.0) {
892  v[0] = fs.vtk.pos_source[0]/norm;
893  v[1] = fs.vtk.pos_source[1]/norm;
894  v[2] = fs.vtk.pos_source[2]/norm;
895  }else{
896  v[0] = 0.0;
897  v[1] = 0.0;
898  v[2] = 1.0;
899  }
900 
901  // source location vector / perpendicular vectors
902  //
903  // vectorproduct(vector1, vector2, product)
904  // calculates the vector product of vector1 and vector2
905  // product(1) = vector1(2)*vector2(3) - vector1(3)*vector2(2)
906  // product(2) = - vector1(1)*vector2(3) + vector1(3)*vector2(1)
907  // product(3) = vector1(1)*vector2(2) - vector1(2)*vector2(1)
908  double p1[3],p2[3];
909  if (norm > 0) {
910  p1[0] = 0.0;
911  p1[1] = 0.0;
912  p1[2] = 1.0;
913 
914  p2[0] = 0.0;
915  p2[1] = 1.0;
916  p2[2] = 0.0;
917  }else{
918  p1[0] = 0.0;
919  p1[1] = 1.0;
920  p1[2] = 0.0;
921 
922  p2[0] = 1.0;
923  p2[1] = 0.0;
924  p2[2] = 0.0;
925  }
926  double pn1[3],pn2[3];
927  pn1[0] = v[1]*p1[2] - v[2]*p1[1];
928  pn1[1] = -v[0]*p1[2] + v[2]*p1[0];
929  pn1[2] = v[0]*p1[1] - v[1]*p1[0];
930 
931  pn2[0] = v[1]*p2[2] - v[2]*p2[1];
932  pn2[1] = -v[0]*p2[2] + v[2]*p2[0];
933  pn2[2] = v[0]*p2[1] - v[1]*p2[0];
934 
935  // flips normal depending on location of source
936  if (fabs(v[0]-xmin) < fabs(v[0]-xmax) ) {
937  pn1[0] *= -1.0;
938  pn1[1] *= -1.0;
939  pn1[2] *= -1.0;
940  }
941  if (fabs(v[1]-ymin) < fabs(v[1]-ymax) ) {
942  pn2[0] *= -1.0;
943  pn2[1] *= -1.0;
944  pn2[2] *= -1.0;
945  }

971  fs.vtk.clip2->Update();
972

974  fs.vtk.merger = vtkAppendFilter::New();
975  fs.vtk.merger->MergePointsOn();

978  fs.vtk.merger->Update();
979
