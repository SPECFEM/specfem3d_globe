132  topo_ep(ilon,jlat) = tmp(3)
133  thickness_ep(ilon,jlat,1:3) = tmp(4:6)
134  vp_ep(ilon,jlat,1:3) = tmp(7:9)
135  vs_ep(ilon,jlat,1:3) = tmp(10:12)
136  rho_ep(ilon,jlat,1:3) = tmp(13:15)

147  subroutine model_epcrust(lat,lon,dep,vp,vs,rho,moho,found_crust,elem_in_crust)
148 
149  use constants
150  use model_epcrust_par
151 
152  implicit none
153 
154  ! INPUT & OUTPUT
155  double precision,intent(in):: lat, lon, dep
156  double precision,intent(out):: vp, vs, rho, moho
157  logical,intent(out) :: found_crust
158  logical,intent(in) :: elem_in_crust
159 
160  ! INTERIOR
161  integer:: ilon, jlat, k
162  double precision :: z0 , topo, basement, conrad, moho_top, scaleval
163  double precision,dimension(3):: zsmooth, vpsmooth, vssmooth, rhosmooth
164  double precision,dimension(NTHETA_EP*NPHI_EP) :: x1,y1,weight
165  double precision:: weightl
166  double precision:: cut, min_sed
167 
168  !if (lat < EPCRUST_LAT_MIN .or. lat > EPCRUST_LAT_MAX &
169  ! .or. lon < EPCRUST_LON_MIN .or. lon > EPCRUST_LON_MAX) then
170  ! stop 'incorrect enter EPCRUST model, check lat and lon'
171  !endif
172 
173  vp = zero
174  vs = zero
175  rho = zero
176 
177  if (.not. flag_smooth_epcrust) then
178  call ilon_jlat(lon,lat,ilon,jlat)
179  z0 = topo_ep(ilon,jlat)
180  zsmooth(:) = thickness_ep(ilon,jlat,:)
181  vpsmooth(:) = vp_ep(ilon,jlat,:)
182  vssmooth(:) = vs_ep(ilon,jlat,:)
183  rhosmooth(:) = rho_ep(ilon,jlat,:)
184  else
185  call epcrust_smooth_base(lon,lat,x1,y1,weight)
186  z0 = zero
187  zsmooth(:) = zero
188  vpsmooth(:) = zero
189  vssmooth(:) = zero
190  rhosmooth(:) = zero
191 
192  do k = 1,ntheta_ep*nphi_ep
193  call ilon_jlat(x1(k),y1(k),ilon,jlat)
194  weightl=weight(k)
195 
196  z0=z0+weightl*topo_ep(ilon,jlat)
197  zsmooth(:)=zsmooth(:)+weightl*thickness_ep(ilon,jlat,:)
198  vpsmooth(:)=vpsmooth(:)+weightl*vp_ep(ilon,jlat,:)
199  vssmooth(:)=vssmooth(:)+weightl*vs_ep(ilon,jlat,:)
200  rhosmooth(:)=rhosmooth(:)+weightl*rho_ep(ilon,jlat,:)
201  enddo
202  endif
203 
204  !topo=(R_EARTH_KM+z0)/R_EARTH_KM
205  !basement=(R_EARTH_KM+z0-zsmooth(1))/R_EARTH_KM
206  !conrad=(R_EARTH_KM+z0-zsmooth(1)-zsmooth(2))/R_EARTH_KM
207  !moho_top=(R_EARTH_KM+z0-zsmooth(1)-zsmooth(2)-zsmooth(3))/R_EARTH_KM
208 
209  topo=(r_earth_km+z0)/r_earth_km
210  basement=(r_earth_km-zsmooth(1))/r_earth_km
211  conrad=(r_earth_km-zsmooth(1)-zsmooth(2))/r_earth_km
212  moho_top=(r_earth_km-zsmooth(1)-zsmooth(2)-zsmooth(3))/r_earth_km
213  min_sed=1.0 - minimum_sediment_thickness/r_earth_km
214 
215  found_crust=.true.
216 
217  if (dep > basement .and. include_sediments_in_crust &
218  .and. zsmooth(1) >= minimum_sediment_thickness) then ! Hejun Zhu add minimum sediment thickness
219  vp=vpsmooth(1)
220  vs=vssmooth(1)
221  rho=rhosmooth(1)
222  else if (dep > conrad) then
223  vp=vpsmooth(2)
224  vs=vssmooth(2)
225  rho=rhosmooth(2)
226  else if (dep > moho_top .or. elem_in_crust) then
227  vp=vpsmooth(3)
228  vs=vssmooth(3)
229  rho=rhosmooth(3)
230  else
231  found_crust=.false.
232  endif
233 
234  if (found_crust) then
235  scaleval=dsqrt(pi*grav*rhoav)
236  vp=vp*1000.d0/(r_earth*scaleval)
237  vs=vs*1000.d0/(r_earth*scaleval)
238  rho=rho*1000.d0/rhoav
239  endif
240 
241  !moho = -(z0-zsmooth(1)-zsmooth(2)-zsmooth(3))/R_EARTH_KM
242  moho = (zsmooth(1)+zsmooth(2)+zsmooth(3))/r_earth_km
243 
244  ! Hejun Zhu, delete moho thickness less than 7 km
245  cut=7.0/r_earth_km
246  if (moho < cut) then
247  moho = cut
248  endif
249 
250  end subroutine model_epcrust

256  subroutine epcrust_smooth_base(x,y,x1,y1,weight)
257 
258  use constants
259  use model_epcrust_par, only: ntheta_ep,nphi_ep,cap_degree_ep
260 
261  implicit none
262 
263  ! INPUT & OUTPUT
264  double precision:: x, y
265  double precision,dimension(NTHETA_EP*NPHI_EP):: x1,y1,weight
266 
267  ! INTERIOR
268  double precision:: CAP,dtheta,dphi,cap_area,dweight,pi_over_nphi,total,wght
269  double precision:: theta,phi,sint,cost,sinp,cosp
270  double precision:: r_rot,theta_rot,phi_rot
271  double precision,dimension(3,3):: rotation_matrix
272  double precision,dimension(3):: xx,xc
273  integer:: i,j,k,itheta,iphi
274 
275  x1(:)=zero
276  y1(:)=zero
277  weight(:)=zero
278 
279  if (cap_degree_ep < tinyval) then
280  print *, 'Error cap:', cap_degree_ep
281  print *, 'lat/lon:', x,y
282  stop 'Error cap_degree too small'
283  endif
284 
285  cap=cap_degree_ep * degrees_to_radians
286  dtheta=0.5d0*cap/dble(ntheta_ep)
287  dphi=two_pi/dble(nphi_ep)
288 
289  cap_area=two_pi*(1.0d0-dcos(cap))
290  dweight=cap/dble(ntheta_ep)*dphi/cap_area
291  pi_over_nphi=pi/dble(nphi_ep)
292 
293  phi=x*degrees_to_radians
294  theta=(90.0d0-y)*degrees_to_radians
295 
296  sint=dsin(theta)
297  cost=dcos(theta)
298  sinp=dsin(phi)
299  cosp=dcos(phi)
300 
301  rotation_matrix(1,1)=cosp*cost
302  rotation_matrix(1,2)=-sinp
303  rotation_matrix(1,3)=cosp*sint
304  rotation_matrix(2,1)=sinp*cost
305  rotation_matrix(2,2)=cosp
306  rotation_matrix(2,3)=sinp*sint
307  rotation_matrix(3,1)=-sint
308  rotation_matrix(3,2)=zero
309  rotation_matrix(3,3)=cost
310 
311  i = 0
312  total=0.0d0
313  do itheta = 1,ntheta_ep
314  theta=dble(2*itheta-1)*dtheta
315  cost=dcos(theta)
316  sint=dsin(theta)
317  wght=sint*dweight
318  do iphi = 1,nphi_ep
319  i=i+1
320  weight(i)=wght
321 
322  total=total+weight(i)
323  phi=dble(2*iphi-1)*pi_over_nphi
324  cosp=dcos(phi)
325  sinp=dsin(phi)
326 
327  xc(1)=sint*cosp
328  xc(2)=sint*sinp
329  xc(3)=cost
330  do j = 1,3
331  xx(j)=0.0d0
332  do k = 1,3
333  xx(j)=xx(j)+rotation_matrix(j,k)*xc(k)
334  enddo
335  enddo
336  call xyz_2_rthetaphi_dble(xx(1),xx(2),xx(3),r_rot,theta_rot,phi_rot)
337  call reduce(theta_rot,phi_rot)
338  x1(i)=phi_rot*radians_to_degrees
339  y1(i)=(pi_over_two-theta_rot)*radians_to_degrees
340  if (x1(i) > 180.d0) x1(i)=x1(i)-360.d0
341  enddo
342  enddo
343 
344  if (abs(total-1.0d0) > 0.001d0) then
345  print *,'Error cap:',total,cap_degree_ep
346  stop
347  endif
348 
349  end subroutine epcrust_smooth_base
