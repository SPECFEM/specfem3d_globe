72  use specfem_par

77  implicit none

81  integer :: num_elements
82 
83  ! inverse arrays use 1D indexing for better compiler vectorization
84  ! only used for Deville routines and FORCE_VECTORIZATION)
85 
86  ! checks if used
87  if (use_deville_products_val) then
88 #ifdef FORCE_VECTORIZATION
89  use_inversed_arrays = .true.
90 #else
91  use_inversed_arrays = .false.
92 #endif
93  else
94  use_inversed_arrays = .false.
95  endif
96 
97  ! uses local array to store all element contributions
98  if (use_deville_products_val) then
99  ! note: we use allocate for sum_terms arrays rather than defining within subroutine compute_forces_**_Dev() itself
100  ! as it will crash when using OpenMP and operating systems with small stack sizes
101  ! e.g. see http://stackoverflow.com/questions/22649827/illegal-instruction-error-when-running-openmp-in-gfortran-mac
102  allocate(sum_terms_crust_mantle(ndim,ngllx,nglly,ngllz,nspec_crust_mantle), &
103  sum_terms_inner_core(ndim,ngllx,nglly,ngllz,nspec_inner_core), &
104  sum_terms_outer_core(ngllx,nglly,ngllz,nspec_outer_core), &
105  stat=ier)
106  if (ier /= 0) stop 'Error allocating sum_terms arrays'
107  sum_terms_crust_mantle(:,:,:,:,:) = 0._custom_real
108  sum_terms_inner_core(:,:,:,:,:) = 0._custom_real
109  sum_terms_outer_core(:,:,:,:) = 0._custom_real
110  endif
111 
112  ! inverse table
113  ! this helps to speedup the assembly, especially with OpenMP (or on MIC) threading
114  if (use_inversed_arrays) then
115  ! allocating arrays
116  allocate(ibool_inv_tbl_crust_mantle(ngllx*nglly*ngllz*nspec_crust_mantle,2), &
117  ibool_inv_tbl_inner_core(ngllx*nglly*ngllz*nspec_inner_core,2), &
118  ibool_inv_tbl_outer_core(ngllx*nglly*ngllz*nspec_outer_core,2),stat=ier)
119  if (ier /= 0) stop 'Error allocating ibool_inv_tbl arrays'
120 
121  allocate(ibool_inv_st_crust_mantle(nglob_crust_mantle+1,2), &
122  ibool_inv_st_inner_core(nglob_inner_core+1,2), &
123  ibool_inv_st_outer_core(nglob_outer_core+1,2),stat=ier)
124  if (ier /= 0) stop 'Error allocating ibool_inv_st arrays'
125 
126  allocate(phase_iglob_crust_mantle(nglob_crust_mantle,2), &
127  phase_iglob_inner_core(nglob_inner_core,2), &
128  phase_iglob_outer_core(nglob_outer_core,2),stat=ier)
129  if (ier /= 0) stop 'Error allocating phase_iglob arrays'
130 
131  ! initializing
132  num_globs_crust_mantle(:) = 0
133  num_globs_inner_core(:) = 0
134  num_globs_outer_core(:) = 0
135 
136  ibool_inv_tbl_crust_mantle(:,:) = 0
137  ibool_inv_tbl_inner_core(:,:) = 0
138  ibool_inv_tbl_outer_core(:,:) = 0
139 
140  ibool_inv_st_crust_mantle(:,:) = 0
141  ibool_inv_st_inner_core(:,:) = 0
142  ibool_inv_st_outer_core(:,:) = 0
143 
144  phase_iglob_crust_mantle(:,:) = 0
145  phase_iglob_inner_core(:,:) = 0
146  phase_iglob_outer_core(:,:) = 0

149  ! loops over phases
150  ! (1 == outer elements / 2 == inner elements)
151  do iphase = 1,2
152  ! crust mantle
153  if (iphase == 1) then
154  ! outer elements (iphase=1)
155  num_elements = nspec_outer_crust_mantle
156  else
157  ! inner elements (iphase=2)
158  num_elements = nspec_inner_crust_mantle
159  endif

161  num_elements,phase_ispec_inner_crust_mantle, &
162  ibool_crust_mantle,phase_iglob_crust_mantle, &
163  ibool_inv_tbl_crust_mantle, ibool_inv_st_crust_mantle, &
164  num_globs_crust_mantle)
165 
166  ! inner core
167  if (iphase == 1) then
168  ! outer elements (iphase=1)
169  num_elements = nspec_outer_inner_core
170  else
171  ! inner elements (iphase=2)
172  num_elements = nspec_inner_inner_core
173  endif
174  call make_inv_table(iphase,nglob_inner_core,nspec_inner_core, &
175  num_elements,phase_ispec_inner_inner_core, &
176  ibool_inner_core,phase_iglob_inner_core, &
177  ibool_inv_tbl_inner_core, ibool_inv_st_inner_core, &
178  num_globs_inner_core,idoubling_inner_core)
179 
180  ! outer core
181  if (iphase == 1) then
182  ! outer elements (iphase=1)
183  num_elements = nspec_outer_outer_core
184  else
185  ! inner elements (iphase=2)
186  num_elements = nspec_inner_outer_core
187  endif
188  call make_inv_table(iphase,nglob_outer_core,nspec_outer_core, &
189  num_elements,phase_ispec_inner_outer_core, &
190  ibool_outer_core,phase_iglob_outer_core, &
191  ibool_inv_tbl_outer_core, ibool_inv_st_outer_core, &
192  num_globs_outer_core)
193  enddo
194 
195  ! user output
196  if (myrank == 0) then
197  write(imain,*)" inverse table of ibool done"
198  call flush_imain()
199  endif
200  endif

202  ! synchronizes processes
203  call synchronize_all()

207  subroutine make_inv_table(iphase,nglob,nspec, &
208  phase_nspec,phase_ispec,ibool,phase_iglob, &
209  ibool_inv_tbl,ibool_inv_st,num_globs,idoubling)
210 
211  implicit none
212 
213  ! arguments
214  integer,intent(in) :: iphase
215  integer,intent(in) :: nglob
216  integer,intent(in) :: nspec
217  integer,intent(in) :: phase_nspec
218  integer, dimension(:,:),intent(in) :: phase_ispec
219  integer, dimension(:,:,:,:),intent(in) :: ibool
220 
221  integer, dimension(:,:),intent(inout) :: phase_iglob
222  integer, dimension(:,:),intent(inout) :: ibool_inv_tbl
223  integer, dimension(:,:),intent(inout) :: ibool_inv_st
224  integer, dimension(:),intent(inout) :: num_globs
225 
226  integer,dimension(:),optional :: idoubling
227 
228  ! local parameters
229  integer, dimension(:), allocatable :: ibool_inv_num
230  integer, dimension(:,:), allocatable :: ibool_inv_tbl_tmp
231  integer :: num_alloc_ibool_inv_tbl,num_alloc_ibool_inv_tbl_theor
232  integer :: num_used_ibool_inv_tbl
233  integer :: ip, iglob, ispec_p, ispec, iglob_p, ier
234  integer :: inum
235 #ifdef FORCE_VECTORIZATION
236  integer :: ijk
237 #else
238  integer :: i,j,k
239 #endif
240  logical :: is_inner_core
241 
242  ! tolerance number of shared degrees per node
243  integer, parameter :: N_TOL = 20
244 
245  ! checks if anything to do (e.g., no outer elements for single process simulations)
246  if (phase_nspec == 0) return
247 
248  ! checks if inner core region
249  if (present(idoubling)) then
250  is_inner_core = .true.
251  else
252  is_inner_core = .false.
253  endif
254 
255  ! allocates temporary arrays
256  allocate(ibool_inv_num(nglob),stat=ier)
257  if (ier /= 0) stop 'Error allocating ibool_inv_num array'
258 
259  ! gets valence of global degrees of freedom for current phase (inner/outer) elements
260  ibool_inv_num(:) = 0
261  do ispec_p = 1,phase_nspec
262  ispec = phase_ispec(ispec_p,iphase)
263 
264  ! exclude fictitious elements in central cube
265  if (is_inner_core) then
266  if (idoubling(ispec) == iflag_in_fictitious_cube) cycle
267  endif
268 
269  do_loop_ijk
270  iglob = ibool(index_ijk,ispec)
271  ! increases valence counter
272  ibool_inv_num(iglob) = ibool_inv_num(iglob) + 1
273  ENDdo_loop_ijk
274  enddo
275 
276  ! gets maximum valence value
277  num_alloc_ibool_inv_tbl = maxval(ibool_inv_num(:))
278 
279  ! theoretical number of maximum shared degrees per node
280  num_alloc_ibool_inv_tbl_theor = n_tol*(ngllx*nglly*ngllz*nspec/nglob+1)
281 
282  ! checks valence
283  if (num_alloc_ibool_inv_tbl < 1 .or. num_alloc_ibool_inv_tbl > num_alloc_ibool_inv_tbl_theor) then
284  print *,'Error invalid maximum valence:'
285  print *,'valence value = ',num_alloc_ibool_inv_tbl,' - theoretical maximum = ',num_alloc_ibool_inv_tbl_theor
286  stop 'Error invalid maximum valence value'
287  endif
288  ! debug
289  !print *,myrank,'maximum shared degrees theoretical = ',num_alloc_ibool_inv_tbl_theor ! regional_Greece_small example: 40
290  !print *,myrank,'maximum shared degrees from array = ',maxval(ibool_inv_num(:)) ! regional_Greece_small example: 8 and 16
291 
292  allocate(ibool_inv_tbl_tmp(num_alloc_ibool_inv_tbl,nglob),stat=ier)
293  if (ier /= 0) stop 'Error allocating ibool_inv_tbl_tmp array'
294 
295  !---- make temporary array of inv. table : ibool_inv_tbl_tmp
296  ibool_inv_tbl_tmp(:,:) = 0
297  ibool_inv_num(:) = 0
298  do ispec_p = 1,phase_nspec
299  ispec = phase_ispec(ispec_p,iphase)
300 
301  ! exclude fictitious elements in central cube
302  if (is_inner_core) then
303  if (idoubling(ispec) == iflag_in_fictitious_cube) cycle
304  endif
305 
306  do_loop_ijk
307 
308  iglob = ibool(index_ijk,ispec)
309 
310  ! increases counter
311  ibool_inv_num(iglob) = ibool_inv_num(iglob) + 1
312 
313  ! inverse table
314  ! sets 1D index of local GLL point (between 1 and NGLLCUBE)
315 #ifdef FORCE_VECTORIZATION
316  inum = ijk
317 #else
318  inum = i + (j-1)*nglly + (k-1)*nglly*ngllz
319 #endif
320  ! sets 1D index in local ibool array
321  ibool_inv_tbl_tmp(ibool_inv_num(iglob),iglob) = inum + ngllx*nglly*ngllz*(ispec-1)
322 
323  ENDdo_loop_ijk
324 
325  enddo
326 
327  !---- packing : ibool_inv_tbl_tmp -> ibool_inv_tbl
328  ip = 0
329  iglob_p = 0
330  num_used_ibool_inv_tbl = 0
331  do iglob = 1, nglob
332  if (ibool_inv_num(iglob) /= 0) then
333  iglob_p = iglob_p + 1
334 
335  phase_iglob(iglob_p,iphase) = iglob
336 
337  ! sets start index of table entry for this global node
338  ibool_inv_st(iglob_p,iphase) = ip + 1
339 
340  ! sets maximum of used valence
341  if ( ibool_inv_num(iglob) > num_used_ibool_inv_tbl ) num_used_ibool_inv_tbl = ibool_inv_num(iglob)
342 
343  ! loops over valence
344  do inum = 1, ibool_inv_num(iglob)
345  ! increases total counter
346  ip = ip + 1
347  ! maps local 1D index in ibool array
348  ibool_inv_tbl(ip,iphase) = ibool_inv_tbl_tmp(inum,iglob)
349  enddo
350  endif
351  enddo
352  ! sets last entry in start index table
353  ibool_inv_st(iglob_p+1,iphase) = ip + 1
354 
355  ! total number global nodes in this phase (inner/outer)
356  num_globs(iphase) = iglob_p
357 
358  ! checks
359  if ( num_used_ibool_inv_tbl > num_alloc_ibool_inv_tbl ) then
360  print *,"Error invalid inverse table setting:"
361  print *," num_alloc_ibool_inv_tbl = ",num_alloc_ibool_inv_tbl
362  print *," num_used_ibool_inv_tbl = ",num_used_ibool_inv_tbl
363  print *,"invalid value encountered: num_used_ibool_inv_tbl > num_alloc_ibool_inv_tbl"
364  print *,"#### Program exits... ##########"
365  call exit_mpi(myrank,'Error making inverse table for optimized arrays')
366  endif
367 
368  ! debug
369  !if (myrank == 0) then
370  ! print *,'ibool_inv_tbl: '
371  ! do iglob_p = 1,200
372  ! print *,' ',iglob_p,'table = ',(ibool_inv_tbl(ip,iphase), &
373  ! ip = ibool_inv_st(iglob_p,iphase),ibool_inv_st(iglob_p+1,iphase)-1)
374  ! enddo
375  !endif
376 
377  ! frees memory
378  deallocate(ibool_inv_num)
379  deallocate(ibool_inv_tbl_tmp)
380 
381  end subroutine make_inv_table
382 
383  end subroutine prepare_timerun_ibool_inv_tbl
