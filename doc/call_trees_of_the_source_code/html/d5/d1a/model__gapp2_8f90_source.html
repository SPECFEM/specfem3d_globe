174  subroutine mantle_gapmodel(radius,theta,phi,dvs,dvp,drho)
175 
176  use constants
177  use gapp2_mantle_model_constants
178 
179  implicit none
180 
181  integer id,ia,io,icon
182  real d,dtheta,dphi
183 
184  double precision radius,theta,phi,dvs,dvp,drho
185 
186 ! factor to convert perturbations in shear speed to perturbations in density
187  double precision, parameter :: SCALE_VS = 1.40d0
188  double precision, parameter :: SCALE_RHO = 0.0d0
189 
190  double precision, parameter :: R_EARTH_ = r_earth_km
191  double precision, parameter :: ZERO_ = 0.d0
192 
193 !.....................................
194 
195  dvs = zero_
196  dvp = zero_
197  drho = zero_
198 
199  ! increments in latitude/longitude (in rad)
200  dtheta = dela * degrees_to_radians
201  dphi = delo * degrees_to_radians
202 
203  ! depth given in km
204  d=r_earth_-radius*r_earth_
205 
206  call d2id(d,nnr,dep,id,icon)
207  if (icon /= 0) then
208  write(6,*)icon
209  write(6,*) radius,theta,phi,dvp,dvs,drho
210  endif
211 
212  ! latitude
213  if (theta >= pi) then
214  ia = na
215  else
216  ia = int(theta / dtheta) + 1
217  endif
218  ! longitude
219  if (phi < 0.0d0) phi = phi + 2.*pi
220  io=int(phi / dphi) + 1
221  if (io > no) io=io-no
222 
223  ! velocity and density perturbations
224  dvp = vp3(ia,io,id)/100.d0
225  dvs = scale_vs*dvp
226  drho = scale_rho*dvs
227 
228  end subroutine mantle_gapmodel

234  subroutine d2id(d,mr,di,id,icon)
235 !.................................................................
236 ! radial section index for a given depth d
237 !.................................................................
238 ! d i depth(km)
239 ! mr i number of radial division
240 ! di i depth table
241 ! id o depth section index for d
242 ! shallow .... di(id-1) <= d < di(id) .... deep
243 ! icon o condition code
244 ! 0:normal, -99:above the surface, 99:below the cmb
245 !.................................................................
246  integer i, mr, id, icon
247  real d,dmax,dmin
248  real di(0:mr)
249  icon = 0
250  dmax=di(mr)
251  dmin=di(0)
252  if (d > dmax) then
253  icon=99
254  else if (d < dmin) then
255  icon=-99
256  else if (d == dmax) then
257  id=mr+1
258  else
259  do i = 0, mr
260  if (d < di(i)) then
261  id=i
262  goto 900
263  endif
264  enddo
265  endif
266 900 continue
267 
268 !..................................................................
269  return
270 
271  end subroutine d2id
