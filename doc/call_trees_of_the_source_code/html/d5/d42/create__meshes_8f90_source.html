28  subroutine create_meshes()
29 
30  use meshfem3d_par
31  implicit none
32 
33  ! local parameters
34  integer :: ipass
35  integer :: ier
36 
37  ! user output
38  if (myrank == 0) then
39  write(imain,*) 'Radial Meshing parameters:'
40  write(imain,*) ' CHUNK WIDTH XI/ETA =', angular_width_xi_in_degrees,'/',angular_width_eta_in_degrees
41  write(imain,*) ' NEX XI/ETA =', nex_xi,'/',nex_eta
42  write(imain,*)
43  write(imain,*) ' NER_CRUST: ', ner_crust
44  write(imain,*) ' NER_80_MOHO: ', ner_80_moho
45  write(imain,*) ' NER_220_80: ', ner_220_80
46  write(imain,*) ' NER_400_220: ', ner_400_220
47  write(imain,*) ' NER_600_400: ', ner_600_400
48  write(imain,*) ' NER_670_600: ', ner_670_600
49  write(imain,*) ' NER_771_670: ', ner_771_670
50  write(imain,*) ' NER_TOPDDOUBLEPRIME_771: ', ner_topddoubleprime_771
51  write(imain,*) ' NER_CMB_TOPDDOUBLEPRIME: ', ner_cmb_topddoubleprime
52  write(imain,*) ' NER_OUTER_CORE: ', ner_outer_core
53  write(imain,*) ' NER_TOP_CENTRAL_CUBE_ICB:', ner_top_central_cube_icb
54  write(imain,*) ' SUPPRESS_CRUSTAL_MESH: ', suppress_crustal_mesh
55  write(imain,*)
56  write(imain,*) ' R_CENTRAL_CUBE = ', sngl(r_central_cube/1000.d0),' km'
57  write(imain,*)
58  write(imain,*) 'Mesh resolution:'
59  write(imain,*) ' DT = ',dt
60  write(imain,*) ' Minimum period = ', &
61  max(angular_width_eta_in_degrees,angular_width_xi_in_degrees)/90.0 * 256.0/min(nex_eta,nex_xi) * 17.0,' (s)'
62  write(imain,*)
63  write(imain,*) ' MIN_ATTENUATION_PERIOD = ',min_attenuation_period
64  write(imain,*) ' MAX_ATTENUATION_PERIOD = ',max_attenuation_period
65  write(imain,*)
66  call flush_imain()
67  endif
68 
69  ! get addressing for this process
70  ichunk = ichunk_slice(myrank)
71  iproc_xi = iproc_xi_slice(myrank)
72  iproc_eta = iproc_eta_slice(myrank)
73 
74  ! volume of the final mesh, and Earth mass computed in the final mesh
75  ! and gravity integrals
76  volume_total = zero
77  earth_mass_total = zero
78  earth_center_of_mass_x_total = zero
79  earth_center_of_mass_y_total = zero
80  earth_center_of_mass_z_total = zero
81 
82  ! make sure everybody is synchronized
83  call synchronize_all()
84 
85  !----
86  !---- loop on all the regions of the mesh
87  !----
88 
89  ! number of regions in full Earth
90  do iregion_code = 1,max_num_regions
91 
92  if (myrank == 0) then
93  write(imain,*)
94  write(imain,*) '*******************************************'
95  write(imain,*) 'creating mesh in region ',iregion_code
96  select case (iregion_code)
97  case (iregion_crust_mantle)
98  write(imain,*) 'this region is the crust and mantle'
99  case (iregion_outer_core)
100  write(imain,*) 'this region is the outer core'
101  case (iregion_inner_core)
102  write(imain,*) 'this region is the inner core'
103  case default
104  call exit_mpi(myrank,'incorrect region code')
105  end select
106  write(imain,*) '*******************************************'
107  write(imain,*)
108  call flush_imain()
109  endif
110 
111  ! compute maximum number of points
112  npointot = nspec(iregion_code) * ngllx * nglly * ngllz
113 
114  ! use dynamic allocation to allocate memory for arrays
115  allocate(idoubling(nspec(iregion_code)), &
116  ibool(ngllx,nglly,ngllz,nspec(iregion_code)), &
117  xstore(ngllx,nglly,ngllz,nspec(iregion_code)), &
118  ystore(ngllx,nglly,ngllz,nspec(iregion_code)), &
119  zstore(ngllx,nglly,ngllz,nspec(iregion_code)), &
120  stat=ier)
121  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating memory for arrays')
122 
123  ! this for non blocking MPI
124  allocate(is_on_a_slice_edge(nspec(iregion_code)), &
125  stat=ier)
126  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating is_on_a_slice_edge array')
127 
128 
129  ! create all the regions of the mesh
130  ! perform two passes in this part to be able to save memory
131  do ipass = 1,2
132  call create_regions_mesh(iregion_code, &
133  nspec(iregion_code),nglob(iregion_code),npointot, &
134  nex_per_proc_xi,nex_per_proc_eta, &
135  nspec2dmax_xmin_xmax(iregion_code),nspec2dmax_ymin_ymax(iregion_code), &
136  nspec2d_bottom(iregion_code),nspec2d_top(iregion_code), &
137  mod(iproc_xi_slice(myrank),2),mod(iproc_eta_slice(myrank),2), &
138  ipass)
139 
140  ! If we're in the request stage of CEM, exit.
141  if (cem_request) exit
142 
143  enddo
144 
145  ! deallocate arrays used for that region
146  deallocate(idoubling)
147  deallocate(ibool)
148  deallocate(xstore)
149  deallocate(ystore)
150  deallocate(zstore)
151 
152  ! this for non blocking MPI
153  deallocate(is_on_a_slice_edge)
154 
155  ! make sure everybody is synchronized
156  call synchronize_all()
157 
158  ! end of loop on all the regions
159  enddo
160 
161  end subroutine create_meshes
