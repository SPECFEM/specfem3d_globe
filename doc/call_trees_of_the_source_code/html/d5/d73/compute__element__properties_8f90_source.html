321  subroutine compute_element_gll_locations(xelm,yelm,zelm,ispec,nspec, &
322  xstore,ystore,zstore,shape3d)
323 
324  use constants
325 
326  implicit none
327 
328  integer :: ispec,nspec
329 
330  double precision,dimension(NGNOD) :: xelm,yelm,zelm
331 
332  double precision,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: xstore,ystore,zstore
333 
334  double precision,dimension(NGNOD,NGLLX,NGLLY,NGLLZ) :: shape3D
335 
336  ! local parameters
337  double precision :: xmesh,ymesh,zmesh
338  integer :: i,j,k,ia
339 
340  do k = 1,ngllz
341  do j = 1,nglly
342  do i = 1,ngllx
343 
344  xmesh = zero
345  ymesh = zero
346  zmesh = zero
347 
348  ! interpolates the location using 3D shape functions
349  do ia = 1,ngnod
350 
351  xmesh = xmesh + shape3d(ia,i,j,k)*xelm(ia)
352  ymesh = ymesh + shape3d(ia,i,j,k)*yelm(ia)
353  zmesh = zmesh + shape3d(ia,i,j,k)*zelm(ia)
354 
355  enddo
356 
357  ! stores mesh coordinates
358  xstore(i,j,k,ispec) = xmesh
359  ystore(i,j,k,ispec) = ymesh
360  zstore(i,j,k,ispec) = zmesh
361 
362  enddo
363  enddo
364  enddo
365 
366  end subroutine compute_element_gll_locations
