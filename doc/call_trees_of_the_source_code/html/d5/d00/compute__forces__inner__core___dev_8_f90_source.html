415  subroutine mxm5_3comp_singlea(A,n1,B1,B2,B3,C1,C2,C3,n3)
416 
417 ! we can force inlining (Intel compiler)
418 !DIR$ ATTRIBUTES FORCEINLINE :: mxm5_3comp_singleA
419 ! cray
420 !DIR$ INLINEALWAYS mxm5_3comp_singleA
421 
422 ! 3 different arrays for x/y/z-components, 2-dimensional arrays (25,5)/(5,25), same B matrix for all 3 component arrays
423 
424  use constants_solver, only: custom_real
425 
426 #ifdef XSMM
427  use my_libxsmm, only: libxsmm_smm_5_25_5
428 #endif
429 
430  implicit none
431 
432  integer,intent(in) :: n1,n3
433  real(kind=CUSTOM_REAL),dimension(n1,5),intent(in) :: A
434  real(kind=CUSTOM_REAL),dimension(5,n3),intent(in) :: B1,B2,B3
435  real(kind=CUSTOM_REAL),dimension(n1,n3),intent(out) :: C1,C2,C3
436 
437  ! local parameters
438  integer :: i,j
439 
440 #ifdef XSMM
441  ! matrix-matrix multiplication C = alpha A * B + beta C
442  ! with A(n1,n2) 5x5-matrix, B(n2,n3) 5x25-matrix and C(n1,n3) 5x25-matrix
443  ! static version using MNK="5 25, 5" ALPHA=1 BETA=0
444  call libxsmm_smm_5_25_5(a=a, b=b1, c=c1, pa=a, pb=b2, pc=c2)
445  call libxsmm_smm_5_25_5(a=a, b=b2, c=c2, pa=a, pb=b3, pc=c3)
446  call libxsmm_smm_5_25_5(a=a, b=b3, c=c3, pa=a, pb=b1, pc=c1) ! with dummy prefetch
447  return
448 #endif
449 
450  ! matrix-matrix multiplication
451  do j = 1,n3
452 !dir$ ivdep
453 !dir$ SIMD
454  do i = 1,n1
455  c1(i,j) = a(i,1) * b1(1,j) &
456  + a(i,2) * b1(2,j) &
457  + a(i,3) * b1(3,j) &
458  + a(i,4) * b1(4,j) &
459  + a(i,5) * b1(5,j)
460 
461  c2(i,j) = a(i,1) * b2(1,j) &
462  + a(i,2) * b2(2,j) &
463  + a(i,3) * b2(3,j) &
464  + a(i,4) * b2(4,j) &
465  + a(i,5) * b2(5,j)
466
