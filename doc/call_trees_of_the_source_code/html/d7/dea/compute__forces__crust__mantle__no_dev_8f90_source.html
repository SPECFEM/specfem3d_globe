431  gyl = gravity_pre_store(2,iglob) ! minus_g*sin_theta*sin_phi * rho
432  gzl = gravity_pre_store(3,iglob) ! minus_g*cos_theta * rho
433 
434  ! Cartesian components of gradient of gravitational acceleration
435  ! get displacement and multiply by density to compute G tensor
436  sx_l = displ_crust_mantle(1,iglob)
437  sy_l = displ_crust_mantle(2,iglob)
438  sz_l = displ_crust_mantle(3,iglob)
439 
440  ! compute G tensor from s . g and add to sigma (not symmetric)
441  sigma_xx = sigma_xx + sy_l * gyl + sz_l * gzl
442  sigma_yy = sigma_yy + sx_l * gxl + sz_l * gzl
443  sigma_zz = sigma_zz + sx_l * gxl + sy_l * gyl
444 
445  sigma_xy = sigma_xy - sx_l * gyl
446  sigma_yx = sigma_yx - sy_l * gxl
447 
448  sigma_xz = sigma_xz - sx_l * gzl
449  sigma_zx = sigma_zx - sz_l * gxl
450 
451  sigma_yz = sigma_yz - sy_l * gzl
452  sigma_zy = sigma_zy - sz_l * gyl
453 
454  hxxl = gravity_h(1,iglob) ! minus_g_over_radius*(cos_phi_sq*cos_theta_sq + sin_phi_sq) + cos_phi_sq*minus_dg*sin_theta_sq * rho
455  hyyl = gravity_h(2,iglob) ! minus_g_over_radius*(cos_phi_sq + cos_theta_sq*sin_phi_sq) + minus_dg*sin_phi_sq*sin_theta_sq * rho
456  hzzl = gravity_h(3,iglob) ! cos_theta_sq*minus_dg + minus_g_over_radius*sin_theta_sq * rho
457  hxyl = gravity_h(4,iglob) ! cos_phi*minus_dg_plus_g_over_radius*sin_phi*sin_theta_sq * rho
458  hxzl = gravity_h(5,iglob) ! cos_phi*cos_theta*minus_dg_plus_g_over_radius*sin_theta * rho
459  hyzl = gravity_h(6,iglob) ! cos_theta*minus_dg_plus_g_over_radius*sin_phi*sin_theta * rho
460 
461  ! precompute vector
462  factor = jacobianl * wgll_cube(i,j,k)
463  rho_s_h(1,i,j,k) = factor * (sx_l * hxxl + sy_l * hxyl + sz_l * hxzl)
464  rho_s_h(2,i,j,k) = factor * (sx_l * hxyl + sy_l * hyyl + sz_l * hyzl)
465  rho_s_h(3,i,j,k) = factor * (sx_l * hxzl + sy_l * hyzl + sz_l * hzzl)
466  endif ! end of section with gravity terms
467 
468  ! form dot product with test vector, non-symmetric form
469  tempx1(i,j,k) = jacobianl * (sigma_xx*xixl + sigma_yx*xiyl + sigma_zx*xizl)
470  tempy1(i,j,k) = jacobianl * (sigma_xy*xixl + sigma_yy*xiyl + sigma_zy*xizl)
471  tempz1(i,j,k) = jacobianl * (sigma_xz*xixl + sigma_yz*xiyl + sigma_zz*xizl)
472 
473  tempx2(i,j,k) = jacobianl * (sigma_xx*etaxl + sigma_yx*etayl + sigma_zx*etazl)
474  tempy2(i,j,k) = jacobianl * (sigma_xy*etaxl + sigma_yy*etayl + sigma_zy*etazl)
475  tempz2(i,j,k) = jacobianl * (sigma_xz*etaxl + sigma_yz*etayl + sigma_zz*etazl)
476 
477  tempx3(i,j,k) = jacobianl * (sigma_xx*gammaxl + sigma_yx*gammayl + sigma_zx*gammazl)
478  tempy3(i,j,k) = jacobianl * (sigma_xy*gammaxl + sigma_yy*gammayl + sigma_zy*gammazl)
479  tempz3(i,j,k) = jacobianl * (sigma_xz*gammaxl + sigma_yz*gammayl + sigma_zz*gammazl)
480 
481  enddo ! NGLLX
482  enddo ! NGLLY
483  enddo ! NGLLZ
484 
485  do k = 1,ngllz
486  do j = 1,nglly
487  do i = 1,ngllx
488 
489  tempx1l = 0._custom_real
490  tempy1l = 0._custom_real
491  tempz1l = 0._custom_real
492 
493  tempx2l = 0._custom_real
494  tempy2l = 0._custom_real
495  tempz2l = 0._custom_real
496 
497  tempx3l = 0._custom_real
498  tempy3l = 0._custom_real
499  tempz3l = 0._custom_real
500 
501  do l = 1,ngllx
502  fac1 = hprimewgll_xx(l,i)
503  tempx1l = tempx1l + tempx1(l,j,k)*fac1
504  tempy1l = tempy1l + tempy1(l,j,k)*fac1
505  tempz1l = tempz1l + tempz1(l,j,k)*fac1
506 !!! can merge these loops because NGLLX = NGLLY = NGLLZ enddo
507 
508 !!! can merge these loops because NGLLX = NGLLY = NGLLZ do l = 1,NGLLY
509  fac2 = hprimewgll_yy(l,j)
510  tempx2l = tempx2l + tempx2(i,l,k)*fac2
511  tempy2l = tempy2l + tempy2(i,l,k)*fac2
512  tempz2l = tempz2l + tempz2(i,l,k)*fac2
513 !!! can merge these loops because NGLLX = NGLLY = NGLLZ enddo
514 
515 !!! can merge these loops because NGLLX = NGLLY = NGLLZ do l = 1,NGLLZ
516  fac3 = hprimewgll_zz(l,k)
517  tempx3l = tempx3l + tempx3(i,j,l)*fac3
518  tempy3l = tempy3l + tempy3(i,j,l)*fac3
519  tempz3l = tempz3l + tempz3(i,j,l)*fac3
520  enddo
521 
522  fac1 = wgllwgll_yz(j,k)
523  fac2 = wgllwgll_xz(i,k)
524  fac3 = wgllwgll_xy(i,j)
525 
526  sum_terms(1,i,j,k) = - (fac1*tempx1l + fac2*tempx2l + fac3*tempx3l)
527  sum_terms(2,i,j,k) = - (fac1*tempy1l + fac2*tempy2l + fac3*tempy3l)
528  sum_terms(3,i,j,k) = - (fac1*tempz1l + fac2*tempz2l + fac3*tempz3l)
529 
530  if (gravity_val) then
531  sum_terms(1,i,j,k) = sum_terms(1,i,j,k) + rho_s_h(1,i,j,k)
532  sum_terms(2,i,j,k) = sum_terms(2,i,j,k) + rho_s_h(2,i,j,k)
533  sum_terms(3,i,j,k) = sum_terms(3,i,j,k) + rho_s_h(3,i,j,k)
534  endif
535 
536  enddo ! NGLLX
537  enddo ! NGLLY
538  enddo ! NGLLZ
539 
540  ! sum contributions from each element to the global mesh and add gravity terms
541  do k = 1,ngllz
542  do j = 1,nglly
543  do i = 1,ngllx
544  iglob = ibool(i,j,k,ispec)
545  accel_crust_mantle(1,iglob) = accel_crust_mantle(1,iglob) + sum_terms(1,i,j,k)
546  accel_crust_mantle(2,iglob) = accel_crust_mantle(2,iglob) + sum_terms(2,i,j,k)
547  accel_crust_mantle(3,iglob) = accel_crust_mantle(3,iglob) + sum_terms(3,i,j,k)
548  enddo
549  enddo
550  enddo
551 
552 ! update memory variables based upon the Runge-Kutta scheme
553 ! convention for attenuation
554 ! term in xx = 1
555 ! term in yy = 2
556 ! term in xy = 3
557 ! term in xz = 4
558 ! term in yz = 5
559 ! term in zz not computed since zero trace
560 ! This is because we only implement Q_\mu attenuation and not Q_\kappa.
561 ! Note that this does *NOT* imply that there is no attenuation for P waves
562 ! because for Q_\kappa = infinity one gets (see for instance Dahlen and Tromp (1998)
563 ! equation (9.59) page 350): Q_\alpha = Q_\mu * 3 * (V_p/V_s)^2 / 4
564 ! therefore Q_\alpha is not zero; for instance for V_p / V_s = sqrt(3)
565 ! we get Q_\alpha = (9 / 4) * Q_\mu = 2.25 * Q_\mu
566 
567  if (attenuation_val .and. .not. partial_phys_dispersion_only_val) then
568  ! updates R_memory
569  if (use_lddrk) then
570  call compute_element_att_memory_cm_lddrk(ispec,r_xx,r_yy,r_xy,r_xz,r_yz, &
571  r_xx_lddrk,r_yy_lddrk,r_xy_lddrk,r_xz_lddrk,r_yz_lddrk, &
572  att1_val,att2_val,att3_val,vnspec,factor_common, &
573  c44store,muvstore, &
574  epsilondev_loc, &
575  deltat)
576  else
577  call compute_element_att_memory_cm(ispec,r_xx,r_yy,r_xy,r_xz,r_yz, &
578  att1_val,att2_val,att3_val,vnspec,factor_common, &
579  alphaval,betaval,gammaval, &
580  c44store,muvstore, &
581  epsilondev_xx,epsilondev_yy,epsilondev_xy, &
582  epsilondev_xz,epsilondev_yz, &
583  epsilondev_loc)
584  endif
585  endif
586 
587  ! save deviatoric strain for Runge-Kutta scheme
588  if (compute_and_store_strain) then
589  !epsilondev(:,:,:,:,ispec) = epsilondev_loc(:,:,:,:)
590  do k = 1,ngllz
591  do j = 1,nglly
592  do i = 1,ngllx
593  epsilondev_xx(i,j,k,ispec) = epsilondev_loc(i,j,k,1)
594  epsilondev_yy(i,j,k,ispec) = epsilondev_loc(i,j,k,2)
595  epsilondev_xy(i,j,k,ispec) = epsilondev_loc(i,j,k,3)
596  epsilondev_xz(i,j,k,ispec) = epsilondev_loc(i,j,k,4)
597  epsilondev_yz(i,j,k,ispec) = epsilondev_loc(i,j,k,5)
598  enddo
599  enddo
600  enddo
601  endif
602 
603  enddo ! spectral element loop NSPEC_CRUST_MANTLE
604 
605  end subroutine compute_forces_crust_mantle_nodev
