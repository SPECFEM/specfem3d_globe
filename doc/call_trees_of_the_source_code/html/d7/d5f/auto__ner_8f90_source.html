52  reference_model_prem,reference_model_iasp91,reference_model_ak135f_no_mud, &
53  reference_model_1066a,reference_model_1dref,reference_model_jp1d,reference_model_sea1d
54  use shared_compute_parameters, only: reference_1d_model
55

57

63  double precision :: RADIAL_LEN_RATIO_CENTRAL_CUBE
64  double precision :: RADIUS_INNER_CORE

100  case (reference_model_1066a)

108  case (reference_model_jp1d)
109  radius_inner_core = 1217.0d0

113  radius_inner_core = 1217.1d0
114  p_velocity_max = 11.09142d0 ! vp

118  ! relative minimum distance between two GLL points
119  ! the roots x_i are given by the first derivative of the Legendre Polynomial: P_n-1'(x_i) = 0
120  !
121  ! note: the x_i interval is between [-1,1], thus relative to the full length, we divide by 2
122  !
123  ! formulas:
124  ! see: https://en.wikipedia.org/wiki/Gaussian_quadrature -> section Gauss-Lobatto rules
125  ! http://mathworld.wolfram.com/LobattoQuadrature.html
126  !
127  ! numerical values:
128  ! see: http://keisan.casio.com/exec/system/1280801905
129 
130  select case (ngllx)
131  case (2)
132  min_gll_point_spacing = 0.5d0 * ( 1.d0 - 1.0 ) ! 1.0
133 
134  case (3)
135  min_gll_point_spacing = 0.5d0 * ( 1.d0 - 0.0 ) ! 0.5
136 
137  case (4)
138  min_gll_point_spacing = 0.5d0 * ( 1.d0 - sqrt(1.d0 / 5.d0) ) ! 0.2764
139 
140  case (5)
141  min_gll_point_spacing = 0.5d0 * ( 1.d0 - sqrt(3.d0 / 7.d0) ) ! 0.1726
142 
143  case (6)
144  min_gll_point_spacing = 0.5d0 * ( 1.d0 - sqrt(1.d0/21.d0*(7.d0 + 2.d0 * sqrt(7.d0))) ) !0.117472
145 
146  case (7)
147  min_gll_point_spacing = 0.5d0 * ( 1.d0 - 0.830223896278566929872 ) ! 0.084888
148 
149  case (8)
150  min_gll_point_spacing = 0.5d0 * ( 1.d0 - 0.8717401485096066153374 ) ! 0.0641299
151 
152  case (9)
153  min_gll_point_spacing = 0.5d0 * ( 1.d0 - 0.8997579954114601573123 ) ! 0.050121
154 
155  case (10)
156  min_gll_point_spacing = 0.5d0 * ( 1.d0 - 0.9195339081664588138289 ) ! 0.040233
157 
158  case default
159  stop 'auto_time_stepping: NGLLX > 10 value not supported yet! please consider adding it...'
160 
161  end select
162 
163  ! element at inner core
164  ! size in horizontal direction
165  elem_size = radial_len_ratio_central_cube * ((width * degrees_to_radians) * radius_inner_core) / &
166  ( dble(nex_max) / doubling_inner_core )
167 
168  ! minimum grid point spacing
169  min_grid_dx = elem_size * min_gll_point_spacing
170 
171  ! estimated time step
172  dt = maximum_stability_condition * min_grid_dx / p_velocity_max
173 
174  end subroutine auto_time_stepping

179  subroutine auto_attenuation_periods(WIDTH, NEX_MAX, MIN_ATTENUATION_PERIOD, MAX_ATTENUATION_PERIOD)
180 
181  use constants, only: n_sls,ngllx
182 
183  implicit none
184 
185  double precision,intent(in) :: WIDTH
186 
187  integer, intent(in) :: NEX_MAX
188  integer, intent(out) :: MIN_ATTENUATION_PERIOD, MAX_ATTENUATION_PERIOD
189 
190  ! local parameters
191  double precision :: TMP
192  double precision :: GLL_SPACING
193  double precision :: S_VELOCITY_MIN
194  double precision :: THETA(5)
195 
196  ! factor degree to km
197  double precision,parameter :: DEG2KM = 111.d0
198 
199  ! required points per wavelength
200  double precision,parameter :: PTS_PER_WAVELENGTH = 4.d0
201 
202  ! safety check
203  if (n_sls < 2 .or. n_sls > 5) then
204  stop 'N_SLS must be greater than 1 or less than 6'
205  endif
206 
207  ! average spacing between GLL points
208  gll_spacing = dble(ngllx - 1)
209 
210  ! minimum velocity (Vs)
211  s_velocity_min = 2.25d0
212 
213  ! Compute Min Attenuation Period
214  !
215  ! width of element in km = (Angular width in degrees / NEX_MAX) * degrees to km
216  tmp = width / dble(nex_max) * deg2km
217 
218  ! average grid node spacing in km = Width of an element in km / spacing for GLL point
219  tmp = tmp / gll_spacing
220 
221  ! minimum resolved wavelength (for fixed number of points per wavelength)
222  tmp = tmp * pts_per_wavelength
223 
224  ! The Minimum attenuation period = (minimum wavelength) / V_min
225  tmp = tmp/s_velocity_min
226 
227  ! The Minimum attenuation period (integer)
228  min_attenuation_period = int(tmp)
229 
230  ! THETA defines the width of the Attenuation Range in Decades
231  ! The number defined here were determined by minimizing
232  ! the "flatness" of the absorption spectrum. Each THETA
233  ! is defined for a particular N_SLS (constants.h)
234  ! THETA(2) is for N_SLS = 2
235  theta(1) = 0.00d0
236  theta(2) = 0.75d0
237  theta(3) = 1.75d0
238  theta(4) = 2.25d0
239  theta(5) = 2.85d0
240 
241  ! Compute Max Attenuation Period
242  !
243  ! The max attenuation period for 3 SLS is optimally
244  ! 1.75 decades from the min attenuation period, see THETA above
245  tmp = tmp * 10.0d0**theta(n_sls)
246 
247  max_attenuation_period = int(tmp)
248 
249  ! debug
250  !print *,'attenuation range min/max: ',MIN_ATTENUATION_PERIOD,MAX_ATTENUATION_PERIOD
251 
252  end subroutine auto_attenuation_periods

398  subroutine auto_optimal_ner(NUM_REGIONS, width, NEX, r, scaling, NER, rt, rb)
399 
400  use constants, only: degrees_to_radians
401 
402  implicit none
403 
404  integer,intent(in) :: NUM_REGIONS
405  integer,intent(in) :: NEX
406  double precision,intent(in) :: width ! Width of the Chunk in Degrees
407 
408  integer, dimension(NUM_REGIONS-1),intent(inout) :: NER ! Elements per Region - IN-N-OUT - Yummy !
409  integer, dimension(NUM_REGIONS) ,intent(in) :: scaling ! Element Doubling - INPUT
410  double precision, dimension(NUM_REGIONS) ,intent(in) :: r ! Radius - INPUT
411  double precision, dimension(NUM_REGIONS-1),intent(out) :: rt ! Ratio at Top - OUTPUT
412  double precision, dimension(NUM_REGIONS-1),intent(out) :: rb ! Ratio at Bottom - OUTPUT
413 
414  ! local parameters
415  double precision :: dr, w, ratio, xi, ximin, wt, wb
416  integer :: ner_test
417  integer :: i
418 
419  ! Find optimal elements per region
420  do i = 1,num_regions-1
421  dr = r(i) - r(i+1) ! Radial Length of Region
422  wt = width * degrees_to_radians * r(i) / (nex*1.0d0 / scaling(i)*1.0d0) ! Element Width Top
423  wb = width * degrees_to_radians * r(i+1) / (nex*1.0d0 / scaling(i)*1.0d0) ! Element Width Bottom
424  w = (wt + wb) * 0.5d0 ! Average Width of Region
425  ner_test = ner(i) ! Initial solution
426  ratio = (dr / ner_test) / w ! Aspect Ratio of Element
427  xi = dabs(ratio - 1.0d0) ! Aspect Ratio should be near 1.0
428  ximin = 1.e7 ! Initial Minimum
429 
430  !debug
431  !print *,'region ',i,'element ratio: ',ratio,'xi = ',xi,'width = ',w
432 
433  ! increases NER to reach vertical/horizontal element ratio of about 1
434  do while(xi <= ximin)
435  ner(i) = ner_test ! Found a better solution
436  ximin = xi !
437  ner_test = ner_test + 1 ! Increment ner_test and
438  ratio = (dr / ner_test) / w ! look for a better
439  xi = dabs(ratio - 1.0d0) ! solution
440  enddo
441  rt(i) = dr / ner(i) / wt ! Find the Ratio of Top
442  rb(i) = dr / ner(i) / wb ! and Bottom for completeness
443 
444  !debug
445  !print *,'region ',i,'element ratio: top = ',rt(i),'bottom = ',rb(i)
446  enddo
447 
448  end subroutine auto_optimal_ner

454  subroutine find_r_central_cube(nex_xi_in, rcube, nex_eta_in)
455 
456  implicit none
457 
458  integer,intent(in) :: nex_xi_in
459  double precision,intent(out) :: rcube
460  integer,intent(out) :: nex_eta_in
461 
462  ! local parameters
463  integer, parameter :: NBNODE = 8
464  double precision, parameter :: alpha = 0.41d0
465 
466  integer :: npts
467  integer :: nex_xi

472  double precision :: elem(nbnode+1, 2)

475  double precision :: max_edgemax, min_edgemin

478  nex_xi = nex_xi_in / 16

485  rcube = rcube_test

491

493

499 
500  nspec = nspec_cube + nspec_chunks
501  do ispec = 1,nspec
502  call get_element(points, ispec, npts, elem)
503  call get_size_min_max(elem, edgemax, edgemin)
504  aspect_ratio = edgemax / edgemin
505  max_edgemax = max(max_edgemax, edgemax)
506  min_edgemin = min(min_edgemin, edgemin)
507  max_aspect_ratio = max(max_aspect_ratio, aspect_ratio)
508  enddo
509  xi = (max_edgemax / min_edgemin)
510 ! xi = abs(rcube_test - 981.0d0) / 45.0d0
511 ! write(*,'(a,5(f14.4,2x))')'rcube, xi, ximin:-',rcube_test, xi, min_edgemin,max_edgemax,max_aspect_ratio
512  deallocate(points)
513 
514  if (xi < ximin) then
515  ximin = xi
516  rcube = rcube_test
517  nex_eta_in = nex_eta
518  endif
519  rcube_test = rcube_test + rcubestep
520  enddo
521 
522  end subroutine find_r_central_cube

528  subroutine compute_nex(nex_xi, rcube, alpha, ner)
529 
530  use constants, only: pi,pi_over_two,pi_over_four
531 
532  implicit none
533 
534  double precision, parameter :: RICB_KM = 1221.0d0
535 
536  integer nex_xi, ner
537  double precision rcube, alpha
538  integer ix
539  double precision ratio_x, factx, xi
540  double precision x, y
541  double precision surfx, surfy
542  double precision dist_cc_icb, somme, dist_moy
543 
544  somme = 0.0d0
545 
546  do ix = 0,nex_xi/2,1
547  ratio_x = (ix * 1.0d0) / ( nex_xi * 1.0d0)
548  factx = 2.0d0 * ratio_x - 1.0d0
549  xi = pi_over_two * factx
550  x = (rcube / sqrt(2.0d0)) * factx
551  y = (rcube / sqrt(2.0d0)) * (1 + cos(xi) * alpha / pi_over_two)
552 
553  surfx = ricb_km * cos(3 * pi_over_four - ratio_x * pi_over_two)
554  surfy = ricb_km * sin(3 * pi_over_four - ratio_x * pi_over_two)
555 
556  dist_cc_icb = sqrt((surfx -x)**2 + (surfy - y)**2)
557  if (ix /= nex_xi/2) then
558  dist_cc_icb = dist_cc_icb * 2
559  endif
560  somme = somme + dist_cc_icb
561  enddo
562  dist_moy = somme / (nex_xi + 1)

707  subroutine compute_coordinate(ix,iy,nbx, nby, rcube, ic, alpha, x, y)
708 
709  use constants, only: pi_over_two,pi_over_four
710 
711  implicit none
712 
713  double precision, parameter :: RICB_KM = 1221.0d0
714 
715  integer ix, iy, nbx, nby, ic
716  double precision rcube, alpha
717  double precision x, y
718 
719  double precision ratio_x, ratio_y
720  double precision factx, xi

724  double precision temp
725 
726  ratio_x = (ix * 1.0d0) / (nbx * 1.0d0)
727  ratio_y = (iy * 1.0d0) / (nby * 1.0d0)
728 
729  factx = 2.0d0 * ratio_x - 1.0d0
730  xi = pi_over_two * factx
731 
732  xcc = (rcube / sqrt(2.0d0)) * factx
733  ycc = (rcube / sqrt(2.0d0)) * (1 + cos(xi) * alpha / pi_over_two)
734 
735  xsurf = ricb_km * cos(3.0d0 * pi_over_four - ratio_x * pi_over_two)
736  ysurf = ricb_km * sin(3.0d0 * pi_over_four - ratio_x * pi_over_two)
737 
738  deltax = xsurf - xcc
739  deltay = ysurf - ycc
740 
741  x = xsurf - ratio_y * deltax
742  y = ysurf - ratio_y * deltay
743 
744  if (ic == 1) then
745  temp = x
746  x = y
747  y = temp
748  else if (ic == 2) then
749  x = -x
750  y = -y
751  else if (ic == 3) then
752  temp = x
753  x = -y
754  y = temp
755  endif
756 
757  end subroutine compute_coordinate
