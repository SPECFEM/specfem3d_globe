108  subroutine read_crust_sh_model()
109 
110  use constants, only: zero,iin,imain,max_string_len,hugeval
111 
112  use model_full_sh_crust_par
113 
114  implicit none
115 
116  ! local parameters
117  integer :: l,lmax,nm,ipar,ier
118  character(len=32) :: modelname
119  character(len=MAX_STRING_LEN) :: rootdir
120  ! for moho statistics
121  integer :: i,j
122  real(kind=4) :: shcof(nsh_40)
123  real(kind=4) :: work1(ns_40+1)
124  real(kind=4) :: work2(ns_40+1)
125  real(kind=4) :: work3(ns_40+1)
126  real(kind=4) :: xlat,xlon
127  double precision :: moho,moho_min,moho_max
128 
129  ! user output
130  write(imain,*)
131  write(imain,*) 'incorporating crustal model: crust_sh '
132  write(imain,*)
133  call flush_imain()
134 
135  ! initialize
136  do l = 1,nsh_40
137  crust_sh_v_vsh(l) = zero
138  crust_sh_v_vsv(l) = zero
139  crust_sh_v_vph(l) = zero
140  crust_sh_v_vpv(l) = zero
141  crust_sh_v_eta(l) = zero
142  crust_sh_v_rho(l) = zero
143  crust_sh_v_moho(l) = zero
144  enddo
145 
146  ! user output
147  write(imain,*) ' reading crustal spherical harmonic model from full_sh:'
148  call flush_imain()
149 
150  ! root directory
151  rootdir = 'DATA/full_sphericalharmonic_model/'
152 
153  do ipar = 1,7
154 
155  select case(ipar)
156  case (1)
157  ! VSV degree 40 max model 1 layer
158  modelname = 'CVSV.dat'
159  case (2)
160  ! VSH
161  modelname = 'CVSH.dat'
162  case (3)
163  ! VPV
164  modelname = 'CVPV.dat'
165  case (4)
166  ! VPH
167  modelname = 'CVPH.dat'
168  case (5)
169  ! ETA
170  modelname = 'CETA.dat'
171  case (6)
172  ! RHO
173  modelname = 'CRHO.dat'
174  case (7)
175  ! MOHO degree 40 model
176  modelname = 'MOHO.dat'
177  case default
178  stop 'Invalid parameter for crust file in full_sh'
179  end select
180 
181  ! opens model file
182  open(unit=iin,file=trim(rootdir)//trim(modelname),status='old',action='read',iostat=ier)
183  if (ier /= 0) then
184  print *,'Error opening file: ',trim(rootdir)//trim(modelname)
185  call exit_mpi(0,'Error opening file full_sphericalharmonic model file')
186  endif
187 
188  read(iin,*) lmax
189 
190  ! check maximum degree
191  if (lmax > ns_40) then
192  print *,'Error full_sh model file: lmax too big ',lmax,' for file ',trim(rootdir)//trim(modelname)
193  call exit_mpi(0,'lmax too big in full_sh model file')
194  endif
195 
196  ! user output
197  write(imain,*) ' name ',trim(modelname),': lmax = ',lmax
198  call flush_imain()
199 
200  ! reads in spherical harmonics coefficients
201  nm = (lmax+1)**2
202 
203  select case(ipar)
204  case (1)
205  read(iin,*) (crust_sh_v_vsv(l),l=1,nm)
206  case (2)
207  read(iin,*) (crust_sh_v_vsh(l),l=1,nm)
208  case (3)
209  read(iin,*) (crust_sh_v_vpv(l),l=1,nm)
210  case (4)
211  read(iin,*) (crust_sh_v_vph(l),l=1,nm)
212  case (5)
213  read(iin,*) (crust_sh_v_eta(l),l=1,nm)
214  case (6)
215  read(iin,*) (crust_sh_v_rho(l),l=1,nm)
216  case (7)
217  read(iin,*) (crust_sh_v_moho(l),l=1,nm)
218  end select
219 
220  close(iin)
221 
222  enddo
223 
224  ! moho thickness statistics
225  ! gets min/max value for a set of sample locations (regular lat/lon grid points)
226  moho_min = hugeval
227  moho_max = -hugeval
228  do i = -90,90
229  do j = -180,180
230  ! lat/lon in degrees
231  xlat = i * 1.0
232  xlon = j * 1.0
233  call ylm(xlat,xlon,ns_40,shcof,work1,work2,work3)
234 
235  ! gets moho depth (in km)
236  moho = zero
237  do l = 1,nsh_40
238  moho = moho + crust_sh_v_moho(l)*dble(shcof(l))
239  enddo
240 
241  ! checks
242  if (moho <= zero) then
243  print *,'Error in crustal sh model: invalid moho depth ',moho,' at lat/lon = ',xlat,xlon
244  stop 'Error invalid moho depth in full_sh model'
245  endif
246 
247  ! statistics
248  if (moho > moho_max) moho_max = moho
249  if (moho < moho_min) moho_min = moho
250  enddo
251  enddo
252 
253  ! user output
254  write(imain,*)
255  write(imain,*) ' Moho crustal thickness min/max = ',sngl(moho_min),sngl(moho_max),' km'
256  call flush_imain()
257 
258  end subroutine read_crust_sh_model

264  subroutine crust_sh(lat,lon,r,vpvc,vphc,vsvc,vshc,etac,rhoc,moho,found_crust,elem_in_crust)
265 
266 ! gets crustal value for location lat/lon/r
267 
268  use constants, only: zero,pi,grav,rhoav,r_earth,r_earth_km
269 
270  use model_full_sh_crust_par
271 
272  implicit none
273 
274  double precision,intent(in) :: lat,lon,r
275  double precision,intent(out) :: vpvc,vphc,vsvc,vshc,etac,rhoc,moho
276  logical,intent(out) :: found_crust
277  logical,intent(in) :: elem_in_crust
278 
279  ! local parameters
280  double precision :: depth,scaleval
281  integer :: l
282  real(kind=4) :: shcof(nsh_40)
283  real(kind=4) :: work1(ns_40+1)
284  real(kind=4) :: work2(ns_40+1)
285  real(kind=4) :: work3(ns_40+1)
286  real(kind=4) :: xlat,xlon
287 
288  ! gets coefficient for location
289  xlat = sngl(lat)
290  xlon = sngl(lon)
291  call ylm(xlat,xlon,ns_40,shcof,work1,work2,work3)
292 
293  ! calculates model values
294  vsvc = zero !km/s
295  vshc = zero !km/s
296  vpvc = zero !km/s
297  vphc = zero !km/s
298  rhoc = zero !g/cm^3
299  moho = zero !km
300  etac = zero
301  do l = 1,nsh_40
302  vsvc = vsvc + crust_sh_v_vsv(l)*dble(shcof(l))
303  vshc = vshc + crust_sh_v_vsh(l)*dble(shcof(l))
304  vpvc = vpvc + crust_sh_v_vpv(l)*dble(shcof(l))
305  vphc = vphc + crust_sh_v_vph(l)*dble(shcof(l))
306  etac = etac + crust_sh_v_eta(l)*dble(shcof(l))
307  rhoc = rhoc + crust_sh_v_rho(l)*dble(shcof(l))
308  moho = moho + crust_sh_v_moho(l)*dble(shcof(l))
309  enddo
310 
311  ! check values
312  found_crust = .true.
313  if (vsvc <= zero) found_crust = .false.
314  if (vshc <= zero) found_crust = .false.
315  if (vpvc <= zero) found_crust = .false.
316  if (vphc <= zero) found_crust = .false.
317  if (etac < 0.6d0) found_crust = .false.
318  if (etac > 1.4d0) found_crust = .false.
319  if (rhoc <= zero) found_crust = .false.
320  if (moho <= zero) found_crust = .false.
321 
322  ! checks if crustal value found
323  if (.not. found_crust) then
324  print *,'Error: crustal value not found in full_sh model! position is at lat/lon/r = ',lat,lon,r
325  stop 'crustal value not found in full_sh model, please check...'
326  endif
327 
328  ! checks if position is in crust
329  found_crust = .false.
330  if (elem_in_crust) then
331  ! forces points to have crustal values
332  found_crust = .true.
333  else
334  ! checks if depth of position above/below moho
335  depth = (1.d0 - r) * r_earth_km
336  !debug
337  !print *,'sh crust: depth = ',depth,' moho = ',moho
338  if (depth <= moho) then
339  found_crust = .true.
340  endif
341  endif
342 
343  ! scales (non-dimensionalizes) values
344  moho = moho * 1000.d0/r_earth
345  if (found_crust) then
346  scaleval = dsqrt(pi*grav*rhoav)
347  vsvc = vsvc * 1000.d0/(r_earth*scaleval)
348  vshc = vshc * 1000.d0/(r_earth*scaleval)
349  vpvc = vpvc * 1000.d0/(r_earth*scaleval)
350  vphc = vphc * 1000.d0/(r_earth*scaleval)
351  rhoc = rhoc * 1000.d0/rhoav
352  endif
353 
354  end subroutine crust_sh

830  double precision function mantle_sh_rsple(I1,I2,X,Y,Q,S)
831 
832  implicit none
833

835 ! using the cubic spline coefficients computed by rspln and saved in Q.
836 ! If S is outside the interval (x(i1),x(i2)) rsple extrapolates
837 ! using the first or last interpolation polynomial. The arrays must
838 ! be dimensioned at least - x(i2), y(i2), and q(3,i2).

845 
846  i = 1
847  ii = i2-1

850  i = max0(i,i1)
851  i = min0(i,ii)
852 
853  ! SEE IF X IS INCREASING OR DECREASING.
854  if (x(i2)-x(i1) < 0) goto 1
855  if (x(i2)-x(i1) >= 0) goto 2
856 
857  ! X IS DECREASING. CHANGE I AS NECESSARY.
858 1 if (s-x(i) <= 0) goto 3
859  if (s-x(i) > 0) goto 4
860 
861 4 i = i-1
862 
863  if (i-i1 < 0) goto 11
864  if (i-i1 == 0) goto 6
865  if (i-i1 > 0) goto 1
866 
867 3 if (s-x(i+1) < 0) goto 5
868  if (s-x(i+1) >= 0) goto 6
869 
870 5 i = i+1
871 
872  if (i-ii < 0) goto 3
873  if (i-ii == 0) goto 6
874  if (i-ii > 0) goto 7
875 
876  ! X IS INCREASING. CHANGE I AS NECESSARY.
877 2 if (s-x(i+1) <= 0) goto 8
878  if (s-x(i+1) > 0) goto 9
879 
880 9 i = i+1
881 
882  if (i-ii < 0) goto 2
883  if (i-ii == 0) goto 6
884  if (i-ii > 0) goto 7
885 
886 8 if (s-x(i) < 0) goto 10
887  if (s-x(i) >= 0) goto 6
888 
889 10 i = i-1
890  if (i-i1 < 0) goto 11
891  if (i-i1 == 0) goto 6
892  if (i-i1 > 0) goto 8
893 
894 7 i = ii
895  goto 6
896 
897 11 i = i1
898

900 6 h = s-x(i)

910  subroutine mantle_sh_rspln(I1,I2,X,Y,Q,F)
911 
912  implicit none
913 
914 ! The subroutine rspln computes cubic spline interpolation coefficients
915 ! for y(x) between grid points i1 and i2 saving them in q.The
916 ! interpolation is continuous with continuous first and second
917 ! derivatives. It agrees exactly with y at grid points and with the
918 ! three point first derivatives at both end points (i1 and i2).
919 ! X must be monotonic but if two successive values of x are equal
920 ! a discontinuity is assumed and separate interpolation is done on
921 ! each strictly monotonic segment. The arrays must be dimensioned at
922 ! least - x(i2), y(i2), q(3,i2), and f(3,i2).
923 ! F is working storage for rspln.
924 
925  integer :: i1,i2
926  double precision :: X(21),Y(21),Q(3,21),F(3,21)
927 
928  integer :: i,j,k,j1,j2
929  double precision :: y0,a0,b0,b1,h,h2,ha,h2a,h3a,h2b
930  double precision :: YY(3),small
931 
932  equivalence(yy(1),y0)
933 
934  data small/1.0d-08/,yy/0.0d0,0.0d0,0.0d0/
935 
936  j1 = i1+1
937  y0 = 0.0d0
938 
939  ! BAIL OUT IF THERE ARE LESS THAN TWO POINTS TOTAL
940  if (i2-i1 < 0) return
941  if (i2-i1 == 0) goto 17
942  if (i2-i1 > 0) goto 8
943 
944 8 a0 = x(j1-1)
945 
946  ! SEARCH FOR DISCONTINUITIES.
947  do 3 i = j1,i2
948  b0 = a0
949  a0 = x(i)
950  if (dabs((a0-b0)/dmax1(a0,b0)) < small) goto 4
951 
952 3 continue
953 
954 17 j1 = j1-1
955  j2 = i2-2
956  goto 5
957 
958 4 j1 = j1-1
959  j2 = i-3
960 
961  ! SEE IF THERE ARE ENOUGH POINTS TO INTERPOLATE (AT LEAST THREE).
962 5 if (j2+1-j1 < 0) goto 9
963  if (j2+1-j1 == 0) goto 10
964  if (j2+1-j1 > 0) goto 11
965 
966  ! ONLY TWO POINTS. USE LINEAR INTERPOLATION.
967 10 j2 = j2+2
968  y0 = (y(j2)-y(j1))/(x(j2)-x(j1))
969  do j = 1,3
970  q(j,j1) = yy(j)
971  q(j,j2) = yy(j)
972  enddo
973  goto 12
974 
975  ! MORE THAN TWO POINTS. DO SPLINE INTERPOLATION.
976 11 a0 = 0.
977  h = x(j1+1)-x(j1)
978  h2 = x(j1+2)-x(j1)
979  y0 = h*h2*(h2-h)
980  h = h*h
981  h2 = h2*h2
982 
983  ! CALCULATE DERIVITIVE AT NEAR END.
984  b0 = (y(j1)*(h-h2)+y(j1+1)*h2-y(j1+2)*h)/y0
985  b1 = b0
986 
987  ! EXPLICITLY REDUCE BANDED MATRIX TO AN UPPER BANDED MATRIX.
988  do i = j1,j2
989  h = x(i+1)-x(i)
990  y0 = y(i+1)-y(i)
991  h2 = h*h
992  ha = h-a0
993  h2a = h-2.0d0*a0
994  h3a = 2.0d0*h-3.0d0*a0
995  h2b = h2*b0
996  q(1,i) = h2/ha
997  q(2,i) = -ha/(h2a*h2)
998  q(3,i) = -h*h2a/h3a
999  f(1,i) = (y0-h*b0)/(h*ha)
1000  f(2,i) = (h2b-y0*(2.0d0*h-a0))/(h*h2*h2a)
1001  f(3,i) = -(h2b-3.0d0*y0*ha)/(h*h3a)
1002  a0 = q(3,i)
1003  b0 = f(3,i)
1004  enddo
1005 
1006  ! TAKE CARE OF LAST TWO ROWS.
1007  i = j2+1
1008  h = x(i+1)-x(i)
1009  y0 = y(i+1)-y(i)
1010  h2 = h*h
1011  ha = h-a0
1012  h2a = h*ha
1013  h2b = h2*b0-y0*(2.0d0*h-a0)
1014  q(1,i) = h2/ha
1015  f(1,i) = (y0-h*b0)/h2a
1016  ha = x(j2)-x(i+1)
1017  y0 = -h*ha*(ha+h)
1018  ha = ha*ha
1019 
1020  ! CALCULATE DERIVATIVE AT FAR END.
1021  y0 = (y(i+1)*(h2-ha)+y(i)*ha-y(j2)*h2)/y0
1022  q(3,i) = (y0*h2a+h2b)/(h*h2*(h-2.0d0*a0))
1023  q(2,i) = f(1,i)-q(1,i)*q(3,i)
1024 
1025  ! SOLVE UPPER BANDED MATRIX BY REVERSE ITERATION.
1026  do j = j1,j2
1027  k = i-1
1028  q(1,i) = f(3,k)-q(3,k)*q(2,i)
1029  q(3,k) = f(2,k)-q(2,k)*q(1,i)
1030  q(2,k) = f(1,k)-q(1,k)*q(3,k)
1031  i = k
1032  enddo
1033  q(1,i) = b1
1034 
1035  ! FILL IN THE LAST POINT WITH A LINEAR EXTRAPOLATION.
1036 9 j2 = j2+2
1037  do j=1,3
1038  q(j,j2) = yy(j)
1039  enddo
1040 
1041  ! SEE IF THIS DISCONTINUITY IS THE LAST.
1042 12 if (j2-i2 < 0) then
1043  goto 6
1044  else
1045  return
1046  endif
1047 
1048  ! NO. GO BACK FOR MORE.
1049 6 j1 = j2+2
1050  if (j1-i2 <= 0) goto 8
1051  if (j1-i2 > 0) goto 7
1052 
1053  ! THERE IS ONLY ONE POINT LEFT AFTER THE LATEST DISCONTINUITY.
1054 7 do j = 1,3
1055  q(j,i2) = yy(j)
1056  enddo
1057 
1058  end subroutine mantle_sh_rspln
