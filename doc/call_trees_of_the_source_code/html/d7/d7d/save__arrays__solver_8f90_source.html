322  subroutine save_arrays_solver_meshfiles(nspec)
323 
324 ! outputs model files in binary format
325 
326  use constants
327 
328  use meshfem3d_models_par, only: &
329  transverse_isotropy,attenuation,attenuation_3d,attenuation_1d_with_3d_storage
330 
331  use create_regions_mesh_par2, only: &
332  rhostore,kappavstore,kappahstore,muvstore,muhstore,eta_anisostore, &
333  qmu_store, &
334  prname
335 
336  implicit none
337 
338  integer :: nspec
339 
340  ! local parameters
341  integer :: i,j,k,ispec,ier
342  real(kind=CUSTOM_REAL) :: scaleval1,scaleval2
343  real(kind=CUSTOM_REAL),dimension(:,:,:,:),allocatable :: temp_store
344 
345  ! scaling factors to re-dimensionalize units
346  scaleval1 = sngl( sqrt(pi*grav*rhoav)*(r_earth/1000.0d0) )
347  scaleval2 = sngl( rhoav/1000.0d0 )
348 
349  ! uses temporary array
350  allocate(temp_store(ngllx,nglly,ngllz,nspec),stat=ier)
351  if (ier /= 0) stop 'Error allocating temp_store array'
352  temp_store(:,:,:,:) = 0._custom_real
353 
354  ! isotropic model
355  ! vp
356  open(unit=iout,file=prname(1:len_trim(prname))//'vp.bin', &
357  status='unknown',form='unformatted',action='write',iostat=ier)
358  if (ier /= 0 ) call exit_mpi(myrank,'Error opening vp.bin file')
359 
360  temp_store(:,:,:,:) = sqrt((kappavstore(:,:,:,:) + 4.0_custom_real * muvstore(:,:,:,:)/3.0_custom_real)/rhostore(:,:,:,:)) &
361  * scaleval1
362  write(iout) temp_store
363  close(iout)
364  ! vs
365  open(unit=iout,file=prname(1:len_trim(prname))//'vs.bin', &
366  status='unknown',form='unformatted',action='write',iostat=ier)
367  if (ier /= 0 ) call exit_mpi(myrank,'Error opening vs.bin file')
368 
369  temp_store(:,:,:,:) = sqrt( muvstore(:,:,:,:)/rhostore(:,:,:,:) )*scaleval1
370  write(iout) temp_store
371  close(iout)
372  ! rho
373  open(unit=iout,file=prname(1:len_trim(prname))//'rho.bin', &
374  status='unknown',form='unformatted',action='write',iostat=ier)
375  if (ier /= 0 ) call exit_mpi(myrank,'Error opening rho.bin file')
376 
377  temp_store(:,:,:,:) = rhostore(:,:,:,:) * scaleval2
378  write(iout) temp_store
379  close(iout)
380 
381  ! transverse isotropic model
382  if (transverse_isotropy) then
383  ! vpv
384  open(unit=iout,file=prname(1:len_trim(prname))//'vpv.bin', &
385  status='unknown',form='unformatted',action='write',iostat=ier)
386  if (ier /= 0 ) call exit_mpi(myrank,'Error opening vpv.bin file')
387 
388  temp_store(:,:,:,:) = sqrt((kappavstore(:,:,:,:) + 4.0_custom_real * muvstore(:,:,:,:)/3.0_custom_real)/rhostore(:,:,:,:)) &
389  * scaleval1
390  write(iout) temp_store
391  close(iout)
392 
393  ! vph
394  open(unit=iout,file=prname(1:len_trim(prname))//'vph.bin', &
395  status='unknown',form='unformatted',action='write',iostat=ier)
396  if (ier /= 0 ) call exit_mpi(myrank,'Error opening vph.bin file')
397 
398  temp_store(:,:,:,:) = sqrt((kappahstore(:,:,:,:) + 4.0_custom_real * muhstore(:,:,:,:)/3.0_custom_real)/rhostore(:,:,:,:)) &
399  * scaleval1
400  write(iout) temp_store
401  close(iout)
402 
403  ! vsv
404  open(unit=iout,file=prname(1:len_trim(prname))//'vsv.bin', &
405  status='unknown',form='unformatted',action='write',iostat=ier)
406  if (ier /= 0 ) call exit_mpi(myrank,'Error opening vsv.bin file')
407 
408  temp_store(:,:,:,:) = sqrt( muvstore(:,:,:,:)/rhostore(:,:,:,:) )*scaleval1
409  write(iout) temp_store
410  close(iout)
411 
412  ! vsh
413  open(unit=iout,file=prname(1:len_trim(prname))//'vsh.bin', &
414  status='unknown',form='unformatted',action='write',iostat=ier)
415  if (ier /= 0 ) call exit_mpi(myrank,'Error opening vsh.bin file')
416 
417  temp_store(:,:,:,:) = sqrt( muhstore(:,:,:,:)/rhostore(:,:,:,:) )*scaleval1
418  write(iout) temp_store
419  close(iout)
420 
421  ! rho
422  open(unit=iout,file=prname(1:len_trim(prname))//'rho.bin', &
423  status='unknown',form='unformatted',action='write',iostat=ier)
424  if (ier /= 0 ) call exit_mpi(myrank,'Error opening rho.bin file')
425 
426  temp_store(:,:,:,:) = rhostore(:,:,:,:) * scaleval2
427  write(iout) temp_store
428  close(iout)
429 
430  ! eta
431  open(unit=iout,file=prname(1:len_trim(prname))//'eta.bin', &
432  status='unknown',form='unformatted',action='write',iostat=ier)
433  if (ier /= 0 ) call exit_mpi(myrank,'Error opening eta.bin file')
434  write(iout) eta_anisostore
435  close(iout)
436  endif ! TRANSVERSE_ISOTROPY
437 
438  ! shear attenuation
439  if (attenuation) then
440  ! saves Qmu_store to full CUSTOM_REAL array
441  if (attenuation_3d .or. attenuation_1d_with_3d_storage) then
442  ! attenuation arrays are fully 3D
443  temp_store(:,:,:,:) = qmu_store(:,:,:,:)
444  else
445  ! attenuation array dimensions: Q_mustore(1,1,1,nspec)
446  do ispec = 1,nspec
447  do k = 1,ngllz
448  do j = 1,nglly
449  do i = 1,ngllx
450  temp_store(i,j,k,ispec) = qmu_store(1,1,1,ispec)
451  enddo
452  enddo
453  enddo
454  enddo
455  endif
456 
457  ! Qmu
458  open(unit=iout,file=prname(1:len_trim(prname))//'qmu.bin', &
459  status='unknown',form='unformatted',action='write',iostat=ier)
460  if (ier /= 0 ) call exit_mpi(myrank,'Error opening qmu.bin file')
461  write(iout) temp_store
462  close(iout)
463  endif ! ATTENUATION
464 
465  ! frees temporary memory
466  deallocate(temp_store)
467 
468  end subroutine save_arrays_solver_meshfiles
