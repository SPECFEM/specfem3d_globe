170 
171  ! compute typical size of elements at the surface
172  typical_size_squared = two_pi * r_unit_sphere / (4.0 * nex_xi_val)
173 
174  ! use 10 times the distance as a criterion for source detection
175  typical_size_squared = (10. * typical_size_squared)**2
176 
177  ! initializes source mask
178  if (save_source_mask .and. simulation_type == 3) then
179  allocate(mask_source(ngllx,nglly,ngllz,nspec),stat=ier)
180  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating mask source array')
181  mask_source(:,:,:,:) = 1.0_custom_real
182  endif
183 
184  ! appends receiver locations to sr.vtk file
185  if (myrank == 0) then
186  open(iout_vtk,file=trim(output_files)//'/sr_tmp.vtk', &
187  position='append',status='old',iostat=ier)
188  if (ier /= 0 ) call exit_mpi(myrank,'Error opening and appending sources to file sr_tmp.vtk')
189  endif
190 
191  ! loop on all the sources
192  ! gather source information in subsets to reduce memory requirements
193 
194  ! loop over subsets of sources
195  do isources_already_done = 0, nsources, nsources_subset_max
196 
197  ! the size of the subset can be the maximum size, or less (if we are in the last subset,
198  ! or if there are fewer sources than the maximum size of a subset)
199  nsources_subset_current_size = min(nsources_subset_max, nsources - isources_already_done)
200 
201  ! allocate arrays specific to each subset
202  allocate(final_distance_source_subset(nsources_subset_current_size), &
203  ispec_selected_source_subset(nsources_subset_current_size), &
204  xi_source_subset(nsources_subset_current_size), &
205  eta_source_subset(nsources_subset_current_size), &
206  gamma_source_subset(nsources_subset_current_size), &
207  x_found_source(nsources_subset_current_size), &
208  y_found_source(nsources_subset_current_size), &
209  z_found_source(nsources_subset_current_size),stat=ier)
210  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating temporary source arrays')
211 
212  ! arrays to collect data
213  if (myrank == 0) then
214  allocate(ispec_selected_source_all(nsources_subset_current_size,0:nproctot_val-1), &
215  xi_source_all(nsources_subset_current_size,0:nproctot_val-1), &
216  eta_source_all(nsources_subset_current_size,0:nproctot_val-1), &
217  gamma_source_all(nsources_subset_current_size,0:nproctot_val-1), &
218  final_distance_source_all(nsources_subset_current_size,0:nproctot_val-1), &
219  x_found_source_all(nsources_subset_current_size,0:nproctot_val-1), &
220  y_found_source_all(nsources_subset_current_size,0:nproctot_val-1), &
221  z_found_source_all(nsources_subset_current_size,0:nproctot_val-1),stat=ier)
222  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating temporary source arrays for gather')
223  else
224  ! dummy arrays
225  allocate(ispec_selected_source_all(1,1), &
226  xi_source_all(1,1), &
227  eta_source_all(1,1), &
228  gamma_source_all(1,1), &
229  final_distance_source_all(1,1), &
230  x_found_source_all(1,1), &
231  y_found_source_all(1,1), &
232  z_found_source_all(1,1),stat=ier)
233  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating temporary source dummy arrays for gather')
234  endif
235  ! use -1 as a flag to detect if gather fails for some reason
236  ispec_selected_source_all(:,:) = -1
237 
238  ! make sure we clean the subset array before the gather
239  ispec_selected_source_subset(:) = 0
240  final_distance_source_subset(:) = hugeval
241  final_distance_source_all(:,:) = hugeval
242 
243  ! loop over sources within this subset
244  do isource_in_this_subset = 1,nsources_subset_current_size
245 
246  ! mapping from source number in current subset to real source number in all the subsets
247  isource = isource_in_this_subset + isources_already_done
248 
249  ! convert geographic latitude lat (degrees) to geocentric colatitude theta (radians)
250  call lat_2_geocentric_colat_dble(lat(isource),theta)
251 
252  phi = long(isource)*degrees_to_radians
253  call reduce(theta,phi)
254 
255  ! get the moment tensor
256  mrr = moment_tensor(1,isource)
257  mtt = moment_tensor(2,isource)
258  mpp = moment_tensor(3,isource)
259  mrt = moment_tensor(4,isource)
260  mrp = moment_tensor(5,isource)
261  mtp = moment_tensor(6,isource)
262 
263  ! convert from a spherical to a Cartesian representation of the moment tensor
264  st=dsin(theta)
265  ct=dcos(theta)
266  sp=dsin(phi)
267  cp=dcos(phi)
268 
269  mxx(isource)=st*st*cp*cp*mrr+ct*ct*cp*cp*mtt+sp*sp*mpp &
270  +2.0d0*st*ct*cp*cp*mrt-2.0d0*st*sp*cp*mrp-2.0d0*ct*sp*cp*mtp
271  myy(isource)=st*st*sp*sp*mrr+ct*ct*sp*sp*mtt+cp*cp*mpp &
272  +2.0d0*st*ct*sp*sp*mrt+2.0d0*st*sp*cp*mrp+2.0d0*ct*sp*cp*mtp
273  mzz(isource)=ct*ct*mrr+st*st*mtt-2.0d0*st*ct*mrt
274  mxy(isource)=st*st*sp*cp*mrr+ct*ct*sp*cp*mtt-sp*cp*mpp &
275  +2.0d0*st*ct*sp*cp*mrt+st*(cp*cp-sp*sp)*mrp+ct*(cp*cp-sp*sp)*mtp
276  mxz(isource)=st*ct*cp*mrr-st*ct*cp*mtt &
277  +(ct*ct-st*st)*cp*mrt-ct*sp*mrp+st*sp*mtp
278  myz(isource)=st*ct*sp*mrr-st*ct*sp*mtt &
279  +(ct*ct-st*st)*sp*mrt+ct*cp*mrp-st*cp*mtp
280 
281  ! record three components for each station
282  do iorientation = 1,3
283 
284  ! North
285  if (iorientation == 1) then
286  stazi = 0.d0
287  stdip = 0.d0
288  ! East
289  else if (iorientation == 2) then
290  stazi = 90.d0
291  stdip = 0.d0
292  ! Vertical
293  else if (iorientation == 3) then
294  stazi = 0.d0
295  stdip = - 90.d0
296  else
297  call exit_mpi(myrank,'incorrect orientation')
298  endif
299 
300  ! get the orientation of the seismometer
301  thetan=(90.0d0+stdip)*degrees_to_radians
302  phin=stazi*degrees_to_radians
303 
304  ! we use the same convention as in Harvard normal modes for the orientation
305 
306  ! vertical component
307  n(1) = dcos(thetan)
308  ! N-S component
309  n(2) = - dsin(thetan)*dcos(phin)
310  ! E-W component
311  n(3) = dsin(thetan)*dsin(phin)
312 
313  ! get the Cartesian components of n in the model: nu
314  nu_source(iorientation,1,isource) = n(1)*st*cp+n(2)*ct*cp-n(3)*sp
315  nu_source(iorientation,2,isource) = n(1)*st*sp+n(2)*ct*sp+n(3)*cp
316  nu_source(iorientation,3,isource) = n(1)*ct-n(2)*st
317 
318  enddo
319 
320  ! normalized source radius
321  r0 = r_unit_sphere
322 
323  ! finds elevation of position
324  if (topography) then
325  call get_topo_bathy(lat(isource),long(isource),elevation,ibathy_topo)
326  r0 = r0 + elevation/r_earth
327  endif
328  if (ellipticity) then
329  dcost = dcos(theta)
330 ! this is the Legendre polynomial of degree two, P2(cos(theta)), see the discussion above eq (14.4) in Dahlen and Tromp (1998)
331  p20 = 0.5d0*(3.0d0*dcost*dcost-1.0d0)
332  radius = r0 - depth(isource)*1000.0d0/r_earth
333 ! get ellipticity using spline evaluation
334  call spline_evaluation(rspl,espl,espl2,nspl,radius,ell)
335 ! this is eq (14.4) in Dahlen and Tromp (1998)
336  r0 = r0*(1.0d0-(2.0d0/3.0d0)*ell*p20)
337  endif
338 
339  ! subtracts source depth (given in km)
340  r_target_source = r0 - depth(isource)*1000.0d0/r_earth
341 
342  ! compute the Cartesian position of the source
343  x_target_source = r_target_source*dsin(theta)*dcos(phi)
344  y_target_source = r_target_source*dsin(theta)*dsin(phi)
345  z_target_source = r_target_source*dcos(theta)
346 
347  ! set distance to huge initial value
348  distmin_squared = hugeval
349 
350  ! flag to check that we located at least one target element
351  located_target = .false.
352  ix_initial_guess_source = 0
353  iy_initial_guess_source = 0
354  iz_initial_guess_source = 0
355 
356  do ispec = 1,nspec
357 
358  ! exclude elements that are too far from target
359  if (use_distance_criterion) then
360  iglob = ibool(midx,midy,midz,ispec)
361  dist_squared = (x_target_source - dble(xstore(iglob)))**2 &
362  + (y_target_source - dble(ystore(iglob)))**2 &
363  + (z_target_source - dble(zstore(iglob)))**2
364  ! we compare squared distances instead of distances themselves to significantly speed up calculations
365  if (dist_squared > typical_size_squared) cycle
366  endif
367 
368  ! define the interval in which we look for points
369  if (use_force_point_source) then
370  ! force sources will be put on an exact GLL point
371  imin = 1
372  imax = ngllx
373 
374  jmin = 1
375  jmax = nglly
376 
377  kmin = 1
378  kmax = ngllz
379 
380  else
381  ! double-couple CMTSOLUTION
382  ! loop only on points inside the element
383  ! exclude edges to ensure this point is not shared with other elements
384  imin = 2
385  imax = ngllx - 1
386 
387  jmin = 2
388  jmax = nglly - 1
389 
390  kmin = 2
391  kmax = ngllz - 1
392  endif
393 
394  do k = kmin,kmax
395  do j = jmin,jmax
396  do i = imin,imax
397 
398  ! keep this point if it is closer to the receiver
399  iglob = ibool(i,j,k,ispec)
400  dist_squared = (x_target_source - dble(xstore(iglob)))**2 &
401  + (y_target_source - dble(ystore(iglob)))**2 &
402  + (z_target_source - dble(zstore(iglob)))**2
403  ! we compare squared distances instead of distances themselves to significantly speed up calculations
404  if (dist_squared < distmin_squared) then
405  distmin_squared = dist_squared
406  ispec_selected_source_subset(isource_in_this_subset) = ispec
407  ix_initial_guess_source = i
408  iy_initial_guess_source = j
409  iz_initial_guess_source = k
410  located_target = .true.
411  endif
412 
413  enddo
414  enddo
415  enddo
416 
417  ! calculates a Gaussian mask around source point
418  if (save_source_mask .and. simulation_type == 3) then
419  call calc_mask_source(mask_source,ispec,nspec,typical_size_squared, &
420  x_target_source,y_target_source,z_target_source, &
421  ibool,xstore,ystore,zstore,nglob)
422  endif
423 
424  ! end of loop on all the elements in current slice
425  enddo
426 
427  ! *******************************************
428  ! find the best (xi,eta,gamma) for the source
429  ! *******************************************
430 
431  ! if we have not located a target element, the source is not in this slice
432  ! therefore use first element only for fictitious iterative search
433  if (.not. located_target) then
434  ispec_selected_source_subset(isource_in_this_subset) = 1
435  ix_initial_guess_source = midx
436  iy_initial_guess_source = midy
437  iz_initial_guess_source = midz
438  endif
439 
440  ! for point sources, the location will be exactly at a GLL point
441  ! otherwise this tries to find best location
442  if (use_force_point_source) then
443  ! store xi,eta,gamma and x,y,z of point found
444  ! note: they have range [1.0d0,NGLLX/Y/Z], used for point sources
445  ! see e.g. in compute_add_sources.f90
446  xi_source_subset(isource_in_this_subset) = dble(ix_initial_guess_source)
447  eta_source_subset(isource_in_this_subset) = dble(iy_initial_guess_source)
448  gamma_source_subset(isource_in_this_subset) = dble(iz_initial_guess_source)
449 
450  iglob = ibool(ix_initial_guess_source,iy_initial_guess_source, &
451  iz_initial_guess_source,ispec_selected_source_subset(isource_in_this_subset))
452  x_found_source(isource_in_this_subset) = xstore(iglob)
453  y_found_source(isource_in_this_subset) = ystore(iglob)
454  z_found_source(isource_in_this_subset) = zstore(iglob)
455 
456  ! compute final distance between asked and found (converted to km)
457  final_distance_source_subset(isource_in_this_subset) = &
458  dsqrt((x_target_source-x_found_source(isource_in_this_subset))**2 + &
459  (y_target_source-y_found_source(isource_in_this_subset))**2 + &
460  (z_target_source-z_found_source(isource_in_this_subset))**2)*r_earth/1000.d0
461 
462  else
463 
464  ! define coordinates of the control points of the element
465  do ia = 1,ngnod
466 
467  iax = 0
468  if (iaddx(ia) == 0) then
469  iax = 1
470  else if (iaddx(ia) == 1) then
471  iax = midx
472  else if (iaddx(ia) == 2) then
473  iax = ngllx
474  else
475  call exit_mpi(myrank,'incorrect value of iaddx')
476  endif
477 
478  iay = 0
479  if (iaddy(ia) == 0) then
480  iay = 1
481  else if (iaddy(ia) == 1) then
482  iay = midy
483  else if (iaddy(ia) == 2) then
484  iay = nglly
485  else
486  call exit_mpi(myrank,'incorrect value of iaddy')
487  endif
488 
489  iaz = 0
490  if (iaddr(ia) == 0) then
491  iaz = 1
492  else if (iaddr(ia) == 1) then
493  iaz = midz
494  else if (iaddr(ia) == 2) then
495  iaz = ngllz
496  else
497  call exit_mpi(myrank,'incorrect value of iaddr')
498  endif
499 
500  iglob = ibool(iax,iay,iaz,ispec_selected_source_subset(isource_in_this_subset))
501  xelm(ia) = dble(xstore(iglob))
502  yelm(ia) = dble(ystore(iglob))
503  zelm(ia) = dble(zstore(iglob))
504 
505  enddo
506 
507  ! use initial guess in xi, eta and gamma
508  xi = xigll(ix_initial_guess_source)
509  eta = yigll(iy_initial_guess_source)
510  gamma = zigll(iz_initial_guess_source)
511 
512  ! iterate to solve the non linear system
513  do iter_loop = 1,num_iter
514 
515  ! recompute Jacobian for the new point
516  call recompute_jacobian(xelm,yelm,zelm,xi,eta,gamma,x,y,z, &
517  xix,xiy,xiz,etax,etay,etaz,gammax,gammay,gammaz)
518 
519  ! compute distance to target location
520  dx = - (x - x_target_source)
521  dy = - (y - y_target_source)
522  dz = - (z - z_target_source)
523 
524  ! compute increments
525  dxi = xix*dx + xiy*dy + xiz*dz
526  deta = etax*dx + etay*dy + etaz*dz
527  dgamma = gammax*dx + gammay*dy + gammaz*dz
528 
529  ! impose limit on increments
530  if (abs(dxi) > 0.3d0 ) dxi = sign(1.0d0,dxi)*0.3d0
531  if (abs(deta) > 0.3d0 ) deta = sign(1.0d0,deta)*0.3d0
532  if (abs(dgamma) > 0.3d0 ) dgamma = sign(1.0d0,dgamma)*0.3d0
533 
534  ! update values
535  xi = xi + dxi
536  eta = eta + deta
537  gamma = gamma + dgamma
538 
539  ! impose that we stay in that element
540  ! (useful if user gives a source outside the mesh for instance)
541  ! we can go slightly outside the [1,1] segment since with finite elements
542  ! the polynomial solution is defined everywhere
543  ! can be useful for convergence of iterative scheme with distorted elements
544  if (xi > 1.10d0) xi = 1.10d0
545  if (xi < -1.10d0) xi = -1.10d0
546  if (eta > 1.10d0) eta = 1.10d0
547  if (eta < -1.10d0) eta = -1.10d0
548  if (gamma > 1.10d0) gamma = 1.10d0
549  if (gamma < -1.10d0) gamma = -1.10d0
550 
551  enddo
552

807  subroutine calc_mask_source(mask_source,ispec,NSPEC,typical_size_squared, &
808  x_target_source,y_target_source,z_target_source, &
809  ibool,xstore,ystore,zstore,nglob)
810 
811 ! calculate a Gaussian function mask in the crust_mantle region
812 ! which is 0 around the source locations and 1 everywhere else
813 
814  use constants
815 
816  implicit none
817 
818  integer :: ispec,NSPEC,NGLOB
819 
820  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLY,NGLLZ,NSPEC) :: mask_source

827  ! local parameters
828  integer i,j,k,iglob
829  double precision dist_squared,sigma_squared

831  ! standard deviation for Gaussian
832  ! (removes factor of 100 added for search radius from typical_size_squared)

837  do j = 1,nglly

848  mask_source(i,j,k,ispec) = mask_source(i,j,k,ispec) &
849  * ( 1.0_custom_real - exp( - dist_squared / sigma_squared ) )

851  enddo
852  enddo
853  enddo
