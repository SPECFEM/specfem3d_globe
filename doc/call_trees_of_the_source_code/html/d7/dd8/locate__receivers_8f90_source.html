739  if (eta < -1.10d0) eta = -1.10d0

741  if (gamma < -1.10d0) gamma = -1.10d0

790  call gather_all_dp(gamma_receiver_subset,nrec_subset_current_size,gamma_receiver_all,nrec_subset_current_size,nproctot_val)

1019 !
1020 ! Implementation of a Heap Sort Routine
1021 ! Input
1022 ! n = Input
1023 ! Length of arrays
1024 ! X_in = Input
1025 ! Vector to be sorted
1026 ! dimension(n)
1027 ! Y = Output
1028 ! Sorted Indices of vector X
1029 !
1030 ! Example:
1031 ! D = [ 4.0 3.0 1.0 2.0 ] on Input
1032 ! Y = [ 1 2 3 4 ] Computed Internally (in order)
1033 !
1034 ! X = [ 1.0 2.0 3.0 4.0 ] Computed Internally
1035 ! Y = [ 3 4 2 1 ] on Output
1036 !
1037  subroutine heap_sort_distances(N, X_in, Y)
1038 
1039  implicit none
1040  integer, intent(in) :: N
1041  double precision, dimension(N), intent(in) :: X_in
1042  integer, dimension(N), intent(out) :: Y
1043 
1044  ! local parameters
1045  double precision, dimension(N) :: X
1046  double precision :: tmp
1047  integer :: itmp
1048  integer :: i
1049 
1050  do i = 1,n
1051  y(i) = i
1052  x(i) = x_in(i)
