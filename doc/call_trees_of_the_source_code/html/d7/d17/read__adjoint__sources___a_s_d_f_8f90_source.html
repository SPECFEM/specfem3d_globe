28 subroutine read_adjoint_sources_asdf(adj_source_name, adj_source, index_start, index_end)

31 
32  use iso_c_binding, only: c_null_char
33 
34  implicit none
35 
36  integer :: itime, offset, nsamples
37  integer :: index_start, index_end
38  real(kind=CUSTOM_REAL), dimension(*),intent(out) :: adj_source ! NSTEP block size
39  character(len=*) :: adj_source_name
40  !--- Error variable
41  integer ier
42 
43  ! Fortran/C index convension. Meaning Fortran starts at 1 and C starts at 0 so
44  ! we need to subtract 1 for the C subroutine read_partial_waveform_f
45  offset = index_start - 1 ! the value to start reading from
46  nsamples = index_end - index_start + 1 ! this is how many points we want to read in from the adjoint source
47 
48  ! print *, myrank, " myrank ", trim(adj_source_name)
49  ! print *, " offset, ", offset
50  ! print *, " nsamples ", nsamples
51  ! print *, adj_source_name, " reading"
52 
53  call asdf_read_partial_waveform_f(current_asdf_handle, "AuxiliaryData/AdjointSources/"//&
54  trim(adj_source_name) // c_null_char, offset, nsamples, adj_source, ier)
55 
56  if (ier /= 0) then
57  print *,'Error reading adjoint source: ',trim(adj_source_name)
58  print *,'rank ',myrank,' - time step: ',itime,' index_start:',index_start,' index_end: ',index_end
59  print *,' ',trim(adj_source_name)//'has wrong length, please check with your simulation duration'
60  call exit_mpi(myrank,'Adjoint source '//trim(adj_source_name)//' has wrong length, please check with your simulation duration')
61  endif
62 
63 end subroutine read_adjoint_sources_asdf

100

103  ! print *, trim(adj_filename), adjoint_source_exists
104 
105  if (adjoint_source_exists == 0) then
106  !adjoint source not found
107  !stops simulation
108  call exit_mpi(myrank,'adjoint source '//trim(adj_filename)//' not found, please check STATIONS_ADJOINT file and ASDF file')
109  endif
110 
111  ! checks length of file
112  call asdf_get_num_elements_from_path_f(current_asdf_handle, &
113  "AuxiliaryData/AdjointSources/" // trim(adj_filename) // c_null_char, nsamples_infered, ier)
114 
115  ! print *, trim(adj_filename), nsamples_infered
116 
117  ! checks length
118  if (nsamples_infered /= nstep) then
119  print *,'adjoint source error: ',trim(adj_filename),' has length',nsamples_infered,' but should be',nstep
120  call exit_mpi(myrank, &
121  'file '//trim(adj_filename)//' length is wrong, please check your adjoint sources and your simulation duration')
122  endif
123 
124  ! updates counter for found files
125  nadj_sources_found = nadj_sources_found + 1
126 
127  enddo

129 end subroutine check_adjoint_sources_asdf
