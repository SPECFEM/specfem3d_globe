393 subroutine define_adios_global_1d_real_generic(adios_group, group_size_inc, array_name, local_dim)
394 
395  use adios_write_mod, only: adios_define_var
396 
397  implicit none
398  ! Parameters
399  integer(kind=8), intent(in) :: adios_group
400  character(len=*), intent(in) :: array_name
401  integer, intent(in) :: local_dim
402  integer(kind=8), intent(inout) :: group_size_inc
403  ! Variables
404  integer(kind=8) :: var_id
405 
406  ! Define the dimensions of the array. local_dim used as a dummy
407  ! variable to call the integer routine.
408  call define_adios_global_dims_1d(adios_group, group_size_inc, trim(array_name),local_dim)
409 
410  call adios_define_var(adios_group, "array", trim(array_name), 5, &
411  trim(array_name) // "/local_dim", &
412  trim(array_name) // "/global_dim", &
413  trim(array_name) // "/offset", var_id)
414 
415  group_size_inc = group_size_inc + local_dim*4
