97  subroutine find_regular_grid_slice_number(slice_number, GRID)

99  use constants, only: custom_real,pi,degrees_to_radians, &
100  nm_kl_reg_layer,kl_reg_min_lat,kl_reg_min_lon
101 
102  use specfem_par, only: myrank, addressing, &
103  nchunks_val, nproc_xi_val, nproc_eta_val
104 
105  implicit none
106

108 
109  type kl_reg_grid_variables
110  sequence
111  real dlat
112  real dlon
113  integer nlayer
114  real rlayer(nm_kl_reg_layer)
115  integer ndoubling(nm_kl_reg_layer)
116  integer nlat(nm_kl_reg_layer)
117  integer nlon(nm_kl_reg_layer)
118  integer npts_total
119  integer npts_before_layer(nm_kl_reg_layer+1)
120  end type kl_reg_grid_variables
121  type(kl_reg_grid_variables), intent(in) :: grid
122 
123  real(kind=CUSTOM_REAL) :: xi_width, eta_width
124  integer :: nproc, ilayer, isp, ilat, ilon, k, chunk_isp
125  integer :: iproc_xi, iproc_eta
126  real :: lat,lon,th,ph,x,y,z,xik,etak,xi_isp,eta_isp,xi1,eta1
127 
128  ! assuming 6 chunks full global simulations right now
129  if (nchunks_val /= 6 .or. nproc_xi_val /= nproc_eta_val) then
130  call exit_mpi(myrank, 'Only deal with 6 chunks at this moment')
131  endif
132 
133  xi_width=pi/2; eta_width=pi/2; nproc=nproc_xi_val
134  ilayer = 0
135 
136  do isp = 1,grid%npts_total
137  if (isp == grid%npts_before_layer(ilayer+1)+1) ilayer=ilayer+1
138  ilat = (isp - grid%npts_before_layer(ilayer) - 1) / grid%nlon(ilayer)
139  ilon = (isp - grid%npts_before_layer(ilayer)) - ilat * grid%nlon(ilayer)
140 
141  ! (lat,lon,radius) for isp point
142  lat = kl_reg_min_lat + ilat * grid%dlat * grid%ndoubling(ilayer)
143  th = (90 - lat) * degrees_to_radians
144  lon = kl_reg_min_lon + (ilon - 1) * grid%dlon * grid%ndoubling(ilayer)
145  ph = lon * degrees_to_radians
146  x = sin(th) * cos(ph); y = sin(th) * sin(ph); z = cos(th)
147 
148  ! figure out slice number
149  chunk_isp = 1; xi_isp = 0; eta_isp = 0
150  do k = 1, nchunks_val
151  call chunk_map(k, x, y, z, xik, etak)
152  if (abs(xik) <= pi/4 .and. abs(etak) <= pi/4) then
153  chunk_isp = k; xi_isp = xik; eta_isp = etak; exit
154  endif
155  enddo
156  xi1 = xi_isp / xi_width * 2; eta1 = eta_isp / eta_width * 2
157  iproc_xi = floor((xi1+1)/2 * nproc)
158  iproc_eta = floor((eta1+1)/2 * nproc)
159  slice_number(isp) = addressing(chunk_isp, iproc_xi, iproc_eta)
160  enddo
161 
162  end subroutine find_regular_grid_slice_number

204  double precision, dimension(NGLLX), intent(in) :: xigll

239  typical_size_squared = two_pi * r_unit_sphere / (4.*nex_xi)
240 
241  ! use 10 times the distance as a criterion for source detection
242  typical_size_squared = (10. * typical_size_squared)**2
243 
244  do ipoint = 1, npoints_slice
245  isp = points_slice(ipoint)
246  do ilayer = 1, grid%nlayer
247  if (isp <= grid%npts_before_layer(ilayer+1)) exit
248  enddo
249 
250  ilat = (isp - grid%npts_before_layer(ilayer) - 1) / grid%nlon(ilayer)
251  ilon = (isp - grid%npts_before_layer(ilayer)) - ilat * grid%nlon(ilayer)
252 
253  ! (lat,lon,radius) for isp point
254  lat = kl_reg_min_lat + ilat * grid%dlat * grid%ndoubling(ilayer)
255  lon = kl_reg_min_lon + (ilon - 1) * grid%dlon * grid%ndoubling(ilayer)
256  ! convert radius to meters and then scale
257  radius = grid%rlayer(ilayer) * 1000.0 / r_earth
258  ! (x,y,z) for isp point
259  th = (90 - lat) * degrees_to_radians; ph = lon * degrees_to_radians
260  x_target = radius * sin(th) * cos(ph)
261  y_target = radius * sin(th) * sin(ph)
262  z_target = radius * cos(th)
263 
264  ! first exclude elements too far away
265  locate_target = .false.
266  distmin_squared = hugeval
267 
268  do ispec = 1,nspec

271  + (y_target - ystore(iglob))**2 &
272  + (z_target - zstore(iglob))**2
273  ! we compare squared distances instead of distances themselves to significantly speed up calculations
274  if (dist_squared > typical_size_squared) cycle
275 
276  locate_target = .true.
277  ! loop only on points inside the element
278  ! exclude edges to ensure this point is not
279  ! shared with other elements
280  ! can be improved if we have a better algorithm of determining if a point

282 
283  do k = 2, ngllz-1
284  do j = 2, nglly-1
285  do i = 2, ngllx-1
286  iglob = ibool(i,j,k,ispec)
287  dist_squared = (x_target - xstore(iglob))**2 &
288  + (y_target - ystore(iglob))**2 &
289  + (z_target - zstore(iglob))**2
290  if (dist_squared < distmin_squared) then
291  ix_in = i
292  iy_in = j
293  iz_in = k
294  ispec_in = ispec
295  distmin_squared = dist_squared
296  endif
297  enddo
298  enddo
299  enddo
300  enddo
301 
302  if (.not. locate_target) then
303  print *, 'Looking for point', isp, ilayer, ilat, ilon, lat, lon, x_target, y_target, z_target, myrank
304  call exit_mpi(myrank, 'Error in point_source() array')
305  endif
306 
307  xi = xigll(ix_in)
308  eta = yigll(iy_in)
309  gamma = zigll(iz_in)
310  ispec_reg(ipoint) = ispec_in
311 
312  ! anchors
313  do ia = 1, ngnod
314  iglob = ibool(iaddx(ia), iaddy(ia), iaddr(ia), ispec_in)
315  xelm(ia) = dble(xstore(iglob))
316  yelm(ia) = dble(ystore(iglob))
317  zelm(ia) = dble(zstore(iglob))
318  enddo
319 
320  ! iterate to solve the nonlinear system
321  do iter_loop = 1,num_iter
322 
323  ! recompute Jacobian for the new point
324  call recompute_jacobian(xelm,yelm,zelm, xi,eta,gamma, x,y,z, &
325  xix,xiy,xiz, etax,etay,etaz, gammax,gammay,gammaz)
326 
327  ! compute distance to target location
328  dx = - (x - x_target)
329  dy = - (y - y_target)
330  dz = - (z - z_target)
331 
332  ! compute increments
333  dxi = xix*dx + xiy*dy + xiz*dz
334  deta = etax*dx + etay*dy + etaz*dz
335  dgamma = gammax*dx + gammay*dy + gammaz*dz
336 
337  ! update values
338  xi = xi + dxi
339  eta = eta + deta
340  gamma = gamma + dgamma
341 
342  ! impose that we stay in that element
343  ! (useful if user gives a source outside the mesh for instance)
344  if (xi > 1.d0) xi = 1.d0
345  if (xi < -1.d0) xi = -1.d0
346  if (eta > 1.d0) eta = 1.d0
347  if (eta < -1.d0) eta = -1.d0
348  if (gamma > 1.d0) gamma = 1.d0
349  if (gamma < -1.d0) gamma = -1.d0
350 
351  enddo
352 
353  ! store l(xi),l(eta),l(gamma)
354  call lagrange_any2(xi, ngllx, xigll, hxir)
355  call lagrange_any2(eta, nglly, yigll, hetar)
356  call lagrange_any2(gamma, ngllz, zigll, hgammar)
357  hxir_reg(:,ipoint) = hxir
358  hetar_reg(:,ipoint) = hetar
359  hgammar_reg(:,ipoint) = hgammar
360 
361  enddo ! ipoint
362 
363  end subroutine locate_regular_points

418  subroutine lagrange_any2(xi,NGLL,xigll,h)
419 
420 ! subroutine to compute the Lagrange interpolants based upon the GLL points
421 ! and their first derivatives at any point xi in [-1,1]
422 
423  implicit none
424 
425  double precision, intent(in) :: xi
426  integer, intent(in) :: NGLL
427  double precision, dimension(NGLL), intent(in) :: xigll
428  double precision, dimension(NGLL), intent(out) :: h
429 
430  integer :: dgr,i
431  double precision :: prod1,prod2
432 
433  do dgr = 1,ngll
434  prod1 = 1.0d0
435  prod2 = 1.0d0
436 
437  do i = 1,ngll
438  if (i /= dgr) then
439  prod1 = prod1 * (xi - xigll(i))
440  prod2 = prod2 * (xigll(dgr) - xigll(i))
441  endif
442  enddo
443 
444  h(dgr) = prod1 / prod2
445  enddo
446 
447  end subroutine lagrange_any2

451  subroutine chunk_map(k,xx,yy,zz,xi,eta)
452 
453  ! this program get the xi,eta for (xx,yy,zz)
454  ! point under the k'th chunk coordinate
455  ! transformation
456 
457  use constants
458 
459  implicit none
460 
461  integer, intent(in) :: k
462  real, intent(in) :: xx, yy, zz
463  real, intent(out) :: xi, eta
464 
465  real :: x, y, z
466  real, parameter :: EPS=1e-6
467 
468  x = xx; y = yy; z = zz
469  if (0 <= x .and. x < eps) x = eps
470  if (-eps < x .and. x < 0) x = -eps
471  if (0 <= y .and. y < eps) y = eps
472  if (-eps < y .and. y < 0) y = -eps
473  if (0 <= z .and. z < eps) z = eps
474  if (-eps < z .and. z < 0) z = -eps
475 
476  if (k == chunk_ab) then
477  xi = atan(y/z); eta = atan(-x/z)
478  if (z < 0) xi = 10
479  else if (k == chunk_ac) then
480  xi = atan(-z/y); eta = atan(x/y)
481  if (y > 0) xi = 10
482  else if (k == chunk_bc) then
483  xi = atan(-z/x); eta = atan(-y/x)
484  if (x > 0) xi = 10
485  else if (k == chunk_ac_antipode) then
486  xi = atan(-z/y); eta = atan(-x/y)
487  if (y < 0) xi = 10
488  else if (k == chunk_bc_antipode) then
489  xi = atan(z/x); eta = atan(-y/x)
490  if (x < 0) xi = 10
491  else if (k == chunk_ab_antipode) then
492  xi = atan(y/z); eta = atan(x/z)
493  if (z > 0) xi = 10
494  else
495  stop 'chunk number k < 6'
496  endif
497 
498  xi = eps * nint(xi / eps)
499  eta = eps * nint(eta / eps)
500  end subroutine chunk_map
