62  if (.not. exist) then

65  endif
66  ! makes sure all processes have same flag
67  call any_all_l(exist,exist_all)
68  if (.not. exist_all) then
69  print *,'old kernels do not exist: ',trim(m_file)
70  call exit_mpi(myrank,'flags old model not consistent')
71  endif

81  call exit_mpi(myrank,'file not found')
82  endif

84  close(iin)
85
