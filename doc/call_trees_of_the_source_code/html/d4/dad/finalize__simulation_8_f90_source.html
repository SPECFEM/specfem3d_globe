28  subroutine finalize_simulation()
29 
30  use specfem_par

35

37 
38 #ifdef XSMM
39  use my_libxsmm
40 #endif
41 
42  implicit none
43 
44  ! synchronize all processes, waits until all processes have written their seismograms
45  call synchronize_all()
46 
47  ! user output
48  if (myrank == 0) then
49  write(imain,*) 'finalizing simulation'
50  call flush_imain()
51  endif
52  call synchronize_all()
53 
54  ! closes Stacey absorbing boundary snapshots
55  if (absorbing_conditions) then
56  ! crust mantle
57  if (nspec2d_xmin_crust_mantle > 0 .and. save_stacey) then
58  call close_file_abs(0)
59  endif
60 
61  if (nspec2d_xmax_crust_mantle > 0 .and. save_stacey) then
62  call close_file_abs(1)
63  endif
64 
65  if (nspec2d_ymin_crust_mantle > 0 .and. save_stacey) then
66  call close_file_abs(2)
67  endif
68 
69  if (nspec2d_ymax_crust_mantle > 0 .and. save_stacey) then
70  call close_file_abs(3)
71  endif
72 
73  ! outer core
74  if (nspec2d_xmin_outer_core > 0 .and. save_stacey) then
75  call close_file_abs(4)
76  endif
77 
78  if (nspec2d_xmax_outer_core > 0 .and. save_stacey) then
79  call close_file_abs(5)
80  endif
81 
82  if (nspec2d_ymin_outer_core > 0 .and. save_stacey) then
83  call close_file_abs(6)
84  endif
85 
86  if (nspec2d_ymax_outer_core > 0 .and. save_stacey) then
87  call close_file_abs(7)
88  endif
89 
90  if (nspec2d_zmin_outer_core > 0 .and. save_stacey) then
91  call close_file_abs(8)
92  endif
93 
94  ! frees memory
95  deallocate(absorb_xmin_crust_mantle, &
96  absorb_xmax_crust_mantle, &
97  absorb_ymin_crust_mantle, &
98  absorb_ymax_crust_mantle, &
99  absorb_xmin_outer_core, &
100  absorb_xmax_outer_core, &
101  absorb_ymin_outer_core, &
102  absorb_ymax_outer_core, &
103  absorb_zmin_outer_core)
104  endif
105 
106  ! save/read the surface movie using the same c routine as we do for absorbing boundaries (file ID is 9)
107  if (noise_tomography /= 0) then
108  call close_file_abs(9)
109  endif
110 
111  ! save files to local disk or tape system if restart file
112  call save_forward_arrays()
113 
114  ! dump kernel arrays
115  call save_kernels()
116 
117  ! VTK visualization
118  if (vtk_mode) then
119  ! closes VTK window
120  if (myrank == 0 ) call finish_vtkwindow()
121  endif
122 
123  ! adios finalizes
124  if (adios_enabled) then
125  call finalize_adios()
126  endif
127 
128  ! asdf finalizes
129  if (simulation_type == 2 .or. simulation_type == 3 &
130  .and. read_adjsrc_asdf) then
131  call asdf_cleanup()
132  endif
133 
134 #ifdef XSMM
135  ! finalizes LIBXSMM
136  call libxsmm_finalize()
137 #endif
138 
139  ! frees dynamically allocated memory
140  call finalize_simulation_cleanup()
141 
142  ! closes the main output file
143  if (myrank == 0) then
144  write(imain,*)
145  write(imain,*) 'End of the simulation'
146  write(imain,*)
147  call flush_imain()
148  close(imain)
149  endif
150 
151  ! synchronizes all the processes to make sure everybody has finished
152  call synchronize_all()
153 
154  end subroutine finalize_simulation
