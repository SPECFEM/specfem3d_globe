28  subroutine count_number_of_sources(NSOURCES)
29 
30 ! count the total number of sources in the CMTSOLUTION file
31 ! there are NLINES_PER_CMTSOLUTION_SOURCE lines per source in that file
32 
33  use constants
34  use shared_parameters, only: number_of_simultaneous_runs
35 
36  implicit none
37 
38  integer, intent(out) :: NSOURCES
39 
40  integer :: ios,icounter
41 
42  character(len=MAX_STRING_LEN) :: CMTSOLUTION_FILE, path_to_add
43  character(len=MAX_STRING_LEN) :: dummystring
44 
45  cmtsolution_file = 'DATA/CMTSOLUTION'
46 
47  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
48  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
49  cmtsolution_file = path_to_add(1:len_trim(path_to_add))//cmtsolution_file(1:len_trim(cmtsolution_file))
50  endif
51 
52  open(unit=iin,file=trim(cmtsolution_file),status='old',action='read',iostat=ios)
53  if (ios /= 0) then
54  print *,'Error opening CMTSOLUTION file: ',trim(cmtsolution_file)
55  stop 'Error opening CMTSOLUTION file'
56  endif
57 
58  icounter = 0
59  do while(ios == 0)
60  read(iin,"(a)",iostat=ios) dummystring
61  if (ios == 0) icounter = icounter + 1
62  enddo
63  close(iin)
64 
65  if (mod(icounter,nlines_per_cmtsolution_source) /= 0) &
66  stop 'total number of lines in CMTSOLUTION file should be a multiple of NLINES_PER_CMTSOLUTION_SOURCE'
67 
68  nsources = icounter / nlines_per_cmtsolution_source
69 
70  if (nsources < 1) stop 'need at least one source in CMTSOLUTION file'
71 
72  end subroutine count_number_of_sources
