286  subroutine get_model_check_idoubling(r_prem,x,y,z,rmin,rmax,idoubling, &
287  ricb,rcmb,rtopddoubleprime, &
288  r220,r670)
289 
290  use meshfem3d_models_par
291 
292  implicit none
293 
294  integer idoubling
295 
296  double precision r_prem,rmin,rmax,x,y,z
297 
298  double precision RICB,RCMB,RTOPDDOUBLEPRIME,R670,R220
299  double precision r_m,r,theta,phi
300 
301  ! compute real physical radius in meters
302  r_m = r_prem * r_earth
303 
304  ! checks layers
305  if (abs(rmax - rmin ) < tinyval) then
306  ! there's probably an error
307  print *,'Error layer radius min/max:',rmin,rmax
308  print *,' point radius: ',r_prem
309  call exit_mpi(myrank,'Error in get_model_check_idoubling() layer radius')
310  endif
311 
312 
313  ! check flags to make sure we correctly honor the discontinuities
314  ! we use strict inequalities since r has been slightly changed in mesher
315 
316  !
317  !--- inner core
318  !
319  if (r_m >= 0.d0 .and. r_m < ricb) then
320  if (idoubling /= iflag_inner_core_normal .and. &
321  idoubling /= iflag_middle_central_cube .and. &
322  idoubling /= iflag_bottom_central_cube .and. &
323  idoubling /= iflag_top_central_cube .and. &
324  idoubling /= iflag_in_fictitious_cube) then
325  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
326  print *,'Error point r/lat/lon:',r_m,90.0 - theta/degrees_to_radians,phi/degrees_to_radians
327  print *,' idoubling/IFLAG: ',idoubling,iflag_inner_core_normal,'-to-',iflag_in_fictitious_cube
328  call exit_mpi(myrank,'Error in get_model_check_idoubling() wrong doubling flag for inner core point')
329  endif
330  !
331  !--- outer core
332  !
333  else if (r_m > ricb .and. r_m < rcmb) then
334  if (idoubling /= iflag_outer_core_normal) then
335  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
336  print *,'Error point r/lat/lon:',r_m,90.0 - theta/degrees_to_radians,phi/degrees_to_radians
337  print *,' idoubling/IFLAG: ',idoubling,iflag_outer_core_normal
338  call exit_mpi(myrank,'Error in get_model_check_idoubling() wrong doubling flag for outer core point')
339  endif
340  !
341  !--- D" at the base of the mantle
342  !
343  else if (r_m > rcmb .and. r_m < rtopddoubleprime) then
344  if (idoubling /= iflag_mantle_normal) then
345  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
346  print *,'Error point r/lat/lon:',r_m,90.0 - theta/degrees_to_radians,phi/degrees_to_radians
347  print *,' dprime radius/RCMB/RTOPDDOUBLEPRIME:',r_m, rcmb,rtopddoubleprime
348  print *,' idoubling/IFLAG: ',idoubling,iflag_mantle_normal
349  call exit_mpi(myrank,'Error in get_model_check_idoubling() wrong doubling flag for D" point')
350  endif
351  !
352  !--- mantle: from top of D" to d670
353  !
354  else if (r_m > rtopddoubleprime .and. r_m < r670) then
355  if (idoubling /= iflag_mantle_normal) then
356  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
357  print *,'Error point r/lat/lon:',r_m,90.0 - theta/degrees_to_radians,phi/degrees_to_radians
358  print *,' idoubling/IFLAG: ',idoubling,iflag_mantle_normal
359  call exit_mpi(myrank,'Error in get_model_check_idoubling() wrong doubling flag for top D" to d670 point')
360  endif
361 
362  !
363  !--- mantle: from d670 to d220
364  !
365  else if (r_m > r670 .and. r_m < r220) then
366  if (idoubling /= iflag_670_220) then
367  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
368  print *,'Error point r/lat/lon:',r_m,90.0 - theta/degrees_to_radians,phi/degrees_to_radians
369  print *,' idoubling/IFLAG: ',idoubling,iflag_670_220
370  call exit_mpi(myrank,'Error in get_model_check_idoubling() wrong doubling flag for d670 to d220 point')
371  endif
372 
373  !
374  !--- mantle and crust: from d220 to MOHO and then to surface
375  !
376  else if (r_m > r220) then
377  if (idoubling /= iflag_220_80 .and. idoubling /= iflag_80_moho .and. idoubling /= iflag_crust) then
378  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
379  print *,'Error point r/lat/lon:',r_m,90.0 - theta/degrees_to_radians,phi/degrees_to_radians
380  print *,' idoubling/IFLAG: ',idoubling,iflag_220_80,iflag_80_moho,iflag_crust
381  call exit_mpi(myrank,'Error in get_model_check_idoubling() wrong doubling flag for d220 to Moho to surface point')
382  endif
383 
384  endif
385 
386  end subroutine get_model_check_idoubling
