286  subroutine setup_timesteps()
287 
288  use specfem_par
289  implicit none
290 
291  ! local parameters
292  logical :: is_initial_guess
293 
294  ! checks if set by initial guess from read_compute_parameters() routine
295  if (ntstep_between_output_seismos == nstep) then
296  is_initial_guess = .true.
297  else
298  is_initial_guess = .false.
299  endif
300 
301  ! from initial guess in read_compute_parameters:
302  ! compute total number of time steps, rounded to next multiple of 100
303  ! NSTEP = 100 * (int(RECORD_LENGTH_IN_MINUTES * 60.d0 / (100.d0*DT)) + 1)
304  !
305  ! adds initial t0 time to update number of time steps and reach full record length
306  if (abs(t0) > 0.d0) then
307  ! note: for zero length, nstep has minimal of 5 timesteps for testing
308  ! we won't extend this
309  !
310  ! careful: do not use RECORD_LENGTH_IN_MINUTES here, as it is only read by the master process
311  ! when reading the parameter file, but it is not broadcasted to all other processes
312  ! NSTEP gets broadcasted, so we work with this values
313  if (nstep /= 5) then
314  ! extend by bulk of 100 steps to account for half-duration rise time
315  nstep = nstep + 100 * (int( abs(t0) / (100.d0*dt)) + 1)
316  endif
317  endif
318 
319  ! if doing benchmark runs to measure scaling of the code for a limited number of time steps only
320  if (do_benchmark_run_only) nstep = nstep_for_benchmark
321 
322  ! checks length for symmetry in case of noise simulations
323  if (noise_tomography /= 0) then
324  if (mod(nstep+1,2) /= 0) then
325  print *,'Error noise simulation: invalid time steps = ',nstep,', NSTEP + 1 must be a multiple of 2 due to branch symmetry'
326  call exit_mpi(myrank,'Error noise simulation: number of timesteps must be symmetric, due to +/- branches')
327  endif
328  endif
329 
330  ! time loop increments end
331  it_end = nstep
332 
333  ! subsets used to save seismograms must not be larger than the whole time series,
334  ! otherwise we waste memory
335  if (ntstep_between_output_seismos > nstep .or. is_initial_guess) ntstep_between_output_seismos = nstep
336 
337  ! subsets used to save adjoint sources must not be larger than the whole time series,
338  ! otherwise we waste memory
339  if (simulation_type == 2 .or. simulation_type == 3) then
340  if (ntstep_between_read_adjsrc > nstep) ntstep_between_read_adjsrc = nstep
341  endif
342 
343  ! buffering with undo_attenuation
344  nt_dump_attenuation = nt_dump_attenuation_val
345  if (undo_attenuation) then
346  ! makes sure buffer size is not too big for total time length
347  !
348  ! note: NSTEP must not be a multiple of NT_DUMP_ATTENUATION.
349  ! the value from the header file NT_DUMP_ATTENUATION_VAL gives the optimal (maximum) number of time steps for buffering
350  if (nstep < nt_dump_attenuation) nt_dump_attenuation = nstep
351  endif
352 
353  ! debug
354  !if (myrank == 0 ) print *,'setup time steps = ',NSTEP,' t0 = ',t0,' DT = ',DT
355 
356  end subroutine setup_timesteps

869  subroutine setup_sources_receivers_adjindx(NSTEP,NSTEP_SUB_ADJ, &
870  ntstep_between_read_adjsrc, &
871  iadjsrc,iadjsrc_len,iadj_vec)
872 
873  use constants
874 
875  implicit none
876 
877  integer NSTEP,NSTEP_SUB_ADJ,NTSTEP_BETWEEN_READ_ADJSRC
878 
879  integer, dimension(NSTEP_SUB_ADJ,2) :: iadjsrc ! to read input in chunks
880  integer, dimension(NSTEP_SUB_ADJ) :: iadjsrc_len
881  integer, dimension(NSTEP) :: iadj_vec
882 
883  ! local parameters
884  integer :: iadj_block,it,it_sub_adj
885  integer :: istart,iend
886 
887  ! initializes
888  iadjsrc(:,:) = 0
889  iadjsrc_len(:) = 0
890 
891  ! setting up chunks of NTSTEP_BETWEEN_READ_ADJSRC to read adjoint source traces
892  ! i.e. as an example: total length NSTEP = 3000, chunk length NTSTEP_BETWEEN_READ_ADJSRC= 1000
893  ! then it will set first block from 2001 to 3000,
894  ! second block from 1001 to 2000 and so on...
895  !
896  ! see routine: compute_arrays_source_adjoint()
897  ! how we read in the adjoint source trace in blocks/chunk sizes
898  !
899  ! see routine: compute_add_sources_adjoint()
900  ! how the adjoint source is added to the (adjoint) acceleration field
901  !counts blocks
902  ! block number
903  ! e.g. increases from 1 (case it=1-1000), 2 (case it=1001-2000) to 3 (case it=2001-3000)
904  it_sub_adj = 0
905  iadj_block = 1
906  do it = 1,nstep
907  ! we are at the edge of a block
908  if (mod(it-1,ntstep_between_read_adjsrc) == 0) then
909  ! sets it_sub_adj subset number
910  it_sub_adj = iadj_block
911 
912  ! block start time ( e.g. 2001)
913  istart = nstep-it_sub_adj*ntstep_between_read_adjsrc+1
914  ! final adj src array
915  ! e.g. will be from 1000 to 1, but doesn't go below 1 in cases where NSTEP isn't
916  ! a multiple of NTSTEP_BETWEEN_READ_ADJSRC
917  if (istart < 1 ) istart = 1
918 
919  ! block end time (e.g. 3000)
920  iend = nstep-(it_sub_adj-1)*ntstep_between_read_adjsrc
921 
922  iadjsrc(iadj_block,1) = istart
923  iadjsrc(iadj_block,2) = iend
924 
925  ! actual block length
926  iadjsrc_len(iadj_block) = iend - istart + 1
927 
928  ! increases block number
929  iadj_block = iadj_block + 1
930  endif
931 
932  ! time stepping for adjoint sources:
933  ! adjoint time step that corresponds to time step in simulation (it).
934  ! note, that adjoint source has to be time-reversed with respect to the forward wavefield
935  ! e.g.: first block 1 has iadjsrc_len = 1000 with start at 2001 and end at 3000
936  ! so iadj_vec(1) = 1000 - 0, iadj_vec(2) = 1000 - 1, ..., to iadj_vec(1000) = 1000 - 999 = 1
937  ! then for block 2, iadjsrc_len = 1000 with start at 1001 and end at 2000
938  ! so iadj_vec(1001) = 1000 - 0, iadj_vec(1002) = 1000 - 1, .. and so on again down to 1
939  ! then block 3 and your guess is right now... iadj_vec(2001) to iadj_vec(3000) is 1000 down to 1. :)
940  iadj_vec(it) = iadjsrc_len(it_sub_adj) - mod(it-1,ntstep_between_read_adjsrc)
941 
942  ! checks that index is non-negative
943  if (iadj_vec(it) < 1 ) iadj_vec(it) = 1
944  enddo
945 
946  end subroutine setup_sources_receivers_adjindx
