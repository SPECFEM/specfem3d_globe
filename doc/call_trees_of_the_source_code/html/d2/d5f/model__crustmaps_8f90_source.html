29 ! General Crustmaps
30 !
31 ! combines Crust2.0 and EUcrust07 for moho depths; the crustal maps
32 ! interpolate the crustal velocities from Crust2.0 onto the more detailed EUcrust
33 ! crustal depths where ever they are defined.
34 
35 ! current crustmaps (cmaps) take sediment thickness
36 ! and moho depths from EUcrust07 if possible and interpolate corresponding
37 ! velocity/densities given from Crust2.0.
38 !
39 ! main author: Matthias Meschede (meschede AT princeton DOT edu)
40 !--------------------------------------------------------------------------------------------------

123  subroutine read_general_crustmap()
124 
125  use constants
126  use model_crustmaps_par
127 
128  implicit none
129 
130  ! local parameters
131  integer :: i,k,l
132  double precision :: moho,moho_min,moho_max
133 
134  ! user output
135  write(imain,*)
136  write(imain,*) 'incorporating crustal model: crustMap'
137  write(imain,*)
138  call flush_imain()
139 
140  do l = 1,nlayers_crustmap
141  ! file index: from 3 to 7
142  i = l + 2
143 
144  call read_general_crustmap_layer(thickness(:,:,l), 't', i)
145  call read_general_crustmap_layer(density(:,:,l), 'r', i)
146  call read_general_crustmap_layer(velocp(:,:,l), 'p', i)
147  call read_general_crustmap_layer(velocs(:,:,l), 's', i)
148  enddo
149 
150  ! North pole / South pole
151  thicknessnp(:) = zero
152  thicknesssp(:) = zero
153  densitynp(:) = zero
154  densitysp(:) = zero
155  velocpnp(:) = zero
156  velocpsp(:) = zero
157  velocsnp(:) = zero
158  velocssp(:) = zero
159 
160  ! compute average values for North and South pole
161  do l = 1,nlayers_crustmap
162  do i=1,360*crustmap_resolution
163  thicknessnp(l) = thicknessnp(l)+thickness(1,i,l)
164  thicknesssp(l) = thicknesssp(l)+thickness(180*crustmap_resolution,i,l)
165  densitynp(l) = densitynp(l)+density(1,i,l)
166  densitysp(l) = densitysp(l)+density(180*crustmap_resolution,i,l)
167  velocpnp(l) = velocpnp(l)+velocp(1,i,l)
168  velocpsp(l) = velocpsp(l)+velocp(180*crustmap_resolution,i,l)
169  velocsnp(l) = velocsnp(l)+velocs(1,i,l)
170  velocssp(l) = velocssp(l)+velocs(180*crustmap_resolution,i,l)
171  enddo
172  thicknessnp(l) = thicknessnp(l)/360.0/dble(crustmap_resolution)
173  thicknesssp(l) = thicknesssp(l)/360.0/dble(crustmap_resolution)
174  densitynp(l) = densitynp(l)/360.0/dble(crustmap_resolution)
175  densitysp(l) = densitysp(l)/360.0/dble(crustmap_resolution)
176  velocpnp(l) = velocpnp(l)/360.0/dble(crustmap_resolution)
177  velocpsp(l) = velocpsp(l)/360.0/dble(crustmap_resolution)
178  velocsnp(l) = velocsnp(l)/360.0/dble(crustmap_resolution)
179  velocssp(l) = velocssp(l)/360.0/dble(crustmap_resolution)
180 
181 ! print *,'thicknessnp(',l,')',thicknessnp(l)
182  enddo
183 
184  ! moho thickness = sediment (index 1+2) + upper crust (index 3) + middle crust (index 4) + lower crust (index 5)
185  moho_min = hugeval
186  moho_max = -hugeval
187  do k = 1,180*crustmap_resolution
188  do i = 1,360*crustmap_resolution
189  moho = sum(thickness(k,i,:)) !thickness(:,:,1) + thickness(:,:,2)+ thickness(:,:,3) + thickness(:,:,4) + thickness(:,:,5)
190  ! limit moho thickness
191  if (moho > moho_max) moho_max = moho
192  if (moho < moho_min) moho_min = moho
193  enddo
194  enddo
195 
196  ! user output
197  write(imain,*) ' Moho crustal thickness (without ice) min/max = ',sngl(moho_min),sngl(moho_max),' km'
198  write(imain,*)
199  call flush_imain()
200 
201  end subroutine read_general_crustmap

244  subroutine model_crustmaps(lat,lon,x,vp,vs,rho,moho,found_crust,elem_in_crust)
245 
246 ! Matthias Meschede
247 ! read smooth crust2.0 model (0.25 degree resolution) with eucrust
248 ! based on software routines provided with the crust2.0 model by Bassin et al.
249 
250  use constants
251  use model_crustmaps_par
252 
253  implicit none
254 
255  double precision,intent(in) :: lat,lon,x
256  double precision,intent(out) :: vp,vs,rho,moho
257  logical,intent(out) :: found_crust
258  logical,intent(in) :: elem_in_crust
259 
260  double precision :: h_sed,h_uc
261  double precision :: x3,x4,x5,x6,x7,scaleval
262  double precision,dimension(NLAYERS_CRUSTMAP) :: vps,vss,rhos,thicks
263 
264  call read_crustmaps(lat,lon,vps,vss,rhos,thicks)
265 
266  x3 = (r_earth-thicks(1)*1000.0d0)/r_earth
267  h_sed = thicks(1) + thicks(2)
268  x4 = (r_earth-h_sed*1000.0d0)/r_earth
269  h_uc = h_sed + thicks(3)
270  x5 = (r_earth-h_uc*1000.0d0)/r_earth
271  x6 = (r_earth-(h_uc+thicks(4))*1000.0d0)/r_earth
272  x7 = (r_earth-(h_uc+thicks(4)+thicks(5))*1000.0d0)/r_earth
273 
274  found_crust = .true.
275 ! if (x > x3 .and. INCLUDE_SEDIMENTS_IN_CRUST .and. h_sed > MINIMUM_SEDIMENT_THICKNESS) then
276  if (x > x3 .and. include_sediments_in_crust) then
277  vp = vps(1)
278  vs = vss(1)
279  rho = rhos(1)
280 ! else if (x > x4 .and. INCLUDE_SEDIMENTS_IN_CRUST .and. h_sed > MINIMUM_SEDIMENT_THICKNESS) then
281  else if (x > x4 .and. include_sediments_in_crust) then
282  vp = vps(2)
283  vs = vss(2)
284  rho = rhos(2)
285  else if (x > x5) then
286  vp = vps(3)
287  vs = vss(3)
288  rho = rhos(3)
289  else if (x > x6) then
290  vp = vps(4)
291  vs = vss(4)
292  rho = rhos(4)
293  else if (x > x7 .or. elem_in_crust) then
294  vp = vps(5)
295  vs = vss(5)
296  rho = rhos(5)
297  else
298  found_crust = .false.
299  endif
300 
301  ! non-dimensionalize
302  scaleval = dsqrt(pi*grav*rhoav)
303 
304  if (found_crust) then
305  vp = vp*1000.0d0/(r_earth*scaleval)
306  vs = vs*1000.0d0/(r_earth*scaleval)
307  rho = rho*1000.0d0/rhoav
308  else
309  ! takes ficticious values
310  vp = 20.0*1000.0d0/(r_earth*scaleval)
311  vs = 20.0*1000.0d0/(r_earth*scaleval)
312  rho = 20.0*1000.0d0/rhoav
313  endif
314 
315  moho = (h_uc+thicks(4)+thicks(5))*1000.0d0/r_earth
316 
317  end subroutine model_crustmaps

323  subroutine read_crustmaps(lat,lon,velp,vels,rhos,thicks)
324 
325 ! crustal vp and vs in km/s, layer thickness in km
326 
327  use constants
328  use model_crustmaps_par
329 
330  implicit none
331 
332  ! argument variables
333  double precision,intent(in) :: lat,lon
334  double precision,intent(out) :: rhos(5),thicks(5),velp(5),vels(5)
335 
336  !-------------------------------
337  ! work-around to avoid Jacobian problems when stretching mesh elements;
338  ! one could also try to slightly change the shape of the doubling element bricks (which cause the problem)...
339  !
340  ! defines a "critical" region to have at least a 1-degree smoothing;
341  ! critical region can lead to negative Jacobians for mesh stretching when CAP smoothing is too small
342  double precision,parameter :: LAT_CRITICAL_EUROPE = 50.0d0
343  double precision,parameter :: LON_CRITICAL_EUROPE = 22.0d0
344  double precision,parameter :: CRITICAL_RANGE_EUROPE = 50.0d0
345 
346  ! defines a "critical" region around the andes to have at least a 1-degree smoothing;
347  ! critical region can lead to negative Jacobians for mesh stretching when CAP smoothing is too small
348  double precision,parameter :: LAT_CRITICAL_ANDES = -20.0d0
349  double precision,parameter :: LON_CRITICAL_ANDES = -70.0d0
350  double precision,parameter :: CRITICAL_RANGE_ANDES = 70.0d0
351 
352  ! sampling rate for CAP points
353  integer, parameter :: NTHETA = 4
354  integer, parameter :: NPHI = 20
355  !-------------------------------
356 
357  ! local variables
358  double precision,dimension(NTHETA*NPHI) :: xlon,xlat,weight
359  double precision,dimension(NLAYERS_CRUSTMAP) :: rhol,thickl,velpl,velsl
360 
361  double precision :: lat_used,lon_used
362  double precision :: weightup,weightleft,weightul,weightur,weightll,weightlr
363  double precision :: weightl,cap_degree,dist
364  double precision :: h_sed
365  integer :: num_points
366  integer :: i,ipoin,iupcolat,ileftlng,irightlng
367 
368 ! get integer colatitude and longitude of crustal cap
369 ! -90 < lat < 90 -180 < lon < 180
370  if (lat > 90.0d0 .or. lat < -90.0d0 .or. lon > 180.0d0 .or. lon < -180.0d0) &
371  write(*,*) lat,' ',lon, ' error in latitude/longitude range in crust'
372 
373  ! avoid rounding at poles
374  lat_used = lat
375  lon_used = lon
376  if (lat == 90.0d0) lat_used = 89.9999d0
377  if (lat == -90.0d0) lat_used = -89.9999d0
378  if (lon == 180.0d0) lon_used = 179.9999d0
379  if (lon == -180.0d0) lon_used = -179.9999d0
380 
381  ! by defaults uses only 1 point location
382  num_points = 1
383 
384  ! initializes cap arrays (to avoid compiler warning)
385  xlat(:) = 0.d0
386  xlon(:) = 0.d0
387  weight(:) = 0.d0
388 
389  ! checks if inside/outside of critical region for mesh stretching
390  if (smooth_crust_even_more) then
391 
392  dist = dsqrt( (lon_used - lon_critical_europe)**2 + (lat_used - lat_critical_europe )**2 )
393  if (dist < critical_range_europe) then
394  ! sets up smoothing points
395  ! by default uses CAP smoothing with crustmap resolution, e.g. 1/4 degree
396  cap_degree = 1.d0 / crustmap_resolution
397 
398  ! increases cap smoothing degree
399  ! scales between -1 at center and 0 at border
400  dist = dist / critical_range_europe - 1.0d0
401  ! shifts value to 1 at center and 0 to the border with exponential decay
402  dist = 1.0d0 - exp( - dist*dist*10.0d0 )
403  ! increases smoothing degree inside of critical region
404  cap_degree = cap_degree + dist
405 
406  ! gets smoothing points and weights
407  call smooth_weights_cap_vardegree(lon_used,lat_used,xlon,xlat,weight,cap_degree,ntheta,nphi)
408  num_points = ntheta*nphi
409  endif
410 
411  dist = dsqrt( (lon_used - lon_critical_andes)**2 + (lat_used - lat_critical_andes )**2 )
412  if (dist < critical_range_andes) then
413  ! sets up smoothing points
414  ! by default uses CAP smoothing with crustmap resolution, e.g. 1/4 degree
415  cap_degree = 1.d0 / crustmap_resolution
416 
417  ! increases cap smoothing degree
418  ! scales between -1 at center and 0 at border
419  dist = dist / critical_range_andes - 1.0d0
420  ! shifts value to 1 at center and 0 to the border with exponential decay
421  dist = 1.0d0 - exp( - dist*dist*10.0d0 )
422  ! increases smoothing degree inside of critical region
423  cap_degree = cap_degree + dist
424 
425  ! gets smoothing points and weights
426  call smooth_weights_cap_vardegree(lon_used,lat_used,xlon,xlat,weight,cap_degree,ntheta,nphi)
427  num_points = ntheta*nphi
428  endif
429 
430  endif
431 
432  ! initializes
433  velp(:) = 0.0d0
434  vels(:) = 0.0d0
435  rhos(:) = 0.0d0
436  thicks(:) = 0.0d0
437 
438  ! loops over weight points
439  do ipoin = 1,num_points
440  ! checks if more than one weighting points are taken
441  if (num_points > 1) then
442  lat_used = xlat(ipoin)
443  lon_used = xlon(ipoin)
444  ! weighting value
445  weightl = weight(ipoin)
446  else
447  weightl = 1.0d0
448  endif
449 
450  ! gets crust value indices
451  call ibilinearmap(lat_used,lon_used,iupcolat,ileftlng,weightup,weightleft)
452 
453  ! interpolates location and crust values
454  if (iupcolat == 0) then
455  weightup=weightup*2
456  else if (iupcolat == 180*crustmap_resolution) then
457  weightup=2*weightup-1
458  endif
459 
460  if (ileftlng == 360*crustmap_resolution) then
461  irightlng = 1
462  else
463  irightlng = ileftlng+1
464  endif
465 
466  weightul=weightup*weightleft
467  weightur=weightup*(1.0-weightleft)
468  weightll=(1.0-weightup)*weightleft
469  weightlr=(1.0-weightup)*(1.0-weightleft)
470 
471  if (iupcolat == 0) then
472  ! North pole
473  do i = 1,nlayers_crustmap
474  thickl(i)=weightul*thicknessnp(i)+weightur*thicknessnp(i)+&
475  weightll*thickness(1,ileftlng,i)+weightlr*thickness(1,irightlng,i)
476 
477  rhol(i)=weightul*densitynp(i)+weightur*densitynp(i)+&
478  weightll*density(1,ileftlng,i)+weightlr*density(1,irightlng,i)
479  velpl(i)=weightul*velocpnp(i)+weightur*velocpnp(i)+&
480  weightll*velocp(1,ileftlng,i)+weightlr*velocp(1,irightlng,i)
481  velsl(i)=weightul*velocsnp(i)+weightur*velocsnp(i)+&
482  weightll*velocs(1,ileftlng,i)+weightlr*velocs(1,irightlng,i)
483  enddo
484  else if (iupcolat == 180*crustmap_resolution) then
485  ! South pole
486  do i = 1,nlayers_crustmap
487  thickl(i)=weightul*thickness(iupcolat,ileftlng,i)+weightur*thickness(iupcolat,irightlng,i)+&
488  weightll*thicknesssp(i)+weightlr*thicknesssp(i)
489  rhol(i)=weightul*density(iupcolat,ileftlng,i)+weightur*density(iupcolat,irightlng,i)+&
490  weightll*densitysp(i)+weightlr*densitysp(i)
491  velpl(i)=weightul*velocp(iupcolat,ileftlng,i)+weightur*velocp(iupcolat,irightlng,i)+&
492  weightll*velocpsp(i)+weightlr*velocpsp(i)
493  velsl(i)=weightul*velocs(iupcolat,ileftlng,i)+weightur*velocs(iupcolat,irightlng,i)+&
494  weightll*velocssp(i)+weightlr*velocssp(i)
495  enddo
496  else
497  do i = 1,nlayers_crustmap
498  thickl(i)=weightul*thickness(iupcolat,ileftlng,i)+weightur*thickness(iupcolat,irightlng,i)+&
499  weightll*thickness(iupcolat+1,ileftlng,i)+weightlr*thickness(iupcolat+1,irightlng,i)
500  rhol(i)=weightul*density(iupcolat,ileftlng,i)+weightur*density(iupcolat,irightlng,i)+&
501  weightll*density(iupcolat+1,ileftlng,i)+weightlr*density(iupcolat+1,irightlng,i)
502  velpl(i)=weightul*velocp(iupcolat,ileftlng,i)+weightur*velocp(iupcolat,irightlng,i)+&
503  weightll*velocp(iupcolat+1,ileftlng,i)+weightlr*velocp(iupcolat+1,irightlng,i)
504  velsl(i)=weightul*velocs(iupcolat,ileftlng,i)+weightur*velocs(iupcolat,irightlng,i)+&
505  weightll*velocs(iupcolat+1,ileftlng,i)+weightlr*velocs(iupcolat+1,irightlng,i)
506  ! thicks(i)=1.0
507  ! rhos(i)=1.0
508  ! velp(i)=1.0
509  ! vels(i)=1.0i
510  enddo
511  endif
512 
513  ! sediment thickness
514  h_sed = thickl(1) + thickl(2)
515 
516  ! takes upper crust value if sediment too thin
517  if (h_sed < minimum_sediment_thickness) then
518  velpl(1) = velpl(3)
519  velpl(2) = velpl(3)
520  velsl(1) = velsl(3)
521  velsl(2) = velsl(3)
522  rhol(1) = rhol(3)
523  rhol(2) = rhol(3)
524  endif
525 
526  ! total, smoothed values
527  rhos(:) = rhos(:) + weightl*rhol(:)
528  thicks(:) = thicks(:) + weightl*thickl(:)
529  velp(:) = velp(:) + weightl*velpl(:)
530  vels(:) = vels(:) + weightl*velsl(:)
531  enddo
532 
533  end subroutine read_crustmaps

537  subroutine ibilinearmap(lat,lng,iupcolat,ileftlng,weightup,weightleft)
538 
539  use constants
540  use model_crustmaps_par, only: crustmap_resolution
541 
542  implicit none
543 
544  ! argument variables
545  double precision weightup,weightleft
546  double precision lat,lng, xlng
547  double precision buffer
548  integer iupcolat
549  integer ileftlng
