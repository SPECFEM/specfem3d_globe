194  endif
195

197  is_okay = .false.
198  do j = 1,dummy_i(ineighbour)
199  if (test_interfaces(j,ineighbour) == iproc) then
200  ! checks if same number of interface points with this neighbour
201  if (test_interfaces_nibool(j,ineighbour) == ipoints) then
202  is_okay = .true.
203  else
204  print *,'Error ',iproc,'neighbour ',ineighbour,' points =',ipoints
205  print *,' ineighbour has points = ',test_interfaces_nibool(j,ineighbour)
206  print *
207  call exit_mpi(myrank,'Error ineighbour points differ')
208  endif
209  exit
210  endif
211  enddo
212  if (.not. is_okay) then
213  print *,'Error ',iproc,' neighbour not found: ',ineighbour
214  print *,'iproc ',iproc,' interfaces:'
215  print *,test_interfaces(1:dummy_i(iproc),iproc)
216  print *,'ineighbour ',ineighbour,' interfaces:'
217  print *,test_interfaces(1:dummy_i(ineighbour),ineighbour)
218  print *
219  call exit_mpi(myrank,'Error ineighbour not found')
220  endif
221  enddo
222  enddo
223 
224  ! user output
225  write(imain,*) ' MPI addressing maximum interfaces:',maxval(dummy_i)
226  write(imain,*) ' MPI addressing : all interfaces okay'
227  write(imain,*)
228  call flush_imain()
229

231  deallocate(test_interfaces)

314

320  call exit_mpi(myrank,'Error test MPI crust mantle valence')

323 
324  ! checks within slice

332

335  ! checks

345  endif
