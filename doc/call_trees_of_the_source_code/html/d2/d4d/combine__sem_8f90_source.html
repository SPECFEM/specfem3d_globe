28 ! XCOMBINE_SEM
29 !
30 ! USAGE
31 ! mpirun -np NPROC bin/xcombine_sem KERNEL_NAMES INPUT_FILE OUTPUT_DIR
32 !
33 !
34 ! COMMAND LINE ARGUMENTS
35 ! KERNEL_NAMES - one or more material parameter names separated by commas
36 ! INPUT_FILE - text file containing list of kernel directories
37 ! OUTPUT_PATH - directory to which summed kernels are written
38 !
39 !
40 ! DESCRIPTION
41 ! For each name in KERNEL_NAMES, sums kernels from directories specified in
42 ! INPUT_FILE. Writes the resulting sum to OUTPUT_DIR.
43 !
44 ! INPUT_FILE is a text file containing a list of absolute or relative paths to
45 ! kernel directories, one directory per line.
46 !
47 ! KERNEL_NAMES is comma-delimited list of kernel names,
48 ! e.g.'alpha_kernel,beta_kernel,rho_kernel'.
49 !
50 ! This program's primary use case is to sum kernels. It can be used though on
51 ! any "reg1" array, i.e. any array of dimension
52 ! (NGLLX,NGLLY,NGLLZ,NSPEC_CRUST_MANTLE).
53 !
54 ! This is a parallel program -- it must be invoked with mpirun or other
55 ! appropriate utility. Operations are performed in embarrassingly-parallel
56 ! fashion.

170 subroutine sum_kernel(kernel_name,kernel_paths,output_dir,npath)
171 
172  use postprocess_par
173 
174  implicit none
175 
176  character(len=*),parameter :: reg = '_reg1_'
177 
178  character(len=MAX_STRING_LEN) :: output_dir,kernel_name,kernel_paths(max_kernel_paths)
179  integer :: ipath,npath
180 
181  ! local parameters
182  character(len=MAX_STRING_LEN) :: k_file
183  real(kind=CUSTOM_REAL), dimension(:,:,:,:),allocatable :: kernel,total_kernel
184  double precision :: norm,norm_sum
185  integer :: ier
186 
187  ! initializes arrays
188  allocate(kernel(ngllx,nglly,ngllz,nspec_crust_mantle), &
189  total_kernel(ngllx,nglly,ngllz,nspec_crust_mantle),stat=ier)
190  if (ier /= 0) stop 'Error allocating kernel arrays'
191 
192  ! loops over all paths
193  total_kernel = 0._custom_real
194  do ipath = 1, npath
195  if (myrank == 0) then
196  write(*,*) 'reading in event kernel for: ',trim(kernel_name)
197  write(*,*) ' ',ipath, ' out of ', npath
198  endif
199 
200  ! read kernel
201  kernel = 0._custom_real
202  write(k_file,'(a,i6.6,a)') trim(kernel_paths(ipath))//'/proc',myrank,reg//trim(kernel_name)//'.bin'
203  open(iin,file=trim(k_file),status='old',form='unformatted',action='read',iostat=ier)
204  if (ier /= 0) then
205  write(*,*) ' kernel not found: ',trim(k_file)
206  stop 'Error kernel file not found'
207  endif
208  read(iin) kernel
209  close(iin)
210 
211  norm = sum( kernel * kernel )
212  call sum_all_dp(norm,norm_sum)
213  if (myrank == 0) then
214  print *,' norm kernel: ',sqrt(norm_sum)
215  print *
216  endif
217 
218  ! keep track of sum
219  total_kernel = total_kernel + kernel
220  enddo
221 
222 
223  ! write sum
224  if (myrank == 0) write(*,*) 'writing out summed kernel for: ',trim(kernel_name)
225  write(k_file,'(a,i6.6,a)') trim(output_dir)//'/'//'proc',myrank,reg//trim(kernel_name)//'.bin'
226 
227  open(iout,file=trim(k_file),form='unformatted',status='unknown',action='write',iostat=ier)
228  if (ier /= 0) then
229  write(*,*) 'Error kernel not written: ',trim(k_file)
230  stop 'Error kernel write'
231  endif
232  write(iout) total_kernel
233  close(iout)
234 
235  if (myrank == 0) write(*,*)
236  deallocate(kernel,total_kernel)
237 
238 end subroutine sum_kernel
