283  subroutine model_attenuation_storage(Qmu, tau_e, rw, AM_S)
284 
285  use constants
286  use meshfem3d_models_par, only: model_attenuation_storage_var
287 
288  implicit none
289 
290  type(model_attenuation_storage_var) am_s
291 
292  double precision :: Qmu
293  double precision, dimension(N_SLS) :: tau_e
294  integer :: rw
295 
296  ! local parameters
297  double precision :: Qmu_new
298  integer :: Qtmp
299  integer, save :: first_time_called = 1
300 
301  if (first_time_called == 1) then
302  first_time_called = 0
303  am_s%Q_resolution = 10**attenuation_comp_resolution
304  am_s%Q_max = attenuation_comp_maximum
305  qtmp = am_s%Q_resolution * am_s%Q_max
306  allocate(am_s%tau_e_storage(n_sls, qtmp))
307  allocate(am_s%Qmu_storage(qtmp))
308  am_s%Qmu_storage(:) = -1
309  endif
310 
311  if (qmu < 0.0d0 .or. qmu > am_s%Q_max) then
312  write(imain,*) 'Error attenuation_storage()'
313  write(imain,*) 'Attenuation Value out of Range: ', qmu
314  write(imain,*) 'Attenuation Value out of Range: Min, Max ', 0, am_s%Q_max
315  call flush_imain()
316  ! stop
317  call exit_mpi(myrank, 'Attenuation Value out of Range')
318  endif
319 
320  if (rw > 0 .and. qmu == 0.0d0) then
321  qmu = 0.0d0;
322  tau_e(:) = 0.0d0;
323  return
324  endif
325  ! Generate index for Storage Array
326  ! and Recast Qmu using this index
327 
328  ! by default: resolution is Q_resolution = 10
329  ! converts Qmu to an array integer index:
330  ! e.g. Qmu = 150.31 -> Qtmp = 150.31 * 10 = int( 1503.10 ) = 1503
331  qtmp = int(qmu * dble(am_s%Q_resolution))
332 
333  ! rounds to corresponding double value:
334  ! e.g. Qmu_new = dble( 1503 ) / dble(10) = 150.30
335  ! but Qmu_new is not used any further...
336  qmu_new = dble(qtmp) / dble(am_s%Q_resolution)
337 
338  if (rw > 0) then
339  ! READ
340  if (am_s%Qmu_storage(qtmp) > 0) then
341  ! READ SUCCESSFUL
342  tau_e(:) = am_s%tau_e_storage(:, qtmp)
343  qmu = am_s%Qmu_storage(qtmp)
344  rw = 1
345  else
346  ! READ NOT SUCCESSFUL
347  rw = -1
348  endif
349  else
350  ! WRITE SUCCESSFUL
351  am_s%tau_e_storage(:,qtmp) = tau_e(:)
352  am_s%Qmu_storage(qtmp) = qmu
353  rw = 1
354  endif
355 
356  end subroutine model_attenuation_storage

383  subroutine attenuation_tau_sigma(tau_s, n, min_period, max_period)
384  ! Set the Tau_sigma (tau_s) to be equally spaced in log10 frequency
385 
386  use constants
387 
388  implicit none
389 
390  integer n
391  double precision tau_s(n)
392  double precision min_period, max_period
393  double precision f1, f2
394  double precision exp1, exp2
395  double precision dexpval
396  integer i
397 
398  f1 = 1.0d0 / max_period
399  f2 = 1.0d0 / min_period
400 
401  exp1 = log10(f1)
402  exp2 = log10(f2)
403 
404  dexpval = (exp2-exp1) / ((n*1.0d0) - 1)
405  do i = 1,n
406  tau_s(i) = 1.0 / (pi * 2.0d0 * 10**(exp1 + (i - 1)* 1.0d0 *dexpval))
407  enddo
408 
409  end subroutine attenuation_tau_sigma

415  subroutine attenuation_invert_by_simplex(t2, t1, n, Q_real, omega_not, tau_s, tau_e, AS_V)
416 
417  use constants
418  use meshfem3d_models_par, only: attenuation_simplex_variables
419 
420  implicit none
421 
422  type(attenuation_simplex_variables) :: as_v
423 
424  ! Input / Output
425  double precision t1, t2
426  double precision Q_real
427  double precision omega_not
428  integer n
429  double precision, dimension(n) :: tau_s, tau_e
430 
431  ! Internal
432  integer i, iterations, err,prnt
433  double precision f1, f2, exp1,exp2,dexpval, min_value
434  double precision, allocatable, dimension(:) :: f
435  integer, parameter :: nf = 100
436  double precision, external :: attenuation_eval
437 
438  ! Values to be passed into the simplex minimization routine
439  iterations = -1
440  min_value = -1.0e-4
441  err = 0
442  prnt = 0
443 
444  allocate(f(nf))
445  ! Determine the min and max frequencies
446  f1 = 1.0d0 / t1
447  f2 = 1.0d0 / t2
448 
449  ! Determine the exponents of the frequencies
450  exp1 = log10(f1)
451  exp2 = log10(f2)
452 
453  if (f2 < f1 .or. q_real < 0.0d0 .or. n < 1) then
454  call exit_mpi(myrank, 'frequencies flipped or Q less than zero or N_SLS < 0')
455  endif
456 
457  ! Determine the Source frequency
458  omega_not = 1.0e+03 * 10.0d0**(0.5 * (log10(f1) + log10(f2)))
459 
460  ! Determine the Frequencies at which to compare solutions
461  ! The frequencies should be equally spaced in log10 frequency
462  do i = 1,nf
463  f(i) = exp1 + ((i-1)*1.0d0 * (exp2-exp1) / ((nf-1)*1.0d0))
464  enddo
465 
466  ! Set the Tau_sigma (tau_s) to be equally spaced in log10 frequency
467  dexpval = (exp2-exp1) / ((n*1.0d0) - 1)
468  do i = 1,n
469  tau_s(i) = 1.0 / (pi * 2.0d0 * 10**(exp1 + (i - 1)* 1.0d0 *dexpval))
470  enddo
471 
472  ! Shove the parameters into the module
473  call attenuation_simplex_setup(nf,n,f,q_real,tau_s,as_v)
474 
475  ! Set the Tau_epsilon (tau_e) to an initial value at omega*tau = 1
476  ! tan_delta = 1/Q = (tau_e - tau_s)/(2 * sqrt(tau e*tau_s))
477  ! if we assume tau_e =~ tau_s
478  ! we get the equation below
479  do i = 1,n
480  tau_e(i) = tau_s(i) + (tau_s(i) * 2.0d0/q_real)
481  enddo
482 
483  ! Run a simplex search to determine the optimum values of tau_e
484  call fminsearch(attenuation_eval, tau_e, n, iterations, min_value, prnt, err,as_v)
485  if (err > 0) then
486  write(*,*)'Search did not converge for an attenuation of ', q_real
487  write(*,*)' Iterations: ', iterations
488  write(*,*)' Min Value: ', min_value
489  write(*,*)' Aborting program'
490  call exit_mpi(myrank,'attenuation_simplex: Search for Strain relaxation times did not converge')
491  endif
492  deallocate(f)
493 
494  call attenuation_simplex_finish(as_v)
495 
496  end subroutine attenuation_invert_by_simplex

546 ! - Computes the Moduli (Maxwell Solid) for a series of
547 ! Standard Linear Solids
548 ! - Computes M1 and M2 parameters after Dahlen and Tromp pp.203
549 ! here called B and A after Liu et al. 1976
550 ! - Another formulation uses Kelvin-Voigt Solids and computes
551 ! Compliences J1 and J2 after Dahlen and Tromp pp.203
552 !
553 ! Input
554 ! nf = Number of Frequencies
555 ! nsls = Number of Standard Linear Solids
556 ! f = Frequencies (in log10 of frequencies)
557 ! dimension(nf)
558 ! tau_s = Tau_sigma Stress relaxation time (see References)
559 ! dimension(nsls)
560 ! tau_e = Tau_epsilon Strain relaxation time (see References)
561 ! dimension(nsls)!
562 ! Output
563 ! B = Real Moduli ( M2 Dahlen and Tromp pp.203 )
564 ! dimension(nf)
565 ! A = Imaginary Moduli ( M1 Dahlen and Tromp pp.203 )
566 ! dimension(nf)
567 !
568 ! Dahlen and Tromp, 1998
569 ! Theoretical Global Seismology
570 !
571 ! Liu et al. 1976
572 ! Velocity dispersion due to anelasticity: implications for seismology and mantle composition
573 ! Geophys, J. R. asts. Soc, Vol 47, pp. 41-58
574  subroutine attenuation_maxwell(nf,nsls,f,tau_s,tau_e,B,A)
575 
576  use constants
577 
578  implicit none
579 
580  ! Input
581  integer nf, nsls
582  double precision, dimension(nf) :: f
583  double precision, dimension(nsls) :: tau_s, tau_e
584  ! Output
585  double precision, dimension(nf) :: A,B
586 
587  integer i,j
