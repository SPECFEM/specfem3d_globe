29 ! JP3D
30 !
31 ! 3D japan Vp velocity model
32 !
33 ! based on:
34 !
35 ! Program ----- veljp3d.f -----
36 !
37 ! This program is used to calculate 3-D P-wave velocity
38 ! distribution beneath the Japan Islands which is obtained
39 ! by a simultaneous inversion of arrival time data from local,
40 ! regional and teleseismic events. For details, see "Deep
41 ! structure of the Japan subduction zone as derived from local,
42 ! regional, and teleseismic events" by Zhao, Hasegawa & Kanamori,
43 ! JGR, 99, 22313-22329, 1994.
44 !
45 ! The meaningful range of this model is as follows:
46 ! latitude : 32 - 45 N
47 ! longitude: 130-145 E
48 ! depth : 0 - 500 km
49 !
50 ! Dapeng Zhao
51 ! Dept. of Earth & Planet. Sci
52 ! Washington University
53 ! St. Louis, MO 63130
54 ! U.S.A.
55 ! dapeng@izu.wustl.edu

246  subroutine model_jp3d_iso_zhao(radius,theta,phi,vp,vs,dvp,dvs,rho,found_crust)
247 
248  use constants
249  use model_jp3d_par
250 
251  implicit none
252 
253  logical found_crust
254  double precision :: radius,theta,phi,vp,vs,dvs,dvp,rho
255  double precision :: PE,RE,HE,H1,H2,H3,scaleval
256  integer :: LAY
257 
258 
259  found_crust = .false.
260 
261  pe = theta
262  re = phi
263  he = (one - radius)*r_earth_km
264 ! calculate depths of the Conrad, the Moho and
265 ! the plate boundary beneath the location (PHI,RAM)
266  CALL hlay(pe,re,h1,1)
267  CALL hlay(pe,re,h2,2)
268  CALL hlay(pe,re,h3,3)
269 ! when LAY = 1, the focus is in the upper crust;
270 ! when LAY = 2, the focus is in the lower crust;
271 ! when LAY = 3, the focus is in the mantle wedge;
272 ! when LAY = 4, the focus is beneath the plate boundary.
273  if (he <= h1) then
274  lay = 1
275  found_crust = .true.
276  else if (he > h1 .and. he <= h2) then
277  lay = 2
278  found_crust = .true.
279  else if (he > h2 .and. he <= h3) then
280  lay = 3
281  ELSE
282  lay = 4
283  endif
284 
285  CALL vel1d(he,vp,lay,1)
286  CALL vel1d(he,vs,lay,2)
287  CALL vel3(pe,re,he,dvp,lay)
288 
289  dvp = 0.01d0*dvp
290  dvs = 1.5d0*dvp
291  vp = vp*(1.0d0+dvp)
292  vs = vs*(1.0d0+dvs)
293 
294 ! determine rho
295  if (lay == 1) then
296  rho=2.6
297  endif
298  if (lay == 2) then
299  rho=2.9
300  endif
301  if (lay > 2) then
302  rho=3.3+(vs-4.4)*0.66667
303  endif
304 ! non-dimensionalize
305 ! time scaling (s^{-1}) is done with scaleval
306  scaleval=dsqrt(pi*grav*rhoav)
307  rho=rho*1000.0d0/rhoav
308  vp=vp*1000.0d0/(r_earth*scaleval)
309  vs=vs*1000.0d0/(r_earth*scaleval)
310 
311  end subroutine model_jp3d_iso_zhao

459

468 
469  jp3dm_p = 90.0-pe/degrees_to_radians

471  jp3dm_h = he
472  if (lay <= 3) then
473  CALL prhf(jp3dm_iploca,jp3dm_irloca,jp3dm_ihloca,jp3dm_pla,jp3dm_rla,jp3dm_hla, &

478  ELSE
479  endif
480 
481  jp3dm_wv(1) = jp3dm_pf1*jp3dm_rf1*jp3dm_hf1
482  jp3dm_wv(2) = jp3dm_pf*jp3dm_rf1*jp3dm_hf1
483  jp3dm_wv(3) = jp3dm_pf1*jp3dm_rf*jp3dm_hf1

488  jp3dm_wv(8) = jp3dm_pf*jp3dm_rf*jp3dm_hf

492  CALL vabps(mpa,mra,mha,jp3dm_velap,v)

540  subroutine prhf(IPLOCX,IRLOCX,IHLOCX,PLX,RLX,HLX, &
541  pnx,rnx,hnx,mpx,mrx,mhx,mkx)
542 
543  use constants
544  use model_jp3d_par
545 
546  implicit none
547 
548  integer :: MPX,MRX,MHX,MKX
549  integer :: IPLOCX(mkx),IRLOCX(mkx),IHLOCX(mkx)
550  double precision :: PNX(mpx),RNX(mrx),HNX(mhx)
551  double precision :: PLX,RLX,HLX
552 
553  CALL limit(pnx(1),pnx(mpx),jp3dm_p)
554  CALL limit(rnx(1),rnx(mrx),jp3dm_r)
555  CALL limit(hnx(1),hnx(mhx),jp3dm_h)
556  CALL intmap(jp3dm_p*100.0,iplocx,mkx,plx,jp3dm_ip)
557  CALL intmap(jp3dm_r*100.0,irlocx,mkx,rlx,jp3dm_jp)
558  CALL intmap(jp3dm_h,ihlocx,mkx,hlx,jp3dm_kp)
559  jp3dm_ip1 = jp3dm_ip+1
560  jp3dm_jp1 = jp3dm_jp+1
561  jp3dm_kp1 = jp3dm_kp+1
562  jp3dm_pd = pnx(jp3dm_ip1)-pnx(jp3dm_ip)
563  jp3dm_rd = rnx(jp3dm_jp1)-rnx(jp3dm_jp)
564  jp3dm_hd = hnx(jp3dm_kp1)-hnx(jp3dm_kp)
565  jp3dm_pf = (jp3dm_p-pnx(jp3dm_ip))/jp3dm_pd
566  jp3dm_rf = (jp3dm_r-rnx(jp3dm_jp))/jp3dm_rd
567  jp3dm_hf = (jp3dm_h-hnx(jp3dm_kp))/jp3dm_hd
