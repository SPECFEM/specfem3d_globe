29 ! JP3D
30 !
31 ! 3D japan Vp velocity model
32 !
33 ! based on:
34 !
35 ! Program ----- veljp3d.f -----
36 !
37 ! This program is used to calculate 3-D P-wave velocity
38 ! distribution beneath the Japan Islands which is obtained
39 ! by a simultaneous inversion of arrival time data from local,
40 ! regional and teleseismic events. For details, see "Deep
41 ! structure of the Japan subduction zone as derived from local,
42 ! regional, and teleseismic events" by Zhao, Hasegawa & Kanamori,
43 ! JGR, 99, 22313-22329, 1994.
44 !
45 ! The meaningful range of this model is as follows:
46 ! latitude : 32 - 45 N
47 ! longitude: 130-145 E
48 ! depth : 0 - 500 km
49 !
50 ! Dapeng Zhao
51 ! Dept. of Earth & Planet. Sci
52 ! Washington University
53 ! St. Louis, MO 63130
54 ! U.S.A.
55 ! dapeng@izu.wustl.edu

57 !

245  subroutine model_jp3d_iso_zhao(radius,theta,phi,vp,vs,dvp,dvs,rho,found_crust)
246 
247  use constants
248  use model_jp3d_par
249 
250  implicit none
251 
252  logical found_crust
253  double precision :: radius,theta,phi,vp,vs,dvs,dvp,rho
254  double precision :: PE,RE,HE,H1,H2,H3,scaleval
255  integer :: LAY
256 
257 
258  found_crust = .false.
259 
260  pe = theta
261  re = phi
262  he = (one - radius)*r_earth_km
263 ! calculate depths of the Conrad, the Moho and
264 ! the plate boundary beneath the location (PHI,RAM)
265  CALL hlay(pe,re,h1,1)
266  CALL hlay(pe,re,h2,2)
267  CALL hlay(pe,re,h3,3)
268 ! when LAY = 1, the focus is in the upper crust;
269 ! when LAY = 2, the focus is in the lower crust;
270 ! when LAY = 3, the focus is in the mantle wedge;
271 ! when LAY = 4, the focus is beneath the plate boundary.
272  if (he <= h1) then
273  lay = 1
274  found_crust = .true.
275  else if (he > h1 .and. he <= h2) then
276  lay = 2
277  found_crust = .true.
278  else if (he > h2 .and. he <= h3) then
279  lay = 3
280  ELSE
281  lay = 4
282  endif
283 
284  CALL vel1d(he,vp,lay,1)
285  CALL vel1d(he,vs,lay,2)
286  CALL vel3(pe,re,he,dvp,lay)
287 
288  dvp = 0.01d0*dvp
289  dvs = 1.5d0*dvp
290  vp = vp*(1.0d0+dvp)
291  vs = vs*(1.0d0+dvs)
292 
293 ! determine rho
294  if (lay == 1) then
295  rho=2.6
296  endif
297  if (lay == 2) then
298  rho=2.9
299  endif
300  if (lay > 2) then
301  rho=3.3+(vs-4.4)*0.66667
302  endif
303 ! non-dimensionalize
304 ! time scaling (s^{-1}) is done with scaleval
305  scaleval=dsqrt(pi*grav*rhoav)
306  rho=rho*1000.0d0/rhoav
307  vp=vp*1000.0d0/(r_earth*scaleval)
308  vs=vs*1000.0d0/(r_earth*scaleval)
309 
310  end subroutine model_jp3d_iso_zhao

457  subroutine vel3(PE,RE,HE,V,LAY)
458 
459  use constants
460  use model_jp3d_par
461 
462  implicit none
463 
464  double precision :: PE,RE,HE,V
465 
466  integer :: LAY
467 
468  jp3dm_p = 90.0-pe/degrees_to_radians
469  jp3dm_r = re/degrees_to_radians
470  jp3dm_h = he
471  if (lay <= 3) then
472  CALL prhf(jp3dm_iploca,jp3dm_irloca,jp3dm_ihloca,jp3dm_pla,jp3dm_rla,jp3dm_hla, &
473  jp3dm_pna,jp3dm_rna,jp3dm_hna,mpa,mra,mha,mka)
474  else if (lay == 4) then
475  CALL prhf(jp3dm_iplocb,jp3dm_irlocb,jp3dm_ihlocb,jp3dm_plb,jp3dm_rlb,jp3dm_hlb, &
476  jp3dm_pnb,jp3dm_rnb,jp3dm_hnb,mpb,mrb,mhb,mkb)
477  ELSE
478  endif
479 
480  jp3dm_wv(1) = jp3dm_pf1*jp3dm_rf1*jp3dm_hf1
481  jp3dm_wv(2) = jp3dm_pf*jp3dm_rf1*jp3dm_hf1
482  jp3dm_wv(3) = jp3dm_pf1*jp3dm_rf*jp3dm_hf1
483  jp3dm_wv(4) = jp3dm_pf*jp3dm_rf*jp3dm_hf1

485  jp3dm_wv(6) = jp3dm_pf*jp3dm_rf1*jp3dm_hf
486  jp3dm_wv(7) = jp3dm_pf1*jp3dm_rf*jp3dm_hf
487  jp3dm_wv(8) = jp3dm_pf*jp3dm_rf*jp3dm_hf

491  CALL vabps(mpa,mra,mha,jp3dm_velap,v)

493  CALL vabps(mpb,mrb,mhb,jp3dm_velbp,v)

539  subroutine prhf(IPLOCX,IRLOCX,IHLOCX,PLX,RLX,HLX, &
540  pnx,rnx,hnx,mpx,mrx,mhx,mkx)
541 
542  use constants
543  use model_jp3d_par
544 
545  implicit none
546 
547  integer :: MPX,MRX,MHX,MKX
548  integer :: IPLOCX(mkx),IRLOCX(mkx),IHLOCX(mkx)
549  double precision :: PNX(mpx),RNX(mrx),HNX(mhx)
550  double precision :: PLX,RLX,HLX
551 
552  CALL limit(pnx(1),pnx(mpx),jp3dm_p)
553  CALL limit(rnx(1),rnx(mrx),jp3dm_r)
554  CALL limit(hnx(1),hnx(mhx),jp3dm_h)
555  CALL intmap(jp3dm_p*100.0,iplocx,mkx,plx,jp3dm_ip)
556  CALL intmap(jp3dm_r*100.0,irlocx,mkx,rlx,jp3dm_jp)
557  CALL intmap(jp3dm_h,ihlocx,mkx,hlx,jp3dm_kp)
558  jp3dm_ip1 = jp3dm_ip+1
559  jp3dm_jp1 = jp3dm_jp+1
560  jp3dm_kp1 = jp3dm_kp+1
561  jp3dm_pd = pnx(jp3dm_ip1)-pnx(jp3dm_ip)
562  jp3dm_rd = rnx(jp3dm_jp1)-rnx(jp3dm_jp)
563  jp3dm_hd = hnx(jp3dm_kp1)-hnx(jp3dm_kp)
564  jp3dm_pf = (jp3dm_p-pnx(jp3dm_ip))/jp3dm_pd
565  jp3dm_rf = (jp3dm_r-rnx(jp3dm_jp))/jp3dm_rd
566  jp3dm_hf = (jp3dm_h-hnx(jp3dm_kp))/jp3dm_hd
