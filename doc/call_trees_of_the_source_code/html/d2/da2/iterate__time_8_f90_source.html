31 
32  use specfem_par
33  use specfem_par_crustmantle

40  ! timing
41  double precision, external :: wtime
42 
43  ! for EXACT_UNDOING_TO_DISK
44  integer :: ispec,iglob,i,j,k
45 
46  !---- create a Gnuplot script to display the energy curve in log scale
47  if (output_energy .and. myrank == 0) then
48  open(unit=iout_energy,file=trim(output_files)//'plot_energy.gnu',status='unknown',action='write')
49  write(iout_energy,*) 'set term wxt'

62  write(iout_energy,*) 'plot "energy.dat" us 1:4 t "Total Energy" w l lc 4'
63  write(iout_energy,*) 'pause -1 "Hit any key..."'
64  close(iout_energy)
65  endif
66 
67  ! open the file in which we will store the energy curve
68  if (output_energy .and. myrank == 0) &
69  open(unit=iout_energy,file=trim(output_files)//'energy.dat',status='unknown',action='write')
70 
71 !
72 ! s t a r t t i m e i t e r a t i o n s
73 !
74 
75  ! synchronize all processes to make sure everybody is ready to start time loop
76  call synchronize_all()
77  if (myrank == 0) write(imain,*) 'All processes are synchronized before time loop'
78 
79  if (myrank == 0) then
80  write(imain,*)
81  write(imain,*) 'Starting time iteration loop...'
82  write(imain,*)
83  call flush_imain()
84  endif
85 
86  ! create an empty file to monitor the start of the simulation
87  if (myrank == 0) then
88  open(unit=iout,file=trim(output_files)//'/starttimeloop.txt',status='unknown',action='write')
89  write(iout,*) 'hello, starting time loop'
90  close(iout)
91  endif
92 
93  ! initialize variables for writing seismograms
94  seismo_offset = it_begin-1
95  seismo_current = 0
96 
97  ! get MPI starting time
98  time_start = wtime()
99 
100  ! *********************************************************
101  ! ************* MAIN LOOP OVER THE TIME STEPS *************
102  ! *********************************************************
103 
104  if (exact_undoing_to_disk) call setup_exact_undoing_to_disk()
105 
106  ! time loop
107  do it = it_begin,it_end
108 
109 
110  ! simulation status output and stability check
111  if (mod(it,ntstep_between_output_info) == 0 .or. it == it_begin + 4 .or. it == it_end) then
112  call check_stability()
113  if (i_am_running_on_a_slow_node) goto 100
114  endif
115 
116  do istage = 1, nstage_time_scheme ! is equal to 1 if Newmark because only one stage then
117 
118  if (use_lddrk) then
119  ! update displacement using Runge-Kutta time scheme
120  call update_displ_lddrk()
121  else
122  ! update displacement using Newmark time scheme
123  call update_displ_newmark()
124  endif
125 
126  ! acoustic solver for outer core
127  ! (needs to be done first, before elastic one)
128  call compute_forces_acoustic()
129 
130  ! elastic solver for crust/mantle and inner core
131  call compute_forces_viscoelastic()
132 
133  enddo ! end of very big external loop on istage for all the stages of the LDDRK time scheme (only one stage if Newmark)
134 
135  ! save the forward run to disk for the alpha kernel only
136  if (exact_undoing_to_disk .and. simulation_type == 1) then
137  do ispec = 1, nspec_crust_mantle
138  do k = 1, ngllz
139  do j = 1, nglly
140  do i = 1, ngllx
141  iglob = ibool_crust_mantle(i,j,k,ispec)
142  if (integer_mask_ibool_exact_undo(iglob) /= -1) &
143  buffer_for_disk(integer_mask_ibool_exact_undo(iglob),it_of_this_exact_subset) = &
144  eps_trace_over_3_crust_mantle(i,j,k,ispec)
145  enddo
146  enddo
147  enddo
148  enddo
149  if (it_of_this_exact_subset == it_exact_subset_end) then
150  do it_of_this_exact_subset = 1, it_exact_subset_end
151  write(ifile_for_exact_undoing,rec=it_exact_subset_offset+it_of_this_exact_subset) &
152  buffer_for_disk(:,it_of_this_exact_subset)
153  enddo
154  it_of_this_exact_subset = 1
155  it_exact_subset_offset = it_exact_subset_offset + it_exact_subset_end
156  it_exact_subset_end = min(nstep_for_exact_undoing, it_end - it_exact_subset_offset)
157  else
158  it_of_this_exact_subset = it_of_this_exact_subset + 1
159  endif
160  endif
161 
162  ! kernel simulations (forward and adjoint wavefields)
163  if (simulation_type == 3) then
164 
165  if (.not. exact_undoing_to_disk) then
166  ! note: we step back in time (using time steps - DT ), i.e. wavefields b_displ_..() are time-reversed here
167 
168  ! reconstructs forward wavefields based on last stored wavefield data
169 
170  ! note: NSTAGE_TIME_SCHEME is equal to 1 if Newmark because only one stage then
171  do istage = 1, nstage_time_scheme
172 
173  if (use_lddrk) then
174  ! update displacement using Runge-Kutta time scheme
175  call update_displ_lddrk_backward()
176  else
177  ! update displacement using Newmark time scheme
178  call update_displ_newmark_backward()
179  endif
180 
181  ! acoustic solver for outer core
182  ! (needs to be done first, before elastic one)
183  call compute_forces_acoustic_backward()
184 
185  ! elastic solver for crust/mantle and inner core
186  call compute_forces_viscoelastic_backward()
187 
188  enddo
189 
190  ! restores last time snapshot saved for backward/reconstruction of wavefields
191  ! note: this is done here after the Newmark time scheme, otherwise the indexing for sources
192  ! and adjoint sources will become more complicated
193  ! that is, index it for adjoint sources will match index NSTEP - 1 for backward/reconstructed wavefields
194  if (it == 1) then
195  call read_forward_arrays()
196  endif
197 
198  else ! of if (.not. EXACT_UNDOING_TO_DISK)
199 
200  ! read the forward run from disk for the alpha kernel only
201  it_of_this_exact_subset = it_of_this_exact_subset + 1
202  if (it_of_this_exact_subset > it_exact_subset_end) then
203  it_exact_subset_offset = it_exact_subset_offset + it_exact_subset_end
204  it_exact_subset_end = min(nstep_for_exact_undoing, it_end - it_exact_subset_offset)
205  do it_of_this_exact_subset = 1, it_exact_subset_end
206  ! here we time revert the forward run by reading time step NSTEP - it + 1
207  ! but here, it == it_exact_subset_offset + it_of_this_exact_subset
208  read(ifile_for_exact_undoing,rec=nstep-it_exact_subset_offset-it_of_this_exact_subset+1) &
209  buffer_for_disk(:,it_of_this_exact_subset)
210  enddo
211  it_of_this_exact_subset = 1
212  endif
213 
214  do ispec = 1, nspec_crust_mantle
215  do k = 1, ngllz
216  do j = 1, nglly
217  do i = 1, ngllx
218  iglob = ibool_crust_mantle(i,j,k,ispec)
219  if (integer_mask_ibool_exact_undo(iglob) /= -1) then
220  b_eps_trace_over_3_crust_mantle(i,j,k,ispec) = &
221  buffer_for_disk(integer_mask_ibool_exact_undo(iglob),it_of_this_exact_subset)
222  endif
223  enddo
224  enddo
225  enddo
226  enddo
227 
228  endif ! of if (.not. EXACT_UNDOING_TO_DISK)
229 
230  endif ! kernel simulations
231 
232  ! calculating gravity field at current timestep
233  if (gravity_simulation) call gravity_timeseries()
234 
235  ! write the seismograms with time shift (GPU_MODE transfer included)
236  call write_seismograms()
237 
238  ! adjoint simulations: kernels
239  ! attention: for GPU_MODE and ANISOTROPIC_KL it is necessary to use resort_array (see lines 265-268)
240  if (simulation_type == 3) then
241  call compute_kernels()
242  endif
243 
244  ! outputs movie files
245  if (movie_surface .or. movie_volume) call write_movie_output()
246 
247  ! first step of noise tomography, i.e., save a surface movie at every time step
248  ! modified from the subroutine 'write_movie_surface'
249  if (noise_tomography == 1) then
250  call noise_save_surface_movie()
251  endif
252 
253  ! updates VTK window
254  if (vtk_mode) then
255  call it_update_vtkwindow()
256  endif
257 
258  !
259  !---- end of time iteration loop
260  !
261  enddo ! end of main time loop
262 
263  100 continue
264 
265 
266  if (simulation_type == 3 .and. gpu_mode) then
267  ! attention: cijkl_kl_crust_mantle is sorted differently on GPU and CPU
268  call resort_array(mesh_pointer)
269  endif
270 
271  ! close the huge file that contains a dump of all the time steps to disk
272  if (exact_undoing_to_disk) call finish_exact_undoing_to_disk()
273 
274  ! user output of runtime
275  call print_elapsed_time()
276 
277  ! Transfer fields from GPU card to host for further analysis
278  if (gpu_mode) call it_transfer_from_gpu()
279 
280 !---- close energy file
281  if (output_energy .and. myrank == 0) close(iout_energy)
282 
283  end subroutine iterate_time

289  subroutine it_transfer_from_gpu()
290 
291 ! transfers fields on GPU back onto CPU
292 
293  use specfem_par
294  use specfem_par_crustmantle
295  use specfem_par_innercore
296  use specfem_par_outercore
297  use specfem_par_noise
298 
299  implicit none
300 
301  ! to store forward wave fields
302  if (simulation_type == 1 .and. save_forward) then
303 
304  call transfer_fields_cm_from_device(ndim*nglob_crust_mantle, &
305  displ_crust_mantle,veloc_crust_mantle,accel_crust_mantle, &
306  mesh_pointer)
307  call transfer_fields_ic_from_device(ndim*nglob_inner_core, &
308  displ_inner_core,veloc_inner_core,accel_inner_core, &
309  mesh_pointer)
310  call transfer_fields_oc_from_device(nglob_outer_core, &
311  displ_outer_core,veloc_outer_core,accel_outer_core, &
312  mesh_pointer)
313 
314  call transfer_strain_cm_from_device(mesh_pointer,eps_trace_over_3_crust_mantle, &
315  epsilondev_xx_crust_mantle,epsilondev_yy_crust_mantle, &
316  epsilondev_xy_crust_mantle,epsilondev_xz_crust_mantle, &

318  call transfer_strain_ic_from_device(mesh_pointer,eps_trace_over_3_inner_core, &
319  epsilondev_xx_inner_core,epsilondev_yy_inner_core, &
320  epsilondev_xy_inner_core,epsilondev_xz_inner_core, &
321  epsilondev_yz_inner_core)
322 
323  if (rotation_val) then
324  call transfer_rotation_from_device(mesh_pointer,a_array_rotation,b_array_rotation)
325  endif
326 
327  else if (simulation_type == 3) then
328 
329  ! note: for kernel simulations (SIMULATION_TYPE == 3), attenuation is
330  ! only mimicking effects on phase shifts, but not on amplitudes.
331  ! flag PARTIAL_PHYS_DISPERSION_ONLY will have to be set to true in this case.
332  !
333  ! arrays b_R_xx, ... are not used when PARTIAL_PHYS_DISPERSION_ONLY is set,
334  ! therefore no need to transfer arrays from GPU to CPU
335  !if (ATTENUATION) then
336  !endif
337 
338  ! to store kernels
339  ! inner core
340  call transfer_kernels_ic_to_host(mesh_pointer, &
341  rho_kl_inner_core, &
342  alpha_kl_inner_core, &
343  beta_kl_inner_core,nspec_inner_core)
344  ! outer core
345  call transfer_kernels_oc_to_host(mesh_pointer, &
346  rho_kl_outer_core, &
347  alpha_kl_outer_core,nspec_outer_core)
348  ! crust/mantle
349  call transfer_kernels_cm_to_host(mesh_pointer, &
350  rho_kl_crust_mantle,alpha_kl_crust_mantle,beta_kl_crust_mantle, &
351  nspec_crust_mantle)
352  ! full anisotropic kernel
353  if (anisotropic_kl) then
354  call transfer_kernels_ani_cm_to_host(mesh_pointer,cijkl_kl_crust_mantle,nspec_crust_mantle)
355  endif
356 
357  ! specific noise strength kernel
358  if (noise_tomography == 3) then
359  call transfer_kernels_noise_to_host(mesh_pointer,sigma_kl_crust_mantle,nspec_crust_mantle)
360  endif
361 
362  ! approximative Hessian for preconditioning kernels
363  if (approximate_hess_kl) then
364  call transfer_kernels_hess_cm_tohost(mesh_pointer,hess_kl_crust_mantle,nspec_crust_mantle)
365  endif
366 
367  endif
368 
369  ! from here on, no gpu data is needed anymore
370  ! frees allocated memory on GPU
371  call prepare_cleanup_device(mesh_pointer,nchunks_val)
372 
373  end subroutine it_transfer_from_gpu

427  vtkdata_numpoints_all,nproctot_val)

432  call gatherv_all_r(vtkdata,data_size, &
433  dummy,vtkdata_points_all,vtkdata_offset_all, &
434  1,nproctot_val)
435  endif
436  else
437  ! serial run
438  ! updates VTK window
439  call visualize_vtkdata(it,currenttime,vtkdata)
440  endif
441 
442  endif
443 
444  end subroutine it_update_vtkwindow

463  subroutine setup_exact_undoing_to_disk()
464 
465  use specfem_par
466  use specfem_par_crustmantle
467 
468  implicit none
469 
470  ! local parameters
471  integer :: ispec,iglob,i,j,k
472  integer :: counter,record_length
473  real(kind=CUSTOM_REAL) :: radius
474  character(len=MAX_STRING_LEN) :: outputname
475 
476  ! checks if anything to do
477  if (.not. exact_undoing_to_disk) return
478 
479  ! checks flags
480  if (gpu_mode) call exit_mpi(myrank,'EXACT_UNDOING_TO_DISK not supported for GPUs')
481 
482  if (undo_attenuation) &
483  call exit_mpi(myrank,'EXACT_UNDOING_TO_DISK needs UNDO_ATTENUATION to be off because it computes the kernel directly instead')
484 
485  if (simulation_type == 1 .and. .not. save_forward) &
486  call exit_mpi(myrank,'EXACT_UNDOING_TO_DISK requires SAVE_FORWARD if SIMULATION_TYPE == 1')
487 
488  if (anisotropic_kl) call exit_mpi(myrank,'EXACT_UNDOING_TO_DISK requires ANISOTROPIC_KL to be turned off')
489 
490  if (simulation_type /= 1 .and. simulation_type /= 3) &
491  call exit_mpi(myrank,'EXACT_UNDOING_TO_DISK can only be used with SIMULATION_TYPE == 1 or SIMULATION_TYPE == 3')
492 
493 
494 !! DK DK determine the largest value of iglob that we need to save to disk,
495 !! DK DK since we save the upper part of the mesh only in the case of surface-wave kernels
496  ! crust_mantle
497  allocate(integer_mask_ibool_exact_undo(nglob_crust_mantle))
498  integer_mask_ibool_exact_undo(:) = -1
499 
500  counter = 0
501  do ispec = 1, nspec_crust_mantle
502  do k = 1, ngllz
503  do j = 1, nglly
504  do i = 1, ngllx
505  iglob = ibool_crust_mantle(i,j,k,ispec)
506  ! xstore ystore zstore have previously been converted to r theta phi in rstore
507  radius = rstore_crust_mantle(1,iglob) ! radius r (normalized)
508  ! save that element only if it is in the upper part of the mesh
509  if (radius >= r670 / r_earth) then
510  ! if this point has not yet been found before
511  if (integer_mask_ibool_exact_undo(iglob) == -1) then
512  ! create a new unique point
513  counter = counter + 1
514  integer_mask_ibool_exact_undo(iglob) = counter
515  endif
516  endif
517  enddo
518  enddo
519  enddo
520  enddo
521 
522  ! allocate the buffer used to dump a single time step
523  allocate(buffer_for_disk(counter,nstep_for_exact_undoing))
524 
525  ! open the file in which we will dump all the time steps (in a single file)
526  write(outputname,"('huge_dumps/proc',i6.6,'_huge_dump_of_all_time_steps.bin')") myrank
527  inquire(iolength=record_length) buffer_for_disk(:,1)
528  ! we write to or read from the file depending on the simulation type
529  if (simulation_type == 1) then
530  open(file=outputname, unit=ifile_for_exact_undoing, action='write', status='unknown', &
531  form='unformatted', access='direct', recl=record_length)
532  else if (simulation_type == 3) then
533  open(file=outputname, unit=ifile_for_exact_undoing, action='read', status='old', &
534  form='unformatted', access='direct', recl=record_length)
535  endif
536 
537  if (simulation_type == 1) then
538  it_of_this_exact_subset = 1
539  it_exact_subset_offset = it_begin - 1
540  it_exact_subset_end = min(nstep_for_exact_undoing, it_end - it_begin + 1)
541  else if (simulation_type == 3) then
542  ! Trigger a read at the start of the loop
543  it_of_this_exact_subset = 0
544  it_exact_subset_offset = it_begin - 1
545  it_exact_subset_end = 0
546  endif
547 
548  end subroutine setup_exact_undoing_to_disk
