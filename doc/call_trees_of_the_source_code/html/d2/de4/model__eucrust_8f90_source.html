190  double precision function crust_eu(lat,lon,x,vp,found_crust)
191 
192 ! returns Vp at the specific location lat/lon
193 
194  use constants
195  use model_eucrust_par
196 
197  implicit none
198 
199  double precision :: lat,lon,x,vp !,vs,rho,moho
200  logical :: found_crust
201 
202  double precision :: h_basement,h_uc,h_moho,x3,x4,x5
203  double precision :: scaleval
204 
205  integer :: i,j
206  integer,parameter :: ilons = 242 ! number of different longitudes
207  integer,parameter :: ilats = 149 ! number of different latitudes
208 
209  ! checks region range
210  found_crust = .false.
211  crust_eu = 0.0
212  if (lon < longitude_min .or. lon > longitude_max ) return
213  if (lat < latitude_min .or. lat > latitude_max ) return
214 
215  ! search
216  do i = 1,ilons-1
217  if (lon >= eucrust_lon(i) .and. lon < eucrust_lon(i+1)) then
218  do j = 0,ilats-1
219  if (lat >= eucrust_lat(i+j*ilons) .and. lat < eucrust_lat(i+(j+1)*ilons)) then
220 
221  h_basement = eucrust_basement(i+j*ilons)
222  h_uc = eucrust_ucdepth(i+j*ilons)
223  h_moho = eucrust_mohodepth(i+j*ilons)
224 
225  x3=(r_earth - h_basement*1000.0d0)/r_earth
226  x4=(r_earth - h_uc*1000.0d0)/r_earth
227  x5=(r_earth - h_moho*1000.0d0)/r_earth
228 
229  scaleval = dsqrt(pi*grav*rhoav)
230 
231  if (x > x3 .and. include_sediments_in_crust &
232  .and. h_basement > minimum_sediment_thickness) then
233  ! above sediment basement, returns average upper crust value
234  ! since no special sediment values are given
235  found_crust = .true.
236  vp = eucrust_vp_uppercrust(i+j*ilons) *1000.0d0/(r_earth*scaleval)
237  crust_eu = vp
238  return
239  else if (x > x4) then
240  found_crust = .true.
241  vp = eucrust_vp_uppercrust(i+j*ilons) *1000.0d0/(r_earth*scaleval)
242  crust_eu = vp
243  return
244  else if (x > x5) then
245  found_crust = .true.
246  vp = eucrust_vp_lowercrust(i+j*ilons) *1000.0d0/(r_earth*scaleval)
247  crust_eu = vp
248  return
249  endif
250  return
251  endif
252  enddo
253  endif
254  enddo
255 
256  end function crust_eu

261  subroutine eu_cap_smoothing(lat,lon,radius,value,found)
262 
263 ! smooths with a cap of size CAP (in degrees)
264 ! using NTHETA points in the theta direction (latitudinal)
265 ! and NPHI in the phi direction (longitudinal).
266 ! The cap is rotated to the North Pole.
267 
268  use constants
269  use model_eucrust_par
270 
271  implicit none
272 
273  ! argument variables
274  double precision lat,lon,radius
275  double precision :: value
276  logical :: found
277 
278  integer, parameter :: NTHETA = 4
279  integer, parameter :: NPHI = 10
280  double precision, parameter :: CAP = 1.0d0 * degrees_to_radians ! 1 degree smoothing
281 
282  double precision,external :: crust_eu
283 
284  ! local variables
285  integer i,j,k !,icolat,ilon,ier
286  integer itheta,iphi,npoints
287  double precision theta,phi,sint,cost,sinp,cosp,dtheta,dphi,cap_area,wght,total,valuel
288  double precision r_rot,theta_rot,phi_rot
289  double precision rotation_matrix(3,3),x(3),xc(3)
290  double precision xlon(ntheta*nphi),xlat(ntheta*nphi),weight(ntheta*nphi)
291 
292  ! get integer colatitude and longitude of crustal cap
293  ! -90 < lat < 90 -180 < lon < 180
294  if (lat > 90.0d0 .or. lat < -90.0d0 .or. lon > 180.0d0 .or. lon < -180.0d0) &
295  stop 'Error in latitude/longitude range in crust'
296  if (lat == 90.0d0) lat=89.9999d0
297  if (lat == -90.0d0) lat=-89.9999d0
298  if (lon == 180.0d0) lon=179.9999d0
299  if (lon == -180.0d0) lon=-179.9999d0
300 
301  !call icolat_ilon(lat,lon,icolat,ilon)
302  !crustaltype=abbreviation(icolat,ilon)
303  !call get_crust_structure(crustaltype,velp,vels,rho,thick,code,thlr,velocp,velocs,dens,ier)
304 
305  ! uncomment the following line to use as is, without smoothing
306  ! value = func(lat,lon,x,value,found)
307  ! return
308 
309  theta = (90.0-lat)*degrees_to_radians
310  phi = lon*degrees_to_radians
311 
312  sint = sin(theta)
313  cost = cos(theta)
314  sinp = sin(phi)
315  cosp = cos(phi)
316 
317  ! set up rotation matrix to go from cap at North pole
318  ! to cap around point of interest
319  rotation_matrix(1,1) = cosp*cost
320  rotation_matrix(1,2) = -sinp
321  rotation_matrix(1,3) = cosp*sint
322  rotation_matrix(2,1) = sinp*cost
323  rotation_matrix(2,2) = cosp
324  rotation_matrix(2,3) = sinp*sint
325  rotation_matrix(3,1) = -sint
326  rotation_matrix(3,2) = 0.0
327  rotation_matrix(3,3) = cost
328 
329  dtheta = cap/dble(ntheta)
330  dphi = two_pi/dble(nphi)
331  cap_area = two_pi*(1.0-cos(cap))
332 
333  ! integrate over a cap at the North pole
334  i = 0
335  total = 0.0
336  do itheta = 1,ntheta
337 
338  theta = 0.5*dble(2*itheta-1)*cap/dble(ntheta)
339  cost = cos(theta)
340  sint = sin(theta)
341  wght = sint*dtheta*dphi/cap_area
342 
343  do iphi = 1,nphi
344 
345  i = i+1
346  !get the weight associated with this integration point (same for all phi)
347  weight(i) = wght
348  total = total + weight(i)
349  phi = dble(2*iphi-1)*pi/dble(nphi)
350  cosp = cos(phi)
351  sinp = sin(phi)
352  ! x,y,z coordinates of integration point in cap at North pole
353  xc(1) = sint*cosp
354  xc(2) = sint*sinp
355  xc(3) = cost
356  ! get x,y,z coordinates in cap around point of interest
357  do j=1,3
358  x(j) = 0.0
359  do k=1,3
360  x(j) = x(j)+rotation_matrix(j,k)*xc(k)
361  enddo
362  enddo
363  ! get latitude and longitude (degrees) of integration point
364  call xyz_2_rthetaphi_dble(x(1),x(2),x(3),r_rot,theta_rot,phi_rot)
365  call reduce(theta_rot,phi_rot)
366  xlat(i) = (pi_over_two-theta_rot)*radians_to_degrees
367  xlon(i) = phi_rot*radians_to_degrees
368  if (xlon(i) > 180.0) xlon(i) = xlon(i)-360.0
369 
370  enddo
371 
372  enddo
373 
374  if (abs(total-1.0) > 0.001) stop 'Error in cap integration for crust2.0'
375 
376  npoints = i
377 
378  ! at this point:
379  !
380  ! xlat(i),xlon(i) are point locations to be used for interpolation
381  ! with weights weight(i)
382 
383  ! integrates value
384  value = 0.0d0
385  do i = 1,npoints
386  valuel = crust_eu(xlat(i),xlon(i),radius,value,found)
387  value = value + weight(i)*valuel
388  enddo
389 
390  if (abs(value) < tinyval) found = .false.
391 
392  end subroutine eu_cap_smoothing
