191  double precision function crust_eu(lat,lon,x,vp,found_crust)
192 
193 ! returns Vp at the specific location lat/lon
194 
195  use constants
196  use model_eucrust_par
197 
198  implicit none
199 
200  double precision :: lat,lon,x,vp !,vs,rho,moho
201  logical :: found_crust
202 
203  double precision :: h_basement,h_uc,h_moho,x3,x4,x5
204  double precision :: scaleval
205 
206  integer :: i,j
207  integer,parameter :: ilons = 242 ! number of different longitudes
208  integer,parameter :: ilats = 149 ! number of different latitudes
209 
210  ! checks region range
211  found_crust = .false.
212  crust_eu = 0.0
213  if (lon < longitude_min .or. lon > longitude_max ) return
214  if (lat < latitude_min .or. lat > latitude_max ) return
215 
216  ! search
217  do i = 1,ilons-1
218  if (lon >= eucrust_lon(i) .and. lon < eucrust_lon(i+1)) then
219  do j = 0,ilats-1
220  if (lat >= eucrust_lat(i+j*ilons) .and. lat < eucrust_lat(i+(j+1)*ilons)) then
221 
222  h_basement = eucrust_basement(i+j*ilons)
223  h_uc = eucrust_ucdepth(i+j*ilons)
224  h_moho = eucrust_mohodepth(i+j*ilons)
225 
226  x3=(r_earth - h_basement*1000.0d0)/r_earth
227  x4=(r_earth - h_uc*1000.0d0)/r_earth
228  x5=(r_earth - h_moho*1000.0d0)/r_earth
229 
230  scaleval = dsqrt(pi*grav*rhoav)
231 
232  if (x > x3 .and. include_sediments_in_crust &
233  .and. h_basement > minimum_sediment_thickness) then
234  ! above sediment basement, returns average upper crust value
235  ! since no special sediment values are given
236  found_crust = .true.
237  vp = eucrust_vp_uppercrust(i+j*ilons) *1000.0d0/(r_earth*scaleval)
238  crust_eu = vp
239  return
240  else if (x > x4) then
241  found_crust = .true.
242  vp = eucrust_vp_uppercrust(i+j*ilons) *1000.0d0/(r_earth*scaleval)
243  crust_eu = vp
244  return
245  else if (x > x5) then
246  found_crust = .true.
247  vp = eucrust_vp_lowercrust(i+j*ilons) *1000.0d0/(r_earth*scaleval)
248  crust_eu = vp
249  return
250  endif
251  return
252  endif
253  enddo
254  endif
255  enddo
256 
257  end function crust_eu

262  subroutine eu_cap_smoothing(lat,lon,radius,value,found)
263 
264 ! smooths with a cap of size CAP (in degrees)
265 ! using NTHETA points in the theta direction (latitudinal)
266 ! and NPHI in the phi direction (longitudinal).
267 ! The cap is rotated to the North Pole.
268 
269  use constants
270  use model_eucrust_par
271 
272  implicit none
273 
274  ! argument variables
275  double precision lat,lon,radius
276  double precision :: value
277  logical :: found
278 
279  integer, parameter :: NTHETA = 4
280  integer, parameter :: NPHI = 10
281  double precision, parameter :: CAP = 1.0d0 * degrees_to_radians ! 1 degree smoothing
282 
283  double precision,external :: crust_eu
284 
285  ! local variables
286  integer i,j,k !,icolat,ilon,ier
287  integer itheta,iphi,npoints
288  double precision theta,phi,sint,cost,sinp,cosp,dtheta,dphi,cap_area,wght,total,valuel
289  double precision r_rot,theta_rot,phi_rot
290  double precision rotation_matrix(3,3),x(3),xc(3)
291  double precision xlon(ntheta*nphi),xlat(ntheta*nphi),weight(ntheta*nphi)
292 
293  ! get integer colatitude and longitude of crustal cap
294  ! -90 < lat < 90 -180 < lon < 180
295  if (lat > 90.0d0 .or. lat < -90.0d0 .or. lon > 180.0d0 .or. lon < -180.0d0) &
296  stop 'Error in latitude/longitude range in crust'
297  if (lat == 90.0d0) lat=89.9999d0
298  if (lat == -90.0d0) lat=-89.9999d0
299  if (lon == 180.0d0) lon=179.9999d0
300  if (lon == -180.0d0) lon=-179.9999d0
301 
302  !call icolat_ilon(lat,lon,icolat,ilon)
303  !crustaltype=abbreviation(icolat,ilon)
304  !call get_crust_structure(crustaltype,velp,vels,rho,thick,code,thlr,velocp,velocs,dens,ier)
305 
306  ! uncomment the following line to use as is, without smoothing
307  ! value = func(lat,lon,x,value,found)
308  ! return
309 
310  theta = (90.0-lat)*degrees_to_radians
311  phi = lon*degrees_to_radians
312 
313  sint = sin(theta)
314  cost = cos(theta)
315  sinp = sin(phi)
316  cosp = cos(phi)
317 
318  ! set up rotation matrix to go from cap at North pole
319  ! to cap around point of interest
320  rotation_matrix(1,1) = cosp*cost
321  rotation_matrix(1,2) = -sinp
322  rotation_matrix(1,3) = cosp*sint
323  rotation_matrix(2,1) = sinp*cost
324  rotation_matrix(2,2) = cosp
325  rotation_matrix(2,3) = sinp*sint
326  rotation_matrix(3,1) = -sint
327  rotation_matrix(3,2) = 0.0
328  rotation_matrix(3,3) = cost
329 
330  dtheta = cap/dble(ntheta)
331  dphi = two_pi/dble(nphi)
332  cap_area = two_pi*(1.0-cos(cap))
333 
334  ! integrate over a cap at the North pole
335  i = 0
336  total = 0.0
337  do itheta = 1,ntheta
338 
339  theta = 0.5*dble(2*itheta-1)*cap/dble(ntheta)
340  cost = cos(theta)
341  sint = sin(theta)
342  wght = sint*dtheta*dphi/cap_area
343 
344  do iphi = 1,nphi
345 
346  i = i+1
347  !get the weight associated with this integration point (same for all phi)
348  weight(i) = wght
349  total = total + weight(i)
350  phi = dble(2*iphi-1)*pi/dble(nphi)
351  cosp = cos(phi)
352  sinp = sin(phi)
353  ! x,y,z coordinates of integration point in cap at North pole
354  xc(1) = sint*cosp
355  xc(2) = sint*sinp
356  xc(3) = cost
357  ! get x,y,z coordinates in cap around point of interest
358  do j=1,3
359  x(j) = 0.0
360  do k=1,3
361  x(j) = x(j)+rotation_matrix(j,k)*xc(k)
362  enddo
363  enddo
364  ! get latitude and longitude (degrees) of integration point
365  call xyz_2_rthetaphi_dble(x(1),x(2),x(3),r_rot,theta_rot,phi_rot)
366  call reduce(theta_rot,phi_rot)
367  xlat(i) = (pi_over_two-theta_rot)*radians_to_degrees
368  xlon(i) = phi_rot*radians_to_degrees
369  if (xlon(i) > 180.0) xlon(i) = xlon(i)-360.0
370 
371  enddo
372 
373  enddo
374 
375  if (abs(total-1.0) > 0.001) stop 'Error in cap integration for crust2.0'
376 
377  npoints = i
378 
379  ! at this point:
380  !
381  ! xlat(i),xlon(i) are point locations to be used for interpolation
382  ! with weights weight(i)
383 
384  ! integrates value
385  value = 0.0d0
386  do i = 1,npoints
387  valuel = crust_eu(xlat(i),xlon(i),radius,value,found)
388  value = value + weight(i)*valuel
389  enddo
390 
391  if (abs(value) < tinyval) found = .false.
392 
393  end subroutine eu_cap_smoothing
