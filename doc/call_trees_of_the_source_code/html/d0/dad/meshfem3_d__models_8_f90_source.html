239  subroutine meshfem3d_models_get1d_val(myrank,iregion_code,idoubling, &
240  r_prem,rho,vpv,vph,vsv,vsh,eta_aniso, &
241  qkappa,qmu,ricb,rcmb, &
242  rtopddoubleprime,r80,r120,r220,r400,r600,r670,r771, &
243  rmoho,rmiddle_crust,rocean)
244 ! reference model values
245 !
246 ! for a given location radius (r_prem, which is the point's radius with tolerance factor),
247 ! this calculates density and velocities
248 !
249 ! note: if CRUSTAL is set, it strips the 1-D crustal profile and mantle gets expanded
250 ! up to the surface.
251 ! only exception is JP1D...
252 !
253 ! routine returns: rho,vpv,vph,vsv,vsh,eta_aniso,Qkappa,Qmu
254 
255  use meshfem3d_models_par
256 
257  implicit none
258 
259  integer myrank,iregion_code,idoubling
260  double precision r_prem,rho
261  double precision vpv,vph,vsv,vsh,eta_aniso
262  double precision Qkappa,Qmu
263  double precision RICB,RCMB,RTOPDDOUBLEPRIME,R80,R120,R220,R400, &
264  R600,R670,R771,RMOHO,RMIDDLE_CRUST,ROCEAN
265 
266  ! local parameters
267  double precision drhodr,vp,vs
268 
269 !---
270 !
271 ! ADD YOUR MODEL HERE
272 !
273 !---
274 
275  ! gets 1-D reference model parameters
276  select case (reference_1d_model)
277 
278  case (reference_model_prem)
279  ! PREM (by Dziewonski & Anderson) - used also as background for 3D models
280  if (transverse_isotropy) then
281  ! get the anisotropic PREM parameters
282  call model_prem_aniso(myrank,r_prem,rho,vpv,vph,vsv,vsh,eta_aniso, &
283  qkappa,qmu,idoubling,crustal,one_crust,ricb,rcmb,rtopddoubleprime, &
284  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
285  else
286  ! isotropic model
287  call model_prem_iso(myrank,r_prem,rho,drhodr,vp,vs,qkappa,qmu,idoubling,crustal, &
288  one_crust,.true.,ricb,rcmb,rtopddoubleprime, &
289  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
290  endif
291 
292  case (reference_model_1dref)
293  ! 1D-REF also known as STW105 (by Kustowski et al.) - used also as background for 3D models
294  call model_1dref(r_prem,rho,vpv,vph,vsv,vsh,eta_aniso,qkappa,qmu,iregion_code,crustal)
295  if (.not. transverse_isotropy) then
296  if (.not. isotropic_3d_mantle) then
297  ! this case here is only executed for 1D_ref_iso
298  ! calculates isotropic values
299  vp = sqrt(((8.d0+4.d0*eta_aniso)*vph*vph + 3.d0*vpv*vpv &
300  + (8.d0 - 8.d0*eta_aniso)*vsv*vsv)/15.d0)
301  vs = sqrt(((1.d0-2.d0*eta_aniso)*vph*vph + vpv*vpv &
302  + 5.d0*vsh*vsh + (6.d0+4.d0*eta_aniso)*vsv*vsv)/15.d0)
303  endif
304  endif
305 
306  case (reference_model_1066a)
307  ! 1066A (by Gilbert & Dziewonski) - pure isotropic model, used in 1D model mode only
308  call model_1066a(r_prem,rho,vp,vs,qkappa,qmu,iregion_code)
309 
310  case (reference_model_ak135f_no_mud)
311  ! AK135 (by Kennett et al. ) - pure isotropic model, used in 1D model mode only
312  call model_ak135(r_prem,rho,vp,vs,qkappa,qmu,iregion_code)
313 
314  case (reference_model_iasp91)
315  ! IASP91 (by Kennett & Engdahl) - pure isotropic model, used in 1D model mode only
316  call model_iasp91(myrank,r_prem,rho,vp,vs,qkappa,qmu,idoubling, &
317  one_crust,.true.,ricb,rcmb,rtopddoubleprime, &
318  r771,r670,r400,r220,r120,rmoho,rmiddle_crust)
319 
320  case (reference_model_jp1d)
321  !JP1D (by Zhao et al.) - pure isotropic model, used also as background for 3D models
322  call model_jp1d(myrank,r_prem,rho,vp,vs,qkappa,qmu,idoubling, &
323  .true.,ricb,rcmb,rtopddoubleprime, &
324  r670,r220,r771,r400,r80,rmoho,rmiddle_crust)
325 
326  case (reference_model_sea1d)
327  ! SEA1D (by Lebedev & Nolet) - pure isotropic model, used also as background for 3D models
328  call model_sea1d(r_prem,rho,vp,vs,qkappa,qmu,iregion_code)
329 
330  case default
331  stop 'unknown 1D reference Earth model in meshfem3D_models_get1D_val()'
332 
333  end select
334 
335  ! needs to set vpv,vph,vsv,vsh and eta_aniso for isotropic models
336  if (.not. transverse_isotropy) then
337  ! in the case of s362iso we want to save the anisotropic constants for the Voight average
338  if (.not. (reference_1d_model == reference_model_1dref .and. isotropic_3d_mantle)) then
339  vpv = vp
340  vph = vp
341  vsv = vs
342  vsh = vs
343  eta_aniso = 1.d0
344  endif
345  endif ! TRANSVERSE_ISOTROPY
346 
347  end subroutine meshfem3d_models_get1d_val

354  subroutine meshfem3d_models_get3dmntl_val(iregion_code,r_prem,rho,dvp, &
355  vpv,vph,vsv,vsh,eta_aniso, &
356  rcmb,r670,rmoho, &
357  xmesh,ymesh,zmesh,r, &
358  c11,c12,c13,c14,c15,c16,c22,c23,c24,c25,c26, &
359  c33,c34,c35,c36,c44,c45,c46,c55,c56,c66 &
360 #ifdef CEM
361  ,ispec,i,j,k &
362 #endif
363  )
364 
365  use meshfem3d_models_par
366 
367  implicit none
368 
369  integer, intent (in) :: iregion_code
370  double precision :: r_prem
371  double precision :: rho,dvp
372  double precision :: vpv,vph,vsv,vsh,eta_aniso
373 
374  double precision :: RCMB,R670,RMOHO
375  double precision :: xmesh,ymesh,zmesh,r
376 
377  ! the 21 coefficients for an anisotropic medium in reduced notation
378  double precision :: c11,c12,c13,c14,c15,c16,c22,c23,c24,c25,c26,c33, &
379  c34,c35,c36,c44,c45,c46,c55,c56,c66
380 
381 #ifdef CEM
382  ! CEM needs these to determine iglob
383  integer, intent (in) :: ispec, i, j, k
384 #endif
385 
386  ! local parameters
387  double precision :: r_used,r_dummy,theta,phi
388  double precision :: dvs,drho,vp,vs
389  double precision :: dvpv,dvph,dvsv,dvsh,deta
390  double precision :: lat,lon
391 
392  real(kind=4) :: xcolat,xlon,xrad
393  real(kind=4) :: xdvpv,xdvph,xdvsv,xdvsh
394 
395  logical :: found_crust,suppress_mantle_extension
396 
397  ! initializes perturbation values
398  dvs = zero
399  dvp = zero
400  drho = zero
401 
402  dvpv = zero
403  dvph = zero
404  dvsv = zero
405  dvsh = zero
406  deta = zero
407 
408  xdvpv = 0.
409  xdvph = 0.
410  xdvsv = 0.
411  xdvsh = 0.
412 
413  r_used = zero
414  suppress_mantle_extension = .false.
415 
416  ! gets point's theta/phi
417  call xyz_2_rthetaphi_dble(xmesh,ymesh,zmesh,r_dummy,theta,phi)
418  call reduce(theta,phi)
419 
420 !---
421 !
422 ! ADD YOUR MODEL HERE
423 !
424 !---
425 
426  ! sets flag when mantle should not be extended to surface
427  if (r_prem >= rmoho/r_earth .and. .not. crustal) then
428  suppress_mantle_extension = .true.
429  endif
430 
431  ! gets parameters for isotropic 3D mantle model
432  !
433  ! note: there can be transverse isotropy in the mantle, but only Lam'e parameters
434  ! like kappav,kappah,muv,muh and eta_aniso are used for these simulations
435  !
436  ! note: in general, models here make use of perturbation values with respect to their
437  ! corresponding 1-D reference models
438  if (isotropic_3d_mantle .and. r_prem > rcmb/r_earth .and. .not. suppress_mantle_extension) then
439 
440  ! extend 3-D mantle model above the Moho to the surface before adding the crust
441  if (r_prem > rcmb/r_earth .and. r_prem < rmoho/r_earth) then
442  ! GLL point is in mantle region, takes exact location
443  r_used = r
444  else ! else if (r_prem >= RMOHO/R_EARTH) then
445  if (crustal) then
446  ! GLL point is above moho
447  ! takes radius slightly below moho radius, this will then "extend the mantle up to the surface";
448  ! crustal values will be superimposed later on
449  r_used = 0.999999d0*rmoho/r_earth
450  endif
451  endif
452 
453  ! gets model parameters
454  select case (three_d_model)
455 
456  case (three_d_model_s20rts)
457  ! s20rts
458  call mantle_s20rts(r_used,theta,phi,dvs,dvp,drho)
459  vpv = vpv*(1.0d0+dvp)
460  vph = vph*(1.0d0+dvp)
461  vsv = vsv*(1.0d0+dvs)
462  vsh = vsh*(1.0d0+dvs)
463  rho = rho*(1.0d0+drho)
464 
465  case (three_d_model_s40rts)
466  ! s40rts
467  call mantle_s40rts(r_used,theta,phi,dvs,dvp,drho)
468  vpv = vpv*(1.0d0+dvp)
469  vph = vph*(1.0d0+dvp)
470  vsv = vsv*(1.0d0+dvs)
471  vsh = vsh*(1.0d0+dvs)
472  rho = rho*(1.0d0+drho)
473 
474  case(three_d_model_mantle_sh)
475  ! full_sh model
476  lat = (pi/2.0d0-theta)*180.0d0/pi
477  lon = phi*180.0d0/pi
478  if (lon > 180.0d0) lon = lon - 360.0d0
479 
480  call mantle_sh(lat,lon,r_used,dvpv,dvph,dvsv,dvsh,deta,drho)
481  vpv = vpv*(1.0d0+dvpv)
482  vph = vph*(1.0d0+dvph)
483  vsv = vsv*(1.0d0+dvsv)
484  vsh = vsh*(1.0d0+dvsh)
485  eta_aniso = eta_aniso*(1.0d0+deta)
486  rho = rho*(1.0d0+drho)
487 
488  case (three_d_model_sea99_jp3d)
489  ! sea99 + jp3d1994
490  call model_sea99_s(r_used,theta,phi,dvs)
491  vsv = vsv*(1.0d0+dvs)
492  vsh = vsh*(1.0d0+dvs)
493  ! use Lebedev model sea99 as background and add vp & vs perturbation from Zhao 1994 model jp3d
494  if (theta >= (pi/2.d0 - jp3d_lat_max*degrees_to_radians) .and. theta <= (pi/2.d0 - jp3d_lat_min*degrees_to_radians) &
495  .and. phi >= jp3d_lon_min*degrees_to_radians .and. phi <= jp3d_lon_max*degrees_to_radians) then
496  ! makes sure radius is fine
497  if (r_used > (r_earth - jp3d_dep_max*1000.d0)/r_earth) then
498  call model_jp3d_iso_zhao(r_used,theta,phi,vp,vs,dvp,dvs,rho,found_crust)
499  vpv = vpv*(1.0d0+dvp)
500  vph = vph*(1.0d0+dvp)
501  vsv = vsv*(1.0d0+dvs)
502  vsh = vsh*(1.0d0+dvs)
503  endif
504  endif
505 
506  case (three_d_model_sea99)
507  ! sea99 Vs-only
508  call model_sea99_s(r_used,theta,phi,dvs)
509  vsv = vsv*(1.0d0+dvs)
510  vsh = vsh*(1.0d0+dvs)
511 
512  case (three_d_model_jp3d)
513  ! jp3d1994
514  if (theta >= (pi/2.d0 - jp3d_lat_max*degrees_to_radians) .and. theta <= (pi/2.d0 - jp3d_lat_min*degrees_to_radians) &
515  .and. phi >= jp3d_lon_min*degrees_to_radians .and. phi <= jp3d_lon_max*degrees_to_radians) then
516  if (r_used > (r_earth - jp3d_dep_max*1000.d0)/r_earth) then
517  call model_jp3d_iso_zhao(r_used,theta,phi,vp,vs,dvp,dvs,rho,found_crust)
518  vpv = vpv*(1.0d0+dvp)
519  vph = vph*(1.0d0+dvp)
520  vsv = vsv*(1.0d0+dvs)
521  vsh = vsh*(1.0d0+dvs)
522  endif
523  endif
524 
525  case (three_d_model_s362ani,three_d_model_s362wmani, &
526  three_d_model_s362ani_prem,three_d_model_s29ea)
527  ! 3D Harvard models s362ani, s362wmani, s362ani_prem and s2.9ea
528  xcolat = sngl(theta*180.0d0/pi)
529  xlon = sngl(phi*180.0d0/pi)
530  xrad = sngl(r_used*r_earth_km)
531  call model_s362ani_subshsv(xcolat,xlon,xrad,xdvsh,xdvsv,xdvph,xdvpv)
532 
533  ! to use speed values from the 1D reference model but with 3D mesh variations
534  if (use_1d_reference) then
535  ! sets all 3D variations in the mantle to zero
536  xdvpv = 0.d0
537  xdvph = 0.d0
538  xdvsv = 0.d0
539  xdvsh = 0.d0
540  endif
541 
542  if (transverse_isotropy) then
543  vpv = vpv*(1.0d0+dble(xdvpv))
544  vph = vph*(1.0d0+dble(xdvph))
545  vsv = vsv*(1.0d0+dble(xdvsv))
546  vsh = vsh*(1.0d0+dble(xdvsh))
547  else
548  vpv = vpv+xdvpv
549  vph = vph+xdvph
550  vsv = vsv+xdvsv
551  vsh = vsh+xdvsh
552  vp = sqrt(((8.d0+4.d0*eta_aniso)*vph*vph + 3.d0*vpv*vpv &
553  + (8.d0 - 8.d0*eta_aniso)*vsv*vsv)/15.d0)
554  vs = sqrt(((1.d0-2.d0*eta_aniso)*vph*vph + vpv*vpv &
555  + 5.d0*vsh*vsh + (6.d0+4.d0*eta_aniso)*vsv*vsv)/15.d0)
556  vpv = vp
557  vph = vp
558  vsv = vs
559  vsh = vs
560  eta_aniso=1.0d0
561  endif
562 
563  case (three_d_model_ppm )
564  ! point profile model
565  call model_ppm(r_used,theta,phi,dvs,dvp,drho)
566  vpv = vpv*(1.0d0+dvp)
567  vph = vph*(1.0d0+dvp)
568  vsv = vsv*(1.0d0+dvs)
569  vsh = vsh*(1.0d0+dvs)
570  rho = rho*(1.0d0+drho)
