236  subroutine meshfem3d_models_get1d_val(iregion_code,idoubling, &
237  r_prem,rho,vpv,vph,vsv,vsh,eta_aniso, &
238  qkappa,qmu,ricb,rcmb, &
239  rtopddoubleprime,r80,r120,r220,r400,r600,r670,r771, &
240  rmoho,rmiddle_crust,rocean)
241 ! reference model values
242 !
243 ! for a given location radius (r_prem, which is the point's radius with tolerance factor),
244 ! this calculates density and velocities
245 !
246 ! note: if CRUSTAL is set, it strips the 1-D crustal profile and mantle gets expanded
247 ! up to the surface.
248 ! only exception is JP1D...
249 !
250 ! routine returns: rho,vpv,vph,vsv,vsh,eta_aniso,Qkappa,Qmu
251 
252  use meshfem3d_models_par
253 
254  implicit none
255 
256  integer iregion_code,idoubling
257  double precision r_prem,rho
258  double precision vpv,vph,vsv,vsh,eta_aniso
259  double precision Qkappa,Qmu
260  double precision RICB,RCMB,RTOPDDOUBLEPRIME,R80,R120,R220,R400, &
261  R600,R670,R771,RMOHO,RMIDDLE_CRUST,ROCEAN
262 
263  ! local parameters
264  double precision drhodr,vp,vs
265 
266 !---
267 !
268 ! ADD YOUR MODEL HERE
269 !
270 !---
271 
272  ! gets 1-D reference model parameters
273  select case (reference_1d_model)
274 
275  case (reference_model_prem)
276  ! PREM (by Dziewonski & Anderson) - used also as background for 3D models
277  if (transverse_isotropy) then
278  ! get the anisotropic PREM parameters
279  call model_prem_aniso(r_prem,rho,vpv,vph,vsv,vsh,eta_aniso, &
280  qkappa,qmu,idoubling,crustal,one_crust,ricb,rcmb,rtopddoubleprime, &
281  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
282  else
283  ! isotropic model
284  call model_prem_iso(r_prem,rho,drhodr,vp,vs,qkappa,qmu,idoubling,crustal, &
285  one_crust,.true.,ricb,rcmb,rtopddoubleprime, &
286  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
287  endif
288 
289  case (reference_model_1dref)
290  ! 1D-REF also known as STW105 (by Kustowski et al.) - used also as background for 3D models
291  call model_1dref(r_prem,rho,vpv,vph,vsv,vsh,eta_aniso,qkappa,qmu,iregion_code,crustal)
292  if (.not. transverse_isotropy) then
293  if (.not. isotropic_3d_mantle) then
294  ! this case here is only executed for 1D_ref_iso
295  ! calculates isotropic values
296  vp = sqrt(((8.d0+4.d0*eta_aniso)*vph*vph + 3.d0*vpv*vpv &
297  + (8.d0 - 8.d0*eta_aniso)*vsv*vsv)/15.d0)
298  vs = sqrt(((1.d0-2.d0*eta_aniso)*vph*vph + vpv*vpv &
299  + 5.d0*vsh*vsh + (6.d0+4.d0*eta_aniso)*vsv*vsv)/15.d0)
300  endif
301  endif
302 
303  case (reference_model_1066a)
304  ! 1066A (by Gilbert & Dziewonski) - pure isotropic model, used in 1D model mode only
305  call model_1066a(r_prem,rho,vp,vs,qkappa,qmu,iregion_code)
306 
307  case (reference_model_ak135f_no_mud)
308  ! AK135 (by Kennett et al. ) - pure isotropic model, used in 1D model mode only
309  call model_ak135(r_prem,rho,vp,vs,qkappa,qmu,iregion_code)
310 
311  case (reference_model_iasp91)
312  ! IASP91 (by Kennett & Engdahl) - pure isotropic model, used in 1D model mode only
313  call model_iasp91(r_prem,rho,vp,vs,qkappa,qmu,idoubling, &
314  one_crust,.true.,ricb,rcmb,rtopddoubleprime, &
315  r771,r670,r400,r220,r120,rmoho,rmiddle_crust)
316 
317  case (reference_model_jp1d)
318  !JP1D (by Zhao et al.) - pure isotropic model, used also as background for 3D models
319  call model_jp1d(r_prem,rho,vp,vs,qkappa,qmu,idoubling, &
320  .true.,ricb,rcmb,rtopddoubleprime, &
321  r670,r220,r771,r400,r80,rmoho,rmiddle_crust)
322 
323  case (reference_model_sea1d)
324  ! SEA1D (by Lebedev & Nolet) - pure isotropic model, used also as background for 3D models
325  call model_sea1d(r_prem,rho,vp,vs,qkappa,qmu,iregion_code)
326 
327  case default
328  stop 'unknown 1D reference Earth model in meshfem3D_models_get1D_val()'
329 
330  end select
331 
332  ! needs to set vpv,vph,vsv,vsh and eta_aniso for isotropic models
333  if (.not. transverse_isotropy) then
334  ! in the case of s362iso we want to save the anisotropic constants for the Voight average
335  if (.not. (reference_1d_model == reference_model_1dref .and. isotropic_3d_mantle)) then
336  vpv = vp
337  vph = vp
338  vsv = vs
339  vsh = vs
340  eta_aniso = 1.d0
341  endif
342  endif ! TRANSVERSE_ISOTROPY
343 
344  end subroutine meshfem3d_models_get1d_val

351  subroutine meshfem3d_models_get3dmntl_val(iregion_code,r_prem,rho,dvp, &
352  vpv,vph,vsv,vsh,eta_aniso, &
353  rcmb,r670,rmoho, &
354  xmesh,ymesh,zmesh,r, &
355  c11,c12,c13,c14,c15,c16,c22,c23,c24,c25,c26, &
356  c33,c34,c35,c36,c44,c45,c46,c55,c56,c66 &
357 #ifdef CEM
358  ,ispec,i,j,k &
359 #endif
360  )
361 
362  use meshfem3d_models_par
363 
364  implicit none
365 
366  integer, intent (in) :: iregion_code
367  double precision :: r_prem
368  double precision :: rho,dvp
369  double precision :: vpv,vph,vsv,vsh,eta_aniso
370 
371  double precision :: RCMB,R670,RMOHO
372  double precision :: xmesh,ymesh,zmesh,r
373 
374  ! the 21 coefficients for an anisotropic medium in reduced notation
375  double precision :: c11,c12,c13,c14,c15,c16,c22,c23,c24,c25,c26,c33, &
376  c34,c35,c36,c44,c45,c46,c55,c56,c66
377 
378 #ifdef CEM
379  ! CEM needs these to determine iglob
380  integer, intent (in) :: ispec, i, j, k
381 #endif
382 
383  ! local parameters
384  double precision :: r_used,r_dummy,theta,phi
385  double precision :: dvs,drho,vp,vs
386  double precision :: dvpv,dvph,dvsv,dvsh,deta
387  double precision :: lat,lon
388 
389  real(kind=4) :: xcolat,xlon,xrad
390  real(kind=4) :: xdvpv,xdvph,xdvsv,xdvsh
391 
392  logical :: found_crust,suppress_mantle_extension
393 
394  ! initializes perturbation values
395  dvs = zero
396  dvp = zero
397  drho = zero
398 
399  dvpv = zero
400  dvph = zero
401  dvsv = zero
402  dvsh = zero
403  deta = zero
404 
405  xdvpv = 0.
406  xdvph = 0.
407  xdvsv = 0.
408  xdvsh = 0.
409 
410  r_used = zero
411  suppress_mantle_extension = .false.
412 
413  ! gets point's theta/phi
414  call xyz_2_rthetaphi_dble(xmesh,ymesh,zmesh,r_dummy,theta,phi)
415  call reduce(theta,phi)
416 
417 !---
418 !
419 ! ADD YOUR MODEL HERE
420 !
421 !---
422 
423  ! sets flag when mantle should not be extended to surface
424  if (r_prem >= rmoho/r_earth .and. .not. crustal) then
425  suppress_mantle_extension = .true.
426  endif
427 
428  ! gets parameters for isotropic 3D mantle model
429  !
430  ! note: there can be transverse isotropy in the mantle, but only Lam'e parameters
431  ! like kappav,kappah,muv,muh and eta_aniso are used for these simulations
432  !
433  ! note: in general, models here make use of perturbation values with respect to their
434  ! corresponding 1-D reference models
435  if (isotropic_3d_mantle .and. r_prem > rcmb/r_earth .and. .not. suppress_mantle_extension) then
436 
437  ! extend 3-D mantle model above the Moho to the surface before adding the crust
438  if (r_prem > rcmb/r_earth .and. r_prem < rmoho/r_earth) then
439  ! GLL point is in mantle region, takes exact location
440  r_used = r
441  else ! else if (r_prem >= RMOHO/R_EARTH) then
442  if (crustal) then
443  ! GLL point is above moho
444  ! takes radius slightly below moho radius, this will then "extend the mantle up to the surface";
445  ! crustal values will be superimposed later on
446  r_used = 0.999999d0*rmoho/r_earth
447  endif
448  endif
449 
450  ! gets model parameters
451  select case (three_d_model)
452 
453  case (three_d_model_s20rts)
454  ! s20rts
455  call mantle_s20rts(r_used,theta,phi,dvs,dvp,drho)
456  vpv = vpv*(1.0d0+dvp)
457  vph = vph*(1.0d0+dvp)
458  vsv = vsv*(1.0d0+dvs)
459  vsh = vsh*(1.0d0+dvs)
460  rho = rho*(1.0d0+drho)
461 
462  case (three_d_model_s40rts)
463  ! s40rts
464  call mantle_s40rts(r_used,theta,phi,dvs,dvp,drho)
465  vpv = vpv*(1.0d0+dvp)
466  vph = vph*(1.0d0+dvp)
467  vsv = vsv*(1.0d0+dvs)
468  vsh = vsh*(1.0d0+dvs)
469  rho = rho*(1.0d0+drho)
470 
471  case(three_d_model_mantle_sh)
472  ! full_sh model
473  lat = (pi/2.0d0-theta)*180.0d0/pi
474  lon = phi*180.0d0/pi
475  if (lon > 180.0d0) lon = lon - 360.0d0
476 
477  call mantle_sh(lat,lon,r_used,dvpv,dvph,dvsv,dvsh,deta,drho)
478  vpv = vpv*(1.0d0+dvpv)
479  vph = vph*(1.0d0+dvph)
480  vsv = vsv*(1.0d0+dvsv)
481  vsh = vsh*(1.0d0+dvsh)
482  eta_aniso = eta_aniso*(1.0d0+deta)
483  rho = rho*(1.0d0+drho)
484 
485  case (three_d_model_sea99_jp3d)
486  ! sea99 + jp3d1994
487  call model_sea99_s(r_used,theta,phi,dvs)
488  vsv = vsv*(1.0d0+dvs)
489  vsh = vsh*(1.0d0+dvs)
490  ! use Lebedev model sea99 as background and add vp & vs perturbation from Zhao 1994 model jp3d
491  if (theta >= (pi/2.d0 - jp3d_lat_max*degrees_to_radians) .and. theta <= (pi/2.d0 - jp3d_lat_min*degrees_to_radians) &
492  .and. phi >= jp3d_lon_min*degrees_to_radians .and. phi <= jp3d_lon_max*degrees_to_radians) then
493  ! makes sure radius is fine
494  if (r_used > (r_earth - jp3d_dep_max*1000.d0)/r_earth) then
495  call model_jp3d_iso_zhao(r_used,theta,phi,vp,vs,dvp,dvs,rho,found_crust)
496  vpv = vpv*(1.0d0+dvp)
497  vph = vph*(1.0d0+dvp)
498  vsv = vsv*(1.0d0+dvs)
499  vsh = vsh*(1.0d0+dvs)
500  endif
501  endif
502 
503  case (three_d_model_sea99)
504  ! sea99 Vs-only
505  call model_sea99_s(r_used,theta,phi,dvs)
506  vsv = vsv*(1.0d0+dvs)
507  vsh = vsh*(1.0d0+dvs)
508 
509  case (three_d_model_jp3d)
510  ! jp3d1994
511  if (theta >= (pi/2.d0 - jp3d_lat_max*degrees_to_radians) .and. theta <= (pi/2.d0 - jp3d_lat_min*degrees_to_radians) &
512  .and. phi >= jp3d_lon_min*degrees_to_radians .and. phi <= jp3d_lon_max*degrees_to_radians) then
513  if (r_used > (r_earth - jp3d_dep_max*1000.d0)/r_earth) then
514  call model_jp3d_iso_zhao(r_used,theta,phi,vp,vs,dvp,dvs,rho,found_crust)
515  vpv = vpv*(1.0d0+dvp)
516  vph = vph*(1.0d0+dvp)
517  vsv = vsv*(1.0d0+dvs)
518  vsh = vsh*(1.0d0+dvs)
519  endif
520  endif
521 
522  case (three_d_model_s362ani,three_d_model_s362wmani, &
523  three_d_model_s362ani_prem,three_d_model_s29ea)
524  ! 3D Harvard models s362ani, s362wmani, s362ani_prem and s2.9ea
525  xcolat = sngl(theta*180.0d0/pi)
526  xlon = sngl(phi*180.0d0/pi)
527  xrad = sngl(r_used*r_earth_km)
528  call model_s362ani_subshsv(xcolat,xlon,xrad,xdvsh,xdvsv,xdvph,xdvpv)
529 
530  ! to use speed values from the 1D reference model but with 3D mesh variations
531  if (use_1d_reference) then
532  ! sets all 3D variations in the mantle to zero
533  xdvpv = 0.d0
534  xdvph = 0.d0
535  xdvsv = 0.d0
536  xdvsh = 0.d0
537  endif
538 
539  if (transverse_isotropy) then
540  vpv = vpv*(1.0d0+dble(xdvpv))
541  vph = vph*(1.0d0+dble(xdvph))
542  vsv = vsv*(1.0d0+dble(xdvsv))
543  vsh = vsh*(1.0d0+dble(xdvsh))
544  else
545  vpv = vpv+xdvpv
546  vph = vph+xdvph
547  vsv = vsv+xdvsv
548  vsh = vsh+xdvsh
549  vp = sqrt(((8.d0+4.d0*eta_aniso)*vph*vph + 3.d0*vpv*vpv &
550  + (8.d0 - 8.d0*eta_aniso)*vsv*vsv)/15.d0)
551  vs = sqrt(((1.d0-2.d0*eta_aniso)*vph*vph + vpv*vpv &
552  + 5.d0*vsh*vsh + (6.d0+4.d0*eta_aniso)*vsv*vsv)/15.d0)
553  vpv = vp
554  vph = vp
555  vsv = vs
556  vsh = vs
557  eta_aniso=1.0d0
558  endif
559 
560  case (three_d_model_ppm )
561  ! point profile model
562  call model_ppm(r_used,theta,phi,dvs,dvp,drho)
563  vpv = vpv*(1.0d0+dvp)
564  vph = vph*(1.0d0+dvp)
565  vsv = vsv*(1.0d0+dvs)
566  vsh = vsh*(1.0d0+dvs)
567  rho = rho*(1.0d0+drho)
568 
569  case (three_d_model_gapp2 )
570  ! 3D GAP model (Obayashi)
571  call mantle_gapmodel(r_used,theta,phi,dvs,dvp,drho)
572  vpv = vpv*(1.0d0+dvp)
573  vph = vph*(1.0d0+dvp)
574  vsv = vsv*(1.0d0+dvs)
