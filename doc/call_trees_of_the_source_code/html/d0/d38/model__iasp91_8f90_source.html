38  subroutine model_iasp91(x,rho,vp,vs,Qkappa,Qmu,idoubling,ONE_CRUST,check_doubling_flag, &
39  ricb,rcmb,rtopddoubleprime,r771,r670,r400,r220,r120,rmoho,rmiddle_crust)
40 
41 
42 ! we use the density model of PREM (or close to PREM in the crust)
43 ! because IASP91 does not provide a density model.
44 ! Note that "ttimes" from the official IASP91 package uses a slightly different
45 ! model: scaling of the P wave velocity based on Birch's law. Both options are fine.
46  use constants
47 
48  implicit none
49 
50 ! given a normalized radius x, gives the non-dimensionalized density rho,
51 ! speeds vp and vs, and the quality factors Qkappa and Qmu
52 
53  logical check_doubling_flag
54 
55  integer idoubling
56 
57  double precision x,rho,vp,vs,Qkappa,Qmu,RICB,RCMB,RTOPDDOUBLEPRIME, &
58  R771,R670,R400,R220,R120,RMOHO,RMIDDLE_CRUST
59 
60  logical ONE_CRUST
61 
62  double precision r,scaleval
63 
64  double precision x1,x2
65 
66 ! compute real physical radius in meters
67  r = x * r_earth
68 
69  x1 = r120 / r_earth
70  x2 = rmoho / r_earth
71 
72  ! check flags to make sure we correctly honor the discontinuities
73  ! we use strict inequalities since r has been slightly changed in mesher
74  if (check_doubling_flag) then
75  !
76  !--- inner core
77  !
78  if (r >= 0.d0 .and. r < ricb) then
79 
80  if (idoubling /= iflag_inner_core_normal .and. &
81  idoubling /= iflag_middle_central_cube .and. &
82  idoubling /= iflag_bottom_central_cube .and. &
83  idoubling /= iflag_top_central_cube .and. &
84  idoubling /= iflag_in_fictitious_cube) &
85  call exit_mpi(myrank,'wrong doubling flag for inner core point')
86  !
87  !--- outer core
88  !
89  else if (r > ricb .and. r < rcmb) then
90 
91  if (idoubling /= iflag_outer_core_normal) &
92  call exit_mpi(myrank,'wrong doubling flag for outer core point')
93  !
94  !--- D" at the base of the mantle
95  !
96  else if (r > rcmb .and. r < rtopddoubleprime) then
97 
98  if (idoubling /= iflag_mantle_normal) &
99  call exit_mpi(myrank,'wrong doubling flag for D" point')
100  !
101  !--- mantle: from top of D" to d670
102  !
103  else if (r > rtopddoubleprime .and. r < r670) then
104 
105  if (idoubling /= iflag_mantle_normal) &
106  call exit_mpi(myrank,'wrong doubling flag for top D" to d670 point')
107  !
108  !--- mantle: from d670 to d220
109  !
110  else if (r > r670 .and. r < r220) then
111 
112  if (idoubling /= iflag_670_220) &
113  call exit_mpi(myrank,'wrong doubling flag for d670 to d220 point')
114  !
115  !--- mantle and crust: from d220 to MOHO and then to surface
116  !
117  else if (r > r220) then
118 
119  if (idoubling /= iflag_220_80 .and. idoubling /= iflag_80_moho .and. idoubling /= iflag_crust) &
120  call exit_mpi(myrank,'wrong doubling flag for d220 to Moho to surface point')
121 
122  endif ! r
123 
124  endif ! check_doubling_flag
125 
126 
127  ! assigns model values
128 
129  !
130  !--- inner core
131  !
132  if (r >= 0.d0 .and. r <= ricb) then
133  rho=13.0885d0-8.8381d0*x*x
134  vp=11.24094-4.09689*x**2
135  vs=3.56454-3.45241*x**2
136  qmu=84.6d0
137  qkappa=1327.7d0
138  !
139  !--- outer core
140  !
141  else if (r > ricb .and. r <= rcmb) then
142  rho=12.5815d0-1.2638d0*x-3.6426d0*x*x-5.5281d0*x*x*x
143  vp=10.03904+3.75665*x-13.67046*x**2
144  vs=0.0d0
145  qmu=0.0d0
146  qkappa=57827.0d0
147  !
148  !--- D" at the base of the mantle
149  !
150  else if (r > rcmb .and. r <= rtopddoubleprime) then
151  rho=7.9565d0-6.4761d0*x+5.5283d0*x*x-3.0807d0*x*x*x
152  vp=14.49470-1.47089*x
153  vs=8.16616-1.58206*x
154  qmu=312.0d0
155  qkappa=57827.0d0
156  !
157  !--- mantle: from top of D" to d670
158  !
159  else if (r > rtopddoubleprime .and. r <= r771) then
160  rho=7.9565d0-6.4761d0*x+5.5283d0*x*x-3.0807d0*x*x*x
161  vp=25.1486-41.1538*x+51.9932*x**2-26.6083*x**3
162  vs=12.9303-21.2590*x+27.8988*x**2-14.1080*x**3
163  qmu=312.0d0
164  qkappa=57827.0d0
165  else if (r > r771 .and. r <= r670) then
166  rho=7.9565d0-6.4761d0*x+5.5283d0*x*x-3.0807d0*x*x*x
167  vp=25.96984-16.93412*x
168  vs=20.76890-16.53147*x
169  qmu=312.0d0
170  qkappa=57827.0d0
171  !
172  !--- mantle: above d670
173  !
174  else if (r > r670 .and. r <= r400) then
175  rho=5.3197d0-1.4836d0*x
176  vp=29.38896-21.40656*x
177  vs=17.70732-13.50652*x
178  qmu=143.0d0
179  qkappa=57827.0d0
180 
181  else if (r > r400 .and. r <= r220) then
182  rho=7.1089d0-3.8045d0*x
183  vp=30.78765-23.25415*x
184  vs=15.24213-11.08552*x
185  qmu=143.0d0
186  qkappa=57827.0d0
187 
188  else if (r > r220 .and. r <= r120) then
189  rho=2.6910d0+0.6924d0*x
190  vp=25.41389-17.69722*x
191  vs=5.75020-1.27420*x
192  qmu=80.0d0
193  qkappa=57827.0d0
194 
195  else if (r > r120 .and. r <= rmoho) then
196  vp = 8.78541d0-0.74953d0*x
197  vs = 6.706231d0-2.248585d0*x
198  rho = 3.3713d0 + (3.3198d0-3.3713d0)*(x-x1)/(x2-x1)
199  if (rho < 3.30d0 .or. rho > 3.38d0) stop 'incorrect density computed for IASP91'
200  qmu=600.0d0
201  qkappa=57827.0d0
202 
203  else if (suppress_crustal_mesh) then
204  !! DK DK extend the Moho up to the surface instead of the crust
205  vp = 8.78541d0-0.74953d0*(rmoho / r_earth)
206  vs = 6.706231d0-2.248585d0*(rmoho / r_earth)
207  rho = 3.3198d0
208  qmu=600.0d0
209  qkappa=57827.0d0
210 
211  else if (r > rmoho .and. r <= rmiddle_crust) then
212  vp = 6.5d0
213  vs = 3.75d0
214  rho = 2.92d0
215  qmu=600.0d0
216  qkappa=57827.0d0
217 
218  ! same properties everywhere in PREM crust if we decide to define only one layer in the crust
219  if (one_crust) then
220  vp = 5.8d0
221  vs = 3.36d0
222  rho = 2.72d0
223  qmu=600.0d0
224  qkappa=57827.0d0
225  endif
226 
227  else
228  vp = 5.8d0
229  vs = 3.36d0
230  rho = 2.72d0
231  qmu=600.0d0
232  qkappa=57827.0d0
233  endif
234 
235  ! make sure Vs is zero in the outer core even if roundoff errors on depth
236  ! also set fictitious attenuation for Qkappa to a very high value (attenuation is not used in the fluid)
237  if (idoubling == iflag_outer_core_normal) then
238  vs = 0.d0
239  qkappa = 0.d0
240  qmu = attenuation_comp_maximum
241  endif
242 
243  ! non-dimensionalize
244  ! time scaling (s^{-1}) is done with scaleval
245  scaleval=dsqrt(pi*grav*rhoav)
246  rho=rho*1000.0d0/rhoav
247  vp=vp*1000.0d0/(r_earth*scaleval)
248  vs=vs*1000.0d0/(r_earth*scaleval)
249 
250  end subroutine model_iasp91
