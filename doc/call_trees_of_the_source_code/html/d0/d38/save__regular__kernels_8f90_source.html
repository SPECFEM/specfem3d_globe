205  ! Purpose : compute the kernels for the An coeffs (an_kl)
206  ! from the kernels for Cij (cijkl_kl_local)
207  ! At r,theta,phi fixed
208  ! kernel def : dx = kij * dcij + krho * drho
209  ! = kAn * dAn + krho * drho

211  ! Definition of the input array cij_kl :
212  ! cij_kl(1) = C11 ; cij_kl(2) = C12 ; cij_kl(3) = C13
213  ! cij_kl(4) = C14 ; cij_kl(5) = C15 ; cij_kl(6) = C16
214  ! cij_kl(7) = C22 ; cij_kl(8) = C23 ; cij_kl(9) = C24
215  ! cij_kl(10) = C25 ; cij_kl(11) = C26 ; cij_kl(12) = C33
216  ! cij_kl(13) = C34 ; cij_kl(14) = C35 ; cij_kl(15) = C36
217  ! cij_kl(16) = C44 ; cij_kl(17) = C45 ; cij_kl(18) = C46
218  ! cij_kl(19) = C55 ; cij_kl(20) = C56 ; cij_kl(21) = C66
219  ! where the Cij (Voigt's notation) are defined as function of
220  ! the components of the elastic tensor in spherical coordinates
221  ! by eq. (A.1) of Chen & Tromp, GJI 168 (2007)
222 
223  ! From the relations giving Cij in function of An
224  ! Checked with Min Chen's results (routine build_cij)
225

228  an_kl(3) = -2*cijkl_kl_local(2)+cijkl_kl_local(21) !N
229  an_kl(4) = cijkl_kl_local(16)+cijkl_kl_local(19) !L
230  an_kl(5) = cijkl_kl_local(3)+cijkl_kl_local(8) !F
231

234  !an_kl(7)=2*cijkl_kl_local(4)+2*cijkl_kl_local(9)+2*cijkl_kl_local(13) !Js
235  !an_kl(8)=-2*cijkl_kl_local(14) !Kc
236  !an_kl(9)=-2*cijkl_kl_local(13) !Ks
237  !an_kl(10)=-2*cijkl_kl_local(10)+cijkl_kl_local(18) !Mc
238  !an_kl(11)=2*cijkl_kl_local(4)-cijkl_kl_local(20) !Ms
239  !an_kl(12)=cijkl_kl_local(1)-cijkl_kl_local(7) !Bc
240  !an_kl(13)=-1./2.*(cijkl_kl_local(6)+cijkl_kl_local(11)) !Bs
241  !an_kl(14)=cijkl_kl_local(3)-cijkl_kl_local(8) !Hc
242  !an_kl(15)=-cijkl_kl_local(15) !Hs
243  !an_kl(16)=-cijkl_kl_local(16)+cijkl_kl_local(19) !Gc
244  !an_kl(17)=-cijkl_kl_local(17) !Gs
245  !an_kl(18)=cijkl_kl_local(5)-cijkl_kl_local(10)-cijkl_kl_local(18) !Dc
246  !an_kl(19)=cijkl_kl_local(4)-cijkl_kl_local(9)+cijkl_kl_local(20) !Ds
247  !an_kl(20)=cijkl_kl_local(1)-cijkl_kl_local(2)+cijkl_kl_local(7)-cijkl_kl_local(21) !Ec
248  !an_kl(21)=-cijkl_kl_local(6)+cijkl_kl_local(11) !Es
249 
250  ! K_rho (primary kernel, for a parameterization (A,C,L,N,F,rho) )
251  rhonotprime_kl = rhol * rho_kl / scale_kl_rho
252  !rhonotprime_kl_crust_mantle(ipoint) = rhonotprime_kl_crust_mantle(ipoint) + rhonotprime_kl
253 
254  ! note: transverse isotropic kernels are calculated for ALL elements,
255  ! and not just transverse elements
256  !
257  ! note: the kernels are for relative perturbations (delta ln (m_i) = (m_i - m_0)/m_i )
258  !
259  ! Gets transverse isotropic kernels
260  ! (see Appendix B of Sieminski et al., GJI 171, 2007)
261 
262  ! for parameterization: ( alpha_v, alpha_h, beta_v, beta_h, eta, rho )
263  ! K_alpha_v
264  alphav_kl = 2*c*an_kl(2) * hlagrange
265  alphav_kl_crust_mantle(ipoint) = alphav_kl_crust_mantle(ipoint) &
266  + alphav_kl
267  ! K_alpha_h
268  alphah_kl = (2*a*an_kl(1) + 2*a*eta*an_kl(5)) * hlagrange
269  alphah_kl_crust_mantle(ipoint) = alphah_kl_crust_mantle(ipoint) &
270  + alphah_kl
271  ! K_beta_v
272  betav_kl = (2*l*an_kl(4) - 4*l*eta*an_kl(5)) * hlagrange
273  betav_kl_crust_mantle(ipoint) = betav_kl_crust_mantle(ipoint) &
274  + betav_kl
275  ! K_beta_h
276  betah_kl = 2*n*an_kl(3) * hlagrange
277  betah_kl_crust_mantle(ipoint) = betah_kl_crust_mantle(ipoint) &
278  + betah_kl
279  ! K_eta
280  eta_kl_crust_mantle(ipoint) = eta_kl_crust_mantle(ipoint) + f*an_kl(5) * hlagrange
281  ! K_rhoprime (for a parameterization (alpha_v, ..., rho) )
282  rho_kl_crust_mantle_reg(ipoint) = rho_kl_crust_mantle_reg(ipoint) &
283  + (a*an_kl(1) + c*an_kl(2) &
284  + n*an_kl(3) + l*an_kl(4) + f*an_kl(5)) * hlagrange &
285  + rhonotprime_kl
286

290  betah_sq = muhl / rhol
291  alphav_sq = ( kappal + four_thirds * muvl ) / rhol
292  alphah_sq = ( kappal + four_thirds * muhl ) / rhol
293  bulk_sq = kappal / rhol
294 
295  bulk_c_kl_crust_mantle(ipoint) = bulk_c_kl_crust_mantle(ipoint) + &
296  bulk_sq / alphav_sq * alphav_kl + &
297  bulk_sq / alphah_sq * alphah_kl
298 
299  bulk_betah_kl_crust_mantle(ipoint) = bulk_betah_kl_crust_mantle(ipoint) + &
300  betah_kl + four_thirds * betah_sq / alphah_sq * alphah_kl
301 
302  bulk_betav_kl_crust_mantle(ipoint) = bulk_betav_kl_crust_mantle(ipoint) + &
303  betav_kl + four_thirds * betav_sq / alphav_sq * alphav_kl
304  ! the rest, K_eta and K_rho are the same as above
305  else
306 
307  rho_kl_crust_mantle_reg(ipoint) = rho_kl_crust_mantle_reg(ipoint) + rho_kl
308 
309  endif ! SAVE_TRANSVERSE_KL_ONLY
310 
311  else
312 
313  ! isotropic kernels
314 
315  rhol = rhostore_crust_mantle(i,j,k,ispec)

317  kappal = kappavstore_crust_mantle(i,j,k,ispec)
318

320  rho_kl = - rhol * rho_kl_crust_mantle(i,j,k,ispec)

322  beta_kl = - 2 * mul * beta_kl_crust_mantle(i,j,k,ispec) ! note: beta_kl corresponds to K_mu
323 
324  ! for a parameterization: (rho,mu,kappa) "primary" kernels

326  + rho_kl * scale_kl * hlagrange

328  kappa_kl_crust_mantle(ipoint) = kappa_kl_crust_mantle(ipoint) + alpha_kl * scale_kl * hlagrange
329 
330  ! for a parameterization: (rho,alpha,beta)
331  ! kernels rho^prime, beta, alpha
332  rho_kl_crust_mantle_reg(ipoint) = rho_kl_crust_mantle_reg(ipoint) &
333  + (rho_kl + alpha_kl + beta_kl) * scale_kl * hlagrange
334  beta_kl_crust_mantle_reg(ipoint) = beta_kl_crust_mantle_reg(ipoint) + &
335  2._custom_real * (beta_kl - four_thirds * mul * alpha_kl / kappal) * scale_kl * hlagrange
336  alpha_kl_crust_mantle_reg(ipoint) = alpha_kl_crust_mantle_reg(ipoint) + &
337  2._custom_real * (1._custom_real + four_thirds * mul / kappal) * alpha_kl * scale_kl * hlagrange
338 
339  ! for a parameterization: (rho,bulk, beta)
340  ! where bulk wave speed is c = sqrt( kappa / rho)
341  ! note: rhoprime is the same as for (rho,alpha,beta) parameterization
342  bulk_c_kl_crust_mantle(ipoint) = bulk_c_kl_crust_mantle(ipoint) &
343  + 2._custom_real * alpha_kl * scale_kl * hlagrange
344  bulk_beta_kl_crust_mantle(ipoint) = bulk_beta_kl_crust_mantle(ipoint) &
345  + 2._custom_real * beta_kl * scale_kl * hlagrange
346 
347  endif
348 
349  enddo
350  enddo
351  enddo

358

360  alphah_kl_crust_mantle(ipoint) = - alphah_kl_crust_mantle(ipoint) * scale_kl

362  betah_kl_crust_mantle(ipoint) = - betah_kl_crust_mantle(ipoint) * scale_kl
363  eta_kl_crust_mantle(ipoint) = - eta_kl_crust_mantle(ipoint) * scale_kl
364  rho_kl_crust_mantle_reg(ipoint) = - rho_kl_crust_mantle_reg(ipoint) * scale_kl
365

368  + alphah_kl_crust_mantle(ipoint)
369  beta_kl_crust_mantle_reg(ipoint) = betav_kl_crust_mantle(ipoint) &
370  + betah_kl_crust_mantle(ipoint)
371  !rho_kl_crust_mantle_reg(ipoint) = rho_kl_crust_mantle_reg(ipoint) &
372  ! + rhonotprime_kl_crust_mantle(ipoint) &
373  ! + alpha_kl_crust_mantle_reg(ipoint) &
374  ! + beta_kl_crust_mantle_reg(ipoint)
375  bulk_beta_kl_crust_mantle(ipoint) = bulk_betah_kl_crust_mantle(ipoint) &
376  + bulk_betav_kl_crust_mantle(ipoint)
377  endif
378  endif
379  enddo

394  ! transverse isotropic kernels

396  open(unit=iout,file=trim(prname)//'alphav_kernel.bin',status='unknown',form='unformatted',action='write')
397  write(iout) alphav_kl_crust_mantle

400  write(iout) alphah_kl_crust_mantle

412  write(iout) rho_kl_crust_mantle_reg
413  close(iout)
