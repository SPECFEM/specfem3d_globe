47  ! opens the parameter file: DATA/Par_file
48  call open_parameter_file(ier)
49  if (ier /= 0) stop 'an error occurred while opening the parameter file'

51  ! reads in values
52  call read_value_integer(simulation_type, 'SIMULATION_TYPE', ier)

55  if (ier /= 0) stop 'an error occurred while reading the parameter file: NOISE_TOMOGRAPHY'

59  if (ier /= 0) stop 'an error occurred while reading the parameter file: NCHUNKS'

61  if (nchunks == 6) then

63  angular_width_xi_in_degrees = 90.d0

65  center_latitude_in_degrees = 0.d0
66  center_longitude_in_degrees = 0.d0

73  if (ier /= 0) stop 'an error occurred while reading the parameter file: ANGULAR_WIDTH_ETA...'

75  if (ier /= 0) stop 'an error occurred while reading the parameter file: CENTER_LATITUDE...'
76  call read_value_double_precision(center_longitude_in_degrees, 'CENTER_LONGITUDE_IN_DEGREES', ier)
77  if (ier /= 0) stop 'an error occurred while reading the parameter file: CENTER_LONGITUDE...'
78  call read_value_double_precision(gamma_rotation_azimuth, 'GAMMA_ROTATION_AZIMUTH', ier)
79  if (ier /= 0) stop 'an error occurred while reading the parameter file: GAMMA_ROTATION...'
80  endif
81 
82  ! number of elements at the surface along the two sides of the first chunk
83  call read_value_integer(nex_xi_read, 'NEX_XI', ier)
84  if (ier /= 0) stop 'an error occurred while reading the parameter file: NEX_XI'
85  call read_value_integer(nex_eta_read, 'NEX_ETA', ier)
86  if (ier /= 0) stop 'an error occurred while reading the parameter file: NEX_ETA'
87  call read_value_integer(nproc_xi_read, 'NPROC_XI', ier)
88  if (ier /= 0) stop 'an error occurred while reading the parameter file: NPROC_XI'
89  call read_value_integer(nproc_eta_read, 'NPROC_ETA', ier)
90  if (ier /= 0) stop 'an error occurred while reading the parameter file: NPROC_ETA'
91 
92  ! physical parameters
93  call read_value_logical(oceans, 'OCEANS', ier)
94  if (ier /= 0) stop 'an error occurred while reading the parameter file: OCEANS'
95  call read_value_logical(ellipticity, 'ELLIPTICITY', ier)
96  if (ier /= 0) stop 'an error occurred while reading the parameter file: ELLIPTICITIY'
97  call read_value_logical(topography, 'TOPOGRAPHY', ier)
98  if (ier /= 0) stop 'an error occurred while reading the parameter file: TOPOGRAPHY'
99  call read_value_logical(gravity, 'GRAVITY', ier)
100  if (ier /= 0) stop 'an error occurred while reading the parameter file: GRAVITY'
101  call read_value_logical(rotation, 'ROTATION', ier)
102  if (ier /= 0) stop 'an error occurred while reading the parameter file: ROTATION'
103  call read_value_logical(attenuation, 'ATTENUATION', ier)
104  if (ier /= 0) stop 'an error occurred while reading the parameter file: ATTENUATION'
105

107  if (ier /= 0) stop 'an error occurred while reading the parameter file: ABSORBING_CONDITIONS'

111  if (ier /= 0) stop 'an error occurred while reading the parameter file: MODEL'

113  if (ier /= 0) stop 'an error occurred while reading the parameter file: RECORD_LENGTH_IN_MINUTES'
114 
115  ! attenuation parameters
116  call read_value_logical(partial_phys_dispersion_only, 'PARTIAL_PHYS_DISPERSION_ONLY', ier)

121  if (ier /= 0) stop 'an error occurred while reading the parameter file: MEMORY_INSTALLED_PER_CORE_IN_GB'
122  call read_value_double_precision(percent_of_mem_to_use_per_core, 'PERCENT_OF_MEM_TO_USE_PER_CORE', ier)
123  if (ier /= 0) stop 'an error occurred while reading the parameter file: PERCENT_OF_MEM_TO_USE_PER_CORE'
124 
125  ! mass matrix corrections
126  call read_value_logical(exact_mass_matrix_for_rotation, 'EXACT_MASS_MATRIX_FOR_ROTATION', ier)
127  if (ier /= 0) stop 'an error occurred while reading the parameter file: EXACT_MASS_MATRIX_FOR_ROTATION'
128 
129  ! low-memory Runge-Kutta time scheme
130  call read_value_logical(use_lddrk, 'USE_LDDRK', ier)
131  if (ier /= 0) stop 'an error occurred while reading the parameter file: USE_LDDRK'
132  call read_value_logical(increase_cfl_for_lddrk, 'INCREASE_CFL_FOR_LDDRK', ier)
133  if (ier /= 0) stop 'an error occurred while reading the parameter file: INCREASE_CFL_FOR_LDDRK'
134  call read_value_double_precision(ratio_by_which_to_increase_it, 'RATIO_BY_WHICH_TO_INCREASE_IT', ier)
135  if (ier /= 0) stop 'an error occurred while reading the parameter file: RATIO_BY_WHICH_TO_INCREASE_IT'
136 
137  ! movie options
138  call read_value_logical(movie_surface, 'MOVIE_SURFACE', ier)
139  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_SURFACE'
140  call read_value_logical(movie_volume, 'MOVIE_VOLUME', ier)
141  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_VOLUME'
142  call read_value_logical(movie_coarse,'MOVIE_COARSE', ier)
143  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_COARSE'
144  call read_value_integer(ntstep_between_frames, 'NTSTEP_BETWEEN_FRAMES', ier)
145  if (ier /= 0) stop 'an error occurred while reading the parameter file: NTSTEP_BETWEEN_FRAMES'
146  call read_value_double_precision(hdur_movie, 'HDUR_MOVIE', ier)
147  if (ier /= 0) stop 'an error occurred while reading the parameter file: HDUR_MOVIE'
148  call read_value_integer(movie_volume_type, 'MOVIE_VOLUME_TYPE', ier)
149  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_VOLUME_TYPE'
150  call read_value_double_precision(movie_top_km, 'MOVIE_TOP_KM', ier)
151  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_TOP_KM'
152  call read_value_double_precision(movie_bottom_km, 'MOVIE_BOTTOM_KM', ier)
153  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_BOTTOM_KM'
154  call read_value_double_precision(movie_west_deg, 'MOVIE_WEST_DEG', ier)
155  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_WEST_DEG'
156  call read_value_double_precision(movie_east_deg, 'MOVIE_EAST_DEG', ier)
157  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_EAST_DEG'
158  call read_value_double_precision(movie_north_deg, 'MOVIE_NORTH_DEG', ier)
159  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_NORTH_DEG'
160  call read_value_double_precision(movie_south_deg, 'MOVIE_SOUTH_DEG', ier)
161  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_SOUTH_DEG'
162  call read_value_integer(movie_start, 'MOVIE_START', ier)
163  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_START'
164  call read_value_integer(movie_stop, 'MOVIE_STOP', ier)
165  if (ier /= 0) stop 'an error occurred while reading the parameter file: MOVIE_STOP'
166 
167  ! run checkpointing
168  call read_value_logical(save_mesh_files, 'SAVE_MESH_FILES', ier)
169  if (ier /= 0) stop 'an error occurred while reading the parameter file: SAVE_MESH_FILES'
170  call read_value_integer(number_of_runs, 'NUMBER_OF_RUNS', ier)
171  if (ier /= 0) stop 'an error occurred while reading the parameter file: NUMBER_OF_RUNS'
172  call read_value_integer(number_of_this_run, 'NUMBER_OF_THIS_RUN', ier)
173  if (ier /= 0) stop 'an error occurred while reading the parameter file: NUMBER_OF_THIS_RUN'
174 
175  ! data file output directories
176  call read_value_string(local_path, 'LOCAL_PATH', ier)
177  if (ier /= 0) stop 'an error occurred while reading the parameter file: LOCAL_PATH'
178  call read_value_string(local_tmp_path, 'LOCAL_TMP_PATH', ier)
179  if (ier /= 0) stop 'an error occurred while reading the parameter file: LOCAL_TMP_PATH'
180 
181  ! user output
182  call read_value_integer(ntstep_between_output_info, 'NTSTEP_BETWEEN_OUTPUT_INFO', ier)
183  if (ier /= 0) stop 'an error occurred while reading the parameter file: NTSTEP_BETWEEN_OUTPUT_INFO'
184  call read_value_integer(ntstep_between_output_seismos, 'NTSTEP_BETWEEN_OUTPUT_SEISMOS', ier)

186  call read_value_integer(ntstep_between_read_adjsrc, 'NTSTEP_BETWEEN_READ_ADJSRC', ier)

192  call read_value_logical(output_seismos_sac_alphanum, 'OUTPUT_SEISMOS_SAC_ALPHANUM', ier)
193  if (ier /= 0) stop 'an error occurred while reading the parameter file: OUTPUT_SEISMOS_SAC_ALPHANUM'
194  call read_value_logical(output_seismos_sac_binary, 'OUTPUT_SEISMOS_SAC_BINARY', ier)
195  if (ier /= 0) stop 'an error occurred while reading the parameter file: OUTPUT_SEISMOS_SAC_BINARY'
196  call read_value_logical(output_seismos_asdf, 'OUTPUT_SEISMOS_ASDF', ier)
197  if (ier /= 0) stop 'an error occurred while reading the parameter file: OUTPUT_ASDF'
198  call read_value_logical(rotate_seismograms_rt, 'ROTATE_SEISMOGRAMS_RT', ier)
199  if (ier /= 0) stop 'an error occurred while reading the parameter file: ROTATE_SEISMOGRAMS_RT'
200  call read_value_logical(write_seismograms_by_master, 'WRITE_SEISMOGRAMS_BY_MASTER', ier)
201  if (ier /= 0) stop 'an error occurred while reading the parameter file: WRITE_SEISMOGRAMS_BY_MASTER'
202  call read_value_logical(save_all_seismos_in_one_file, 'SAVE_ALL_SEISMOS_IN_ONE_FILE', ier)

204  call read_value_logical(use_binary_for_large_file, 'USE_BINARY_FOR_LARGE_FILE', ier)
205  if (ier /= 0) stop 'an error occurred while reading the parameter file: USE_BINARY_FOR_LARGE_FILE'
206  call read_value_logical(receivers_can_be_buried, 'RECEIVERS_CAN_BE_BURIED', ier)
207  if (ier /= 0) stop 'an error occurred while reading the parameter file: RECEIVERS_CAN_BE_BURIED'
208  call read_value_logical(print_source_time_function, 'PRINT_SOURCE_TIME_FUNCTION', ier)
209  if (ier /= 0) stop 'an error occurred while reading the parameter file: PRINT_SOURCE_TIME_FUNCTION'
210 
211  ! adjoint kernels
212  call read_value_logical(read_adjsrc_asdf, 'READ_ADJSRC_ASDF', ier)
213  if (ier /= 0) stop 'an error occured while reading the parameter file: READ_ADJSRC_ASDF'
214  call read_value_logical(anisotropic_kl, 'ANISOTROPIC_KL', ier)
215  if (ier /= 0) stop 'an error occurred while reading the parameter file: ANISOTROPIC_KL'
216  call read_value_logical(save_transverse_kl_only, 'SAVE_TRANSVERSE_KL_ONLY', ier)
217  if (ier /= 0) stop 'an error occurred while reading the parameter file: SAVE_TRANSVERSE_KL_ONLY'
218  call read_value_logical(approximate_hess_kl, 'APPROXIMATE_HESS_KL', ier)
219  if (ier /= 0) stop 'an error occurred while reading the parameter file: APPROXIMATE_HESS_KL'
220  call read_value_logical(use_full_tiso_mantle, 'USE_FULL_TISO_MANTLE', ier)
221  if (ier /= 0) stop 'an error occurred while reading the parameter file: USE_FULL_TISO_MANTLE'
222  call read_value_logical(save_source_mask, 'SAVE_SOURCE_MASK', ier)
223  if (ier /= 0) stop 'an error occurred while reading the parameter file: SAVE_SOURCE_MASK'
224  call read_value_logical(save_regular_kl, 'SAVE_REGULAR_KL', ier)
225  if (ier /= 0) stop 'an error occurred while reading the parameter file: SAVE_REGULAR_KL'

227  ! for simultaneous runs from the same batch job
228  call read_value_integer(number_of_simultaneous_runs, 'NUMBER_OF_SIMULTANEOUS_RUNS', ier)
