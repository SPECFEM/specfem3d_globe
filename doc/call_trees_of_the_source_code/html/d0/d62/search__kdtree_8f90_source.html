121  subroutine kdtree_setup()
122 
123  ! sets up the kd-tree structure
124  !
125  ! needs:
126  ! kdtree_num_nodes - total number of points
127  ! kdtree_nodes_location - 3D array of points
128  !
129  ! returns:
130  ! creates internal tree representation
131  !
132 
133  implicit none
134 
135  ! local parameters
136  integer :: npoints
137  integer, dimension(:), allocatable :: points_index
138  double precision,dimension(:,:), pointer :: points_data
139 
140  ! tree statistics
141  integer :: depth
142  integer :: numnodes,maxdepth
143  integer :: i,ier
144 
145  ! timing
146  real :: ct_start,ct_end
147 
148  ! test search
149  double precision, dimension(3) :: xyz_target
150  integer :: ipoint_min
151  double precision :: dist_min
152 
153  !------------------------------------------------------
154 
155  ! debugging: performs a test search
156  logical,parameter :: DEBUG = .false.
157 
158  !------------------------------------------------------
159 
160  ! checks
161  if (kdtree_num_nodes <= 0 ) stop 'Error creating kdtree with zero nodes is invalid'
162  if (.not. allocated(kdtree_nodes_location) ) stop 'Error array kdtree_nodes_location not allocated yet'
163 
164  ! timing
165  call cpu_time(ct_start)
166 
167  ! number of data points
168  npoints = kdtree_num_nodes
169 
170  ! 3D point coordinates
171  points_data => kdtree_nodes_location(:,:)
172 
173  if (be_verbose) then
174  print *,'kd-tree:'
175  print *,' total data points: ',npoints
176  !print *,' box boundaries : x min/max = ',minval(points_data(1,:)),maxval(points_data(1,:))
177  !print *,' y min/max = ',minval(points_data(2,:)),maxval(points_data(2,:))
178  !print *,' z min/max = ',minval(points_data(3,:)),maxval(points_data(3,:))
179  endif
180 
181  ! theoretical number of node for totally balanced tree
182  numnodes = npoints
183  i = npoints
184  do while ( i >= 1 )
185  i = i / 2
186  numnodes = numnodes + i
187  ! integer 32-bit limitation (integer overflow)
188  if (numnodes > 2147483646 - i ) stop 'Error number of nodes might exceed integer limit'
189  enddo
190  if (be_verbose) then
191  print *,' theoretical number of nodes: ',numnodes
192  print *,' tree memory size: ',( numnodes * 32 )/1024./1024.,'MB'
193  endif
194 
195  ! local ordering
196  allocate(points_index(kdtree_num_nodes),stat=ier)
197  if (ier /= 0) stop 'Error allocating array points_index'
198 
199  ! initial point ordering
200  do i = 1,npoints
201  points_index(i) = i
202  enddo
203 
204  ! builds tree structure
205  nullify(kdtree)
206  depth = 0
207  numnodes = 0
208  maxdepth = -1
209 
210  call create_kdtree(npoints,points_data,points_index,kdtree, &
211  depth,1,npoints,numnodes,maxdepth)
212 
213  ! checks root tree node
214  if (.not. associated(kdtree) ) stop 'Error creation of kd-tree failed'
215 
216  if (be_verbose) then
217  print *,' actual number of nodes: ',numnodes
218  ! tree node size: 4 (idim) + 8 (cut_value) + 4 (ipoint) + 2*4 (ibound_**) + 2*4 (left,right) = 32 bytes
219  print *,' tree memory size: ',( numnodes * 32 )/1024./1024.,'MB'
220  print *,' maximum depth : ',maxdepth
221 
222  ! timing
223  call cpu_time(ct_end)
224  print *,' creation timing : ',ct_end - ct_start, '(s)'
225  print *
226  endif
227 
228 
229  ! debugging
230  if (debug) then
231  ! outputs tree
232  if (be_verbose) then
233  numnodes = 0
234  call print_kdtree(npoints,points_data,points_index,kdtree,numnodes)
235  endif
236 
237  ! test search
238  print *,'search tree:'
239  xyz_target(1) = 0.13261298835277557
240  xyz_target(2) = -8.4083788096904755e-002
241  xyz_target(3) = 0.97641450166702271
242 
243  print *,'search : ',xyz_target(:)
244 
245  ipoint_min = -1
246  dist_min = 1.d30
247 
248  call find_nearest_node(npoints,points_data,kdtree,xyz_target,ipoint_min,dist_min)
249 
250  dist_min = sqrt(dist_min)
251  print *,'found : ',ipoint_min,'distance:',dist_min
252 
253  if (ipoint_min < 1 ) stop 'Error search kd-tree found no point'
254 
255  print *,'target : ',xyz_target(:)
256  print *,'nearest : ',points_data(:,ipoint_min)
257  print *
258  ! safety stop
259  stop 'kdtree_setup safety stop'
260  endif
261 
262  ! frees temporary arrays
263  deallocate(points_index)
264  nullify(points_data)
265 
266  end subroutine kdtree_setup

625  depth,ibound_lower,ibound_upper,numnodes,maxdepth)
626 
627  ! creates node in kd-tree structure
628 
629  implicit none
630 
631  integer, intent(in) :: npoints
632  double precision, dimension(3,npoints), intent(in) :: points_data
633  integer,dimension(npoints), intent(inout) :: points_index
634 
635  type(kdtree_node), pointer, intent(inout) :: node
636 
637  integer,intent(in) :: depth

639 
640  integer,intent(inout) :: numnodes,maxdepth
641 
642  ! local parameters
643  double precision :: cut_value
644  double precision :: range,range_max,min,max
645  integer :: i,ier,idim
646  integer :: iloc,ilower,iupper
647  integer :: l,u
648  integer,dimension(:),allocatable :: workindex
649 
650  ! checks if anything to sort
651  if (ibound_lower > ibound_upper) then
652  nullify(node)
653  return
654  endif
655 
656  ! creates new node
657  allocate(node,stat=ier)
658  if (ier /= 0) then
659  print *,'Error creating node: ',numnodes
660  stop 'Error allocating kd-tree node'
661  endif
662 
663  ! initializes new node
664  node%idim = -1
665  node%ipoint = -1
666  node%cut_value = 0.d0
667 
668  nullify(node%left)
669  nullify(node%right)
670 
671  ! tree statistics
672  numnodes = numnodes + 1
673  if (maxdepth < depth) maxdepth = depth
674 
675  !node%id = numnodes
676 
677  ! checks if final node
678  if (ibound_lower == ibound_upper) then
679  node%idim = 0
680  node%ipoint = points_index(ibound_lower)
681  ! done with this node
682  return
683  endif
684 
685  ! sets cut dimension index
686 
687  ! version 1: varies between 1 and 3 depending on depth for 3D data set
688  ! (leads to some unneccessary unbalanced nodes)
689  !idim = mod(depth,3) + 1
690  ! determines cut value
691  ! range in this dimension
692  !min = HUGEVAL
693  !max = - HUGEVAL
694  !do i = ibound_lower,ibound_upper
695  ! iloc = points_index(i)
696  ! val = points_data(idim,iloc)
697  ! if (val < min ) min = val
698  ! if (val > max ) max = val
699  !enddo
700  !min = minval(points_data(idim,points_index(ibound_lower:ibound_upper)))
701  !max = maxval(points_data(idim,points_index(ibound_lower:ibound_upper)))
702  !cut_value = 0.5d0 * ( min + max )
703 
704  ! version 2: selects cut dimension where biggest range is occurring
705  ! (better balances tree)

1309 
1310  implicit none

1313

1315  double precision :: r0,r1

1317 
1318  ! vertical distance (squared)
1319  r0 = sqrt( sum( xyz0(:)**2 ) ) ! length of first position vector
1320  r1 = sqrt( sum( xyz1(:)**2 ) )
1321 
1322  dist_v = (r1 - r0)*(r1 - r0)
1323 
1324  ! only for flat earth with z in depth: dist_v = sqrt( (cz(ispec2)-cz0(ispec))** 2)
1325 
1326  ! epicentral distance
1327  ! (accounting for spherical curvature)
1328  ! calculates distance of circular segment
1329  ! angle between r0 and r1 in radian
1330  ! given by dot-product of two vectors
1331  ratio = sum(xyz0(:)*xyz1(:)) / (r0 * r1)
1332 
1333  ! checks boundaries of ratio (due to numerical inaccuracies)
1334  if (ratio > 1.d0) ratio = 1.d0
1335  if (ratio < -1.d0) ratio = -1.d0
1336 
1337  theta = acos( ratio )
1338 
1339  ! segment length at heigth of r1 (squared)
1340  dist_h = (r1 * theta)*(r1 * theta)
