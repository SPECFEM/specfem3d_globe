28  subroutine add_topography(xelm,yelm,zelm,ibathy_topo)
29 
30  use constants, only: myrank, &

43  ! local parameters
44  double precision :: r,lat,lon,elevation
45  double precision :: x,y,z
46  double precision :: gamma
47 
48  integer :: ia
49 
50  ! we loop on all the points of the element
51  do ia = 1,ngnod
52 
53  x = xelm(ia)
54  y = yelm(ia)
55  z = zelm(ia)
56 
57  ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
58  call xyz_2_rlatlon_dble(x,y,z,r,lat,lon)
59 
60  ! compute elevation at current point
61  call get_topo_bathy(lat,lon,elevation,ibathy_topo)
62 
63  ! non-dimensionalize the elevation, which is in meters
64  elevation = elevation / r_earth
65 
66  ! stretching topography between d220 and the surface
67  gamma = (r - r220/r_earth) / (r_unit_sphere - r220/r_earth)
68 
69  ! add elevation to all the points of that element
70  ! also make sure gamma makes sense
71  if (gamma < -0.02 .or. gamma > 1.02) call exit_mpi(myrank,'incorrect value of gamma for topography')
72 
73  xelm(ia) = x*(one + gamma * elevation / r)
74  yelm(ia) = y*(one + gamma * elevation / r)
75  zelm(ia) = z*(one + gamma * elevation / r)
76 
77  enddo
78 
79  end subroutine add_topography
