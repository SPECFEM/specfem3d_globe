29  subroutine setup_mpi_interfaces(iregion_code)
30 
31  use meshfem3d_par, only: &
32  include_central_cube,myrank,numfaces_shared
33 
34  use create_mpi_interfaces_par
35 
36  use mpi_crust_mantle_par
37  use mpi_outer_core_par
38  use mpi_inner_core_par
39 
40  implicit none
41 
42  integer,intent(in):: iregion_code
43 
44  ! local parameters
45  ! assigns initial maximum arrays
46  ! for global slices, maximum number of neighbor is around 17 ( 8 horizontal, max of 8 on bottom )
47  integer :: MAX_NEIGHBOURS,max_nibool
48  integer, dimension(:),allocatable :: my_neighbours,nibool_neighbours
49  integer, dimension(:,:),allocatable :: ibool_neighbours
50  integer :: ier
51 
52  ! allocates temporary arrays for setup routines
53  ! estimates a maximum size of needed arrays
54  max_neighbours = 8 + ncornerschunks
55  if (include_central_cube ) max_neighbours = max_neighbours + nummsgs_faces
56 
57  allocate(my_neighbours(max_neighbours), &
58  nibool_neighbours(max_neighbours),stat=ier)
59  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating my_neighbours array')
60  my_neighbours(:) = -1
61  nibool_neighbours(:) = 0
62 
63  ! estimates initial maximum ibool array
64  max_nibool = npoin2d_max_all_cm_ic * numfaces_shared &
65  + non_zero_nb_msgs_theor_in_cube*npoin2d_cube_from_slices
66 
67  allocate(ibool_neighbours(max_nibool,max_neighbours), stat=ier)
68  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating ibool_neighbours')
69  ibool_neighbours(:,:) = 0
70 
71  ! sets up MPI interfaces between different processes
72  select case (iregion_code)
73  case (iregion_crust_mantle)
74  ! crust/mantle
75  call setup_mpi_interfaces_cm(max_neighbours,my_neighbours,nibool_neighbours, &
76  max_nibool,ibool_neighbours)
77 
78  case (iregion_outer_core)
79  ! outer core
80  call setup_mpi_interfaces_oc(max_neighbours,my_neighbours,nibool_neighbours, &
81  max_nibool,ibool_neighbours)
82 
83  case (iregion_inner_core)
84  ! inner core
85  call setup_mpi_interfaces_ic(max_neighbours,my_neighbours,nibool_neighbours, &
86  max_nibool,ibool_neighbours)
87  end select
88 
89  ! frees temporary array
90  deallocate(ibool_neighbours)
91  deallocate(my_neighbours,nibool_neighbours)
92 
93  ! frees arrays not needed any further
94  deallocate(iprocfrom_faces,iprocto_faces,imsg_type)
95  deallocate(iproc_master_corners,iproc_worker1_corners,iproc_worker2_corners)
96  deallocate(buffer_send_chunkcorn_scalar,buffer_recv_chunkcorn_scalar)
97  deallocate(buffer_send_chunkcorn_vector,buffer_recv_chunkcorn_vector)
98 
99  select case (iregion_code)
100  case (iregion_crust_mantle)
101  ! crust mantle
102  deallocate(iboolcorner_crust_mantle)
103  deallocate(iboolleft_xi_crust_mantle,iboolright_xi_crust_mantle)
104  deallocate(iboolleft_eta_crust_mantle,iboolright_eta_crust_mantle)
105  deallocate(iboolfaces_crust_mantle)
106  case (iregion_outer_core)
107  ! outer core
108  deallocate(iboolcorner_outer_core)
109  deallocate(iboolleft_xi_outer_core,iboolright_xi_outer_core)
110  deallocate(iboolleft_eta_outer_core,iboolright_eta_outer_core)
111  deallocate(iboolfaces_outer_core)
112  case (iregion_inner_core)
113  ! inner core
114  deallocate(iboolcorner_inner_core)
115  deallocate(iboolleft_xi_inner_core,iboolright_xi_inner_core)
116  deallocate(iboolleft_eta_inner_core,iboolright_eta_inner_core)
117  deallocate(iboolfaces_inner_core)
118  end select
119 
120  ! synchronizes MPI processes
121  call synchronize_all()
122 
123  end subroutine setup_mpi_interfaces

224  ibool_interfaces_crust_mantle(:,:) = 0
225 
226  ! ranks of neighbour processes
227  my_neighbours_crust_mantle(:) = my_neighbours(1:num_interfaces_crust_mantle)
228  ! number of global ibool entries on each interface
229  nibool_interfaces_crust_mantle(:) = nibool_neighbours(1:num_interfaces_crust_mantle)
230  ! global iglob point ids on each interface
231  ibool_interfaces_crust_mantle(:,:) = ibool_neighbours(1:max_nibool_interfaces_cm,1:num_interfaces_crust_mantle)
232  else
233  ! dummy allocation (fortran90 should allow allocate statement with zero array size)
234  max_nibool_interfaces_cm = 0
235  allocate(ibool_interfaces_crust_mantle(0,0),stat=ier)
236  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating dummy array ibool_interfaces_crust_mantle')
237  endif
238 
239  ! debug: outputs MPI interface
240  if (debug) then
241  do i = 1,num_interfaces_crust_mantle
242  write(filename,'(a,i6.6,a,i2.2)') trim(output_files)//'/MPI_points_crust_mantle_proc',myrank, &
243  '_',my_neighbours_crust_mantle(i)
244  call write_vtk_data_points(nglob_crust_mantle, &
245  xstore_glob,ystore_glob,zstore_glob, &
246  ibool_interfaces_crust_mantle(1:nibool_interfaces_crust_mantle(i),i), &
247  nibool_interfaces_crust_mantle(i),filename)
248  enddo
249  call synchronize_all()
250  endif
251 
252  ! checks addressing
253  call test_mpi_neighbours(iregion_crust_mantle, &
254  num_interfaces_crust_mantle,max_nibool_interfaces_cm, &
255  my_neighbours_crust_mantle,nibool_interfaces_crust_mantle, &
256  ibool_interfaces_crust_mantle)
257 
258  ! checks with assembly of test fields
259  call test_mpi_cm()
260 
261  end subroutine setup_mpi_interfaces_cm

473  buffer_send_chunkcorn_scalar,buffer_recv_chunkcorn_scalar, &
474  nummsgs_faces,num_msg_types,ncornerschunks, &
475  nproc_xi,nproc_eta,nglob1d_radial(iregion_inner_core), &
476  nglob2dmax_xmin_xmax(iregion_inner_core),nglob2dmax_ymin_ymax(iregion_inner_core), &
477  nglob2dmax_xy,nchunks)
478 
479  ! debug: idoubling inner core
480  if (debug) then
481  write(filename,'(a,i6.6)') trim(output_files)//'/MPI_idoubling_inner_core_proc',myrank
482  call write_vtk_data_elem_i(nspec_inner_core,nglob_inner_core, &
483  xstore_glob,ystore_glob,zstore_glob, &
484  ibool,idoubling,filename)
485  call synchronize_all()
486  endif
487 
488  ! including central cube
489  if (include_central_cube) then
490  ! user output
491  if (myrank == 0 ) write(imain,*) 'inner core with central cube MPI:'
492 
493  ! test_flag is a scalar, not a vector
494  ndim_assemble = 1
495
