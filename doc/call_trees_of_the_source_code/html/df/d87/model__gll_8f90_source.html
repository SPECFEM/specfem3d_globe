36  subroutine model_gll_broadcast(MGLL_V,NSPEC)
37 
38 ! standard routine to setup model
39 
40  use constants
41  use meshfem3d_models_par, only: transverse_isotropy,model_gll_variables
42  use meshfem3d_par, only: adios_for_models
43 
44  implicit none
45 
46  ! GLL model_variables
47  type(model_gll_variables) mgll_v
48 
49  integer, dimension(MAX_NUM_REGIONS) :: NSPEC
50 
51  ! local parameters
52  double precision :: scaleval
53  real(kind=CUSTOM_REAL) :: minvalue,maxvalue,min_all,max_all
54  integer :: ier
55 
56  ! allocates arrays
57  ! differs for isotropic model or transverse isotropic models
58  if (.not. transverse_isotropy) then
59  ! isotropic model
60  allocate( mgll_v%vp_new(ngllx,nglly,ngllz,nspec(iregion_crust_mantle)), &
61  mgll_v%vs_new(ngllx,nglly,ngllz,nspec(iregion_crust_mantle)), stat=ier)
62  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating vp_new,.. arrays')
63  else
64  ! transverse isotropic model
65  allocate( mgll_v%vpv_new(ngllx,nglly,ngllz,nspec(iregion_crust_mantle)), &
66  mgll_v%vph_new(ngllx,nglly,ngllz,nspec(iregion_crust_mantle)), &
67  mgll_v%vsv_new(ngllx,nglly,ngllz,nspec(iregion_crust_mantle)), &
68  mgll_v%vsh_new(ngllx,nglly,ngllz,nspec(iregion_crust_mantle)), &
69  mgll_v%eta_new(ngllx,nglly,ngllz,nspec(iregion_crust_mantle)), stat=ier)
70  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating vpv_new,.. arrays')
71 
72  endif
73  allocate( mgll_v%rho_new(ngllx,nglly,ngllz,nspec(iregion_crust_mantle)), stat=ier)
74  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating rho_new,.. arrays')
75 
76  ! reads in model files for each process
77  if (adios_for_models) then
78  call read_gll_model_adios(mgll_v,nspec)
79  else
80  call read_gll_model(mgll_v,nspec)
81  endif
82 
83  ! checks velocity range
84  if (.not. transverse_isotropy) then
85 
86  ! isotropic model
87  if (myrank == 0) then
88  write(imain,*)'model GLL: isotropic'
89  call flush_imain()
90  endif
91 
92  ! Vs
93  maxvalue = maxval( mgll_v%vs_new )
94  minvalue = minval( mgll_v%vs_new )
95  call max_all_cr(maxvalue, max_all)
96  call min_all_cr(minvalue, min_all)
97  if (myrank == 0) then
98  write(imain,*) ' vs new min/max: ',min_all,max_all
99  endif
100  ! Vp
101  maxvalue = maxval( mgll_v%vp_new )
102  minvalue = minval( mgll_v%vp_new )
103  call max_all_cr(maxvalue, max_all)
104  call min_all_cr(minvalue, min_all)
105  if (myrank == 0) then
106  write(imain,*) ' vp new min/max: ',min_all,max_all
107  endif
108  ! density
109  maxvalue = maxval( mgll_v%rho_new )
110  minvalue = minval( mgll_v%rho_new )
111  call max_all_cr(maxvalue, max_all)
112  call min_all_cr(minvalue, min_all)
113  if (myrank == 0) then
114  write(imain,*) ' rho new min/max: ',min_all,max_all
115  write(imain,*)
116  call flush_imain()
117  endif
118 
119  else
120 
121  ! transverse isotropic model
122  if (myrank == 0) then
123  write(imain,*)'model GLL: transverse isotropic'
124  endif
125 
126  ! Vsv
127  maxvalue = maxval( mgll_v%vsv_new )
128  minvalue = minval( mgll_v%vsv_new )
129  call max_all_cr(maxvalue, max_all)
130  call min_all_cr(minvalue, min_all)
131  if (myrank == 0) then
132  write(imain,*) ' vsv new min/max: ',min_all,max_all
133  endif
134  ! Vsh
135  maxvalue = maxval( mgll_v%vsh_new )
136  minvalue = minval( mgll_v%vsh_new )
137  call max_all_cr(maxvalue, max_all)
138  call min_all_cr(minvalue, min_all)
139  if (myrank == 0) then
140  write(imain,*) ' vsh new min/max: ',min_all,max_all
141  endif
142  ! Vpv
143  maxvalue = maxval( mgll_v%vpv_new )
144  minvalue = minval( mgll_v%vpv_new )
145  call max_all_cr(maxvalue, max_all)
146  call min_all_cr(minvalue, min_all)
147  if (myrank == 0) then
148  write(imain,*) ' vpv new min/max: ',min_all,max_all
149  endif
150  ! Vph
151  maxvalue = maxval( mgll_v%vph_new )
152  minvalue = minval( mgll_v%vph_new )
153  call max_all_cr(maxvalue, max_all)
154  call min_all_cr(minvalue, min_all)
155  if (myrank == 0) then
156  write(imain,*) ' vph new min/max: ',min_all,max_all
157  endif
158  ! density
159  maxvalue = maxval( mgll_v%rho_new )
160  minvalue = minval( mgll_v%rho_new )
161  call max_all_cr(maxvalue, max_all)
162  call min_all_cr(minvalue, min_all)
163  if (myrank == 0) then
164  write(imain,*) ' rho new min/max: ',min_all,max_all
165  endif
166  ! eta
167  maxvalue = maxval( mgll_v%eta_new )
168  minvalue = minval( mgll_v%eta_new )
169  call max_all_cr(maxvalue, max_all)
170  call min_all_cr(minvalue, min_all)
171  if (myrank == 0) then
172  write(imain,*) ' eta new min/max: ',min_all,max_all
173  write(imain,*)
174  call flush_imain()
175  endif
176 
177  endif
178 
179  ! non-dimensionalizes model values
180  ! (SPECFEM3D_GLOBE uses non-dimensionalized values in subsequent computations)
181  ! scaling values
182  ! (model velocities must be given as km/s)
183  scaleval = dsqrt(pi*grav*rhoav)
184  mgll_v%scale_velocity = 1000.0d0/(r_earth*scaleval)
185  mgll_v%scale_density = 1000.0d0/rhoav
186  if (.not. transverse_isotropy) then
187  ! non-dimensionalize isotropic values
188  mgll_v%vp_new = mgll_v%vp_new * mgll_v%scale_velocity
189  mgll_v%vs_new = mgll_v%vs_new * mgll_v%scale_velocity
190  mgll_v%rho_new = mgll_v%rho_new * mgll_v%scale_density
191  else
192  ! non-dimensionalize
193  ! transverse isotropic model
194  mgll_v%vpv_new = mgll_v%vpv_new * mgll_v%scale_velocity
195  mgll_v%vph_new = mgll_v%vph_new * mgll_v%scale_velocity
196  mgll_v%vsv_new = mgll_v%vsv_new * mgll_v%scale_velocity
197  mgll_v%vsh_new = mgll_v%vsh_new * mgll_v%scale_velocity
198  mgll_v%rho_new = mgll_v%rho_new * mgll_v%scale_density
199  ! eta is already non-dimensional
200  endif
201 
202  end subroutine model_gll_broadcast
