28  subroutine finalize_mesher()
29 
30  use meshfem3d_par
31  use meshfem3d_models_par
32 
33  use manager_adios
34 
35  implicit none
36 
37  ! local parameters
38  ! timing
39  double precision :: tCPU
40  double precision, external :: wtime
41 
42  integer :: NT_DUMP_ATTENUATION_optimal
43  logical, parameter :: PRINT_INFO_TO_SCREEN = .false.
44 
45 
46  !--- print number of points and elements in the mesh for each region
47  if (myrank == 0) then
48 
49  ! center of mass distance
50  distance_to_center_in_km = (sqrt(earth_center_of_mass_x_total**2 + &
51  earth_center_of_mass_y_total**2 + &
52  earth_center_of_mass_z_total**2) / earth_mass_total) / 1000.d0
53 
54  numelem_crust_mantle = nspec(iregion_crust_mantle)
55  numelem_outer_core = nspec(iregion_outer_core)
56  numelem_inner_core = nspec(iregion_inner_core)
57 
58  numelem_total = numelem_crust_mantle + numelem_outer_core + numelem_inner_core
59 
60  ! check volume of chunk
61  write(imain,*)
62  write(imain,*) 'calculated volume: ',volume_total
63  if (nchunks == 6 .and. .not. ellipticity .and. .not. topography) &
64  write(imain,*) ' exact volume: ',(4.0d0/3.0d0)*pi*(r_unit_sphere**3)
65 
66  ! check total Earth mass
67  if (nchunks == 6) then
68  write(imain,*)
69  write(imain,*) 'computed total Earth mass for this density model and mesh: '
70  write(imain,*) ' ',sngl(earth_mass_total),' kg'
71  write(imain,*) ' (should be not too far from 5.974E+24 kg)'
72  write(imain,*)
73  ! take into account the fact that dimensions have been non-dimensionalized by dividing them by R_EARTH
74  write(imain,*) 'average density for this density model and mesh: '
75  write(imain,*) ' ',sngl(earth_mass_total / (volume_total * r_earth**3)),' kg/m3'
76  write(imain,*) ' (should be not too far from 5514 kg/m3)'
77  write(imain,*)
78  write(imain,*) 'position of the center of mass of the Earth for this density model and mesh: '
79  write(imain,*) ' x = ',sngl((earth_center_of_mass_x_total / earth_mass_total) / 1000.d0),' km'
80  write(imain,*) ' y = ',sngl((earth_center_of_mass_y_total / earth_mass_total) / 1000.d0),' km'
81  write(imain,*) ' z = ',sngl((earth_center_of_mass_z_total / earth_mass_total) / 1000.d0),' km'
82  write(imain,*) ' distance to center = ',sngl(distance_to_center_in_km),' km'
83  endif
84 
85  ! info output
86  write(imain,*)
87  write(imain,*) 'Repartition of elements in regions:'
88  write(imain,*) '----------------------------------'
89  write(imain,*)
90  write(imain,*) 'total number of elements in each slice: ',numelem_total
91  write(imain,*)
92  write(imain,*) ' - crust and mantle: ',sngl(100.d0*dble(numelem_crust_mantle)/dble(numelem_total)),' %'
93  write(imain,*) ' - outer core: ',sngl(100.d0*dble(numelem_outer_core)/dble(numelem_total)),' %'
94  write(imain,*) ' - inner core: ',sngl(100.d0*dble(numelem_inner_core)/dble(numelem_total)),' %'
95  write(imain,*)
96  write(imain,*) 'for some mesh statistics, see comments in file OUTPUT_FILES/values_from_mesher.h'
97  write(imain,*)
98 
99  ! load balancing
100  write(imain,*) 'Load balancing = 100 % by definition'
101  write(imain,*)
102 
103  write(imain,*)
104  write(imain,*) 'the time step of the solver will be DT = ',sngl(dt)
105  write(imain,*)
106 
107  ! write information about precision used for floating-point operations
108  if (custom_real == size_real) then
109  write(imain,*) 'using single precision for the calculations'
110  else
111  write(imain,*) 'using double precision for the calculations'
112  endif
113  write(imain,*)
114  write(imain,*) 'smallest and largest possible floating-point numbers are: ',tiny(1._custom_real),huge(1._custom_real)
115  write(imain,*)
116  call flush_imain()
117 
118  ! create include file for the solver
119  call save_header_file(nspec,nglob,nproc,nproctot, &
120  static_memory_size, &
121  nspec2d_top,nspec2d_bottom, &
122  nspec2dmax_ymin_ymax,nspec2dmax_xmin_xmax, &
123  nspecmax_aniso_ic,nspecmax_iso_mantle,nspecmax_tiso_mantle, &
124  nspecmax_aniso_mantle,nspec_crust_mantle_attenuation, &
125  nspec_inner_core_attenuation, &
126  nspec_crust_mantle_str_or_att,nspec_inner_core_str_or_att, &
127  nspec_crust_mantle_str_and_att,nspec_inner_core_str_and_att, &
128  nspec_crust_mantle_strain_only,nspec_inner_core_strain_only, &
129  nspec_crust_mantle_adjoint, &
130  nspec_outer_core_adjoint,nspec_inner_core_adjoint, &
131  nglob_crust_mantle_adjoint,nglob_outer_core_adjoint, &
132  nglob_inner_core_adjoint,nspec_outer_core_rot_adjoint, &
133  nspec_crust_mantle_stacey,nspec_outer_core_stacey, &
134  nglob_crust_mantle_oceans,nspec_outer_core_rotation,nt_dump_attenuation_optimal, &
135  print_info_to_screen)
136 
137  endif ! end of section executed by main process only
138 
139  ! finishes with gravity integrals
140  if (gravity_integrals) call finalize_gravity_integrals()
141 
142  ! deallocate arrays used for mesh generation
143  deallocate(addressing)
144  deallocate(ichunk_slice)
145  deallocate(iproc_xi_slice)
146  deallocate(iproc_eta_slice)
147 
148  ! elapsed time since beginning of mesh generation
149  if (myrank == 0) then
150  tcpu = wtime() - time_start
151  write(imain,*)
152  write(imain,*) 'Elapsed time for mesh generation and buffer creation in seconds = ',tcpu
153  write(imain,"(' Elapsed time for mesh generation and buffer creation in hh:mm:ss = ',i4,' h ',i2.2,' m ',i2.2,' s')") &
154  int(tcpu/3600),int( (tcpu - int(tcpu/3600)*3600)/60 ),int(tcpu - int(tcpu/60) * 60)
155  write(imain,*)
156  write(imain,*) 'End of mesh generation'
157  write(imain,*)
158  call flush_imain()
159 
160  ! close main output file
161  close(imain)
162  endif
163 
164  ! synchronize all the processes to make sure everybody has finished
165  call synchronize_all()
166 
167  if (adios_enabled) then
168  call finalize_adios()
169  endif
170 
171  end subroutine finalize_mesher
