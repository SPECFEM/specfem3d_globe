75  call heap_sort_multi(ninseg(iseg), y(ioff), x(ioff), z(ioff), ibool(ioff), locval(ioff))

96  enddo

98

100  nseg = 0
101  do i = 1,npointot
102  if (ifseg(i)) then
103  nseg = nseg + 1
104  ninseg(nseg) = 1
105  else
106  ninseg(nseg) = ninseg(nseg) + 1
107  endif
108  enddo
109 
110  enddo

115  ! eliminate the multiples by using a single (new) point number for all the points that have the same X Y Z after sorting
116  if (ifseg(i)) ig = ig + 1
117  iglob(locval(i)) = ig
118  enddo
