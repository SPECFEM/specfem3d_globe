83  subroutine model_sea1d(x,rho,vp,vs,Qkappa,Qmu,iregion_code)
84 
85  use constants
86  use model_sea1d_par
87 
88  implicit none
89 
90 ! input:
91 ! radius r: meters
92 
93 ! output:
94 ! density rho: kg/m^3
95 ! compressional wave speed vp: km/s
96 ! shear wave speed vs: km/s
97 
98  double precision :: x,rho,vp,vs,Qmu,Qkappa
99  integer :: iregion_code
100 
101  ! local parameters
102  double precision :: r,frac,scaleval
103  integer :: i
104 
105 !! DK DK implementation of model sea1d below and its radii has not been thoroughly
106 !! DK DK checked yet
107 
108 ! compute real physical radius in meters
109  r = x * r_earth
110 
111  i = 1
112  do while(r >= sea1dm_v_radius_sea1d(i) .and. i /= nr_sea1d)
113  i = i + 1
114  enddo
115 
116 ! make sure we stay in the right region
117  if (iregion_code == iregion_inner_core .and. i > 13) i = 13
118  if (iregion_code == iregion_outer_core .and. i < 15) i = 15
119  if (iregion_code == iregion_outer_core .and. i > 37) i = 37
120  if (iregion_code == iregion_crust_mantle .and. i < 39) i = 39
121 
122  if (i == 1) then
123  rho = sea1dm_v_density_sea1d(i)
124  vp = sea1dm_v_vp_sea1d(i)
125  vs = sea1dm_v_vs_sea1d(i)
126  qmu = sea1dm_v_qmu_sea1d(i)
127  qkappa = sea1dm_v_qkappa_sea1d(i)
128  else
129 
130 ! interpolate from SEA1DM_V_radius_sea1d(i-1) to r using the values at i-1 and i
131  frac = (r-sea1dm_v_radius_sea1d(i-1))/(sea1dm_v_radius_sea1d(i)-sea1dm_v_radius_sea1d(i-1))
132 
133  rho = sea1dm_v_density_sea1d(i-1) + frac * (sea1dm_v_density_sea1d(i)-sea1dm_v_density_sea1d(i-1))
134  vp = sea1dm_v_vp_sea1d(i-1) + frac * (sea1dm_v_vp_sea1d(i)-sea1dm_v_vp_sea1d(i-1))
135  vs = sea1dm_v_vs_sea1d(i-1) + frac * (sea1dm_v_vs_sea1d(i)-sea1dm_v_vs_sea1d(i-1))
136  qmu = sea1dm_v_qmu_sea1d(i-1) + frac * (sea1dm_v_qmu_sea1d(i)-sea1dm_v_qmu_sea1d(i-1))
137  qkappa = sea1dm_v_qkappa_sea1d(i-1) + frac * (sea1dm_v_qkappa_sea1d(i)-sea1dm_v_qkappa_sea1d(i-1))
138 
139  endif
140 
141 ! make sure Vs is zero in the outer core even if roundoff errors on depth
142 ! also set fictitious attenuation to a very high value (attenuation is not used in the fluid)
143  if (iregion_code == iregion_outer_core) then
144  vs = 0.d0
145  qkappa = 3000.d0
146  qmu = 3000.d0
147  endif
148 
149 ! non-dimensionalize
150 ! time scaling (s^{-1}) is done with scaleval
151  scaleval=dsqrt(pi*grav*rhoav)
152  rho=rho*1000.0d0/rhoav
153  vp=vp*1000.0d0/(r_earth*scaleval)
154  vs=vs*1000.0d0/(r_earth*scaleval)
155 
156  end subroutine model_sea1d
