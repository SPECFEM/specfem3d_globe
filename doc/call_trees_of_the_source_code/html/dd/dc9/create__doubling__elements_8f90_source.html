162 ! If there is a doubling at the top of this region, let us add these elements.
163 ! The superbrick implements a symmetric four-to-two doubling and therefore replaces
164 ! a basic regular block of 2 x 2 = 4 elements.
165 ! We have imposed that NEX be a multiple of 16 therefore we know that we can always create
166 ! these 2 x 2 blocks because NEX_PER_PROC_XI / ratio_sampling_array(ilayer) and
167 ! NEX_PER_PROC_ETA / ratio_sampling_array(ilayer) are always divisible by 2.
168 
169  if (use_one_layer_sb) then
170  call define_superbrick_one_layer(x_superbrick,y_superbrick,z_superbrick,ibool_superbrick,iboun_sb)
171  nspec_sb = nspec_superbrick_1l
172  iz_elem = ner(ilayer)
173  step_mult = 2
174  else
175  if (iregion_code == iregion_outer_core .and. ilayer == ilast_region &
176  .and. (cut_superbrick_xi .or. cut_superbrick_eta)) then
177  nspec_sb = nspec_doubling_basicbrick
178  step_mult = 1
179  else
180  call define_superbrick(x_superbrick,y_superbrick,z_superbrick,ibool_superbrick,iboun_sb)
181  nspec_sb = nspec_doubling_superbrick
182  step_mult = 2
183  endif
184  ! the doubling is implemented in the last two radial elements
185  ! therefore we start one element before the last one
186  iz_elem = ner(ilayer) - 1
187  endif
188 
189  ! stores original value
190  ispec0 = ispec
191 
192  ! counts number of elements for this layer
193  nelements = nex_per_proc_xi/(step_mult*ratio_sampling_array(ilayer)) &
194  * nex_per_proc_eta/(step_mult*ratio_sampling_array(ilayer)) &
195  * nspec_sb
196 
197  ! fill mapping to be able to parallelize loops below
198  allocate(map_ispec(nelements),stat=ier)
199  if (ier /= 0) stop 'Error allocating map_ispec'
200 
201  do ix_elem = 1,nex_per_proc_xi,step_mult*ratio_sampling_array(ilayer)
202  do iy_elem = 1,nex_per_proc_eta,step_mult*ratio_sampling_array(ilayer)
203  do ispec_superbrick = 1,nspec_sb
204  ! counts in increasing order 1,2,3,..
205  ielem = ispec_superbrick + nspec_sb * ( (iy_elem-1)/(step_mult*ratio_sampling_array(ilayer)) &
206  + (ix_elem-1)/(step_mult*ratio_sampling_array(ilayer)) &
207  * nex_per_proc_eta/(step_mult*ratio_sampling_array(ilayer)))
208  map_ispec(ielem) = ispec0 + ielem
209  ! check
210  if (map_ispec(ielem) > nspec) call exit_mpi(myrank,'ispec greater than nspec in mesh creation')
211  enddo
212  enddo
213  enddo
214

216 !$OMP PARALLEL DEFAULT(SHARED) &
217 !$OMP PRIVATE(ix_elem,iy_elem,ispec_superbrick,ignod, &

240  endif

243  case_xi = 1
244  else
245  case_xi = 2
246  endif
247  endif
248  endif
249  ! for eta direction
250  if (.not. cut_superbrick_eta) then
251  if (mod((iy_elem-1),(2*step_mult*ratio_sampling_array(ilayer))) == 0) then
252  case_eta = 1
253  else
254  case_eta = 2
255  endif
256  else
257  if (offset_proc_eta == 0) then
258  if (mod((iy_elem-1),(2*step_mult*ratio_sampling_array(ilayer))) == 0) then
259  case_eta = 1
260  else

262  endif
263  else
264  if (mod((iy_elem-1),(2*step_mult*ratio_sampling_array(ilayer))) /= 0) then
265  case_eta = 1
266  else
267  case_eta = 2
268  endif
269  endif
270  endif
271  ! determine the current sub-block
272  if (case_xi == 1) then
273  if (case_eta == 1) then

301  enddo

313  !ispec = ispec + 1
314

327  endif

331  endif
332  !! ymin & ymax
333  if (iy_elem == 1) then

338  impicut_eta(2,ispec) = iboun_sb(ispec_superbrick,4)
339  if (iproc_eta == nproc_eta-1) iboun(4,ispec)= iboun_sb(ispec_superbrick,4)
340  endif

342  if (ilayer == ifirst_region) then
343  iboun(6,ispec)= iboun_sb(ispec_superbrick,6)
344  endif
345  if (ilayer == ilast_region .and. iz_elem == 1) then
346  iboun(5,ispec)= iboun_sb(ispec_superbrick,5)
347  endif
348 
349  ! define the doubling flag of this element
350  idoubling(ispec) = doubling_index(ilayer)
351 
352  ! save the radii of the nodes before modified through compute_element_properties()
353  if (ipass == 2 .and. save_boundary_mesh .and. iregion_code == iregion_crust_mantle) then
354  r1 = sqrt(xelm(1)*xelm(1) + yelm(1)*yelm(1) + zelm(1)*zelm(1))

363 
364  ! compute several rheological and geometrical properties for this spectral element
365  call compute_element_properties(ispec,iregion_code,idoubling,ipass, &
366  xstore,ystore,zstore,nspec, &

380  is_superbrick = .true.
381  call get_jacobian_discontinuities(ispec,ix_elem,iy_elem,rmin,rmax, &
382  r1,r2,r3,r4,r5,r6,r7,r8, &
383  xstore(:,:,:,ispec),ystore(:,:,:,ispec),zstore(:,:,:,ispec),dershape2d_bottom, &
384  ibelm_moho_top,ibelm_moho_bot,ibelm_400_top,ibelm_400_bot,ibelm_670_top,ibelm_670_bot, &
385  normal_moho,normal_400,normal_670,jacobian2d_moho,jacobian2d_400,jacobian2d_670, &
386  ispec2d_moho_top,ispec2d_moho_bot,ispec2d_400_top, &
387  ispec2d_400_bot,ispec2d_670_top,ispec2d_670_bot, &
388  nspec2d_moho,nspec2d_400,nspec2d_670,r_moho,r_400,r_670, &
389  is_superbrick,use_one_layer_sb,ispec_superbrick,nex_eta_moho,honor_1d_spherical_moho)
390  endif
391 
392  ! end of loops on the mesh doubling elements
393  enddo
394  enddo
395  enddo
396 !$OMP enddo
397 !$OMP END PARALLEL
398 
399  ! end index
400  ispec = ispec0 + nelements
401
