266 subroutine read_values_adios(value_handle, var_name, iproc, ir, nspec, data)
267 
268  use constants, only: custom_real,ngllx,nglly,ngllz,iregion_crust_mantle,iregion_inner_core,iregion_outer_core
269 
270  implicit none
271  ! Parameters
272  integer(kind=8), intent(in) :: value_handle
273  character(len=*), intent(in) :: var_name
274  integer, intent(in) :: iproc, ir, nspec
275  real(kind=CUSTOM_REAL), dimension(:,:,:,:), intent(inout) :: data
276  ! Variables
277  integer(kind=8), dimension(1) :: start, count_ad
278  integer(kind=8) :: sel
279  integer :: offset, ier
280  character(len=128) :: data_name
281  character(len=8) :: reg_name
282  logical :: is_kernel
283 
284  ! note: we can either visualize
285  ! wavespeed arrays (in DATABASES_MPI/solver_meshfiles.bp)
286  ! (e.g. rho,vp,vs,vph,vpv,vsh,..)
287  ! or
288  ! sensitivity kernels (in OUTPUT_FILES/kernels.bp)
289  ! (e.g. rho_kl,alpha_kl,beta_kl,alphah_kl,alphav_kl,betah_kl,..)
290  !
291  ! unfortunately, they have different naming conventions for different Earth regions:
292  ! - wavespeed arrays: reg1/vp/, reg2/vp/, reg3/vp/, ..
293  ! - kernels: alpha_kl_crust_mantle/, alpha_kl_outer_core/, alpha_kl_inner_core/,..
294  ! here we try to estimate the type by the ending of the variable name given by the user,
295  ! i.e. if the ending is "_kl" we assume it is a kernel name
296  !
297  is_kernel = .false.
298  if (len_trim(var_name) > 3) then
299  if (var_name(len_trim(var_name)-2:len_trim(var_name)) == '_kl') then
300  is_kernel = .true.
301  endif
302  endif
303 
304  ! determines full data array name
305  if (is_kernel) then
306  ! for kernel name: alpha_kl, .. , **_kl only:
307  ! adds region name to get kernel_name
308  ! for example: var_name = "alpha_kl" -> alpha_kl_crust_mantle
309  !
310  ! note: this must match the naming convention used
311  ! in file save_kernels_adios.F90
312  select case (ir)
313  case (iregion_crust_mantle)
314  data_name = trim(var_name) // "_crust_mantle"
315  case (iregion_outer_core)
316  data_name = trim(var_name) // "_outer_core"
317  case (iregion_inner_core)
318  data_name = trim(var_name) // "_inner_core"
319  case default
320  stop 'Error wrong region code in read_values_adios() routine'
321  end select
322  else
323  ! for wavespeed name: rho,vp,..
324  ! adds region name: var_name = "rho" -> reg1/rho
325  write(reg_name, '(a,i1, a)') trim("reg"), ir, "/"
326  data_name = trim(reg_name) // trim(var_name)
327  endif
328 
329  ! reads in data offset
330  call adios_selection_writeblock(sel, iproc)
331 
332  call adios_schedule_read(value_handle, sel, trim(data_name) // "/offset", 0, 1, offset, ier)
333  if (ier /= 0) then
334  print *
335  print * ,'Error adios: could not read parameter: ',trim(data_name) // "/offset"
336  print * ,' please check if your input data file is correct...'
337  print *
338  stop 'Error adios adios_schedule_read() for data offset'
339  endif
340 
341  call adios_perform_reads(value_handle, ier)
342  if (ier /= 0 ) stop 'Error adios: perform reading data offset failed'
343 
344  ! reads in data array
345  start(1) = offset
346  count_ad(1) = ngllx * nglly * ngllz * nspec
347  call adios_selection_boundingbox (sel , 1, start, count_ad)
348 
349  call adios_schedule_read(value_handle, sel, trim(data_name) // "/array", 0, 1, data, ier)
350  if (ier /= 0 ) stop 'Error adios reading data array'
351 
352  call adios_perform_reads(value_handle, ier)
353  if (ier /= 0 ) stop 'Error adios perform reading of data array'
354 
355 end subroutine read_values_adios
