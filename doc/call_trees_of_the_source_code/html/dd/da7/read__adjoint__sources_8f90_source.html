28  subroutine read_adjoint_sources()
29 
30 ! reads in adjoint source files
31 !
32 ! note: depending on the number of adjoint source files and cluster filesystem,
33 ! this can take a very long time and slow down adjoint simulations considerably.
34 !
35 ! when flag IO_ASYNC_COPY is set, a new parallel thread is created to read in adjoint source files,
36 ! while the main processes can continue computing. we thus overlap file i/o with computations.
37 ! in case the heavy computations take long enough, this overlap should perfectly hide the i/o latency.
38 !
39 ! the costs are then for the additional memory, i.e. buffer array (buffer_sourcearrays), and the
40 ! copying operation of data from the buffer array to the actual adj_sourcearrays array.
41 
42  use specfem_par
43 
44  implicit none
45 
46  ! local parameters
47  integer :: it_sub_adj
48  ! debug timing
49  !double precision, external :: wtime
50  !double precision :: tstart
51 
52 
53  ! we already checked that nadj_rec_local > 0 before calling this routine
54 
55  ! debug timing
56  !tstart = wtime()
57 
58  ! determines chunk_number
59  it_sub_adj = ceiling( dble(it)/dble(ntstep_between_read_adjsrc) )
60 
61  ! debug
62  !print *,'read adjoint sources: it_sub_adj = ',it_sub_adj
63 
64  ! asynchronously reads in adjoint source files
65  if (io_async_copy .and. nstep_sub_adj > 1) then
66  ! handles file input/output thread
67  if (it == it_begin) then
68  ! creates new io thread for reading in sources
69  call read_adj_io_thread(it_sub_adj)
70 
71  ! first chunk of adjoint sources must ready at beginning, so we wait.
72  ! waits for previous read to finish and
73  ! copy over buffered data into tmp_sourcearray
74  call sync_adj_io_thread(adj_sourcearrays)
75 
76  else
77  ! waits for previous read to finish and
78  ! copy over buffered data into tmp_sourcearray
79  call sync_adj_io_thread(adj_sourcearrays)
80  endif
81 
82  ! checks if next chunk necessary
83  if (it_sub_adj < nstep_sub_adj) then
84  ! starts thread to read in next junk
85  it_sub_adj = it_sub_adj + 1
86 
87  ! creates new io thread for reading in sources
88  call read_adj_io_thread(it_sub_adj)
89  endif
90 
91  else
92  ! synchronous read routine
93 
94  ! reads in local adjoint sources
95  call read_adjoint_sources_local(adj_sourcearrays,nadj_rec_local,it_sub_adj)
96 
97  endif
98 
99  ! debug timing
100  !print *,'read adjoint sources: elapsed time = ',wtime() - tstart
101  !print *
102 
103  end subroutine read_adjoint_sources

109  subroutine read_adjoint_sources_local(sourcearrays,nadj_rec_local,it_sub_adj)
110 
111 ! reads in local adjoint source files
112 
113  use specfem_par, only: myrank,nproctot_val, &
114  nrec,islice_selected_rec,station_name,network_name, &
115  xi_receiver,eta_receiver,gamma_receiver,nu,xigll,yigll,zigll, &
116  iadjsrc_len,iadjsrc,nstep_sub_adj, &
117  dt,custom_real,ndim,ngllx,nglly,ngllz,ntstep_between_read_adjsrc,max_string_len,read_adjsrc_asdf
118 
119  implicit none
120 
121  integer,intent(in) :: nadj_rec_local
122  integer,intent(in) :: it_sub_adj
123  real(kind=CUSTOM_REAL), dimension(NDIM,NGLLX,NGLLY,NGLLZ,nadj_rec_local,NTSTEP_BETWEEN_READ_ADJSRC) :: sourcearrays
124 
125  ! local parameters
126  real(kind=CUSTOM_REAL), dimension(:,:,:,:,:), allocatable :: tmp_sourcearray
127  integer :: irec,irec_local,itime,ier
128  character(len=MAX_STRING_LEN) :: adj_source_file
129 
130  ! debug
131  !print *,'reading adjoint sources local:',myrank,' - chunk ',it_sub_adj,'out of ',NSTEP_SUB_ADJ, &
132  ! ' for local adjoint sources = ',nadj_rec_local
133 
134  ! checks chunk number
135  if (it_sub_adj < 1 .or. it_sub_adj > nstep_sub_adj) then
136  print *,'Error reading adjoint sources: chunk number ',it_sub_adj,'is invalid'
137  call exit_mpi(myrank,'Error reading adjoint sources with invalid chunk number')
138  endif
139 
140  ! allocates temporary source array
141  allocate(tmp_sourcearray(ndim,ngllx,nglly,ngllz,ntstep_between_read_adjsrc),stat=ier)
142  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating array tmp_sourcearray')
143 
144  ! initializes
145  tmp_sourcearray(:,:,:,:,:) = 0._custom_real
146 
147  ! counter
148  irec_local = 0
149 
150  ! loops over all adjoint sources
151  do irec = 1, nrec
152  ! checks that the source slice number is okay
153  if (islice_selected_rec(irec) < 0 .or. islice_selected_rec(irec) > nproctot_val-1) then
154  print *,'Error rank ',myrank,': adjoint source slice index ',islice_selected_rec(irec), &
155  ' is out of bounds ',nproctot_val-1
156  call exit_mpi(myrank,'Error adjoint source has wrong source slice number in adjoint simulation')
157  endif
158 
159  ! compute source arrays for adjoint sources within this rank's slice
160  if (myrank == islice_selected_rec(irec)) then
161  ! increases counter
162  irec_local = irec_local + 1
163 
164  ! adjoint source file name **net**.**sta**
165  if (read_adjsrc_asdf) then
166  adj_source_file = trim(network_name(irec))//'_'//trim(station_name(irec))
167  else
168  adj_source_file = trim(network_name(irec))//'.'//trim(station_name(irec))
169  endif
170 
171  ! reads in *.adj files
172  call compute_arrays_source_adjoint(myrank,adj_source_file, &
173  xi_receiver(irec),eta_receiver(irec),gamma_receiver(irec), &
174  nu(:,:,irec),tmp_sourcearray, &
175  xigll,yigll,zigll, &
176  iadjsrc_len(it_sub_adj),iadjsrc,it_sub_adj, &
177  nstep_sub_adj,ntstep_between_read_adjsrc,dt)
178 
179  ! stores source array
180  ! note: the adj_sourcearrays has a time stepping from 1 to NTSTEP_BETWEEN_READ_ADJSRC
181  ! this gets overwritten every time a new block/chunk is read in
182  do itime = 1,ntstep_between_read_adjsrc
183  sourcearrays(:,:,:,:,irec_local,itime) = tmp_sourcearray(:,:,:,:,itime)
184  enddo
185 
186  endif
187  enddo
188 
189  ! checks that number of read sources is valid
190  if (irec_local /= nadj_rec_local) then
191  call exit_mpi(myrank,'irec_local /= nadj_rec_local in adjoint simulation')
192  endif
193 
194  ! frees temporary array
195  deallocate(tmp_sourcearray)
196 
197  end subroutine read_adjoint_sources_local

203  subroutine check_adjoint_sources(irec,nadj_files_found)
204 
205  use specfem_par
206 
207  implicit none
208 
209  integer,intent(in) :: irec
210  ! counter
211  integer,intent(inout) :: nadj_files_found
212 
213  ! local parameters
214  double precision :: junk
215  integer :: icomp,itime
216  integer :: ier
217  character(len=MAX_STRING_LEN) :: filename,adj_source_file,path_to_add
218  character(len=3),dimension(NDIM) :: comp
219  character(len=2) :: bic
220 
221  ! checks **net**.**sta**.**MX**.adj files for correct number of time steps
222  adj_source_file = trim(network_name(irec))//'.'//trim(station_name(irec))
223  ! adjoint source file name **net**.**sta**
224 
225  ! bandwidth code
226  call band_instrument_code(dt,bic)
227  comp(1) = bic(1:2)//'N'
228  comp(2) = bic(1:2)//'E'
229  comp(3) = bic(1:2)//'Z'
230 
231  ! loops over file components E/N/Z
232  do icomp = 1,ndim
233 
234  ! opens adjoint source file for this component
235  filename = 'SEM/'//trim(adj_source_file) // '.'// comp(icomp) // '.adj'
236 
237  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
238  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
239  filename = path_to_add(1:len_trim(path_to_add))//filename(1:len_trim(filename))
240  endif
241 
242  open(unit=iin,file=trim(filename),status='old',action='read',iostat=ier)
243 
244  ! checks if file opens/exists
245  if (ier /= 0) then
246  ! adjoint source file not found
247  ! stops simulation
248  call exit_mpi(myrank, &
249  'file '//trim(filename)//' not found, please check with your STATIONS_ADJOINT file')
250  endif
251 
252  ! checks length of file
253  itime = 0
254  do while(ier == 0)
255  read(iin,*,iostat=ier) junk,junk
256  if (ier == 0 ) itime = itime + 1
257  enddo
258 
259  ! checks length
260  if (itime /= nstep) then
261  print *,'adjoint source error: ',trim(filename),' has length',itime,' but should be',nstep
262  call exit_mpi(myrank, &
263  'file '//trim(filename)//' length is wrong, please check your adjoint sources and your simulation duration')
264  endif
265 
266  ! updates counter for found files
267  nadj_files_found = nadj_files_found + 1
268 
269  ! closes file
270  close(iin)
271 
272  enddo
273 
274  end subroutine check_adjoint_sources
