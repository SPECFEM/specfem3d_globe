56 cl_kernel *setup_ocl_memset (int do_setup) {
57 
58  static int inited = 0;
59  static cl_kernel memset_kern;
60  cl_int errcode;
61 
62  if (do_setup) {
63  if (!inited) {
64  // creates openCL kernel
65  cl_program memset_program = clCreateProgramWithSource(mocl.context, 1,
66  memset_kern_code, 0,
67  clck_(&errcode));
68  clCheck (clBuildProgram (memset_program, 0, NULL, NULL, NULL, NULL));
69  memset_kern = clCreateKernel (memset_program, "memset_uint4", clck_(&errcode));
70  inited = 1;
71  }
72  } else {
73  // releases kernel
74  if (inited) { clCheck(clReleaseKernel (memset_kern)); }
75  }
76 
77  return &memset_kern;
78 }

82 void moclEnqueueFillBuffer (cl_mem *buffer, int val, size_t size_byte) {
83 
84  // creates/gets OpenCL memset kernel
85  cl_kernel *memset_kern = setup_ocl_memset(1);
86 
87  // value to fill buffer
88  cl_int value = val;
89 
90  // gets size as number of integer values
91  int size;
92  size = size_byte / sizeof(cl_int);
93 
94  size_t global_work_size[2];
95  size_t local_work_size[2];
96  cl_uint idx = 0;
97 
98  clCheck (clSetKernelArg (*memset_kern, idx++, sizeof (cl_mem), (void *) buffer));
99  clCheck (clSetKernelArg (*memset_kern, idx++, sizeof (cl_int), (void *) &size));
100  clCheck (clSetKernelArg (*memset_kern, idx++, sizeof (cl_int), (void *) &value));
101 
102  int blocksize = BLOCKSIZE_TRANSFER;
103  int size_padded = ((int) ceil ((double) size / (double) blocksize)) * blocksize;
104 
105  int num_blocks_x, num_blocks_y;
106  get_blocks_xy (size_padded/blocksize, &num_blocks_x, &num_blocks_y);
107 
108  local_work_size[0] = blocksize;
109  local_work_size[1] = 1;
110  global_work_size[0] = num_blocks_x * blocksize;
111  global_work_size[1] = num_blocks_y;
112 
113  //debug
114  //printf("moclEnqueueFillBuffer: size %i value %i - work_size %zu %zu \n",size,value,local_work_size[0],global_work_size[0]);
115 
116  clCheck (clEnqueueNDRangeKernel (mocl.command_queue, *memset_kern, 2, NULL,
117  global_work_size, local_work_size, 0, NULL, NULL));
118  // synchronizes
119  clFinish (mocl.command_queue);
120 }

129 // copies integer array from CPU host to GPU device
130 void gpuCreateCopy_todevice_int (gpu_int_mem *d_array_addr_ptr, int *h_array, size_t size) {
131 
132  TRACE ("gpuCreateCopy_todevice_int");
133 
134 #ifdef USE_OPENCL
135  if (run_opencl) {
136  cl_int errcode;
137 
138  // allocates memory on GPU
139  d_array_addr_ptr->ocl = clCreateBuffer (mocl.context, CL_MEM_READ_WRITE,
140  size * sizeof (int), NULL, clck_(&errcode));
141 
142  // copies values onto GPU
143  clCheck (clEnqueueWriteBuffer (mocl.command_queue, d_array_addr_ptr->ocl, CL_TRUE, 0,
144  size*sizeof (int), h_array, 0, NULL, NULL));
145  }
146 #endif
147 #ifdef USE_CUDA
148  if (run_cuda) {
149  // allocates memory on GPU
150  //
151  // note: cudaMalloc uses a double-pointer, such that it can return an error code in case it fails
152  // we thus pass the address to the pointer above (as void double-pointer) to have it
153  // pointing to the correct pointer of the array here
154  print_CUDA_error_if_any(cudaMalloc((void**)&d_array_addr_ptr->cuda,size*sizeof(int)),12001);
155 
156  // copies values onto GPU
157  //
158  // note: cudaMemcpy uses the pointer to the array, we thus re-cast the value of
159  // the double-pointer above to have the correct pointer to the array
160  print_CUDA_error_if_any(cudaMemcpy((int*) d_array_addr_ptr->cuda,h_array,size*sizeof(int),cudaMemcpyHostToDevice),12002);
161  }
162 #endif
163 }
