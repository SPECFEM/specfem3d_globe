36  it,it_begin,it_end,seismo_current,seismo_offset, seismograms,ntstep_between_output_seismos, &
37  write_seismograms_by_master,output_seismos_asdf, &
38  it_adj_written,moment_der,sloc_der,shdur_der,stshift_der
39

48  ! timing

54 
55  ! compute & store the seismograms only if there is at least one receiver located in this slice
56  if (nrec_local > 0) then
57  ! gets resulting array values onto CPU
58  if (gpu_mode) then
59  ! gets field values from GPU

71  if (gpu_async_copy) then
72  call transfer_seismo_from_device_async(mesh_pointer, &
73  displ_crust_mantle,b_displ_crust_mantle, &

75  ibool_crust_mantle)
76  endif

78  endif

81  select case (simulation_type)
82  case (1)
83  call compute_seismograms(nglob_crust_mantle,displ_crust_mantle, &
84  seismo_current,seismograms)
85  case (2)
86  call compute_seismograms_adjoint(displ_crust_mantle, &
87  eps_trace_over_3_crust_mantle, &
88  epsilondev_xx_crust_mantle,epsilondev_yy_crust_mantle,epsilondev_xy_crust_mantle, &
89  epsilondev_xz_crust_mantle,epsilondev_yz_crust_mantle, &
90  it_adj_written, &
91  moment_der,sloc_der,stshift_der,shdur_der, &
92  seismograms)
93  case (3)
94  call compute_seismograms(nglob_crust_mantle_adjoint,b_displ_crust_mantle, &
95  seismo_current,seismograms)
96  end select
97  endif ! nrec_local

100  if (seismo_current == ntstep_between_output_seismos .or. it == it_end) then
101  ! timing
102  write_time_begin = wtime()
103 
104  ! checks if anything to do
105  ! note: ASDF uses parallel hdf5 that defines the MPI communicator group that the solver is
106  ! run with. this means every processor in the group is needed for write_seismograms

108  ! writes out seismogram files
109  select case (simulation_type)
110  case (1,3)
111  ! forward/reconstructed wavefields
112  call write_seismograms_to_file()
113  case (2)
114  ! adjoint wavefield
115  call write_adj_seismograms(it_adj_written)
116  it_adj_written = it
117  end select
118  endif
119 
120  ! synchronizes processes (waits for all processes to finish writing)
121  call synchronize_all()
122 
123  ! resets current seismogram position
124  seismo_offset = seismo_offset + seismo_current
125  seismo_current = 0
126 
127  ! user output
128  if (myrank == 0) then
129  ! timing
130  write_time = wtime() - write_time_begin
131  ! output
132  write(imain,*)
133  write(imain,*) 'Total number of time steps written: ', it-it_begin+1
134  write(imain,*)
135  if (write_seismograms_by_master) then
136  write(imain,*) 'Writing the seismograms by master proc alone took ',write_time,' seconds'
137  else
138  write(imain,*) 'Writing the seismograms in parallel took ',write_time,' seconds'
139  endif
140  write(imain,*)
141  call flush_imain()
142  endif
143  endif

151 ! write seismograms to files
152  subroutine write_seismograms_to_file()
153 
154  use constants_solver, only: max_string_len,custom_real,ndim,imain,iout,itag
155 
156  use specfem_par, only: &
157  nproctot_val,myrank,nrec,nrec_local, &
158  number_receiver_global,seismograms, &
159  islice_selected_rec, &
160  seismo_offset,seismo_current, &
161  output_seismos_ascii_text, &
162  output_seismos_asdf, &
163  ntstep_between_output_seismos, &
164  save_all_seismos_in_one_file,use_binary_for_large_file, &
165  output_files, &
166  write_seismograms_by_master
167 
168  implicit none
169 
170  ! local parameters
171  real(kind=CUSTOM_REAL), dimension(:,:), allocatable :: one_seismogram
172 
173  integer :: iproc,sender,irec_local,irec,ier,receiver
174  integer :: nrec_local_received
175  integer :: total_seismos
176  integer,dimension(:),allocatable:: islice_num_rec_local
177  character(len=MAX_STRING_LEN) :: sisname
178 
179  ! allocates single station seismogram
180  allocate(one_seismogram(ndim,ntstep_between_output_seismos),stat=ier)
181  if (ier /= 0) call exit_mpi(myrank,'Error while allocating one temporary seismogram')
182 
183  ! write out seismograms: all processes write their local seismograms themselves
184  if (.not. write_seismograms_by_master) then
185 
186  ! all the processes write their local seismograms themselves
187  if (output_seismos_ascii_text .and. save_all_seismos_in_one_file) then
188  write(sisname,'(A,I5.5)') '/all_seismograms_node_',myrank
189 
190  if (use_binary_for_large_file) then
191  if (seismo_offset == 0) then
192  open(unit=iout,file=trim(output_files)//trim(sisname)//'.bin',status='unknown',form='unformatted',action='write')
193  else
194  open(unit=iout,file=trim(output_files)//trim(sisname)//'.bin',status='old', &
195  form='unformatted',position='append',action='write')
196  endif
197  else
198  if (seismo_offset == 0) then
199  open(unit=iout,file=trim(output_files)//trim(sisname)//'.ascii',status='unknown',form='formatted',action='write')
200  else
201  open(unit=iout,file=trim(output_files)//trim(sisname)//'.ascii',status='old', &
202  form='formatted',position='append',action='write')
203  endif
204  endif
205  endif
206 
207  ! initializes the ASDF data structure by allocating arrays
208  if (output_seismos_asdf) call init_asdf_data(nrec_local)
209 
210  ! loop on all the local receivers
211  do irec_local = 1,nrec_local
212 
213  ! get global number of that receiver
214  irec = number_receiver_global(irec_local)
215 
216  one_seismogram(:,:) = seismograms(:,irec_local,:)
217 
218  ! write this seismogram
219  ! note: ASDF data structure is given in module
220  ! stores all traces into ASDF container in case
221  call write_one_seismogram(one_seismogram,irec,irec_local)
222  enddo
223 
224  ! writes out ASDF container to the file
225  if (output_seismos_asdf) then
226  call write_asdf()
227  ! deallocate the container
228  call close_asdf_data()
229  endif
230 
231  ! create one large file instead of one small file per station to avoid file system overload
232  if (output_seismos_ascii_text .and. save_all_seismos_in_one_file) close(iout)
233 
234  else
235  ! WRITE_SEISMOGRAMS_BY_MASTER
236 
237  ! only the master process does the writing of seismograms and
238  ! collects the data from all other processes
239  if (myrank == 0) then
240  ! on the master, gather all the seismograms
241 
242  ! create one large file instead of one small file per station to avoid file system overload
243  if (output_seismos_ascii_text .and. save_all_seismos_in_one_file) then
244  write(sisname,'(A)') '/all_seismograms'
245 
246  if (use_binary_for_large_file) then
247  if (seismo_offset == 0) then
248  open(unit=iout,file=trim(output_files)//trim(sisname)//'.bin',status='unknown',form='unformatted',action='write')
249  else
250  open(unit=iout,file=trim(output_files)//trim(sisname)//'.bin',status='old', &
251  form='unformatted',position='append',action='write')
252  endif
253  else
254  if (seismo_offset == 0) then
255  open(unit=iout,file=trim(output_files)//trim(sisname)//'.ascii',status='unknown',form='formatted',action='write')
256  else
257  open(unit=iout,file=trim(output_files)//trim(sisname)//'.ascii',status='old', &
258  form='formatted',position='append',action='write')
259  endif
260  endif
261 
262  endif
263 
264  ! counts number of local receivers for each slice
265  allocate(islice_num_rec_local(0:nproctot_val-1),stat=ier)
266  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating islice_num_rec_local')
267 
268  islice_num_rec_local(:) = 0
269  do irec = 1,nrec
270  iproc = islice_selected_rec(irec)
271  ! checks iproc value
272  if (iproc < 0 .or. iproc >= nproctot_val) then
273  print *,'Error :',myrank,'iproc = ',iproc,'NPROCTOT = ',nproctot_val
274  call exit_mpi(myrank,'Error iproc in islice_selected_rec')
275  endif
276  ! sums number of receivers for each slice
277  islice_num_rec_local(iproc) = islice_num_rec_local(iproc) + 1
278  enddo
279 
280  total_seismos = 0
281 
282  ! loop on all the slices
283  do iproc = 0,nproctot_val-1
284 
285  ! communicates only with processes that contain local receivers (to minimize MPI chatter)
286  if (islice_num_rec_local(iproc) == 0 ) cycle
287 
288  ! receive except from proc 0, which is me and therefore I already have this value
289  sender = iproc
290  if (iproc == 0) then
291  ! master is current slice
292  nrec_local_received = nrec_local
293  else
294  ! receives info from slave processes
295  call recv_singlei(nrec_local_received,sender,itag)
296  if (nrec_local_received <= 0) call exit_mpi(myrank,'Error while receiving local number of receivers')
297  endif
298  if (nrec_local_received > 0) then
299  do irec_local = 1,nrec_local_received
300  ! receive except from proc 0, which is myself and therefore I already have these values
301  if (iproc == 0) then
302  ! get global number of that receiver
303  irec = number_receiver_global(irec_local)
304  one_seismogram(:,:) = seismograms(:,irec_local,:)
305  else
306  ! receives info from slave processes
307  call recv_singlei(irec,sender,itag)
308  if (irec < 1 .or. irec > nrec) call exit_mpi(myrank,'Error while receiving global receiver number')
309  call recv_cr(one_seismogram,ndim*seismo_current,sender,itag)
310  endif
311 
312  total_seismos = total_seismos + 1
313  ! write this seismogram
314  call write_one_seismogram(one_seismogram,irec,irec_local)
315 
316  enddo
317  endif
318  enddo
319  deallocate(islice_num_rec_local)
320 
321  write(imain,*)
322  write(imain,*) 'Total number of receivers saved is ',total_seismos,' out of ',nrec
323  write(imain,*)
324 
325  if (total_seismos /= nrec) call exit_mpi(myrank,'incorrect total number of receivers saved')
326 
327  ! create one large file instead of one small file per station to avoid file system overload
328  if (save_all_seismos_in_one_file) close(iout)
329 
330  else
331  ! on the nodes, send the seismograms to the master
332  receiver = 0
333  ! only sends if this slice contains receiver stations
334  if (nrec_local > 0) then
335  call send_singlei(nrec_local,receiver,itag)
336  do irec_local = 1,nrec_local
337  ! get global number of that receiver
338  irec = number_receiver_global(irec_local)
339  call send_singlei(irec,receiver,itag)
340 
341  one_seismogram(:,:) = seismograms(:,irec_local,:)
342  call send_cr(one_seismogram,ndim*seismo_current,receiver,itag)
343  enddo
344  endif
345  endif
346  endif ! WRITE_SEISMOGRAMS_BY_MASTER
347 
348  deallocate(one_seismogram)
349 
350  end subroutine write_seismograms_to_file

356  subroutine write_one_seismogram(one_seismogram,irec,irec_local)
357 
358  use constants_solver, only: max_string_len,custom_real,ndim,degrees_to_radians, &
359  max_length_station_name,max_length_network_name
360 
361  use specfem_par, only: &
362  myrank, &
363  station_name,network_name,stlat,stlon, &
364  dt, &
365  seismo_current, &
366  output_seismos_ascii_text,output_seismos_sac_alphanum,output_seismos_asdf, &
367  output_seismos_sac_binary,rotate_seismograms_rt,ntstep_between_output_seismos
368 
369  use specfem_par, only: &
370  cmt_lat => cmt_lat_sac,cmt_lon => cmt_lon_sac
371 
372  implicit none
373 
374  integer :: irec,irec_local
375  real(kind=CUSTOM_REAL), dimension(NDIM,NTSTEP_BETWEEN_OUTPUT_SEISMOS) :: one_seismogram
376 
377  ! local parameters
378  real(kind=CUSTOM_REAL), dimension(5,NTSTEP_BETWEEN_OUTPUT_SEISMOS) :: seismogram_tmp
379  integer :: iorientation,length_station_name,length_network_name
380 
381  character(len=4) :: chn
382  character(len=MAX_STRING_LEN) :: sisname,sisname_big_file
383  character(len=2) :: bic
384 
385  ! variables used for calculation of backazimuth and
386  ! rotation of components if ROTATE_SEISMOGRAMS=.true.
387  integer :: ior_start,ior_end
388  double precision :: backaz
389  double precision :: phi
390  real(kind=CUSTOM_REAL) :: cphi,sphi
391  integer :: isample
392 
393  ! initializes
394  seismogram_tmp(:,:) = 0.0_custom_real
395 
396  ! get band code
397  call band_instrument_code(dt,bic)
398 
399  if (rotate_seismograms_rt) then ! iorientation 1=N,2=E,3=Z,4=R,5=T
400  ior_start=3 ! starting from Z
401  ior_end =5 ! ending with T => ZRT
402  else
403  ior_start=1 ! starting from N
404  ior_end =3 ! ending with Z => NEZ
405  endif
406 
407  do iorientation = ior_start,ior_end ! BS BS changed according to ROTATE_SEISMOGRAMS_RT
408 
409  select case (iorientation)
410  case (1)
411  !chn = 'LHN'
412  chn = bic(1:2)//'N'
413  case (2)
414  !chn = 'LHE'
415  chn = bic(1:2)//'E'
416  case (3)
417  !chn = 'LHZ'
418  chn = bic(1:2)//'Z'
419  case (4)
420  !chn = 'LHR'
421  chn = bic(1:2)//'R'
422  case (5)
423  !chn = 'LHT'
424  chn = bic(1:2)//'T'
425  case default
426  call exit_mpi(myrank,'incorrect channel value')
427  end select
428 
429  if (iorientation == 4 .or. iorientation == 5) then ! LMU BS BS
430 
431  ! BS BS calculate backazimuth needed to rotate East and North
432  ! components to Radial and Transverse components
433  ! (back-azimuth returned in degrees between [0,360])
434  call get_backazimuth(cmt_lat,cmt_lon,stlat(irec),stlon(irec),backaz)
435 
436  phi = backaz
437 
438  ! back azimuth is the incoming direction of a raypath to a receiving station, i.e. the angle of
439  ! the incoming wave front arriving at the station measured between north and the direction to the epicenter in degrees.
440  ! (north corresponds to zero degrees)
441 
442  ! rotation angle phi takes opposite direction; to have radial direction pointing in outgoing direction
443  if (phi > 180.d0) then
444  phi = phi-180.d0
445  else if (phi < 180.d0) then
446  phi = phi+180.d0
447  else if (phi == 180.d0) then
448  phi = 0.d0
449  endif
450 
451  cphi=cos(phi*degrees_to_radians)
452  sphi=sin(phi*degrees_to_radians)
453 
454  ! BS BS do the rotation of the components and put result in
455  ! new variable seismogram_tmp
456  if (iorientation == 4) then ! radial component
457  do isample = 1,seismo_current
458  seismogram_tmp(iorientation,isample) = &
459  cphi * one_seismogram(1,isample) + sphi * one_seismogram(2,isample)
460  enddo
461  else if (iorientation == 5) then ! transverse component
462  do isample = 1,seismo_current
463  seismogram_tmp(iorientation,isample) = &
464  -1*sphi * one_seismogram(1,isample) + cphi * one_seismogram(2,isample)
465  enddo
466  endif
467 
468  else ! keep NEZ components
469  do isample = 1,seismo_current
470  seismogram_tmp(iorientation,isample) = one_seismogram(iorientation,isample)
471  enddo
472 
473  endif
474 
475  ! create the name of the seismogram file for each slice
476  ! file name includes the name of the station and the network
477  length_station_name = len_trim(station_name(irec))
478  length_network_name = len_trim(network_name(irec))
479 
480  ! check that length conforms to standard
481  if (length_station_name < 1 .or. length_station_name > max_length_station_name) &
482  call exit_mpi(myrank,'wrong length of station name')
483 
484  if (length_network_name < 1 .or. length_network_name > max_length_network_name) &
485  call exit_mpi(myrank,'wrong length of network name')
486 
487  ! create the name of the seismogram file using the station name and network name
488  ! using format: **net**.**sta**.channel
489  write(sisname,"('/',a,'.',a,'.',a3,'.sem')") network_name(irec)(1:length_network_name), &
490  station_name(irec)(1:length_station_name),chn
491 
492  ! create this name also for the text line added to the unique big seismogram file
493  write(sisname_big_file,"(a,'.',a,'.',a3,'.sem')") network_name(irec)(1:length_network_name), &
494  station_name(irec)(1:length_station_name),chn
495 
496  ! SAC output format
497  if (output_seismos_sac_alphanum .or. output_seismos_sac_binary ) then
498  call write_output_sac(seismogram_tmp,irec,iorientation,sisname,chn,phi)
499  endif
500 
501  ! ASCII output format
502  if (output_seismos_ascii_text) then
503  call write_output_ascii(seismogram_tmp,iorientation,sisname,sisname_big_file)
504  endif
505 
506  ! ASDF output format
507  if (output_seismos_asdf) then
508  call store_asdf_data(seismogram_tmp,irec_local,irec,chn,iorientation)
509  endif
510 
511  enddo ! do iorientation
512 
513  end subroutine write_one_seismogram

520 
521  subroutine write_adj_seismograms(it_adj_written)
522 
523  use constants, only: max_string_len,custom_real,iout
524 
525  use specfem_par, only: nstep,ntstep_between_output_seismos, &
526  dt,t0,local_tmp_path, &
527  seismograms,number_receiver_global,nrec_local, &
528  it
529 
530  implicit none
531 
532  integer,intent(in) :: it_adj_written
533 
534  ! local parameters
535  integer :: irec,irec_local
536  integer :: iorientation,isample
537 
538  character(len=4) :: chn
539  character(len=MAX_STRING_LEN) :: sisname
540  character(len=2) :: bic
541 
542  ! checks if anything to do
543  if (nrec_local <= 0 ) return
544 
545  call band_instrument_code(dt,bic)
546 
547  do irec_local = 1,nrec_local
548 
549  ! get global number of that receiver
550  irec = number_receiver_global(irec_local)
551 
552  do iorientation = 1,9
553  if (iorientation == 1) then
554  chn = 'SNN'
555  else if (iorientation == 2) then
556  chn = 'SEE'
557  else if (iorientation == 3) then
558  chn = 'SZZ'
559  else if (iorientation == 4) then
560  chn = 'SNE'
561  else if (iorientation == 5) then
562  chn = 'SNZ'
563  else if (iorientation == 6) then
564  chn = 'SEZ'
565  else if (iorientation == 7) then
566  !chn = 'LHN'
567  chn = bic(1:2)//'N'
568  else if (iorientation == 8) then
569  chn = bic(1:2)//'E'
570  else if (iorientation == 9) then
571  chn = bic(1:2)//'Z'
572  endif
573 
574 
575  ! create the name of the seismogram file for each slice
576  ! file name includes the name of the station, the network and the component
577  write(sisname,"(a3,'.',a1,i6.6,'.',a3,'.sem')") '/NT','S',irec,chn
578 
579  ! save seismograms in text format with no subsampling.
580  ! Because we do not subsample the output, this can result in large files
581  ! if the simulation uses many time steps. However, subsampling the output
582  ! here would result in a loss of accuracy when one later convolves
583  ! the results with the source time function
584  if (it <= ntstep_between_output_seismos) then
585  !open new file
586  open(unit=iout,file=local_tmp_path(1:len_trim(local_tmp_path))//sisname(1:len_trim(sisname)), &
