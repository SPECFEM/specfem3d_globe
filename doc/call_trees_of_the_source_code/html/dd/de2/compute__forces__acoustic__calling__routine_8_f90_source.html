28  subroutine compute_forces_acoustic()
29 
30 ! acoustic domains for forward or adjoint simulations (SIMULATION_TYPE == 1 or 2 )
31 
32  use specfem_par
33  use specfem_par_outercore
34  use specfem_par_movie, only: div_displ_outer_core
35  implicit none
36 
37  ! local parameters
38  real(kind=CUSTOM_REAL) :: timeval
39  ! non blocking MPI
40  ! iphase: iphase = 1 is for computing outer elements in the outer_core,
41  ! iphase = 2 is for computing inner elements in the outer core (former icall parameter)
42  integer :: iphase
43 
44  ! compute internal forces in the fluid region
45 
46  ! current simulated time
47  if (use_lddrk) then
48  timeval = real((dble(it-1)*dt+dble(c_lddrk(istage))*dt-t0)*scale_t_inv, kind=custom_real)
49  else
50  timeval = real((dble(it-1)*dt-t0)*scale_t_inv, kind=custom_real)
51  endif
52 
53  ! ****************************************************
54  ! big loop over all spectral elements in the fluid
55  ! ****************************************************
56 
57  ! distinguishes two runs: for elements on MPI interfaces, and elements within the partitions
58  do iphase = 1,2
59 
60  ! first, iphase == 1 for points on MPI interfaces (thus outer elements)
61  ! second, iphase == 2 for points purely inside partition (thus inner elements)
62  !
63  ! compute all the outer elements first, then sends out non blocking MPI communication
64  ! and continues computing inner elements (overlapping)
65 
66  if (.not. gpu_mode) then
67  ! on CPU
68  call compute_forces_outer_core(timeval,deltat,two_omega_earth, &
69  nspec_outer_core_rotation,nglob_outer_core, &
70  a_array_rotation,b_array_rotation, &
71  a_array_rotation_lddrk,b_array_rotation_lddrk, &
72  displ_outer_core,accel_outer_core, &
73  div_displ_outer_core,iphase)
74  else
75  ! on GPU
76  ! includes FORWARD_OR_ADJOINT == 1
77  call compute_forces_outer_core_gpu(mesh_pointer,iphase,timeval,1)
78 
79  ! initiates asynchronous MPI transfer
80  if (nproctot_val > 1) then
81  if (gpu_async_copy .and. iphase == 2) then
82  ! crust/mantle region
83  ! wait for asynchronous copy to finish
84  call sync_copy_from_device(mesh_pointer,iphase,buffer_send_scalar_outer_core,iregion_outer_core,1)
85  ! sends MPI buffers
86  call assemble_mpi_scalar_send_gpu(nproctot_val, &
87  buffer_send_scalar_outer_core,buffer_recv_scalar_outer_core, &
88  num_interfaces_outer_core,max_nibool_interfaces_oc, &
89  nibool_interfaces_outer_core, &
90  my_neighbours_outer_core, &
91  request_send_scalar_oc,request_recv_scalar_oc)
92  endif
93  endif
94  endif
95 
96  ! computes additional contributions to acceleration field
97  if (iphase == 1) then
98 
99  ! Stacey absorbing boundaries
100  if (nchunks_val /= 6 .and. absorbing_conditions) call compute_stacey_oc_forward()
101 
102  ! ****************************************************
103  ! ********** add matching with solid part **********
104  ! ****************************************************
105  call compute_coupling_fluid()
106 
107  endif ! iphase == 1
108 
109  ! assemble all the contributions between slices using MPI
110  if (nproctot_val > 1) then
111  ! in outer core
112  if (iphase == 1) then
113  ! sends out MPI interface data (non-blocking)
114  if (.not. gpu_mode) then
115  ! on CPU
116  call assemble_mpi_scalar_s(nproctot_val,nglob_outer_core, &
117  accel_outer_core, &
118  buffer_send_scalar_outer_core,buffer_recv_scalar_outer_core, &
119  num_interfaces_outer_core,max_nibool_interfaces_oc, &
120  nibool_interfaces_outer_core,ibool_interfaces_outer_core, &
121  my_neighbours_outer_core, &
122  request_send_scalar_oc,request_recv_scalar_oc)
123  else
124  ! on GPU
125  ! sends accel values to corresponding MPI interface neighbors
126 
127  ! preparation of the contribution between partitions using MPI
128  ! transfers MPI buffers to CPU
129  ! note: for asynchronous transfers, this transfers boundary region to host asynchronously. The
130  ! MPI-send is done after compute_forces_outer_core_gpu,
131  ! once the inner element kernels are launched, and the memcpy has finished.
132  call transfer_boun_pot_from_device(mesh_pointer,buffer_send_scalar_outer_core,1)
133 
134  if (.not. gpu_async_copy) then
135  ! for synchronous transfers, sending over MPI can directly proceed
136  ! outer core
137  call assemble_mpi_scalar_send_gpu(nproctot_val, &
138  buffer_send_scalar_outer_core,buffer_recv_scalar_outer_core, &
139  num_interfaces_outer_core,max_nibool_interfaces_oc, &
140  nibool_interfaces_outer_core, &
141  my_neighbours_outer_core, &
142  request_send_scalar_oc,request_recv_scalar_oc)
143  endif
144  endif
145  else
146  ! make sure the last communications are finished and processed
147  ! waits for send/receive requests to be completed and assembles values
148  if (.not. gpu_mode) then
149  ! on CPU
150  call assemble_mpi_scalar_w(nproctot_val,nglob_outer_core, &
151  accel_outer_core, &
152  buffer_recv_scalar_outer_core,num_interfaces_outer_core, &
153  max_nibool_interfaces_oc, &
154  nibool_interfaces_outer_core,ibool_interfaces_outer_core, &
155  request_send_scalar_oc,request_recv_scalar_oc)
156  else
157  ! on GPU
158  if (gpu_async_copy) then
159  ! while inner elements compute "Kernel_2", we wait for MPI to
160  ! finish and transfer the boundary terms to the device asynchronously
161  !
162  ! transfers MPI buffers onto GPU
163  call transfer_boundarypot_to_device(mesh_pointer,nproctot_val,buffer_recv_scalar_outer_core, &
164  num_interfaces_outer_core,max_nibool_interfaces_oc, &
165  request_recv_scalar_oc, &
166  iregion_outer_core,1)
167  endif
168 
169  ! waits for MPI send/receive requests to be completed and assembles values
170  call assemble_mpi_scalar_write_gpu(mesh_pointer,nproctot_val, &
171  buffer_recv_scalar_outer_core, &
172  num_interfaces_outer_core,max_nibool_interfaces_oc, &
173  request_send_scalar_oc,request_recv_scalar_oc, &
174  1) ! -- 1 == fwd accel
175  endif
176  endif ! iphase == 1
177  endif
178 
179  enddo ! iphase
180 
181  ! multiply by the inverse of the mass matrix
182 
183  if (.not. gpu_mode) then
184  ! on CPU
185  call multiply_accel_acoustic(nglob_outer_core,accel_outer_core,rmass_outer_core)
186  else
187  ! on GPU
188  ! includes FORWARD_OR_ADJOINT == 1
189  call multiply_accel_acoustic_gpu(mesh_pointer,1)
190  endif
191 
192 
193  ! time schemes
194  if (use_lddrk) then
195  ! Runge-Kutta scheme
196  call update_veloc_acoustic_lddrk()
197  else
198  ! Newmark time scheme
199  call update_veloc_acoustic_newmark()
200  endif
201 
202  end subroutine compute_forces_acoustic
