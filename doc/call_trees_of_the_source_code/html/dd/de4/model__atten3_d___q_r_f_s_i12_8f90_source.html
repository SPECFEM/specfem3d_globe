30 ! This file contains subroutines to read in and get values for
31 ! 3-D attenuation model QRFSI12 (Dalton, Ekstr"om, & Dziewonski, 2008)
32 !
33 ! C.A. Dalton, G. Ekstr\"om and A.M. Dziewonski, 2008.
34 ! The global attenuation structure of the upper mantle,
35 ! J. Geophys. Res., 113, B05317,10.1029/2006JB004394
36 !
37 ! Last edit: Colleen Dalton, March 25, 2008
38 !
39 ! Q1: what are theta and phi?
40 ! A1: input theta is colatitude in degrees, phi is longitude in degrees

42 ! Q2: units for radius?
43 ! A2: radius is given in km
44 !
45 ! Q3: what to do about core?

67  subroutine model_atten3d_qrfsi12_broadcast()
68 
69 ! standard routine to setup model
70 
71  use constants
72  use model_atten3d_qrfsi12_par
73 
74  implicit none
75 
76  ! local parameters
77  integer :: ier
78 
79  ! allocates model arrays
80  allocate(qrfsi12_q_dqmu(nkq,nsq), &
81  qrfsi12_q_spknt(nkq), &
82  qrfsi12_q_refdepth(ndepths_refq), &
83  qrfsi12_q_refqmu(ndepths_refq), &

94  call bcast_all_dp(qrfsi12_q_refqmu, ndepths_refq)
95 
96  ! user output
97  if (myrank == 0) then
98  write(imain,*)
99  write(imain,*) 'read 3D attenuation model'
100  write(imain,*) ' model: QRFSI12'
101  write(imain,*)
102  call flush_imain()
103  endif
104

111  subroutine read_atten_model_3d_qrfsi12()
112 
113  use constants
114  use model_atten3d_qrfsi12_par
115 
116  implicit none
117 
118  ! local parameters
119  integer :: j,k,l,m,ier
120  integer :: indexval,ll,mm
121  double precision :: v1,v2
122 
123  character(len=MAX_STRING_LEN) :: QRFSI12,QRFSI12_ref
124 
125 ! read in QRFSI12
126 ! hard-wire for now
127  qrfsi12='DATA/QRFSI12/QRFSI12.dat'
128  qrfsi12_ref='DATA/QRFSI12/ref_QRFSI12'
129 
130 ! get the dq model coefficients
131  open(unit=iin,file=qrfsi12,status='old',action='read',iostat=ier)
132  if (ier /= 0) then
133  write(imain,*) 'Error opening "', trim(qrfsi12), '": ', ier
134  call exit_mpi(0, 'Error model QRFSI12')
135  endif
136 
137  do k = 1,nkq
138  read(iin,*) indexval
139  j = 0
140  do l = 0,maxl_q
141  do m = 0,l
142  if (m == 0) then
143  j=j+1
144  read(iin,*)ll,mm,v1
145  qrfsi12_q_dqmu(k,j)=v1
146  else
147  j=j+2
148  read(iin,*)ll,mm,v1,v2
149  ! write(*,*) 'k,l,m,ll,mm:',k,l,m,ll,mm,v1
150  qrfsi12_q_dqmu(k,j-1)=2.*v1
151  qrfsi12_q_dqmu(k,j)=-2.*v2
152  endif
153  enddo
154  enddo
155  enddo
156  close(iin)
157 
158 ! get the depths (km) of the spline knots
159  qrfsi12_q_spknt(1) = 24.4d0
160  qrfsi12_q_spknt(2) = 75.d0
161  qrfsi12_q_spknt(3) = 150.d0
162  qrfsi12_q_spknt(4) = 225.d0
163  qrfsi12_q_spknt(5) = 300.d0
164  qrfsi12_q_spknt(6) = 410.d0
165  qrfsi12_q_spknt(7) = 530.d0
166  qrfsi12_q_spknt(8) = 650.d0
167 
168 ! get the depths and 1/Q values of the reference model
169  open(iin,file=qrfsi12_ref,status='old',action='read',iostat=ier)
170  if (ier /= 0) then
171  write(imain,*) 'Error opening "', trim(qrfsi12_ref), '": ', ier
172  call exit_mpi(0, 'Error model QRFSI12')
173  endif
174 
175  do j = 1,ndepths_refq
176  read(iin,*)qrfsi12_q_refdepth(j),qrfsi12_q_refqmu(j)
177  enddo
178  close(iin)
179 
180 
181  end subroutine read_atten_model_3d_qrfsi12

187  subroutine model_atten3d_qrfsi12(radius,theta,phi,Qmu,idoubling)
188 
189  use constants
190  use model_atten3d_qrfsi12_par
191 
192  implicit none
193 
194  double precision :: radius,theta,phi,Qmu
195  integer :: idoubling
196 
197  ! local parameters
198  integer :: i,j,k,n
199  integer :: ifnd
200  double precision :: smallq,dqmu,smallq_ref
201  real(kind=4) :: splpts(nkq),splcon(nkq),splcond(nkq)
202  real(kind=4) :: depth,ylat,xlon
203  real(kind=4) :: shdep(nsq)
204  real(kind=4) :: wk1(nsq),wk2(nsq),wk3(nsq)
205  real(kind=4) :: xlmvec(nsq**2)
206 
207  double precision, parameter :: rmoho_prem = r_earth_km - 24.4d0
208  double precision, parameter :: rcmb = 3480.0d0
209 
210  ! in Colleen's original code theta refers to the latitude. Here we have redefined theta to be colatitude
211  ! to agree with the rest of specfem
212 
213  ! debug
214  ! print *,'entering QRFSI12 subroutine'
215 
216  ylat=90.0d0-theta
217  xlon=phi
218 
219  ! only checks radius for crust, idoubling is misleading for oceanic crust
220  ! when we want to expand mantle up to surface...
221 
222 ! !if (idoubling == IFLAG_CRUST .or. radius >= rmoho) then
223  if (radius >= rmoho_prem) then
224  ! print *,'QRFSI12: we are in the crust'
225  qmu = 600.0d0
226  else if (idoubling == iflag_inner_core_normal .or. idoubling == iflag_middle_central_cube .or. &
227  idoubling == iflag_bottom_central_cube .or. idoubling == iflag_top_central_cube .or. &
228  idoubling == iflag_in_fictitious_cube) then
229  ! we are in the inner core
230 
231  !debug
232  ! print *,'QRFSI12: we are in the inner core'
233 
234  qmu = 84.6d0
235 
236  else if (idoubling == iflag_outer_core_normal) then
237 
238  ! we are in the outer core
239 
240  !debug
241  ! print *,'QRFSI12: we are in the outer core'
242 
243  qmu = 0.0d0
244 
245  else
246 
247  ! we are in the mantle
248 
249  depth = r_earth_km - radius
250 
251  !debug
252  ! print *,'QRFSI12: we are in the mantle at depth',depth
253 
254  ifnd = 0
255  do i=2,ndepths_refq
256  if (depth >= qrfsi12_q_refdepth(i-1) .and. depth < qrfsi12_q_refdepth(i)) then
257  ifnd=i
258  endif
259  enddo
260  if (ifnd == 0) then
261  write(6,"('problem finding reference Q value at depth: ',f8.3)") depth
262  stop
263  endif
264  smallq_ref=qrfsi12_q_refqmu(ifnd)
265  smallq = smallq_ref
266 
267 ! Colleen's model is only defined between depths of 24.4 and 650km
268  if (depth < 650.d0) then
269  do j = 1,nsq
270  shdep(j) = 0.
271  enddo
272  do n = 1,nkq
273  splpts(n)=qrfsi12_q_spknt(n)
274  enddo
275  call vbspl(depth,nkq,splpts,splcon,splcond)
276  do n = 1,nkq
277  do j = 1,nsq
278  shdep(j)=shdep(j)+(splcon(n)*qrfsi12_q_dqmu(n,j))
279  enddo
280  enddo
281  call ylm(ylat,xlon,maxl_q,xlmvec,wk1,wk2,wk3)
282  dqmu = 0.
283  do k = 1,nsq
284  dqmu=dqmu+xlmvec(k)*shdep(k)
285  enddo
286  smallq = smallq_ref + dqmu
287  endif
288 
289  ! if smallq is small and negative (due to numerical error), Qmu is very large:
290  if (smallq < 0.0d0) smallq = 1.0d0/attenuation_comp_maximum
291 
292  qmu = 1/smallq
293 
294  ! if Qmu is larger than MAX_ATTENUATION_VALUE, set it to ATTENUATION_COMP_MAXIMUM.
295  ! This assumes that this value is high enough that at this point there is almost no attenuation at all.
296  if (qmu >= attenuation_comp_maximum) qmu = 0.99d0*attenuation_comp_maximum
297 
298  endif
299 
300  end subroutine model_atten3d_qrfsi12
