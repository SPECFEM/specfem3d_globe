34 ! this routine detects cut planes along xi
35 ! In principle the left cut plane of the first slice
36 ! and the right cut plane of the last slice are not used
37 ! in the solver except if we want to have periodic conditions
38 
39  use constants
40 
41  implicit none
42 
43  integer :: nspec,myrank
44 
45  logical,dimension(2,nspec) :: iMPIcut_xi
46 
47  integer,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: ibool
48 
49  double precision,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: xstore,ystore,zstore
50 
51  ! logical mask used to create arrays iboolleft_xi and iboolright_xi
52  integer :: npointot
53  logical,dimension(npointot) :: mask_ibool
54 
55  integer, dimension(MAX_NUM_REGIONS,NB_SQUARE_EDGES_ONEDIR) :: NSPEC2D_ETA_FACE
56 
57  integer :: iregion
58  integer :: npoin2D_xi
59 
60  integer :: NGLOB2DMAX_XMIN_XMAX
61  integer, dimension(NGLOB2DMAX_XMIN_XMAX) :: iboolleft_xi,iboolright_xi
62 
63  integer, dimension(NB_SQUARE_EDGES_ONEDIR) :: npoin2D_xi_all
64 
65  ! processor identification
66  character(len=MAX_STRING_LEN) :: prname
67 
68  ! local parameters
69  ! global element numbering
70  integer :: ispec
71  ! MPI cut-plane element numbering
72  integer :: ispecc1,ispecc2,ix,iy,iz
73  integer :: nspec2Dtheor
74  integer :: ier
75 
76  character(len=MAX_STRING_LEN) :: errmsg
77 
78  ! debug: file output
79  logical,parameter :: DEBUG = .false.
80 
81  ! theoretical number of surface elements in the buffers
82  ! cut planes along xi=constant correspond to ETA faces
83  nspec2dtheor = nspec2d_eta_face(iregion,1)
84 
85 ! write the MPI buffers for the left and right edges of the slice
86 ! and the position of the points to check that the buffers are fine
87 
88 !
89 ! determine if the element falls on the left MPI cut plane
90 !
91 
92  if (debug) then
93  ! global point number and coordinates left MPI cut-plane
94  open(unit=iout,file=prname(1:len_trim(prname))//'iboolleft_xi.txt',status='unknown',iostat=ier)
95  if (ier /= 0) then
96  if (myrank == 0) then
97  write(imain,*)
98  write(imain,*) 'Error creating file: '
99  write(imain,*) prname(1:len_trim(prname))//'iboolleft_xi.txt'
100  write(imain,*)
101  write(imain,*) 'please make sure that the directory specified in Par_file as LOCAL_PATH exists'
102  write(imain,*)
103  call flush_imain()
104  endif
105  call exit_mpi(myrank,'Error creating iboolleft_xi.txt, please check your Par_file LOCAL_PATH setting')
106  endif
107  endif
108 
109  ! erase the logical mask used to mark points already found
110  mask_ibool(:) = .false.
111 
112  ! nb of global points shared with the other slice
113  iboolleft_xi(:) = 0
114  npoin2d_xi = 0
115  npoin2d_xi_all(1) = 1
116 
117  ! nb of elements in this cut-plane
118  ispecc1 = 0
119  do ispec = 1,nspec
120  if (impicut_xi(1,ispec)) then
121  ispecc1=ispecc1+1
122  ! loop on all the points in that 2-D element, including edges
123  ix = 1
124  do iy = 1,nglly
125  do iz = 1,ngllz
126  ! select point, if not already selected
127  if (.not. mask_ibool(ibool(ix,iy,iz,ispec))) then
128  mask_ibool(ibool(ix,iy,iz,ispec)) = .true.
129  npoin2d_xi = npoin2d_xi + 1
130 
131  ! fills buffer arrays
132  iboolleft_xi(npoin2d_xi) = ibool(ix,iy,iz,ispec)
133 
134  npoin2d_xi_all(1) = npoin2d_xi_all(1) + 1
135 
136  ! debug file output
137  if (debug) then
138  write(iout,*) ibool(ix,iy,iz,ispec), xstore(ix,iy,iz,ispec), &
139  ystore(ix,iy,iz,ispec),zstore(ix,iy,iz,ispec)
140  endif
141  endif
142  enddo
143  enddo
144  endif
145  enddo
146 
147  if (debug) then
148  ! put flag to indicate end of the list of points
149  write(iout,*) '0 0 0. 0. 0.'
150  ! write total number of points
151  write(iout,*) npoin2d_xi
152  close(iout)
153  endif
154 
155  ! compare number of surface elements detected to analytical value
156  if (ispecc1 /= nspec2dtheor) then
157  write(errmsg,*) 'Error MPI cut-planes detection in xi=left T=',nspec2dtheor,' C=',ispecc1
158  call exit_mpi(myrank,errmsg)
159  endif
160 
161  ! subtract the line that contains the flag after the last point
162  npoin2d_xi_all(1) = npoin2d_xi_all(1) - 1
163  if (npoin2d_xi_all(1) > nglob2dmax_xmin_xmax .or. npoin2d_xi_all(1) /= npoin2d_xi) &
164  call exit_mpi(myrank,'incorrect iboolleft_xi read')
165 
166 
167 !
168 ! determine if the element falls on the right MPI cut plane
169 !
170  nspec2dtheor = nspec2d_eta_face(iregion,2)
171 
172  if (debug) then
173  ! global point number and coordinates right MPI cut-plane
174  open(unit=iout,file=prname(1:len_trim(prname))//'iboolright_xi.txt',status='unknown',iostat=ier)
175  if (ier /= 0 ) call exit_mpi(myrank,'Error creating iboolright_xi.txt for this process')
176  endif
177 
178  ! erase the logical mask used to mark points already found
179  mask_ibool(:) = .false.
180 
181  ! nb of global points shared with the other slice
182  iboolright_xi(:) = 0
183  npoin2d_xi = 0
184  npoin2d_xi_all(2) = 1
185 
186  ! nb of elements in this cut-plane
187  ispecc2 = 0
188  do ispec = 1,nspec
189  if (impicut_xi(2,ispec)) then
190  ispecc2=ispecc2+1
191  ! loop on all the points in that 2-D element, including edges
192  ix = ngllx
193  do iy = 1,nglly
194  do iz = 1,ngllz
195  ! select point, if not already selected
196  if (.not. mask_ibool(ibool(ix,iy,iz,ispec))) then
197  mask_ibool(ibool(ix,iy,iz,ispec)) = .true.
198  npoin2d_xi = npoin2d_xi + 1
199 
200  ! fills buffer arrays
201  iboolright_xi(npoin2d_xi) = ibool(ix,iy,iz,ispec)
202 
203  npoin2d_xi_all(2) = npoin2d_xi_all(2) + 1
204 
205  ! debug file output
206  if (debug) then
207  write(iout,*) ibool(ix,iy,iz,ispec), xstore(ix,iy,iz,ispec), &
208  ystore(ix,iy,iz,ispec),zstore(ix,iy,iz,ispec)
209  endif
210  endif
211  enddo
212  enddo
213  endif
214  enddo
215 
216  if (debug) then
217  ! put flag to indicate end of the list of points
218  write(iout,*) '0 0 0. 0. 0.'
219  ! write total number of points
220  write(iout,*) npoin2d_xi
221  close(iout)
222  endif
223 
224  ! compare number of surface elements detected to analytical value
225  if (ispecc2 /= nspec2dtheor) then
226  write(errmsg,*) 'Error MPI cut-planes detection in xi=right T=',nspec2dtheor,' C=',ispecc2
227  call exit_mpi(myrank,errmsg)
228  endif
229 
230  ! subtract the line that contains the flag after the last point
231  npoin2d_xi_all(2) = npoin2d_xi_all(2) - 1
232  if (npoin2d_xi_all(2) > nglob2dmax_xmin_xmax .or. npoin2d_xi_all(2) /= npoin2d_xi) &
233  call exit_mpi(myrank,'incorrect iboolright_xi read')
234 
235  end subroutine get_mpi_cutplanes_xi
