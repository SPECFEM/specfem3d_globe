28  subroutine write_output_ascii(seismogram_tmp,iorientation,sisname,sisname_big_file)
29 
30 ! save seismograms in text format with no subsampling.
31 ! Because we do not subsample the output, this can result in large files
32 ! if the simulation uses many time steps. However, subsampling the output
33 ! here would result in a loss of accuracy when one later convolves
34 ! the results with the source time function
35 
36  use constants, only: custom_real,size_real,iout,max_string_len
37 
38  use specfem_par, only: &
39  dt,t0,nstep, &
40  seismo_offset,seismo_current, &
41  ntstep_between_output_seismos,output_files,simulation_type, &
42  save_all_seismos_in_one_file,use_binary_for_large_file, &
43  myrank
44 
45  implicit none
46 
47  real(kind=CUSTOM_REAL), dimension(5,NTSTEP_BETWEEN_OUTPUT_SEISMOS) :: seismogram_tmp
48 
49  integer :: iorientation
50  character(len=MAX_STRING_LEN) :: sisname,sisname_big_file
51 
52  ! local parameters
53  integer :: it
54  integer :: ier,isample
55  double precision :: value
56  double precision :: timeval
57  character(len=MAX_STRING_LEN) :: sisname_2
58 
59  ! add .ascii extension to seismogram file name for ASCII seismograms
60  write(sisname_2,"('/',a,'.ascii')") trim(sisname)
61 
62  ! save all seismograms in one large combined file instead of one file per seismogram
63  ! to avoid overloading shared non-local file systems
64 
65  ! create one large file instead of one small file per station to avoid file system overload
66  if (save_all_seismos_in_one_file) then
67  if (use_binary_for_large_file) then
68  write(iout) sisname_big_file
69  else
70  write(iout,*) sisname_big_file(1:len_trim(sisname_big_file))
71  endif
72  else
73  if (seismo_offset == 0) then
74  open(unit=iout,file=trim(output_files)//trim(sisname_2), &
75  status='unknown',action='write',iostat=ier)
76  else
77  open(unit=iout,file=trim(output_files)//trim(sisname_2), &
78  status='old',position='append',action='write',iostat=ier)
79  endif
80  if (ier /= 0) call exit_mpi(myrank,'Error opening file:'//trim(output_files)//trim(sisname_2))
81  endif
82 
83  ! subtract half duration of the source to make sure travel time is correct
84  do isample = 1,seismo_current
85 
86  ! seismogram value
87  value = dble(seismogram_tmp(iorientation,isample))
88 
89  ! current time increment
90  it = seismo_offset + isample
91 
92  ! current time
93  if (simulation_type == 3) then
94  timeval = dble(nstep-it)*dt - t0
95  else
96  timeval = dble(it-1)*dt - t0
97  endif
98 
99  ! writes out to file
100  if (save_all_seismos_in_one_file .and. use_binary_for_large_file) then
101  ! distinguish between single and double precision for reals
102  write(iout) real(timeval, kind=CUSTOM_REAL), real(value, kind=custom_real)
103  else
104  ! distinguish between single and double precision for reals
105  write(iout,*) real(timeval, kind=CUSTOM_REAL), ' ', real(value, kind=custom_real)
106  endif
107  enddo
108 
109  if (.not. save_all_seismos_in_one_file) close(iout)
110 
111  end subroutine write_output_ascii
