28  subroutine assemble_mpi_central_cube_block(ichunk,nb_msgs_theor_in_cube, sender_from_slices_to_cube, &
29  npoin2d_cube_from_slices, &
30  buffer_all_cube_from_slices, buffer_slices, buffer_slices2, &
31  ibool_central_cube, &
32  receiver_cube_from_slices, ibool_inner_core, &
33  idoubling_inner_core, nspec_inner_core, &
34  ibelm_bottom_inner_core, nspec2d_bottom_inner_core,nglob_inner_core, &
35  vector_assemble,ndim_assemble, &
36  iproc_eta,addressing,nchunks,nproc_xi,nproc_eta)
37 
38  ! this version of the routine is based on blocking MPI calls
39 
40  use constants
41 
42  implicit none
43 
44  ! for matching with central cube in inner core
45  integer ichunk, nb_msgs_theor_in_cube, npoin2D_cube_from_slices
46  integer, dimension(nb_msgs_theor_in_cube) :: sender_from_slices_to_cube
47  double precision, dimension(npoin2D_cube_from_slices,NDIM) :: &
48  buffer_slices,buffer_slices2
49  double precision, dimension(nb_msgs_theor_in_cube,npoin2D_cube_from_slices,NDIM) :: &
50  buffer_all_cube_from_slices
51  integer, dimension(nb_msgs_theor_in_cube,npoin2D_cube_from_slices):: ibool_central_cube
52  integer receiver_cube_from_slices
53 
54  ! local to global mapping
55  integer NSPEC_INNER_CORE,NSPEC2D_BOTTOM_INNER_CORE, NGLOB_INNER_CORE
56  integer, dimension(NGLLX,NGLLY,NGLLZ,NSPEC_INNER_CORE) :: ibool_inner_core
57  integer, dimension(NSPEC_INNER_CORE) :: idoubling_inner_core
58  integer, dimension(NSPEC2D_BOTTOM_INNER_CORE) :: ibelm_bottom_inner_core
59 
60  ! vector
61  integer ndim_assemble
62  real(kind=CUSTOM_REAL), dimension(ndim_assemble,NGLOB_INNER_CORE) :: vector_assemble
63 
64  !for addressing of the slices
65  integer, intent(in) :: NCHUNKS,NPROC_XI,NPROC_ETA
66  integer, dimension(NCHUNKS,0:NPROC_XI-1,0:NPROC_ETA-1), intent(in) :: addressing
67  integer, intent(in) :: iproc_eta
68 
69  integer ipoin,idimension, ispec2D, ispec
70  integer i,j,k
71  integer sender,receiver,imsg
72 
73  real(kind=CUSTOM_REAL), dimension(NGLOB_INNER_CORE) :: array_central_cube
74 
75  ! mask
76  logical, dimension(NGLOB_INNER_CORE) :: mask
77 
78  !---
79  !--- now use buffers to assemble mass matrix with central cube once and for all
80  !---
81 
82  ! on chunks AB and AB_ANTIPODE, receive all the messages from slices
83  if (ichunk == chunk_ab .or. ichunk == chunk_ab_antipode) then
84 
85  do imsg = 1,nb_msgs_theor_in_cube-1
86 
87  ! receive buffers from slices
88  sender = sender_from_slices_to_cube(imsg)
89  call recv_dp(buffer_slices,ndim_assemble*npoin2d_cube_from_slices,sender,itag)
90 
91  ! copy buffer in 2D array for each slice
92  buffer_all_cube_from_slices(imsg,:,1:ndim_assemble) = buffer_slices(:,1:ndim_assemble)
93 
94  enddo
95  endif
96 
97  ! send info to central cube from all the slices except those in CHUNK_AB & CHUNK_AB_ANTIPODE
98  if (ichunk /= chunk_ab .and. ichunk /= chunk_ab_antipode) then
99  ! for bottom elements in contact with central cube from the slices side
100  ipoin = 0
101  do ispec2d = 1,nspec2d_bottom_inner_core
102 
103  ispec = ibelm_bottom_inner_core(ispec2d)
104 
105  ! only for DOFs exactly on surface of central cube (bottom of these elements)
106  k = 1
107  do j = 1,nglly
108  do i = 1,ngllx
109  ipoin = ipoin + 1
110  buffer_slices(ipoin,1:ndim_assemble) = dble(vector_assemble(1:ndim_assemble,ibool_inner_core(i,j,k,ispec)))
111  enddo
112  enddo
113  enddo
114 
115  ! send buffer to central cube
116  receiver = receiver_cube_from_slices
117  call send_dp(buffer_slices,ndim_assemble*npoin2d_cube_from_slices,receiver,itag)
118 
119  ! in case NPROC_XI == 1, the other chunks exchange all bottom points with
120  ! CHUNK_AB **and** CHUNK_AB_ANTIPODE
121  if (nproc_xi == 1) then
122  call send_dp(buffer_slices,ndim_assemble*npoin2d_cube_from_slices,addressing(chunk_ab_antipode,0,iproc_eta),itag)
123  endif
124 
125  endif ! end sending info to central cube
126 
127 
128  ! exchange of their bottom faces between chunks AB and AB_ANTIPODE
129  if (ichunk == chunk_ab .or. ichunk == chunk_ab_antipode) then
130 
131  ipoin = 0
132  do ispec = nspec_inner_core, 1, -1
133  if (idoubling_inner_core(ispec) == iflag_bottom_central_cube) then
134  k = 1
135  do j = 1,nglly
136  do i = 1,ngllx
137  ipoin = ipoin + 1
138  buffer_slices(ipoin,1:ndim_assemble) = dble(vector_assemble(1:ndim_assemble,ibool_inner_core(i,j,k,ispec)))
139  enddo
140  enddo
141  endif
142  enddo
143 
144  sender = sender_from_slices_to_cube(nb_msgs_theor_in_cube)
145 
146  call sendrecv_dp(buffer_slices,ndim_assemble*npoin2d_cube_from_slices,receiver_cube_from_slices,itag, &
147  buffer_slices2,ndim_assemble*npoin2d_cube_from_slices,sender,itag)
148 
149  buffer_all_cube_from_slices(nb_msgs_theor_in_cube,:,1:ndim_assemble) = buffer_slices2(:,1:ndim_assemble)
150 
151  endif
152 
153  !--- now we need to assemble the contributions
154 
155  if (ichunk == chunk_ab .or. ichunk == chunk_ab_antipode) then
156 
157  do idimension = 1,ndim_assemble
158  ! erase contributions to central cube array
159  array_central_cube(:) = 0._custom_real
160 
161  ! use indirect addressing to store contributions only once
162  ! distinguish between single and double precision for reals
163  do imsg = 1,nb_msgs_theor_in_cube-1
164  do ipoin = 1,npoin2d_cube_from_slices
165  if (nproc_xi == 1) then
166  if (ibool_central_cube(imsg,ipoin) > 0) then
167  array_central_cube(ibool_central_cube(imsg,ipoin)) = real(buffer_all_cube_from_slices(imsg,ipoin,idimension), &
168  kind=custom_real)
169  endif
170  else
171  array_central_cube(ibool_central_cube(imsg,ipoin)) = real(buffer_all_cube_from_slices(imsg,ipoin,idimension), &
172  kind=custom_real)
173  endif
174  enddo
175  enddo
176  ! add the contribution of AB or AB_ANTIPODE to sum with the external slices on the edges
177  ! use a mask to avoid taking the same point into account several times.
178  mask(:) = .false.
179  do ipoin = 1,npoin2d_cube_from_slices
180  if (nproc_xi == 1) then
181  if (ibool_central_cube(nb_msgs_theor_in_cube,ipoin) > 0) then
182  if (.not. mask(ibool_central_cube(nb_msgs_theor_in_cube,ipoin))) then
183  array_central_cube(ibool_central_cube(nb_msgs_theor_in_cube,ipoin)) = &
184  array_central_cube(ibool_central_cube(nb_msgs_theor_in_cube,ipoin)) + &
185  real(buffer_all_cube_from_slices(nb_msgs_theor_in_cube,ipoin,idimension), kind=custom_real)
186  endif
187  mask(ibool_central_cube(nb_msgs_theor_in_cube,ipoin)) = .true.
188  endif
189  else
190  if (.not. mask(ibool_central_cube(nb_msgs_theor_in_cube,ipoin))) then
191  array_central_cube(ibool_central_cube(nb_msgs_theor_in_cube,ipoin)) = &
192  array_central_cube(ibool_central_cube(nb_msgs_theor_in_cube,ipoin)) + &
193  real(buffer_all_cube_from_slices(nb_msgs_theor_in_cube,ipoin,idimension), kind=custom_real)
194  endif
195  mask(ibool_central_cube(nb_msgs_theor_in_cube,ipoin)) = .true.
196  endif
197  enddo
198 
199  ! suppress degrees of freedom already assembled at top of cube on edges
200  do ispec = 1,nspec_inner_core
201  if (idoubling_inner_core(ispec) == iflag_top_central_cube) then
202  k = ngllz
203  do j = 1,nglly
204  do i = 1,ngllx
205  array_central_cube(ibool_inner_core(i,j,k,ispec)) = 0._custom_real
206  enddo
207  enddo
208  endif
209  enddo
210 
211  ! assemble contributions
212  vector_assemble(idimension,:) = vector_assemble(idimension,:) + array_central_cube(:)
213 
214  ! copy sum back
215  do imsg = 1,nb_msgs_theor_in_cube-1
216  do ipoin = 1,npoin2d_cube_from_slices
217  if (nproc_xi == 1) then
218  if (ibool_central_cube(imsg,ipoin) > 0) then
219  buffer_all_cube_from_slices(imsg,ipoin,idimension) = &
220  vector_assemble(idimension,ibool_central_cube(imsg,ipoin))
221  else
222  buffer_all_cube_from_slices(imsg,ipoin,idimension) = 0._custom_real
223  endif
224  else
225  buffer_all_cube_from_slices(imsg,ipoin,idimension) = &
226  vector_assemble(idimension,ibool_central_cube(imsg,ipoin))
227  endif
228  enddo
229  enddo
230 
231  enddo
232 
233  endif
234 
235  !----------
236 
237  ! receive info from central cube on all the slices except those in CHUNK_AB & CHUNK_AB_ANTIPODE
238  if (ichunk /= chunk_ab .and. ichunk /= chunk_ab_antipode) then
239  ! receive buffers from slices
240  sender = receiver_cube_from_slices
241  call recv_dp(buffer_slices,ndim_assemble*npoin2d_cube_from_slices,sender,itag)
242 
243  ! in case NPROC_XI == 1, the other chunks exchange all bottom points with
244  ! CHUNK_AB **and** CHUNK_AB_ANTIPODE
245  if (nproc_xi == 1) then
246  call recv_dp(buffer_slices2,ndim_assemble*npoin2d_cube_from_slices,addressing(chunk_ab_antipode,0,iproc_eta),itag)
247 
248  buffer_slices = buffer_slices + buffer_slices2
249  endif
250 
251  ! for bottom elements in contact with central cube from the slices side
252  ipoin = 0
253  do ispec2d = 1,nspec2d_bottom_inner_core
254 
255  ispec = ibelm_bottom_inner_core(ispec2d)
256 
257  ! only for DOFs exactly on surface of central cube (bottom of these elements)
258  k = 1
259  do j = 1,nglly
260  do i = 1,ngllx
261  ipoin = ipoin + 1
262 
263  ! distinguish between single and double precision for reals
264  vector_assemble(1:ndim_assemble,ibool_inner_core(i,j,k,ispec)) = real(buffer_slices(ipoin,1:ndim_assemble), &
265  kind=custom_real)
266 
267  enddo
268  enddo
269  enddo
270 
271  endif ! end receiving info from central cube
272 
273  !------- send info back from central cube to slices
274 
275  ! on chunk AB & CHUNK_AB_ANTIPODE, send all the messages to slices
276  if (ichunk == chunk_ab .or. ichunk == chunk_ab_antipode) then
277 
278  do imsg = 1,nb_msgs_theor_in_cube-1
279 
280  ! copy buffer in 2D array for each slice
281  buffer_slices(:,1:ndim_assemble) = buffer_all_cube_from_slices(imsg,:,1:ndim_assemble)
282 
283  ! send buffers to slices
284  receiver = sender_from_slices_to_cube(imsg)
285  call send_dp(buffer_slices,ndim_assemble*npoin2d_cube_from_slices,receiver,itag)
286 
287  enddo
288  endif
289 
290  end subroutine assemble_mpi_central_cube_block
