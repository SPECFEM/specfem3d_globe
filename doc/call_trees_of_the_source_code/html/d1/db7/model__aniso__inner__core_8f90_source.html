30 ! based on scaling factors by Ishii et al. (2002)
31 !

33 ! adds a 3D model or a read_aniso_inner_core_model subroutine

36  subroutine model_aniso_inner_core(x,c11,c33,c12,c13,c44,REFERENCE_1D_MODEL, &
37  vpv,vph,vsv,vsh,rho,eta_aniso)
38 
39  use constants
40 
41  implicit none
42 
43 ! given a normalized radius x, gives non-dimensionalized c11,c33,c12,c13,c44
44 
45  integer REFERENCE_1D_MODEL
46 
47  double precision x,c11,c33,c12,c13,c44
48  double precision rho,vpv,vph,vsv,vsh,eta_aniso
49 
50  ! local parameters
51  double precision vp,vs
52  double precision vpc,vsc,rhoc
53  double precision vp0,vs0,rho0,A0
54  double precision c66
55  double precision scale_fac
56 
57  ! calculates isotropic values from given (transversely isotropic) reference values
58  ! (are non-dimensionalized)
59  vp = sqrt(((8.d0+4.d0*eta_aniso)*vph*vph + 3.d0*vpv*vpv &
60  + (8.d0 - 8.d0*eta_aniso)*vsv*vsv)/15.d0)
61  vs = sqrt(((1.d0-2.d0*eta_aniso)*vph*vph + vpv*vpv &
62  + 5.d0*vsh*vsh + (6.d0+4.d0*eta_aniso)*vsv*vsv)/15.d0)
63 
64  ! scale to dimensions (e.g. used in prem model)
65  scale_fac = r_earth*dsqrt(pi*grav*rhoav)/1000.d0
66  vp = vp * scale_fac
67  vs = vs * scale_fac
68  rho = rho * rhoav/1000.d0
69 
70  select case (reference_1d_model)
71 
72  case (reference_model_iasp91)
73  vpc=11.24094d0-4.09689d0*x*x
74  vsc=3.56454d0-3.45241d0*x*x
75  rhoc=13.0885d0-8.8381d0*x*x
76  ! checks with given values
77  if (abs(vpc-vp) > tinyval .or. abs(vsc-vs) > tinyval .or. abs(rhoc-rho) > tinyval) then
78  stop 'Error isotropic IASP91 values in model_aniso_inner_core() '
79  endif
80 
81  ! values at center
82  vp0=11.24094d0
83  vs0=3.56454d0
84  rho0=13.0885d0
85 
86  case (reference_model_prem)
87  vpc=11.2622d0-6.3640d0*x*x
88  vsc=3.6678d0-4.4475d0*x*x
89  rhoc=13.0885d0-8.8381d0*x*x
90  ! checks
91  if (abs(vpc-vp) > tinyval .or. abs(vsc-vs) > tinyval .or. abs(rhoc-rho) > tinyval) then
92  stop 'Error isotropic PREM values in model_aniso_inner_core() '
93  endif
94 
95  ! values at center
96  vp0=11.2622d0
97  vs0=3.6678d0
98  rho0=13.0885d0
99 
100  case (reference_model_1dref)
101  ! values at center
102  vp0 = 11262.20 / 1000.0d0
103  vs0 = 3667.800 / 1000.0d0
104  rho0 = 13088.480 / 1000.0d0
105 
106  case (reference_model_1066a)
107  ! values at center
108  vp0 = 11.33830
109  vs0 = 3.62980
110  rho0 = 13.429030
111 
112  case (reference_model_ak135f_no_mud)
113  ! values at center
114  vp0 = 11.26220
115  vs0 = 3.667800
116  rho0 = 13.01220
117 
118  case (reference_model_jp1d)
119  ! values at center
120  vp0 = 11.24094
121  vs0 = 3.56454
122  rho0 = 13.0885d0
123 
124  case (reference_model_sea1d)
125  ! values at center
126  vp0 = 11.240940
127  vs0 = 3.564540
128  rho0 = 13.012190
129 
130  case default
131  stop 'unknown 1D reference Earth model in anisotropic inner core'
132 
133  end select
134 
135 ! non-dimensionalization of elastic parameters (GPa--[g/cm^3][(km/s)^2])
136  scale_fac = rhoav*r_earth*r_earth*pi*grav*rhoav
137  scale_fac = 1.d9 / scale_fac
138 
139 ! elastic tensor for hexagonal symmetry in reduced notation:
140 !
141 ! c11 c12 c13 0 0 0
142 ! c12 c11 c13 0 0 0
143 ! c13 c13 c33 0 0 0
144 ! 0 0 0 c44 0 0
145 ! 0 0 0 0 c44 0
146 ! 0 0 0 0 0 c66=(c11-c12)/2
147 !
148 ! in terms of the A, C, L, N and F of Love (1927):
149 !
150 ! c11 = A
151 ! c33 = C
152 ! c12 = A-2N
153 ! c13 = F
154 ! c44 = L
155 ! c66 = N
156 !
157 ! isotropic equivalent:
158 !
159 ! c11 = lambda+2mu
160 ! c33 = lambda+2mu
161 ! c12 = lambda
162 ! c13 = lambda
163 ! c44 = mu
164 ! c66 = mu
165 
166 ! Ishii et al. (2002):
167 !
168 ! alpha = 3.490 % = (C-A)/A0 = (c33-c11)/A0
169 ! beta = 0.988 % = (L-N)/A0 = (c44-c66)/A0
170 ! gamma = 0.881 % = (A-2N-F)/A0 = (c12-c13)/A0
171 ! where A0 is A at the Earth's center
172 !
173 ! assume c11 = lambda+2mu
174 ! c66 = (c11-c12)/2 = mu
175 !
176 ! then c33 = c11 + alpha*A0
177 ! c44 = c66 + beta*A0
178 ! c13 = c12 - gamma*A0
179 ! and c12 = c11 - 2*c66
180 !
181 ! Note: The value vs0, while set above, is unnecessary for the
182 ! calculation below.
183 !
184 ! Steinle-Neumann (2002):
185 !
186 ! r T rho c11 c12 c13 c33 c44 KS mu
187 ! (km) (K) (Mg/m3) (GPa)
188 ! 0 5735 13.09 1693 1253 1364 1813 154 1457 184
189 ! 200 5729 13.08 1689 1251 1362 1809 154 1455 184
190 ! 400 5711 13.05 1676 1243 1353 1795 151 1444 181
191 ! 600 5682 13.01 1661 1232 1341 1779 150 1432 180
192 ! 800 5642 12.95 1638 1214 1321 1755 148 1411 178
193 ! 1000 5590 12.87 1606 1190 1295 1720 146 1383 175
194 ! 1200 5527 12.77 1559 1155 1257 1670 141 1343 169
195 !
196  c11 = rho*vp*vp*scale_fac
197  c66 = rho*vs*vs*scale_fac
198  c12 = c11 - 2.0d0*c66
199 
200  a0 = rho0*vp0*vp0*scale_fac
201 
202  c33 = c11 + 0.0349d0*a0
203  c44 = c66 + 0.00988d0*a0
204  c13 = c12 - 0.00881d0*a0
205 
206  end subroutine model_aniso_inner_core
