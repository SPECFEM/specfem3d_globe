535  subroutine get_timestep_limit_significant_digit(time_step)
536 
537  ! cut at a significant number of digits (2 digits) using 1/2 rounding
538  ! example: 0.0734815 -> 0.0730
539  ! and 0.0737777 -> 0.0735
540 
541  implicit none
542 
543  double precision,intent(inout) :: time_step
544 
545  ! rounding
546  integer :: lpow,ival
547  double precision :: fac_pow,dt_cut
548 
549  ! initializes
550  dt_cut = time_step
551 
552  ! cut at a significant number of digits (2 digits)
553  ! example: 0.0734815 -> lpow = (2 - (-1) = 3
554  lpow = int(2.d0 - log10(dt_cut))
555 
556  ! example: -> factor 10**3
557  fac_pow = 10.d0**(lpow)
558 
559  ! example: -> 73
560  ival = int(fac_pow * dt_cut)
561 
562  ! adds .5-digit (in case): 73.0 -> 0.073
563  if ( (fac_pow * dt_cut - ival) >= 0.5 ) then
564  dt_cut = (dble(ival) + 0.5d0) / fac_pow
565  else
566  dt_cut = dble(ival) / fac_pow
567  endif
568 
569  time_step = dt_cut
570 
571  end subroutine get_timestep_limit_significant_digit
