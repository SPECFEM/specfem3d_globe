29  subroutine parse_kernel_names(kernel_names_comma_delimited,kernel_names,nker)
30 
31  use postprocess_par, only: max_string_len, max_kernel_names
32 
33  implicit none
34 
35  character(len=MAX_STRING_LEN),intent(in) :: kernel_names_comma_delimited
36  character(len=MAX_STRING_LEN),intent(inout) :: kernel_names(max_kernel_names)
37  integer,intent(out) :: nker
38 
39  ! local parameters
40  integer :: iker
41  character(len=MAX_STRING_LEN) :: tmp
42  character,parameter :: delimiter = ','
43 
44  ! gets first name/token
45  iker = 1
46  call strtok(kernel_names_comma_delimited, delimiter, kernel_names(iker))
47 
48  ! null-string as first argument for successive strtok-calls
49  tmp(1:1) = char(0)
50 
51  ! gets next names/tokens (strtok will return null-terminated token when finished)
52  do while (kernel_names(iker)(1:1) /= char(0))
53  ! increases name/token number
54  iker = iker + 1
55  ! gets next successive token (with null-terminated string as first argument)
56  call strtok(tmp, delimiter, kernel_names(iker))
57  enddo
58 
59  ! number of kernel names
60  nker = iker-1
61 
62  ! checks name lengths (e.g. if kernel_name argument is "vsv,vsh," we will have a 3. kernel name with empty string)
63  do iker = 1,nker
64  if (len_trim(kernel_names(iker)) == 0) then
65  print *,'Error encountered kernel name with zero length: kernel name number ',iker,' out of ',nker,' is empty'
66  print *,'Please check your kernel_names argument...'
67  stop 'Error kernel name with zero length'
68  endif
69  enddo
70 
71  end subroutine parse_kernel_names

80  subroutine strtok (source_string, delimiter, token)
81 
82 ! @(#) Tokenize a string in a similar manner to C routine strtok(3c).
83 !
84 ! Usage: First call STRTOK() with the string to tokenize as SOURCE_STRING,
85 ! and the delimiter list used to tokenize SOURCE_STRING in delimiter.
86 !
87 ! then, if the returned value is not equal to char(0), keep calling until it is
88 ! with SOURCE_STRING set to char(0).
89 !
90 ! STRTOK will return a token on each call until the entire line is processed,
91 ! which it signals by returning char(0).
92 !
93 ! Input: source_string = Source string to tokenize.
94 ! delimiter = delimiter string. Used to determine the beginning/end of each token in a string.
95 !
96 ! Output: strtok()
97 !
98 ! LIMITATIONS:
99 ! can not be called with a different string until current string is totally processed, even from different procedures
100 
101  use postprocess_par, only: max_string_len
102 
103  ! PARAMETERS:
104  character(len=MAX_STRING_LEN), intent(in) :: source_string
105  character(len=1), intent(in) :: delimiter
106  character(len=MAX_STRING_LEN), intent(out) :: token
107 
108  ! SAVED VALUES:
109  character(len=MAX_STRING_LEN),save :: saved_string
110  integer,save :: isaved_start ! points to beginning of unprocessed data
111  integer,save :: isource_len ! length of original input string
112 
113  ! LOCAL VALUES:
114  integer :: ibegin ! beginning of token to return
115  integer :: ifinish ! end of token to return
116 
117  ! initialize stored copy of input string and pointer into input string on first call
118  if (source_string(1:1) /= char(0)) then
119  isaved_start = 1 ! beginning of unprocessed data
120  saved_string = source_string ! save input string from first call in series
121  isource_len = len(saved_string) ! length of input string from first call
122  endif
123 
124  token = ''
125  ibegin = isaved_start
126 
127  ! sets first index ibegin to beginning of (next) token
128  do while (.true.)
129  if ( (ibegin <= isource_len) .and. (index(delimiter,saved_string(ibegin:ibegin)) /= 0)) then
130  ! delimiter is encountered, starts with next index (next token)
131  ibegin = ibegin + 1
132  else
133  ! exits do-loop
134  exit
135  endif
136  enddo
137 
138  if (ibegin > isource_len) then
139  token = char(0)
140  return
141  endif
142 
143  ! sets second index ifinish to end of token (including delimiter)
144  ifinish = ibegin
145 
146  do while (.true.)
147  if ((ifinish <= isource_len) .and. (index(delimiter,saved_string(ifinish:ifinish)) == 0)) then
148  ! delimiter is not encountered yet, increases finish index
149  ifinish = ifinish + 1
150  else
151  ! exits do-loop
152  exit
153  endif
154  enddo
155 
156  ! sets token string
157  !strtok = "["//saved_string(ibegin:ifinish-1)//"]"
158  token = saved_string(ibegin:ifinish-1)
159  isaved_start = ifinish
160 
161  end subroutine strtok
