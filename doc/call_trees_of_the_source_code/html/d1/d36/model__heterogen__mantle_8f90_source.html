126  subroutine model_heterogen_mantle(radius,theta,phi,dvs,dvp,drho)
127 
128  use constants
129  use model_heterogen_mantle_par
130 
131  implicit none
132 
133  ! variable declaration
134  double precision :: radius,theta,phi ! input coordinates
135  double precision :: drho,dvp,dvs ! output anomaly values
136 
137  ! local parameters
138  double precision :: x,y,z ! input converted to Cartesian
139  double precision :: x_low,x_high ! x values used to interpolate
140  double precision :: y_low,y_high ! y values used to interpolate
141  double precision :: z_low,z_high ! z values used to interpolate
142  double precision :: delta,delta2 ! weights in record# and in interpolation
143  double precision :: rho1,rho2,rho3,rho4,rho5,rho6,rho7,rho8 ! rho values at the interpolation points
144  double precision :: r_inner,r_outer ! lower and upper domain bounds for r
145  integer :: rec_read ! nr of record to be read from heterogen.dat (direct access file)
146  double precision :: a,b,c ! substitutions in interpolation algorithm (weights)
147 
148  radius = radius*r_earth
149  r_inner = 3.500d6 !lower bound for heterogeneity zone
150 ! NOTE: r_outer NEEDS TO BE (just) SMALLER THAN R_EARTH!!!!!!!!
151  r_outer = r_earth-1.0d1 !6.300d6 !upper bound for heterogeneity zone (lower mantle: e.g. 4.500d6)
152 
153  delta = 2.*r_earth/(real(n_r-1))
154  delta2 = 2.*r_earth/(real(n_r-2))
155  !delta2 = 2.*R_EARTH/(real(N_R))
156 
157  if ((radius >= r_inner) .and. (radius <= r_outer)) then
158  ! convert spherical point to Cartesian point, move origin to corner
159  x = r_earth + radius*sin(theta)*cos(phi)
160  y = r_earth + radius*sin(theta)*sin(phi)
161  z = r_earth + radius*cos(theta)
162 
163  ! determine which points to search for in heterogen.dat
164  ! find x_low,y_low,z_low etc.
165  x_low = floor(x/delta2) + 1
166  x_high = x_low + 1
167  y_low = floor(y/delta2) + 1
168  y_high = y_low + 1
169  z_low = floor(z/delta2) + 1
170  z_high = z_low + 1
171 
172  ! rho1 at: x_low y_low z_low
173  rec_read = int(1+(x_low*n_r*n_r)+(y_low*n_r)+z_low)
174  rho1 = hmm_rho_in(rec_read)
175 
176  ! rho2 at: x_low y_high z_low
177  rec_read = int(1+(x_low*n_r*n_r)+(y_high*n_r)+z_low)
178  rho2 = hmm_rho_in(rec_read)
179 
180  ! rho3 at: x_high y_low z_low
181  rec_read = int(1+(x_high*n_r*n_r)+(y_low*n_r)+z_low)
182  rho3 = hmm_rho_in(rec_read)
183 
184  ! rho4 at: x_high y_high z_low
185  rec_read = int(1+(x_high*n_r*n_r)+(y_high*n_r)+z_low)
186  rho4 = hmm_rho_in(rec_read)
187 
188  ! rho5 at: x_low y_low z_high
189  rec_read = int(1+(x_low*n_r*n_r)+(y_low*n_r)+z_high)
190  rho5 = hmm_rho_in(rec_read)
191 
192  ! rho6 at: x_low y_high z_high
193  rec_read = int(1+(x_low*n_r*n_r)+(y_high*n_r)+z_high)
194  rho6 = hmm_rho_in(rec_read)
195 
196  ! rho7 at: x_high y_low z_high
197  rec_read = int(1+(x_high*n_r*n_r)+(y_low*n_r)+z_high)
198  rho7 = hmm_rho_in(rec_read)
199 
200  ! rho8 at: x_high y_high z_high
201  rec_read = int(1+(x_high*n_r*n_r)+(y_high*n_r)+z_high)
202  rho8 = hmm_rho_in(rec_read)
203 
204  ! perform linear interpolation between the 8 points
205  a = (x-x_low*delta)/delta ! weight for x
206  b = (y-y_low*delta)/delta ! weight for y
207  c = (z-z_low*delta)/delta ! weight for z
208 
209  drho = rho1*(1.-a)*(1.-b)*(1.-c) + rho2*(1.-a)*b*(1.-c) + &
210  rho3*a*(1.-b)*(1.-c) + rho4*a*b*(1.-c) + rho5*(1.-a)*(1.-b)*c + &
211  rho6*(1.-a)*b*c + rho7*a*(1.-b)*c + rho8*a*b*c
212 
213  ! calculate delta vp,vs from the interpolated delta rho
214  dvp = (0.55/0.30)*drho
215  dvs = (1.00/0.30)*drho
216 
217  else !outside of heterogeneity domain
218  drho = 0.
219  dvp = 0.
220  dvs = 0.
221  endif
222 
223  end subroutine model_heterogen_mantle
