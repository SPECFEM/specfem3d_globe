128  subroutine model_heterogen_mantle(radius,theta,phi,dvs,dvp,drho)
129 
130  use constants
131  use model_heterogen_mantle_par
132 
133  implicit none
134 
135  ! variable declaration
136  double precision :: radius,theta,phi ! input coordinates
137  double precision :: drho,dvp,dvs ! output anomaly values
138 
139  ! local parameters
140  double precision :: x,y,z ! input converted to Cartesian
141  double precision :: x_low,x_high ! x values used to interpolate
142  double precision :: y_low,y_high ! y values used to interpolate
143  double precision :: z_low,z_high ! z values used to interpolate
144  double precision :: delta,delta2 ! weights in record# and in interpolation
145  double precision :: rho1,rho2,rho3,rho4,rho5,rho6,rho7,rho8 ! rho values at the interpolation points
146  double precision :: r_inner,r_outer ! lower and upper domain bounds for r
147  integer :: rec_read ! nr of record to be read from heterogen.dat (direct access file)
148  double precision :: a,b,c ! substitutions in interpolation algorithm (weights)
149 
150  radius = radius*r_earth
151  r_inner = 3.500d6 !lower bound for heterogeneity zone
152 ! NOTE: r_outer NEEDS TO BE (just) SMALLER THAN R_EARTH!!!!!!!!
153  r_outer = r_earth-1.0d1 !6.300d6 !upper bound for heterogeneity zone (lower mantle: e.g. 4.500d6)
154 
155  delta = 2.*r_earth/(real(n_r-1))
156  delta2 = 2.*r_earth/(real(n_r-2))
157  !delta2 = 2.*R_EARTH/(real(N_R))
158 
159  if ((radius >= r_inner) .and. (radius <= r_outer)) then
160  ! convert spherical point to Cartesian point, move origin to corner
161  x = r_earth + radius*sin(theta)*cos(phi)
162  y = r_earth + radius*sin(theta)*sin(phi)
163  z = r_earth + radius*cos(theta)
164 
165  ! determine which points to search for in heterogen.dat
166  ! find x_low,y_low,z_low etc.
167  x_low = floor(x/delta2) + 1
168  x_high = x_low + 1
169  y_low = floor(y/delta2) + 1
170  y_high = y_low + 1
171  z_low = floor(z/delta2) + 1
172  z_high = z_low + 1
173 
174  ! rho1 at: x_low y_low z_low
175  rec_read = int(1+(x_low*n_r*n_r)+(y_low*n_r)+z_low)
176  rho1 = hmm_rho_in(rec_read)
177 
178  ! rho2 at: x_low y_high z_low
179  rec_read = int(1+(x_low*n_r*n_r)+(y_high*n_r)+z_low)
180  rho2 = hmm_rho_in(rec_read)
181 
182  ! rho3 at: x_high y_low z_low
183  rec_read = int(1+(x_high*n_r*n_r)+(y_low*n_r)+z_low)
184  rho3 = hmm_rho_in(rec_read)
185 
186  ! rho4 at: x_high y_high z_low
187  rec_read = int(1+(x_high*n_r*n_r)+(y_high*n_r)+z_low)
188  rho4 = hmm_rho_in(rec_read)
189 
190  ! rho5 at: x_low y_low z_high
191  rec_read = int(1+(x_low*n_r*n_r)+(y_low*n_r)+z_high)
192  rho5 = hmm_rho_in(rec_read)
193 
194  ! rho6 at: x_low y_high z_high
195  rec_read = int(1+(x_low*n_r*n_r)+(y_high*n_r)+z_high)
196  rho6 = hmm_rho_in(rec_read)
197 
198  ! rho7 at: x_high y_low z_high
199  rec_read = int(1+(x_high*n_r*n_r)+(y_low*n_r)+z_high)
200  rho7 = hmm_rho_in(rec_read)
201 
202  ! rho8 at: x_high y_high z_high
203  rec_read = int(1+(x_high*n_r*n_r)+(y_high*n_r)+z_high)
204  rho8 = hmm_rho_in(rec_read)
205 
206  ! perform linear interpolation between the 8 points
207  a = (x-x_low*delta)/delta ! weight for x
208  b = (y-y_low*delta)/delta ! weight for y
209  c = (z-z_low*delta)/delta ! weight for z
210 
211  drho = rho1*(1.-a)*(1.-b)*(1.-c) + rho2*(1.-a)*b*(1.-c) + &
212  rho3*a*(1.-b)*(1.-c) + rho4*a*b*(1.-c) + rho5*(1.-a)*(1.-b)*c + &
213  rho6*(1.-a)*b*c + rho7*a*(1.-b)*c + rho8*a*b*c
214 
215  ! calculate delta vp,vs from the interpolated delta rho
216  dvp = (0.55/0.30)*drho
217  dvs = (1.00/0.30)*drho
218 
219  else !outside of heterogeneity domain
220  drho = 0.
221  dvp = 0.
222  dvs = 0.
223  endif
224 
225  end subroutine model_heterogen_mantle
