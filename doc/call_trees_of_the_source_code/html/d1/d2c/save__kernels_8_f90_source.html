168  ! kernel unit [ s / km^3 ]

171  ! final unit : [s km^(-3) GPa^(-1)]

178  if (save_transverse_kl_only) then
179  ! transverse isotropic kernel arrays for file output
180  allocate(alphav_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint), &
181  alphah_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint), &

183  betah_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint), &

185  if (ier /= 0 ) stop 'Error allocating transverse kernels alphav_kl_crust_mantle,...'
186 
187  ! isotropic kernel arrays for file output
188  allocate(bulk_betav_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint), &
189  bulk_betah_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint),stat=ier)
190  if (ier /= 0 ) stop 'Error allocating transverse kernels bulk_betav_kl_crust_mantle,...'
191 
192  ! bulk velocity kernels
193  allocate(bulk_c_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint), &
194  bulk_beta_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint),stat=ier)
195  if (ier /= 0 ) stop 'Error allocating transverse kernels bulk_c_kl_crust_mantle,...'
196  else
197  ! dummy allocation
198  allocate(alphav_kl_crust_mantle(1,1,1,1), &
199  alphah_kl_crust_mantle(1,1,1,1), &
200  betav_kl_crust_mantle(1,1,1,1), &
201  betah_kl_crust_mantle(1,1,1,1), &
202  eta_kl_crust_mantle(1,1,1,1))
203  ! isotropic kernel arrays for file output
204  allocate(bulk_betav_kl_crust_mantle(1,1,1,1), &
205  bulk_betah_kl_crust_mantle(1,1,1,1))
206  ! only dummy
207  allocate(bulk_c_kl_crust_mantle(1,1,1,1), &
208  bulk_beta_kl_crust_mantle(1,1,1,1))
209  endif

233  ! transverse isotropic kernel calculations
234  if (save_transverse_kl_only) then
235  ! note: transverse isotropic kernels are calculated for all elements
236  !
237  ! however, the factors A,C,L,N,F are based only on transverse elements
238  ! in between Moho and 220 km layer, otherwise they are taken from isotropic values
239 
240  rhol = rhostore_crust_mantle(i,j,k,ispec)
241 
242  ! transverse isotropic parameters from compute_force_crust_mantle.f90
243  ! C=rhovpvsq A=rhovphsq L=rhovsvsq N=rhovshsq eta=F/(A - 2 L)
244 
245  ! Get A,C,F,L,N,eta from kappa,mu
246  ! element can have transverse isotropy if between d220 and Moho
247  if (.not. ispec_is_tiso_crust_mantle(ispec)) then
248 
249  ! layer with no transverse isotropy
250  ! A,C,L,N,F from isotropic model
251 
252  mul = muvstore_crust_mantle(i,j,k,ispec)
253  kappal = kappavstore_crust_mantle(i,j,k,ispec)
254  muvl = mul
255  muhl = mul
256 
257  a = kappal + four_thirds * mul
258  c = a
259  l = mul
260  n = mul
261  f = kappal - 2._custom_real/3._custom_real * mul
262  eta = 1._custom_real
263 
264  else
265 
266  ! A,C,L,N,F from transverse isotropic model
267  kappavl = kappavstore_crust_mantle(i,j,k,ispec)
268  kappahl = kappahstore_crust_mantle(i,j,k,ispec)
269  muvl = muvstore_crust_mantle(i,j,k,ispec)
270  muhl = muhstore_crust_mantle(i,j,k,ispec)
271  kappal = kappavl
272 
273  a = kappahl + four_thirds * muhl
274  c = kappavl + four_thirds * muvl
275  l = muvl
276  n = muhl
277  eta = eta_anisostore_crust_mantle(i,j,k,ispec) ! that is F / (A - 2 L)
278  f = eta * ( a - 2._custom_real * l )
279 
280  endif

282  ! note: cijkl_kl_local() is fully anisotropic C_ij kernel components (non-dimensionalized)
283  ! for GLL point at (i,j,k,ispec)
284 
285  ! Purpose : compute the kernels for the An coeffs (an_kl)
286  ! from the kernels for Cij (cijkl_kl_local)
287  ! At r,theta,phi fixed
288  ! kernel def : dx = kij * dcij + krho * drho
289  ! = kAn * dAn + krho * drho
290 
291  ! Definition of the input array cij_kl :
292  ! cij_kl(1) = C11 ; cij_kl(2) = C12 ; cij_kl(3) = C13
293  ! cij_kl(4) = C14 ; cij_kl(5) = C15 ; cij_kl(6) = C16
294  ! cij_kl(7) = C22 ; cij_kl(8) = C23 ; cij_kl(9) = C24
295  ! cij_kl(10) = C25 ; cij_kl(11) = C26 ; cij_kl(12) = C33
296  ! cij_kl(13) = C34 ; cij_kl(14) = C35 ; cij_kl(15) = C36
297  ! cij_kl(16) = C44 ; cij_kl(17) = C45 ; cij_kl(18) = C46
298  ! cij_kl(19) = C55 ; cij_kl(20) = C56 ; cij_kl(21) = C66
299  ! where the Cij (Voigt's notation) are defined as function of
300  ! the components of the elastic tensor in spherical coordinates
301  ! by eq. (A.1) of Chen & Tromp, GJI 168 (2007)
302 
303  ! From the relations giving Cij in function of An
304  ! Checked with Min Chen's results (routine build_cij)
305 
306  an_kl(1) = cijkl_kl_local(1)+cijkl_kl_local(2)+cijkl_kl_local(7) !A
307  an_kl(2) = cijkl_kl_local(12) !C
308  an_kl(3) = -2*cijkl_kl_local(2)+cijkl_kl_local(21) !N
309  an_kl(4) = cijkl_kl_local(16)+cijkl_kl_local(19) !L
310  an_kl(5) = cijkl_kl_local(3)+cijkl_kl_local(8) !F
311

313  !an_kl(6)=2*cijkl_kl_local(5)+2*cijkl_kl_local(10)+2*cijkl_kl_local(14) !Jc
314  !an_kl(7)=2*cijkl_kl_local(4)+2*cijkl_kl_local(9)+2*cijkl_kl_local(13) !Js
315  !an_kl(8)=-2*cijkl_kl_local(14) !Kc
316  !an_kl(9)=-2*cijkl_kl_local(13) !Ks
317  !an_kl(10)=-2*cijkl_kl_local(10)+cijkl_kl_local(18) !Mc

326  !an_kl(19)=cijkl_kl_local(4)-cijkl_kl_local(9)+cijkl_kl_local(20) !Ds

330  ! K_rho (primary kernel, for a parameterization (A,C,L,N,F,rho) )
331  rhonotprime_kl_crust_mantle(i,j,k,ispec) = rhol * rho_kl_crust_mantle(i,j,k,ispec) / scale_kl_rho
332 
333  ! note: transverse isotropic kernels are calculated for ALL elements,
334  ! and not just transverse elements
335  !
336  ! note: the kernels are for relative perturbations (delta ln (m_i) = (m_i - m_0)/m_i )
337  !
338  ! Gets transverse isotropic kernels
339  ! (see Appendix B of Sieminski et al., GJI 171, 2007)
340 
341  ! for parameterization: ( alpha_v, alpha_h, beta_v, beta_h, eta, rho )
342  ! K_alpha_v
343  alphav_kl_crust_mantle(i,j,k,ispec) = 2*c*an_kl(2)
344  ! K_alpha_h

352  ! K_rhoprime (for a parameterization (alpha_v, ..., rho) )
353  rho_kl_crust_mantle(i,j,k,ispec) = a*an_kl(1) + c*an_kl(2) &
354  + n*an_kl(3) + l*an_kl(4) + f*an_kl(5) &
355  + rhonotprime_kl_crust_mantle(i,j,k,ispec)

358  rhonotprime_kl_crust_mantle(i,j,k,ispec) = - rhonotprime_kl_crust_mantle(i,j,k,ispec) * scale_kl

361  alphah_kl_crust_mantle(i,j,k,ispec) = - alphah_kl_crust_mantle(i,j,k,ispec) * scale_kl

365  rho_kl_crust_mantle(i,j,k,ispec) = - rho_kl_crust_mantle(i,j,k,ispec) * scale_kl
366

368  ! where kappa_v = kappa_h = kappa and bulk c = sqrt( kappa / rho )
369  betav_sq = muvl / rhol
370  betah_sq = muhl / rhol
371  alphav_sq = ( kappal + four_thirds * muvl ) / rhol
372  alphah_sq = ( kappal + four_thirds * muhl ) / rhol
373  bulk_sq = kappal / rhol
374

376  bulk_sq / alphav_sq * alphav_kl_crust_mantle(i,j,k,ispec) + &

378

380  betah_kl_crust_mantle(i,j,k,ispec) + &
381  four_thirds * betah_sq / alphah_sq * alphah_kl_crust_mantle(i,j,k,ispec)
382 
383  bulk_betav_kl_crust_mantle(i,j,k,ispec) = &
384  betav_kl_crust_mantle(i,j,k,ispec) + &
385  four_thirds * betav_sq / alphav_sq * alphav_kl_crust_mantle(i,j,k,ispec)
386  ! the rest, K_eta and K_rho are the same as above
387 
388  ! to check: isotropic kernels from transverse isotropic ones
389  alpha_kl_crust_mantle(i,j,k,ispec) = alphav_kl_crust_mantle(i,j,k,ispec) &
390  + alphah_kl_crust_mantle(i,j,k,ispec)
391  beta_kl_crust_mantle(i,j,k,ispec) = betav_kl_crust_mantle(i,j,k,ispec) &
392  + betah_kl_crust_mantle(i,j,k,ispec)

394  ! + alpha_kl_crust_mantle(i,j,k,ispec) &
395  ! + beta_kl_crust_mantle(i,j,k,ispec)

397  + bulk_betav_kl_crust_mantle(i,j,k,ispec)
398 
399  endif ! SAVE_TRANSVERSE_KL_ONLY
400 
401  enddo
402  enddo
403  enddo
404  enddo

409  betav_kl_crust_mantle,betah_kl_crust_mantle, &

859  subroutine save_kernels_source_derivatives()
860 
861  use specfem_par
862 
863  implicit none
864 
865  ! local parameters
866  real(kind=CUSTOM_REAL),parameter :: scale_mass = rhoav * (r_earth**3)
867  integer :: irec_local
868  character(len=MAX_STRING_LEN) :: outputname
869 
870  !scale_mass = RHOAV * (R_EARTH**3)
871 
872  ! computes derivatives
873  do irec_local = 1, nrec_local
874  ! rotate and scale the location derivatives to correspond to dn,de,dz
875  sloc_der(:,irec_local) = matmul(transpose(nu_source(:,:,irec_local)),sloc_der(:,irec_local)) &
876  * scale_displ * scale_t
877 
878  ! rotate scale the moment derivatives to correspond to M[n,e,z][n,e,z]
879  moment_der(:,:,irec_local) = matmul(matmul(transpose(nu_source(:,:,irec_local)),moment_der(:,:,irec_local)), &
880  nu_source(:,:,irec_local)) * scale_t ** 3 / scale_mass
881 
882  ! *nu_source* is the rotation matrix from ECEF to local N-E-UP as defined in src/specfem3D/locate_sources.f90
883 
884 ! From Qinya Liu, Toronto University, Canada:
885 ! these derivatives are basically derivatives of the misfit function phi with respect to
886 ! source parameters, which means, if the nu is the rotation matrix that
887 ! transforms coordinates from the global system (x,y,z) to the local
888 ! coordinate system (N,E,V), e.g., the moment tensor is transformed as
889 !
890 ! M_L = \nu * M_g * \nu^T,
891 !
892 ! then the derivative should be transformed as
893 !
894 ! \partial{\phi}{M_L} = \nu^T \partial{\phi}{M_g} \nu
895 !
896 ! which is in the opposite sense from the transformation of M.
897 
898  ! derivatives for time shift and hduration
899  stshift_der(irec_local) = stshift_der(irec_local) * scale_displ**2
900  shdur_der(irec_local) = shdur_der(irec_local) * scale_displ**2
901  enddo
902 
903  ! writes out kernels to file
904  if (adios_for_kernels) then
905  call write_kernels_source_derivatives_adios()
906  else
907  ! kernel file output
908  do irec_local = 1, nrec_local
909  write(outputname,'(a,i6.6)') trim(output_files)//'/src_frechet.',number_receiver_global(irec_local)
910  open(unit=iout,file=trim(outputname),status='unknown',action='write')
911  !
912  ! r -> z, theta -> -n, phi -> e, plus factor 2 for Mrt,Mrp,Mtp, and 1e-7 to dyne.cm
913  ! Mrr = Mzz
914  ! Mtt = Mnn
915  ! Mpp = Mee
916  ! Mrt = -Mzn
917  ! Mrp = Mze
918  ! Mtp = -Mne
919  ! for consistency, location derivatives are in the order of [Xr,Xt,Xp]
920  ! minus sign for sloc_der(3,irec_local) to get derivative for depth instead of radius
921  write(iout,'(g16.5)') moment_der(3,3,irec_local) * 1e-7
922  write(iout,'(g16.5)') moment_der(1,1,irec_local) * 1e-7
923  write(iout,'(g16.5)') moment_der(2,2,irec_local) * 1e-7
924  write(iout,'(g16.5)') -2*moment_der(1,3,irec_local) * 1e-7
