618 ! stores kernels for isotropic parameterization

621  use specfem_par_crustmantle
622 
623  implicit none
624 
625  ! local parameters
626  real(kind=CUSTOM_REAL) :: scale_kl,scale_kl_ani,scale_kl_rho
627  real(kind=CUSTOM_REAL) :: rhol,mul,kappal
628  real(kind=CUSTOM_REAL) :: rho_kl,alpha_kl,beta_kl
629  integer :: ispec,i,j,k
630  integer :: ier
631 
632  ! primary kernels
633  real(kind=CUSTOM_REAL), dimension(:,:,:,:),allocatable :: &
634  mu_kl_crust_mantle, kappa_kl_crust_mantle, rhonotprime_kl_crust_mantle
635 
636  ! bulk parameterization
637  real(kind=CUSTOM_REAL), dimension(:,:,:,:),allocatable :: &
638  bulk_c_kl_crust_mantle,bulk_beta_kl_crust_mantle
639 
640  ! checks if anything to do
641  if (anisotropic_kl) return
642 
643  ! scaling factors: note that this scaling has been introduced by Qinya Liu (2006)
644  ! with the intent to dimensionalize kernel values to [ s km^(-3) ]
645  !
646  ! kernel unit [ s / km^3 ]
647  scale_kl = scale_t * scale_displ_inv * 1.d9
648  ! For anisotropic kernels
649  ! final unit : [s km^(-3) GPa^(-1)]
650  scale_kl_ani = scale_t**3 / (rhoav*r_earth**3) * 1.d18
651  ! final unit : [s km^(-3) (kg/m^3)^(-1)]
652  scale_kl_rho = scale_t * scale_displ_inv / rhoav * 1.d9

654 
655  ! isotropic kernels
656  !
657  ! allocates temporary arrays
658  ! primary kernels
659  allocate(mu_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint), &
660  kappa_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint), &

662  if (ier /= 0 ) stop 'Error allocating transverse kernels bulk_c_kl_crust_mantle,...'
663

665  allocate(bulk_c_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint), &
666  bulk_beta_kl_crust_mantle(ngllx,nglly,ngllz,nspec_crust_mantle_adjoint),stat=ier)

671  do k = 1, ngllz
672  do j = 1, nglly
673  do i = 1, ngllx
674  ! gets material properties
675  rhol = rhostore_crust_mantle(i,j,k,ispec)
676  mul = muvstore_crust_mantle(i,j,k,ispec)
677  kappal = kappavstore_crust_mantle(i,j,k,ispec)
678

682  beta_kl = - 2 * mul * beta_kl_crust_mantle(i,j,k,ispec) ! note: beta_kl corresponds to K_mu

685  rhonotprime_kl_crust_mantle(i,j,k,ispec) = rho_kl * scale_kl

692  beta_kl_crust_mantle(i,j,k,ispec) = &
693  2._custom_real * (beta_kl - four_thirds * mul * alpha_kl / kappal) * scale_kl
694  alpha_kl_crust_mantle(i,j,k,ispec) = &
695  2._custom_real * (1 + four_thirds * mul / kappal) * alpha_kl * scale_kl
696 
697  ! for a parameterization: (rho,bulk, beta)
698  ! where bulk wave speed is c = sqrt( kappa / rho)
699  ! note: rhoprime is the same as for (rho,alpha,beta) parameterization
700  bulk_c_kl_crust_mantle(i,j,k,ispec) = 2._custom_real * alpha_kl * scale_kl
701  bulk_beta_kl_crust_mantle(i,j,k,ispec ) = 2._custom_real * beta_kl * scale_kl
702 
703  enddo
704  enddo
705  enddo
706  enddo
707 
708  ! writes out kernels to files
709  if (adios_for_kernels) then
710  call write_kernels_cm_iso_adios(mu_kl_crust_mantle, kappa_kl_crust_mantle, rhonotprime_kl_crust_mantle, &
711  bulk_c_kl_crust_mantle,bulk_beta_kl_crust_mantle)
712  else
713 
714  call create_name_database(prname,myrank,iregion_crust_mantle,local_tmp_path)
715 
716  ! primary kernels: (rho,kappa,mu) parameterization
717  open(unit=iout,file=trim(prname)//'rhonotprime_kernel.bin',status='unknown',form='unformatted',action='write')
718  write(iout) rhonotprime_kl_crust_mantle
719  close(iout)
720  open(unit=iout,file=trim(prname)//'kappa_kernel.bin',status='unknown',form='unformatted',action='write')
721  write(iout) kappa_kl_crust_mantle
722  close(iout)
723  open(unit=iout,file=trim(prname)//'mu_kernel.bin',status='unknown',form='unformatted',action='write')
724  write(iout) mu_kl_crust_mantle
725  close(iout)
726 
727  ! (rho, alpha, beta ) parameterization
728  open(unit=iout,file=trim(prname)//'rho_kernel.bin',status='unknown',form='unformatted',action='write')
729  write(iout) rho_kl_crust_mantle
730  close(iout)
731  open(unit=iout,file=trim(prname)//'alpha_kernel.bin',status='unknown',form='unformatted',action='write')
732  write(iout) alpha_kl_crust_mantle
733  close(iout)
734  open(unit=iout,file=trim(prname)//'beta_kernel.bin',status='unknown',form='unformatted',action='write')
735  write(iout) beta_kl_crust_mantle
736  close(iout)
737 
738  ! (rho, bulk, beta ) parameterization, K_rho same as above
739  open(unit=iout,file=trim(prname)//'bulk_c_kernel.bin',status='unknown',form='unformatted',action='write')
740  write(iout) bulk_c_kl_crust_mantle
741  close(iout)
742  open(unit=iout,file=trim(prname)//'bulk_beta_kernel.bin',status='unknown',form='unformatted',action='write')
743  write(iout) bulk_beta_kl_crust_mantle
744  close(iout)
745 
746  endif ! ADIOS_FOR_KERNELS
747 
748  ! cleans up temporary kernel arrays
749  deallocate(bulk_c_kl_crust_mantle,bulk_beta_kl_crust_mantle)
750  deallocate(mu_kl_crust_mantle,kappa_kl_crust_mantle,rhonotprime_kl_crust_mantle)

962  ! computes derivatives
963  do irec_local = 1, nrec_local
964  ! rotate and scale the location derivatives to correspond to dn,de,dz
965  sloc_der(:,irec_local) = matmul(transpose(nu_source(:,:,irec_local)),sloc_der(:,irec_local)) &
966  * scale_displ * scale_t
967 
968  ! rotate scale the moment derivatives to correspond to M[n,e,z][n,e,z]
969  moment_der(:,:,irec_local) = matmul(matmul(transpose(nu_source(:,:,irec_local)),moment_der(:,:,irec_local)), &
970  nu_source(:,:,irec_local)) * scale_t ** 3 / scale_mass

973 
974 ! From Qinya Liu, Toronto University, Canada:
975 ! these derivatives are basically derivatives of the misfit function phi with respect to
976 ! source parameters, which means, if the nu is the rotation matrix that
977 ! transforms coordinates from the global system (x,y,z) to the local
978 ! coordinate system (N,E,V), e.g., the moment tensor is transformed as
979 !
980 ! M_L = \nu * M_g * \nu^T,
981 !
982 ! then the derivative should be transformed as
983 !
984 ! \partial{\phi}{M_L} = \nu^T \partial{\phi}{M_g} \nu
985 !
986 ! which is in the opposite sense from the transformation of M.
987 
988  ! derivatives for time shift and hduration
989  stshift_der(irec_local) = stshift_der(irec_local) * scale_displ**2
990  shdur_der(irec_local) = shdur_der(irec_local) * scale_displ**2

1003  ! Mrr = Mzz
1004  ! Mtt = Mnn
1005  ! Mpp = Mee
1006  ! Mrt = -Mzn
1007  ! Mrp = Mze
1008  ! Mtp = -Mne
1009  ! for consistency, location derivatives are in the order of [Xr,Xt,Xp]
1010  ! minus sign for sloc_der(3,irec_local) to get derivative for depth instead of radius
1011  write(iout,'(g16.5)') moment_der(3,3,irec_local) * 1e-7
1012  write(iout,'(g16.5)') moment_der(1,1,irec_local) * 1e-7
1013  write(iout,'(g16.5)') moment_der(2,2,irec_local) * 1e-7
1014  write(iout,'(g16.5)') -2*moment_der(1,3,irec_local) * 1e-7

1016  write(iout,'(g16.5)') -2*moment_der(1,2,irec_local) * 1e-7

1019  write(iout,'(g16.5)') -sloc_der(3,irec_local)

1021  write(iout,'(g16.5)') stshift_der(irec_local)
1022  write(iout,'(g16.5)') shdur_der(irec_local)
