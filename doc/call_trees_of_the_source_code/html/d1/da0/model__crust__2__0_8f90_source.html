29 ! CRUST 2.0 model by Bassin et al. (2000)
30 !
31 ! C. Bassin, G. Laske, and G. Masters.
32 ! The current limits of resolution for surface wave tomography in North America.
33 ! EOS, 81: F897, 2000.
34 !
35 ! The 7 crustal layers:
36 ! ====================
37 ! 1) ice
38 ! 2) water
39 ! 3) soft sediments
40 ! 4) hard sediments

42 ! 6) middle crust
43 ! 7) lower crust
44 ! + Parameters VP, VS and rho are given explicitly for these 7 layers as well as the mantle below the Moho.
45 !

115  subroutine model_crust_2_0(lat,lon,x,vp,vs,rho,moho,found_crust,elem_in_crust)
116 
117  use constants
118  use model_crust_2_0_par
119 
120  implicit none
121 
122  double precision,intent(in) :: lat,lon,x
123  double precision,intent(out) :: vp,vs,rho,moho
124  logical,intent(out) :: found_crust
125  logical,intent(in) :: elem_in_crust
126 
127  ! local parameters
128  double precision :: thicks_1
129  double precision :: h_sed,h_uc
130  double precision :: x1,x3,x4,x5,x6,x7
131  double precision :: scaleval
132  double precision,dimension(CRUST_NP):: vps,vss,rhos,thicks
133 
134  ! initializes
135  vp = zero
136  vs = zero
137  rho = zero
138  moho = zero
139 
140  ! gets smoothed structure
141  call crust_2_0_capsmoothed(lat,lon,vps,vss,rhos,thicks,abbreviation, &
142  code,crust_thickness,crust_vp,crust_vs,crust_rho)
143 
144  ! note: for seismic wave propagation in general we ignore the water and ice sheets (oceans are re-added later as an ocean load)
145  ! note: but for gravity integral calculations we include the ice
146  if (include_ice_in_crust) then
147  thicks_1 = thicks(1)
148  else
149  thicks_1 = zero
150  endif
151 
152  ! whole sediment thickness (with ice if included)
153  h_sed = thicks_1 + thicks(3) + thicks(4)
154 
155  ! upper crust thickness (including sediments above, and also ice if included)
156  h_uc = h_sed + thicks(5)
157 
158  ! non-dimensionalization factor
159  scaleval = one / r_earth_km
160 
161  ! non-dimensionalize thicknesses (given in km)
162 
163  ! ice
164  x1 = one - thicks_1 * scaleval
165  ! upper sediment
166  x3 = one - (thicks_1 + thicks(3)) * scaleval
167  ! all sediments
168  x4 = one - h_sed * scaleval
169  ! upper crust
170  x5 = one - h_uc * scaleval
171  ! middle crust
172  x6 = one - (h_uc + thicks(6)) * scaleval
173  ! lower crust
174  x7 = one - (h_uc + thicks(6) + thicks(7)) * scaleval
175 
176  ! no matter if found_crust is true or false, compute moho thickness
177  moho = (h_uc + thicks(6) + thicks(7)) * scaleval
178 
179  ! gets corresponding crustal velocities and density
180  found_crust = .true.
181 
182  ! gets corresponding crustal velocities and density
183  if (x > x1 .and. include_ice_in_crust) then
184  vp = vps(1)
185  vs = vss(1)
186  rho = rhos(1)
187  else if (x > x3 .and. include_sediments_in_crust) then
188  vp = vps(3)
189  vs = vss(3)
190  rho = rhos(3)
191  else if (x > x4 .and. include_sediments_in_crust) then
192  vp = vps(4)
193  vs = vss(4)
194  rho = rhos(4)
195  else if (x > x5) then
196  vp = vps(5)
197  vs = vss(5)
198  rho = rhos(5)
199  else if (x > x6) then
200  vp = vps(6)
201  vs = vss(6)
202  rho = rhos(6)
203  else if (x > x7 .or. elem_in_crust) then
204  ! takes lower crustal values only if x is slightly above moho depth or
205  ! if elem_in_crust is set
206  !
207  ! note: it looks like this does distinguish between GLL points at the exact moho boundary,
208  ! where the point is on the interface between both,
209  ! oceanic elements and mantle elements below
210  vp = vps(7)
211  vs = vss(7)
212  rho = rhos(7)
213  else
214  ! note: if x is exactly the moho depth this will return false
215  found_crust = .false.
216  endif
217 
218  ! non-dimensionalize
219  if (found_crust) then
220  scaleval = one / ( r_earth_km * dsqrt(pi*grav*rhoav) )
221  vp = vp * scaleval
222  vs = vs * scaleval
223  rho = rho * 1000.0d0 / rhoav
224  endif
225 
226  end subroutine model_crust_2_0

233 
234  use constants

236 
237  implicit none

241 
242  double precision :: moho

246  character(len=*), parameter :: CNtype2_key_modif = 'DATA/crust2.0/CNtype2_key_modif.txt'

259  call exit_mpi(0,'Error model crust2.0')
260  endif

264  enddo
265  close(iin)
266 
267  open(unit = iin,file=cntype2_key_modif,status='old',action='read',iostat=ier)

270  call exit_mpi(0,'Error model crust2.0')
271  endif

273  h_moho_min = hugeval
274  h_moho_max = -hugeval

276  do ikey = 1,crust_nlo

278  read(iin,*) (crust_vp(i,ikey),i = 1,crust_np)
279  read(iin,*) (crust_vs(i,ikey),i = 1,crust_np)
280  read(iin,*) (crust_rho(i,ikey),i = 1,crust_np)

282 
283  ! thickness = ice (layer index 1) + sediment (index 3+4) + crystalline crust (index 5+6+7)
284  ! crustal thickness without ice
285  ! note: etopo1 has topography including ice ("ice surface" version) and at base of ice sheets ("bedrock" version)
286  ! see: http://www.ngdc.noaa.gov/mgg/global/global.html
287  moho = crust_thickness(3,ikey) + crust_thickness(4,ikey) &
288  + crust_thickness(5,ikey) + crust_thickness(6,ikey) + crust_thickness(7,ikey)
289 
290  ! limit moho thickness
291  if (moho > h_moho_max) h_moho_max = moho
292  if (moho < h_moho_min) h_moho_min = moho
293 
294  enddo
295  close(iin)
296 
297  ! user output
298  write(imain,*) ' Moho crustal thickness (without ice) min/max = ',sngl(h_moho_min),sngl(h_moho_max),' km'
299  write(imain,*)
300  call flush_imain()
301 
302  ! checks
303  if (h_moho_min == hugeval .or. h_moho_max == -hugeval) stop 'incorrect moho depths in read_crust_2_0_model'
304 
305  end subroutine read_crust_2_0_model

311  subroutine crust_2_0_capsmoothed(lat,lon,velp,vels,rho,thick,abbreviation,code,crust_thickness,crust_vp,crust_vs,crust_rho)
312 
313 ! crustal vp and vs in km/s, layer thickness in km
314 !
315 ! crust2.0 gets smoothed with a cap of size CAP using NTHETA points
316 ! in the theta direction and NPHI in the phi direction.
317 ! The cap is first rotated to the North Pole for easier implementation.
318 
319  use constants
320  use model_crust_2_0_par, only: crust_np,crust_nlo,crust_nla
321 
322  implicit none
323 
324  ! sampling rate for CAP points
325  integer, parameter :: NTHETA = 4
326  integer, parameter :: NPHI = 20
327 
328  ! argument variables
329  double precision :: lat,lon
330  double precision,dimension(CRUST_NP) :: rho,thick,velp,vels
331  double precision,dimension(CRUST_NP,CRUST_NLO) :: crust_thickness,crust_vp,crust_vs,crust_rho
332 
333  character(len=2) :: code(crust_nlo)
334  character(len=2) :: abbreviation(crust_nla/2,crust_nla)
335 
336  !-------------------------------
337  ! work-around to avoid Jacobian problems when stretching mesh elements;
338  ! one could also try to slightly change the shape of the doubling element bricks (which cause the problem)...
339  !
340  ! defines a "critical" region around the andes to have at least a 2-degree smoothing;
341  ! critical region can lead to negative Jacobians for mesh stretching when CAP smoothing is too small
342  double precision,parameter :: LAT_CRITICAL_ANDES = -20.0d0
343  double precision,parameter :: LON_CRITICAL_ANDES = -70.0d0
344  double precision,parameter :: CRITICAL_RANGE = 70.0d0
345  !-------------------------------
346 
347  ! local variables
348  double precision :: xlon(ntheta*nphi),xlat(ntheta*nphi),weight(ntheta*nphi)
349  double precision :: rhol(crust_np),thickl(crust_np),velpl(crust_np),velsl(crust_np)
350 
351  double precision :: weightl,cap_degree
352  double precision :: dist
353  double precision :: h_sed
354  integer :: i,icolat,ilon
355  character(len=2) :: crustaltype
356 
357  ! small hash table to convert crustal types to key
358  integer, dimension(128*128) :: crustalhash_to_key
359  integer :: ihash, crustalkey
360 
361  ! fill in the hash table
362  crustalhash_to_key(:) = -1
363  do i = 1,crust_nlo
364  call hash_crustal_type(code(i), ihash)
365  if (crustalhash_to_key(ihash) /= -1) stop 'Error in crust_2_0_CAPsmoothed: hash table collision'
366  crustalhash_to_key(ihash) = i
367  enddo
368 
369  ! checks latitude/longitude
370  if (lat > 90.0d0 .or. lat < -90.0d0 .or. lon > 180.0d0 .or. lon < -180.0d0) then
371  print *,'Error in lat/lon:',lat,lon
372  stop 'Error in latitude/longitude range in crust2.0'
373  endif
374 
375  ! makes sure lat/lon are within crust2.0 range
376  if (lat == 90.0d0) lat=89.9999d0
377  if (lat == -90.0d0) lat=-89.9999d0
378  if (lon == 180.0d0) lon=179.9999d0
379  if (lon == -180.0d0) lon=-179.9999d0
380 
381  ! sets up smoothing points based on cap smoothing
382  cap_degree = cap_smoothing_degree_default
383 
384  ! checks if inside/outside of critical region for mesh stretching
385  if (smooth_crust_even_more) then
386  dist = dsqrt( (lon-lon_critical_andes)**2 + (lat-lat_critical_andes )**2 )
387  if (dist < critical_range) then
388  ! increases cap smoothing degree
389  ! scales between -1 at center and 0 at border
390  dist = dist / critical_range - one
391  ! shifts value to 1 at center and 0 to the border with exponential decay
392  dist = one - exp( - dist*dist*10.0d0 )
393  ! increases smoothing degree inside of critical region to 2 degree
394  cap_degree = cap_degree + dist
395  endif
396  endif
397 
398  ! gets smoothing points and weights
399  call smooth_weights_cap_vardegree(lon,lat,xlon,xlat,weight,cap_degree,ntheta,nphi)
400 
401  ! initializes
402  velp(:) = zero
403  vels(:) = zero
404  rho(:) = zero
405  thick(:) = zero
406 
407  ! loops over weight points
408  do i = 1,ntheta*nphi
409  ! gets lat/lon indices
410  call icolat_ilon(xlat(i),xlon(i),icolat,ilon)
411 
412  crustaltype = abbreviation(icolat,ilon)
413 
414  call hash_crustal_type(crustaltype, ihash)
415  crustalkey = crustalhash_to_key(ihash)
416  if (crustalkey == -1) stop 'Error in retrieving crust type key'
417 
418  ! gets crust values
419  call get_crust_2_0_structure(crustalkey,velpl,velsl,rhol,thickl, &
420  crust_thickness,crust_vp,crust_vs,crust_rho)
421 
422  ! sediment thickness
423  h_sed = thickl(3) + thickl(4)
424 
425  ! takes upper crust value if sediment too thin
426  if (h_sed < minimum_sediment_thickness) then
427  velpl(3) = velpl(5)
428  velpl(4) = velpl(5)
429 
430  velsl(3) = velsl(5)
431  velsl(4) = velsl(5)
432 
433  rhol(3) = rhol(5)
434  rhol(4) = rhol(5)
435  endif
436 
437  ! weighting value
438  weightl = weight(i)
439 
440  ! total, smoothed values
441  rho(:) = rho(:) + weightl*rhol(:)
442  thick(:) = thick(:) + weightl*thickl(:)
443  velp(:) = velp(:) + weightl*velpl(:)
444  vels(:) = vels(:) + weightl*velsl(:)
445  enddo
446 
447  end subroutine crust_2_0_capsmoothed
