28 ! sum_preconditioned_kernels_globe
29 !
30 ! this program can be used for event kernel summation,
31 ! where it sums up transverse isotropic kernel files:
32 !
33 ! - proc***_reg1_bulk_c_kernel.bin
34 ! - proc***_reg1_bulk_betav_kernel.bin
35 ! - proc***_reg1_bulk_betah_kernel.bin
36 ! - proc***_reg1_eta_kernel.bin
37 !
38 ! input file: kernels_list.txt
39 ! lists all event kernel directories which should be summed together
40 !
41 ! input directory: INPUT_KERNELS/
42 ! contains links to all event kernel directories (listed in
43 ! "kernels_list.txt")
44 !
45 ! output directory: OUTPUT_SUM/
46 ! the resulting kernel files will be stored in this directory
47

58 
59  ! ============ program starts here =====================
60  ! initialize the MPI communicator and start the NPROCTOT MPI processes
61  call init_mpi()
62  call world_size(sizeprocs)

64

66  write(*,*) 'sum_preconditioned_kernels_globe:'

71

77  stop 1

88  write(*,*) ' ',nker,' events'

91

162 
163  endif

176 subroutine sum_kernel_pre(kernel_name,kernel_list,nker)
177 
178  use tomography_par
179 
180  implicit none
181 
182  character(len=MAX_STRING_LEN) :: kernel_name,kernel_list(max_kernel_paths)
183  integer :: nker
184 
185  ! local parameters
186  character(len=MAX_STRING_LEN) :: k_file
187  real(kind=CUSTOM_REAL), dimension(:,:,:,:),allocatable :: kernel,hess,total_kernel
188  double precision :: norm,norm_sum
189  integer :: iker,ier
190  real(kind=CUSTOM_REAL), dimension(:,:,:,:),allocatable :: total_hess,mask_source
191 
192  ! initializes arrays
193  allocate(kernel(ngllx,nglly,ngllz,nspec_crust_mantle), &
194  hess(ngllx,nglly,ngllz,nspec_crust_mantle), &
195  total_kernel(ngllx,nglly,ngllz,nspec_crust_mantle),stat=ier)
196  if (ier /= 0) stop 'Error allocating kernel arrays'
197 
198  if (use_hess_sum) then
199  allocate( total_hess(ngllx,nglly,ngllz,nspec_crust_mantle) )
200  total_hess(:,:,:,:) = 0.0_custom_real
201  endif
202 
203  if (use_source_mask) then
204  allocate( mask_source(ngllx,nglly,ngllz,nspec_crust_mantle) )
205  mask_source(:,:,:,:) = 1.0_custom_real
206  endif
207 
208  ! loops over all event kernels
209  total_kernel = 0._custom_real
210  do iker = 1, nker
211  ! user output
212  if (myrank == 0) then
213  write(*,*) 'reading in event kernel for: ',trim(kernel_name)
214  write(*,*) 'and preconditioner: ','hess_kernel'
215  write(*,*) ' ',iker, ' out of ', nker
216  endif
217 
218  ! sensitivity kernel / frechet derivative
219  kernel = 0._custom_real
220  write(k_file,'(a,i6.6,a)') 'INPUT_KERNELS/'//trim(kernel_list(iker)) &
221  //'/proc',myrank,trim(reg)//trim(kernel_name)//'.bin'
222  open(iin,file=trim(k_file),status='old',form='unformatted',action='read',iostat=ier)
223  if (ier /= 0) then
224  write(*,*) ' kernel not found: ',trim(k_file)
225  stop 'Error kernel file not found'
226  endif
227  read(iin) kernel
228  close(iin)
229 
230  ! outputs norm of kernel
231  norm = sum( kernel * kernel )
232  call sum_all_dp(norm,norm_sum)
233  if (myrank == 0) then
234  print *,' norm kernel: ',sqrt(norm_sum)
235  endif
236 
237  ! approximate Hessian
238  hess = 0._custom_real
239  write(k_file,'(a,i6.6,a)') 'INPUT_KERNELS/'//trim(kernel_list(iker)) &
240  //'/proc',myrank,trim(reg)//'hess_kernel.bin'
241  open(iin,file=trim(k_file),status='old',form='unformatted',action='read',iostat=ier)
242  if (ier /= 0) then
243  write(*,*) ' Hessian kernel not found: ',trim(k_file)
244  stop 'Error hess_kernel.bin files not found'
245  endif
246 
247  read(iin) hess
248  close(iin)
249 
250  ! outputs norm of preconditioner
251  norm = sum( hess * hess )
252  call sum_all_dp(norm,norm_sum)
253  if (myrank == 0) then
254  print *,' norm preconditioner: ',sqrt(norm_sum)
255  endif
256 
257  ! note: we take absolute values for Hessian (as proposed by Yang)
258  hess = abs(hess)
259 
260  ! source mask
261  if (use_source_mask) then
262  ! reads in mask
263  write(k_file,'(a,i6.6,a)') 'INPUT_KERNELS/'//trim(kernel_list(iker)) &
264  //'/proc',myrank,trim(reg)//'mask_source.bin'
265  open(iin,file=trim(k_file),status='old',form='unformatted',action='read',iostat=ier)
266  if (ier /= 0) then
267  write(*,*) ' file not found: ',trim(k_file)
268  stop 'Error source mask file not found'
269  endif
270  read(iin) mask_source
271  close(iin)
272 
273  ! masks source elements
274  kernel = kernel * mask_source
275  endif
276 
277  ! precondition
278  if (use_hess_sum) then
279 
280  ! sums up Hessians first
281  total_hess = total_hess + hess
282 
283  else
284 
285  ! inverts Hessian
286  call invert_hess( hess )
287 
288  ! preconditions each event kernel with its Hessian
289  kernel = kernel * hess
290 
291  endif
292 
293  ! sums all kernels from each event
294  total_kernel = total_kernel + kernel
295 
296  if (myrank == 0) print *
297  enddo
298 
299  ! preconditions summed kernels with summed Hessians
300  if (use_hess_sum) then
301 
302  ! inverts Hessian matrix
303  call invert_hess( total_hess )
304 
305  ! preconditions kernel
306  total_kernel = total_kernel * total_hess
307 
308  endif
309 
310  ! stores summed kernels
311  if (myrank == 0) write(*,*) 'writing out summed kernel for: ',trim(kernel_name)
312 
313  ! outputs summed kernel
314  write(k_file,'(a,i6.6,a)') 'OUTPUT_SUM/proc',myrank,trim(reg)//trim(kernel_name)//'.bin'
315  open(iout,file=trim(k_file),form='unformatted',status='unknown',action='write',iostat=ier)
316  if (ier /= 0) then
317  write(*,*) 'Error kernel not written: ',trim(k_file)
318  stop 'Error kernel write'
319  endif
320  write(iout) total_kernel
321  close(iout)
322 
323  ! outputs summed Hessian
324  if (use_hess_sum) then
325  if (myrank == 0) write(*,*) 'writing out summed kernel for: ','hess_inv_kernel'
326  write(k_file,'(a,i6.6,a)') 'OUTPUT_SUM/proc',myrank,trim(reg) // 'hess_inv_kernel' // '.bin'
327  open(iout,file=trim(k_file),form='unformatted',status='unknown',action='write',iostat=ier)
328  if (ier /= 0) then
329  write(*,*) 'Error kernel not written: ',trim(k_file)
330  stop 'Error kernel write'
331  endif
332  write(iout) total_hess
333  close(iout)
334  endif
335 
336  if (myrank == 0) write(*,*)
337 
338  ! frees memory
339  deallocate(kernel,hess,total_kernel)
340  if (use_hess_sum) deallocate(total_hess)
341  if (use_source_mask) deallocate(mask_source)
342 
343 end subroutine sum_kernel_pre

349 subroutine invert_hess( hess_matrix )
350 
351 ! inverts the Hessian matrix
352 ! the approximate Hessian is only defined for diagonal elements: like
353 ! H_nn = \frac{ \partial^2 \chi }{ \partial \rho_n \partial \rho_n }
354 ! on all GLL points, which are indexed (i,j,k,ispec)
355 
356  use tomography_par
357 
358  implicit none
359 
360  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLY,NGLLZ,NSPEC_CRUST_MANTLE) :: hess_matrix
361 
362  ! local parameters
363  real(kind=CUSTOM_REAL) :: maxh,maxh_all
364 
365  ! maximum value of Hessian
366  maxh = maxval( abs(hess_matrix) )
367 
368  ! determines maximum from all slices on master
369  call max_allreduce_cr(maxh,maxh_all)
370 
371  ! user output
372  if (myrank == 0) then
373  print *
374  print *,'Hessian maximum: ',maxh_all
375  print *
376  endif
377 
378  ! normalizes Hessian
379  if (maxh_all < 1.e-18) then
380  ! Hessian is zero, re-initializes
381  hess_matrix = 1.0_custom_real
382  !stop 'Error Hessian too small'
383  else
384  ! since Hessian has absolute values, this scales between [0,1]
385  hess_matrix = hess_matrix / maxh_all
386  endif
387 
388 
389  ! inverts Hessian values
390  where( abs(hess_matrix(:,:,:,:)) > threshold_hess )
391  hess_matrix = 1.0_custom_real / hess_matrix
392  elsewhere
393  hess_matrix = 1.0_custom_real / threshold_hess
394  endwhere
395 
396  ! rescales Hessian
397  !hess_matrix = hess_matrix * maxh_all
398 
399 end subroutine invert_hess
