44 
45  integer :: myrank
46  character(len=*) :: error_msg
47  character(len=80) :: outputname
48 
49  ! write error message to screen
50  write(*,*) error_msg(1:len(error_msg))
51  write(*,*) 'Error detected, aborting MPI... proc ',myrank
52 
53  ! write error message to file
54  write(outputname,"('/error_message',i6.6,'.txt')") myrank
55  open(unit=ierror,file=trim(output_files)//'/'//outputname,status='unknown')
56  write(ierror,*) error_msg(1:len(error_msg))
57  write(ierror,*) 'Error detected, aborting MPI... proc ',myrank
58  close(ierror)
59 
60  ! close output file
61  if (myrank == 0 .and. imain /= istandard_output) close(imain)
62 
63  ! stop all the MPI processes, and exit

67  ! however most compilers do recognize this as an error code stop statement;
68  ! to check stop code in terminal: > echo $?

70

72  !stop 'Error, program ended in exit_MPI'
73
