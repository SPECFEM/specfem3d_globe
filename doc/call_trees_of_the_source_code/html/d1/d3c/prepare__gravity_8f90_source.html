63  ! outer core
64  ! to hold pre-computed vectors (different between gravity or no gravity case)
65  allocate(gravity_pre_store_outer_core(ndim,nglob_outer_core),stat=ier)
66  if (ier /= 0) stop 'Error allocating gravity_grad_ln_density_dr array'
67  gravity_pre_store_outer_core(:,:) = 0._custom_real
68 
69  ! store g, rho and dg/dr=dg using normalized radius in lookup table every 100 m
70  ! get density and velocity from PREM model using dummy doubling flag
71  ! this assumes that the gravity perturbations are small and smooth
72  ! and that we can neglect the 3D model and use PREM every 100 m in all cases
73  ! this is probably a rather reasonable assumption
74  if (gravity_val) then
75  ! gravity term
76  call make_gravity(nspl_gravity,rspl_gravity,gspl,gspl2,one_crust)
77 
78  do int_radius = 1,nrad_gravity
79  radius = dble(int_radius) / (r_earth_km * 10.d0)
80  call spline_evaluation(rspl_gravity,gspl,gspl2,nspl_gravity,radius,g)
81 
82  ! use PREM density profile to calculate gravity (fine for other 1D models)
83  idoubling = 0
84  call model_prem_iso(radius,rho,drhodr,vp,vs,qkappa,qmu,idoubling,.false., &
85  one_crust,.false.,ricb,rcmb,rtopddoubleprime, &
86  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
87 
88  dg = 4.0d0 * rho - 2.0d0 * g / radius
89 
90  minus_gravity_table(int_radius) = - g
91  minus_deriv_gravity_table(int_radius) = - dg
92  density_table(int_radius) = rho
93  minus_rho_g_over_kappa_fluid(int_radius) = - g / vp**2
94  enddo
95 
96  ! make sure fluid array is only assigned in outer core between 1222 and 3478 km
97  ! lookup table is defined every 100 m
98  do int_radius = 1,nrad_gravity
99  radius_km = dble(int_radius) / 10.d0
100  if (radius_km > rcmb/1000.d0 - 3.d0) &
101  minus_rho_g_over_kappa_fluid(int_radius) = minus_rho_g_over_kappa_fluid(nint((rcmb/1000.d0 - 3.d0)*10.d0))
102  if (radius_km < ricb/1000.d0 + 3.d0) &
103  minus_rho_g_over_kappa_fluid(int_radius) = minus_rho_g_over_kappa_fluid(nint((ricb/1000.d0 + 3.d0)*10.d0))
104  enddo
105 
106  ! compute gravity value at CMB and ICB once and for all
107  radius = rcmb / r_earth
108  call spline_evaluation(rspl_gravity,gspl,gspl2,nspl_gravity,radius,g_cmb_dble)
109 
110  radius = ricb / r_earth
111  call spline_evaluation(rspl_gravity,gspl,gspl2,nspl_gravity,radius,g_icb_dble)
112 
113  ! distinguish between single and double precision for reals
114  minus_g_cmb = real(- g_cmb_dble, kind=custom_real)
115  minus_g_icb = real(- g_icb_dble, kind=custom_real)
116 
117  ! pre-computes gradient
118  do iglob = 1,nglob_outer_core
119  ! pre-computes factors
120  ! use mesh coordinates to get theta and phi
121  ! x y z contain r theta phi
122  radius = dble(rstore_outer_core(1,iglob))
123  theta = dble(rstore_outer_core(2,iglob))
124  phi = dble(rstore_outer_core(3,iglob))
125 
126  ! integrated and multiply by rho / Kappa
127  int_radius = nint(10.d0 * radius * r_earth_km)
128  fac = minus_rho_g_over_kappa_fluid(int_radius)
129 
130  ! gravitational acceleration (integrated and multiply by rho / Kappa) in Cartesian coordinates
131  gxl = dsin(theta) * dcos(phi) * fac
132  gyl = dsin(theta) * dsin(phi) * fac
133  gzl = dcos(theta) * fac
134  gravity_pre_store_outer_core(1,iglob) = real(gxl,kind=custom_real)
135  gravity_pre_store_outer_core(2,iglob) = real(gyl,kind=custom_real)
136  gravity_pre_store_outer_core(3,iglob) = real(gzl,kind=custom_real)
137  enddo
138 
139  else
140  ! no gravity
141  ! tabulate d ln(rho)/dr needed for the no gravity fluid potential
142  do int_radius = 1,nrad_gravity
143  radius = dble(int_radius) / (r_earth_km * 10.d0)
144  idoubling = 0
145  call model_prem_iso(radius,rho,drhodr,vp,vs,qkappa,qmu,idoubling,.false., &
146  one_crust,.false.,ricb,rcmb,rtopddoubleprime, &
147  r600,r670,r220,r771,r400,r80,rmoho,rmiddle_crust,rocean)
148 
149  d_ln_density_dr_table(int_radius) = drhodr/rho
150  enddo
151 
152  ! pre-computes gradient
153  do iglob = 1,nglob_outer_core
154  ! pre-computes factors
155  ! use mesh coordinates to get theta and phi
156  ! x y z contain r theta phi
157  radius = rstore_outer_core(1,iglob)
158  theta = rstore_outer_core(2,iglob)
159  phi = rstore_outer_core(3,iglob)
160 
161  ! d ln(rho) / dr
162  int_radius = nint(10.d0 * radius * r_earth_km)
163  fac = d_ln_density_dr_table(int_radius)
164 
165  ! gradient of d ln(rho)/dr in Cartesian coordinates
166  gxl = dsin(theta) * dcos(phi) * fac
167  gyl = dsin(theta) * dsin(phi) * fac
168  gzl = dcos(theta) * fac
169  gravity_pre_store_outer_core(1,iglob) = real(gxl,kind=custom_real)
170  gravity_pre_store_outer_core(2,iglob) = real(gyl,kind=custom_real)
171  gravity_pre_store_outer_core(3,iglob) = real(gzl,kind=custom_real)
172  enddo
173  endif
174 
175  ! crust/mantle
176  if (gravity_val) then
177  ! to hold pre-computed vectors (different between gravity or no gravity case)
178  allocate(gravity_pre_store_crust_mantle(ndim,nglob_crust_mantle), &
179  gravity_h_crust_mantle(6,nglob_crust_mantle),stat=ier)
180  if (ier /= 0) stop 'Error allocating gravity arrays for crust/mantle'
181  gravity_pre_store_crust_mantle(:,:) = 0._custom_real
182  gravity_h_crust_mantle(:,:) = 0._custom_real
183 
184  do iglob = 1,nglob_crust_mantle
185  ! use mesh coordinates to get theta and phi
186  ! x y and z contain r theta and phi
187  radius = dble(rstore_crust_mantle(1,iglob))
188  theta = dble(rstore_crust_mantle(2,iglob))
189  phi = dble(rstore_crust_mantle(3,iglob))
190 
191  cos_theta = dcos(theta)
192  sin_theta = dsin(theta)
193  cos_phi = dcos(phi)
194  sin_phi = dsin(phi)
195 
196  cos_theta_sq = cos_theta*cos_theta
197  sin_theta_sq = sin_theta*sin_theta
198  cos_phi_sq = cos_phi*cos_phi
199  sin_phi_sq = sin_phi*sin_phi
200 
201  ! get g, rho and dg/dr=dg
202  ! spherical components of the gravitational acceleration
203 
204  ! for efficiency replace with lookup table every 100 m in radial direction
205  int_radius = nint(10.d0 * radius * r_earth_km )
206 
207  rho = density_table(int_radius)
208  minus_g = minus_gravity_table(int_radius)
209  minus_dg = minus_deriv_gravity_table(int_radius)
210 
211  ! Cartesian components of the gravitational acceleration
212  ! multiplied by common factor rho
213  gxl = minus_g * sin_theta * cos_phi * rho
214  gyl = minus_g * sin_theta * sin_phi * rho
215  gzl = minus_g * cos_theta * rho
216  gravity_pre_store_crust_mantle(1,iglob) = real(gxl,kind=custom_real)
217  gravity_pre_store_crust_mantle(2,iglob) = real(gyl,kind=custom_real)
218  gravity_pre_store_crust_mantle(3,iglob) = real(gzl,kind=custom_real)
219 
220  ! H-matrix
221  ! Cartesian components of gradient of gravitational acceleration
222  ! obtained from spherical components
223  ! multiplied by common factor rho
224  minus_g_over_radius = minus_g / radius
225  minus_dg_plus_g_over_radius = minus_dg - minus_g_over_radius
226 
227  hxxl = (minus_g_over_radius*(cos_phi_sq*cos_theta_sq + sin_phi_sq) + cos_phi_sq*minus_dg*sin_theta_sq) * rho
228  hyyl = (minus_g_over_radius*(cos_phi_sq + cos_theta_sq*sin_phi_sq) + minus_dg*sin_phi_sq*sin_theta_sq) * rho
229  hzzl = (cos_theta_sq*minus_dg + minus_g_over_radius*sin_theta_sq) * rho
230  hxyl = (cos_phi*minus_dg_plus_g_over_radius*sin_phi*sin_theta_sq) * rho
231  hxzl = (cos_phi*cos_theta*minus_dg_plus_g_over_radius*sin_theta) * rho
232  hyzl = (cos_theta*minus_dg_plus_g_over_radius*sin_phi*sin_theta) * rho
233 
234  gravity_h_crust_mantle(1,iglob) = real(hxxl,kind=custom_real)
235  gravity_h_crust_mantle(2,iglob) = real(hyyl,kind=custom_real)
236  gravity_h_crust_mantle(3,iglob) = real(hzzl,kind=custom_real)
237  gravity_h_crust_mantle(4,iglob) = real(hxyl,kind=custom_real)
238  gravity_h_crust_mantle(5,iglob) = real(hxzl,kind=custom_real)
239  gravity_h_crust_mantle(6,iglob) = real(hyzl,kind=custom_real)
240  enddo
241  else
242  ! dummy allocation
243  allocate(gravity_pre_store_crust_mantle(ndim,1), &
244  gravity_h_crust_mantle(6,1),stat=ier)
245  if (ier /= 0) stop 'Error allocating gravity arrays for crust/mantle'
246  endif
247 
248 
249  ! inner core
250  if (gravity_val) then
251  ! to hold pre-computed vectors (different between gravity or no gravity case)
252  allocate(gravity_pre_store_inner_core(ndim,nglob_inner_core), &
253  gravity_h_inner_core(6,nglob_inner_core),stat=ier)
254  if (ier /= 0) stop 'Error allocating gravity arrays for inner core'
255  gravity_pre_store_inner_core(:,:) = 0._custom_real

262  theta = dble(rstore_inner_core(2,iglob))
263  phi = dble(rstore_inner_core(3,iglob))
264 
265  ! make sure radius is never zero even for points at center of cube

268

270  sin_theta = dsin(theta)

278
