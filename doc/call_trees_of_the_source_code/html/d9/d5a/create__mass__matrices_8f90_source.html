28  subroutine create_mass_matrices(nspec,nglob,idoubling,ibool, &
29  iregion_code,xstore,ystore,zstore, &
30  nspec2d_top,nspec2d_bottom)
31 
32  ! creates rmassx, rmassy, rmassz and rmass_ocean_load
33 
34 !
35 ! ****************************************************************************************************
36 ! IMPORTANT: this routine must *NOT* use flag SIMULATION_TYPE (nor SAVE_FORWARD), i.e. none of the parameters it computes
37 ! should depend on SIMULATION_TYPE, because most users do not recompile the code nor rerun the mesher
38 ! when switching from SIMULATION_TYPE == 1 to SIMULATION_TYPE == 3 and thus the header file created
39 ! by this routine would become wrong in the case of a run with SIMULATION_TYPE == 3 if the code
40 ! was compiled with SIMULATION_TYPE == 1
41 ! ****************************************************************************************************
42 !
43 
44  use constants
45 
46  use meshfem3d_models_par, only: &
47  oceans
48 
49  use meshfem3d_par, only: &
50  myrank,nchunks,absorbing_conditions, &
51  rotation,exact_mass_matrix_for_rotation,include_central_cube
52 
53  use create_regions_mesh_par, only: &
54  wxgll,wygll,wzgll
55 
56  use create_regions_mesh_par2, only: &
57  xixstore,xiystore,xizstore,etaxstore,etaystore,etazstore, &
58  gammaxstore,gammaystore,gammazstore,rhostore,kappavstore, &
59  rmassx,rmassy,rmassz,b_rmassx,b_rmassy, &
60  nglob_xy
61 
62  implicit none
63 
64  integer :: nspec,nglob
65  integer,dimension(nspec) :: idoubling
66  integer,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: ibool
67 
68  integer :: iregion_code
69 
70  ! arrays with the mesh in double precision
71  double precision,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: xstore,ystore,zstore
72 
73  ! Stacey conditions put back
74  integer :: NSPEC2D_TOP,NSPEC2D_BOTTOM
75 
76  ! local parameters
77  double precision :: weight
78  real(kind=CUSTOM_REAL) :: xixl,xiyl,xizl,etaxl,etayl,etazl,gammaxl,gammayl,gammazl,jacobianl
79 
80  integer :: ispec,i,j,k,iglob
81 
82  ! initializes matrices
83  !
84  ! in the case of Stacey boundary conditions, add C*delta/2 contribution to the mass matrix
85  ! on the Stacey edges for the crust_mantle and outer_core regions but not for the inner_core region
86  ! thus the mass matrix must be replaced by three mass matrices including the "C" damping matrix
87  !
88  ! if absorbing_conditions are not set or if NCHUNKS=6, only one mass matrix is needed
89  ! for the sake of performance, only "rmassz" array will be filled and "rmassx" & "rmassy" will be fictitious / unused
90  !
91  ! Now also handle EXACT_MASS_MATRIX_FOR_ROTATION, which requires similar corrections
92 
93  rmassx(:) = 0._custom_real
94  rmassy(:) = 0._custom_real
95  rmassz(:) = 0._custom_real
96 
97  b_rmassx(:) = 0._custom_real
98  b_rmassy(:) = 0._custom_real
99 
100 !----------------------------------------------------------------
101 
102 ! first create the main standard mass matrix with no corrections
103 !$OMP PARALLEL DEFAULT(SHARED) &
104 !$OMP PRIVATE(ispec,i,j,k,iglob,weight, &
105 !$OMP xixl,xiyl,xizl,etaxl,etayl,etazl,gammaxl,gammayl,gammazl,jacobianl)
106 !$OMP DO
107  do ispec = 1,nspec
108 
109  ! suppress fictitious elements in central cube
110  if (idoubling(ispec) == iflag_in_fictitious_cube) cycle
111 
112  do k = 1,ngllz
113  do j = 1,nglly
114  do i = 1,ngllx
115 
116  weight = wxgll(i)*wygll(j)*wzgll(k)
117  iglob = ibool(i,j,k,ispec)
118 
119  ! compute the Jacobian
120  xixl = xixstore(i,j,k,ispec)
121  xiyl = xiystore(i,j,k,ispec)
122  xizl = xizstore(i,j,k,ispec)
123  etaxl = etaxstore(i,j,k,ispec)
124  etayl = etaystore(i,j,k,ispec)
125  etazl = etazstore(i,j,k,ispec)
126  gammaxl = gammaxstore(i,j,k,ispec)
127  gammayl = gammaystore(i,j,k,ispec)
128  gammazl = gammazstore(i,j,k,ispec)
129 
130  jacobianl = 1._custom_real / (xixl*(etayl*gammazl-etazl*gammayl) &
131  - xiyl*(etaxl*gammazl-etazl*gammaxl) &
132  + xizl*(etaxl*gammayl-etayl*gammaxl))
133 
134  ! definition depends if region is fluid or solid
135  select case (iregion_code)
136 
137  case (iregion_crust_mantle, iregion_inner_core)
138  ! distinguish between single and double precision for reals
139  rmassz(iglob) = rmassz(iglob) + &
140  real(dble(rhostore(i,j,k,ispec)) * dble(jacobianl) * weight, kind=custom_real)
141 
142  ! fluid in outer core
143  case (iregion_outer_core)
144 
145  ! no anisotropy in the fluid, use kappav
146 
147  ! distinguish between single and double precision for reals
148  rmassz(iglob) = rmassz(iglob) + &
149  real(dble(jacobianl) * weight * dble(rhostore(i,j,k,ispec)) / dble(kappavstore(i,j,k,ispec)), &
150  kind=custom_real)
151 
152  case default
153  call exit_mpi(myrank,'wrong region code')
154 
155  end select
156 
157  enddo
158  enddo
159  enddo
160  enddo ! of loop on ispec
161 !$OMP enddo
162 !$OMP END PARALLEL
163 
164 ! copy the initial mass matrix if needed
165  if (nglob_xy == nglob) then
166  rmassx(:) = rmassz(:)
167  rmassy(:) = rmassz(:)
168 
169  b_rmassx(:) = rmassz(:)
170  b_rmassy(:) = rmassz(:)
171  endif
172 
173  ! then make the corrections to the copied mass matrices if needed
174  if (rotation .and. exact_mass_matrix_for_rotation) then
175  call create_mass_matrices_rotation(nspec,ibool,idoubling,iregion_code)
176  endif
177 
178  ! absorbing boundaries
179  ! add C*deltat/2 contribution to the mass matrices on the Stacey edges
180  if (nchunks /= 6 .and. absorbing_conditions) then
181  call create_mass_matrices_stacey(nspec,ibool,iregion_code,nspec2d_bottom)
182  endif
183 
184  ! check that mass matrix is positive
185  if (iregion_code == iregion_inner_core .and. include_central_cube) then
186  ! note: in fictitious elements mass matrix is still zero
187  if (minval(rmassz(:)) < 0._custom_real) call exit_mpi(myrank,'negative rmassz matrix term')
188  ! check that the additional mass matrices are positive, if they exist
189  if (nglob_xy == nglob) then
190  if (minval(rmassx) < 0._custom_real) call exit_mpi(myrank,'negative rmassx matrix term')
191  if (minval(rmassy) < 0._custom_real) call exit_mpi(myrank,'negative rmassy matrix term')
192  if (minval(b_rmassx) < 0._custom_real) call exit_mpi(myrank,'negative b_rmassx matrix term')
193  if (minval(b_rmassy) < 0._custom_real) call exit_mpi(myrank,'negative b_rmassy matrix term')
194  endif
195  else
196  ! no fictitious elements, mass matrix must be strictly positive
197  if (minval(rmassz(:)) <= 0._custom_real) call exit_mpi(myrank,'negative rmassz matrix term')
198  ! check that the additional mass matrices are strictly positive, if they exist
199  if (nglob_xy == nglob) then
200  if (minval(rmassx) <= 0._custom_real) call exit_mpi(myrank,'negative rmassx matrix term')
201  if (minval(rmassy) <= 0._custom_real) call exit_mpi(myrank,'negative rmassy matrix term')
202  if (minval(b_rmassx) <= 0._custom_real) call exit_mpi(myrank,'negative b_rmassx matrix term')
203  if (minval(b_rmassy) <= 0._custom_real) call exit_mpi(myrank,'negative b_rmassy matrix term')
204  endif
205  endif
206 
207  ! save ocean load mass matrix as well if oceans
208  if (oceans .and. iregion_code == iregion_crust_mantle) then
209  call create_mass_matrices_ocean_load(nspec,ibool,xstore,ystore,zstore,nspec2d_top)
210  endif
211 
212  end subroutine create_mass_matrices

219  subroutine create_mass_matrices_rotation(nspec,ibool,idoubling,iregion_code)
220 
221 ! in the case of Stacey boundary conditions, add C*deltat/2 contribution to the mass matrix
222 ! on Stacey edges for the crust_mantle and outer_core regions but not for the inner_core region
223 ! thus the mass matrix must be replaced by three mass matrices including the "C" damping matrix
224 !
225 ! only called in case of (ROTATION .and. EXACT_MASS_MATRIX_FOR_ROTATION)
226  use constants
227 
228  use meshfem3d_par, only: &
229  myrank,dt
230 
231  use create_regions_mesh_par, only: &
232  wxgll,wygll,wzgll
233 
234  use create_regions_mesh_par2, only: &
235  xixstore,xiystore,xizstore,etaxstore,etaystore,etazstore, &
236  gammaxstore,gammaystore,gammazstore, &
237  rmassx,rmassy,b_rmassx,b_rmassy
238 
239  use shared_parameters, only: undo_attenuation
240 
241  implicit none
242 
243  integer :: nspec
244 
245  integer,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: ibool
246  integer,dimension(nspec) :: idoubling
247 
248  integer :: iregion_code
249 
250  ! local parameters
251  double precision :: weight
252 
253  real(kind=CUSTOM_REAL) :: xixl,xiyl,xizl,etaxl,etayl,etazl,gammaxl,gammayl,gammazl,jacobianl
254  real(kind=CUSTOM_REAL) :: deltat,two_omega_earth_dt,b_two_omega_earth_dt,scale_t_inv
255 
256  integer :: ispec,i,j,k,iglob
257 
258  ! user output
259  if (myrank == 0) then
260  write(imain,*) ' creates exact mass matrix for rotation'
261  call flush_imain()
262  endif
263 
264  ! use the non-dimensional time step to make the mass matrix correction
265  deltat = real(DT*dsqrt(PI*GRAV*RHOAV), kind=custom_real)
266 
267  scale_t_inv = real(dsqrt(PI*GRAV*RHOAV), kind=custom_real)
268 
269  ! distinguish between single and double precision for reals
270  two_omega_earth_dt = real(2.d0 * TWO_PI / (HOURS_PER_DAY * SECONDS_PER_HOUR * scale_t_inv) * deltat, kind=custom_real)
271 
272  ! reconstructed wavefield
273  if (undo_attenuation) then
274  ! spinning forward
275  b_two_omega_earth_dt = two_omega_earth_dt
276  else
277  ! spinning backward to reconstruct wavefield
278  b_two_omega_earth_dt = - two_omega_earth_dt
279  endif
280 
281  ! definition depends if region is fluid or solid
282  select case (iregion_code)
283 
284  case (iregion_crust_mantle, iregion_inner_core)
285 
286 !$OMP PARALLEL DEFAULT(SHARED) &
287 !$OMP PRIVATE(ispec,i,j,k,iglob,weight, &
288 !$OMP xixl,xiyl,xizl,etaxl,etayl,etazl,gammaxl,gammayl,gammazl,jacobianl)
289 !$OMP DO
290  do ispec = 1,nspec
291 
292  ! suppress fictitious elements in central cube
293  if (idoubling(ispec) == iflag_in_fictitious_cube) cycle
294 
295  do k = 1,ngllz
296  do j = 1,nglly
297  do i = 1,ngllx
298 
299  weight = wxgll(i)*wygll(j)*wzgll(k)
300  iglob = ibool(i,j,k,ispec)
301 
302  ! compute the Jacobian
303  xixl = xixstore(i,j,k,ispec)
304  xiyl = xiystore(i,j,k,ispec)
305  xizl = xizstore(i,j,k,ispec)
306  etaxl = etaxstore(i,j,k,ispec)
307  etayl = etaystore(i,j,k,ispec)
308  etazl = etazstore(i,j,k,ispec)
309  gammaxl = gammaxstore(i,j,k,ispec)
310  gammayl = gammaystore(i,j,k,ispec)
311  gammazl = gammazstore(i,j,k,ispec)
312 
313  jacobianl = 1._custom_real / (xixl*(etayl*gammazl-etazl*gammayl) &
314  - xiyl*(etaxl*gammazl-etazl*gammaxl) &
315  + xizl*(etaxl*gammayl-etayl*gammaxl))
316 
317  ! distinguish between single and double precision for reals
318  rmassx(iglob) = rmassx(iglob) &
319  - two_omega_earth_dt * 0.5_custom_real*real(dble(jacobianl) * weight, kind=custom_real)
320  rmassy(iglob) = rmassy(iglob) &
321  + two_omega_earth_dt * 0.5_custom_real*real(dble(jacobianl) * weight, kind=custom_real)
322 
323  b_rmassx(iglob) = b_rmassx(iglob) &
324  - b_two_omega_earth_dt * 0.5_custom_real*real(dble(jacobianl) * weight, kind=custom_real)
325  b_rmassy(iglob) = b_rmassy(iglob) &
326  + b_two_omega_earth_dt * 0.5_custom_real*real(dble(jacobianl) * weight, kind=custom_real)
327  enddo
328  enddo
329  enddo
330  enddo ! of loop on ispec
331 !$OMP enddo
332 !$OMP END PARALLEL
333 
334  end select
335 
336 
337  end subroutine create_mass_matrices_rotation

344  subroutine create_mass_matrices_stacey(nspec,ibool,iregion_code, &
345  nspec2d_bottom)
346 
347 ! in the case of Stacey boundary conditions, add C*deltat/2 contribution to the mass matrix
348 ! on Stacey edges for the crust_mantle and outer_core regions but not for the inner_core region
349 ! thus the mass matrix must be replaced by three mass matrices including the "C" damping matrix
350 
351  use constants
352 
353  use meshfem3d_par, only: &
354  myrank,dt,nchunks,ichunk, &
355  rotation,exact_mass_matrix_for_rotation
356 
357  use create_regions_mesh_par, only: &
358  wxgll,wygll,wzgll
359 
360  use create_regions_mesh_par2, only: &
361  rmassx,rmassy,rmassz,b_rmassx,b_rmassy, &
362  ibelm_xmin,ibelm_xmax,ibelm_ymin,ibelm_ymax,ibelm_bottom, &
363  normal_xmin,normal_xmax,normal_ymin,normal_ymax, &
364  jacobian2d_xmin,jacobian2d_xmax,jacobian2d_ymin,jacobian2d_ymax, &
365  jacobian2d_bottom, &
366  rho_vp,rho_vs, &
367  nspec2d_xmin,nspec2d_xmax,nspec2d_ymin,nspec2d_ymax, &
368  nimin,nimax,njmin,njmax,nkmin_xi,nkmin_eta
369 
370  implicit none
371 
372  integer :: nspec
373  integer,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: ibool
374 
375  integer :: iregion_code
376 
377  ! Stacey conditions
378  integer :: NSPEC2D_BOTTOM
379 
380  ! local parameters
381  double precision :: weight
382  double precision, dimension(NGLLX,NGLLY) :: wgllwgll_xy
383  double precision, dimension(NGLLX,NGLLZ) :: wgllwgll_xz
384  double precision, dimension(NGLLY,NGLLZ) :: wgllwgll_yz
385 
386  real(kind=CUSTOM_REAL) :: deltat,deltatover2
387  real(kind=CUSTOM_REAL) :: tx,ty,tz,sn
388  real(kind=CUSTOM_REAL) :: nx,ny,nz,vn
389 
390  integer :: ispec,i,j,k,iglob
391  integer :: ispec2D
392 
393  ! user output
394  if (myrank == 0) then
395  write(imain,*) ' updates mass matrix with Stacey boundary corrections'
396  call flush_imain()
397  endif
398 
399  ! checks if we have absorbing boundary arrays
400  if (.not. allocated(nimin) ) call exit_mpi(myrank,'Error Stacey array not allocated')
401 
402  ! use the non-dimensional time step to make the mass matrix correction
403  deltat = real(DT*dsqrt(PI*GRAV*RHOAV), kind=custom_real)
404  deltatover2 = real(0.5d0*deltat, kind=custom_real)
405 
406  ! weights on surfaces
407  do i = 1,ngllx
408  do j = 1,nglly
409  wgllwgll_xy(i,j) = wxgll(i)*wygll(j)
410  enddo
411  enddo
412  do i = 1,ngllx
413  do k = 1,ngllz
414  wgllwgll_xz(i,k) = wxgll(i)*wzgll(k)
415  enddo
416  enddo
417  do j = 1,nglly
418  do k = 1,ngllz
419  wgllwgll_yz(j,k) = wygll(j)*wzgll(k)
420  enddo
421  enddo
422 
423  ! adds contributions to mass matrix to stabilize Stacey conditions
424  select case (iregion_code)
425  case (iregion_crust_mantle)
426 
427  rmassx(:) = rmassz(:)
428  rmassy(:) = rmassz(:)
429 
430  ! xmin
431  ! if two chunks exclude this face for one of them
432  if (nchunks == 1 .or. ichunk == chunk_ac) then
433 
434  do ispec2d = 1,nspec2d_xmin
435 
436  ispec=ibelm_xmin(ispec2d)
437 
438  ! exclude elements that are not on absorbing edges
439  if (nkmin_xi(1,ispec2d) == 0 .or. njmin(1,ispec2d) == 0) cycle
440 
441  i = 1
442  do k = nkmin_xi(1,ispec2d),ngllz
443  do j = njmin(1,ispec2d),njmax(1,ispec2d)
444  iglob=ibool(i,j,k,ispec)
445 
446  nx = normal_xmin(1,j,k,ispec2d)
447  ny = normal_xmin(2,j,k,ispec2d)
448  nz = normal_xmin(3,j,k,ispec2d)
449 
450  vn = deltatover2*(nx+ny+nz)
451 
452  tx = rho_vp(i,j,k,ispec)*vn*nx + rho_vs(i,j,k,ispec)*(deltatover2-vn*nx)
453  ty = rho_vp(i,j,k,ispec)*vn*ny + rho_vs(i,j,k,ispec)*(deltatover2-vn*ny)
454  tz = rho_vp(i,j,k,ispec)*vn*nz + rho_vs(i,j,k,ispec)*(deltatover2-vn*nz)
455 
456  weight = jacobian2d_xmin(j,k,ispec2d)*wgllwgll_yz(j,k)
457 
458  rmassx(iglob) = rmassx(iglob) + real(tx*weight, kind=custom_real)
459  rmassy(iglob) = rmassy(iglob) + real(ty*weight, kind=custom_real)
460  rmassz(iglob) = rmassz(iglob) + real(tz*weight, kind=custom_real)
461  if (rotation .and. exact_mass_matrix_for_rotation) then
462  b_rmassx(iglob) = b_rmassx(iglob) + real(tx*weight, kind=custom_real)
463  b_rmassy(iglob) = b_rmassy(iglob) + real(ty*weight, kind=custom_real)
464  endif
465  enddo
466  enddo
467  enddo
468 
469  endif ! NCHUNKS == 1 .or. ichunk == CHUNK_AC
470 
471  ! xmax
472  ! if two chunks exclude this face for one of them
473  if (nchunks == 1 .or. ichunk == chunk_ab) then
474 
475  do ispec2d = 1,nspec2d_xmax
476 
477  ispec=ibelm_xmax(ispec2d)
478 
479  ! exclude elements that are not on absorbing edges
480  if (nkmin_xi(2,ispec2d) == 0 .or. njmin(2,ispec2d) == 0) cycle
481 
482  i=ngllx
483  do k=nkmin_xi(2,ispec2d),ngllz
484  do j=njmin(2,ispec2d),njmax(2,ispec2d)
485  iglob=ibool(i,j,k,ispec)
486 
487  nx = normal_xmax(1,j,k,ispec2d)
488  ny = normal_xmax(2,j,k,ispec2d)
489  nz = normal_xmax(3,j,k,ispec2d)
490 
491  vn = deltatover2*(nx+ny+nz)
492 
493  tx = rho_vp(i,j,k,ispec)*vn*nx + rho_vs(i,j,k,ispec)*(deltatover2-vn*nx)
494  ty = rho_vp(i,j,k,ispec)*vn*ny + rho_vs(i,j,k,ispec)*(deltatover2-vn*ny)
495  tz = rho_vp(i,j,k,ispec)*vn*nz + rho_vs(i,j,k,ispec)*(deltatover2-vn*nz)
496 
497  weight = jacobian2d_xmax(j,k,ispec2d)*wgllwgll_yz(j,k)
498 
499  rmassx(iglob) = rmassx(iglob) + real(tx*weight, kind=custom_real)
500  rmassy(iglob) = rmassy(iglob) + real(ty*weight, kind=custom_real)
501  rmassz(iglob) = rmassz(iglob) + real(tz*weight, kind=custom_real)
502  if (rotation .and. exact_mass_matrix_for_rotation) then
503  b_rmassx(iglob) = b_rmassx(iglob) + real(tx*weight, kind=custom_real)
504  b_rmassy(iglob) = b_rmassy(iglob) + real(ty*weight, kind=custom_real)
505  endif
506  enddo
507  enddo
508  enddo
509 
510  endif ! NCHUNKS == 1 .or. ichunk == CHUNK_AB
511 
512  ! ymin
513  do ispec2d = 1,nspec2d_ymin
514 
515  ispec=ibelm_ymin(ispec2d)
516 
517  ! exclude elements that are not on absorbing edges
518  if (nkmin_eta(1,ispec2d) == 0 .or. nimin(1,ispec2d) == 0) cycle
519 
520  j = 1
521  do k = nkmin_eta(1,ispec2d),ngllz
522  do i = nimin(1,ispec2d),nimax(1,ispec2d)
523  iglob=ibool(i,j,k,ispec)
524 
525  nx = normal_ymin(1,i,k,ispec2d)
526  ny = normal_ymin(2,i,k,ispec2d)
527  nz = normal_ymin(3,i,k,ispec2d)
528 
529  vn = deltatover2*(nx+ny+nz)

758  if (myrank == 0) then
759  write(imain,*) ' updates mass matrix with ocean load'
760  call flush_imain()
761  endif

764  do_ocean_load = .false.
765

767  ! only for models where 3D crustal stretching was used (even without topography?)
768  if (use_old_version_5_1_5_format) then
769  if (case_3d) then
770  do_ocean_load = .true.

773  ! note: new version:
774  ! for 3D Earth with topography, compute local height of oceans
775  if (topography) then
776  do_ocean_load = .true.
777  endif
778  endif

781  rmass_ocean_load(:) = 0._custom_real
782 
783  ! add contribution of the oceans
784  ! for surface elements exactly at the top of the crust (ocean bottom)
785 !$OMP PARALLEL DEFAULT(SHARED) &
786 !$OMP PRIVATE(ispec2D,ispec,i,j,iglob,x,y,z,r,theta,phi,lat,lon,elevation,height_oceans,weight)
787 !$OMP DO
788  do ispec2d = 1,nspec2d_top
789 
790  ! gets spectral element index
791  ispec = ibelm_top(ispec2d)
792 
793  ! assumes elements are ordered such that k == NGLLZ is the top surface
794  k = ngllz
795 
796  ! loops over surface points
797  do j = 1,nglly
798  do i = 1,ngllx
799 
800  ! for 3D Earth with topography, compute local height of oceans
801  if (do_ocean_load) then
802 
803  ! get coordinates of current point
804  x = xstore(i,j,k,ispec)
805  y = ystore(i,j,k,ispec)
806  z = zstore(i,j,k,ispec)
807

809  ! slightly move points to avoid roundoff problem when exactly on the polar axis
810  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
811 
812  if (.not. use_old_version_5_1_5_format) then
813  ! adds small margins
814  !! DK DK: added a test to only do this if we are on the axis
815  if (abs(theta) > 89.99d0) then

817  phi = phi + 0.0000001d0
818  endif
819  endif

827  if (use_old_version_5_1_5_format) then

832  call geocentric_2_geographic_dble(theta,theta)

841

849

852  height_oceans = thickness_oceans_prem
853  endif

856  weight = wxgll(i) * wygll(j) &
857  * dble(jacobian2d_top(i,j,ispec2d)) &
858  * dble(rho_oceans) * height_oceans

861  iglob = ibool(i,j,k,ispec)
862 
863  ! distinguish between single and double precision for reals
864  rmass_ocean_load(iglob) = rmass_ocean_load(iglob) + real(weight, kind=custom_real)
865 
866  enddo
867  enddo
868  enddo
869 !$OMP enddo
870 !$OMP END PARALLEL
871 
872  ! add regular mass matrix to ocean load contribution
873  rmass_ocean_load(:) = rmass_ocean_load(:) + rmassz(:)
874
