37 
38  use constants

41 
42  use adios_read_mod

45 
46  implicit none
47 
48  ! GLL model_variables
49  type(model_gll_variables) mgll_v
50 
51  integer, dimension(MAX_NUM_REGIONS) :: NSPEC
52  integer :: myrank
53 
54  ! local parameters
55  integer :: local_dim
56  character(len=MAX_STRING_LEN) :: file_name
57 
58  ! ADIOS variables
59  integer :: adios_err
60  integer(kind=8), dimension(1) :: start, count
61 
62  integer(kind=8) :: sel
63 
64  ! only crust and mantle
65  write(file_name,'(a)') trim(pathname_gll_modeldir) // 'model_gll.bp'
66 
67  ! user output
68  if (myrank == 0) then
69  write(imain,*)
70  write(imain,*)'reading in model from ',trim(pathname_gll_modeldir)
71  write(imain,*)'ADIOS file: ',trim(file_name)
72  call flush_imain()
73  endif
74 
75  ! Setup the ADIOS library to read the file
76  call open_file_adios_read(file_name)
77 
78  local_dim = ngllx * nglly * ngllz * nspec(iregion_crust_mantle)
79  start(1) = local_dim*myrank
80  count(1) = local_dim
81  call adios_selection_boundingbox (sel , 1, start, count)
82 
83  ! reads in model for each partition
84  if (.not. transverse_isotropy) then
85  ! isotropic model
86  ! vp mesh
87  call adios_schedule_read(file_handle_adios, sel, "reg1/vp/array", 0, 1, &
88  mgll_v%vp_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
89  call adios_schedule_read(file_handle_adios, sel, "reg1/vs/array", 0, 1, &
90  mgll_v%vs_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
91  else
92  ! transverse isotropic model
93  ! WARNING previously wronly name 'vps' in the adios files
94  ! vp mesh
95  call adios_schedule_read(file_handle_adios, sel, "reg1/vpv/array", 0, 1, &
96  mgll_v%vpv_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
97  call adios_schedule_read(file_handle_adios, sel, "reg1/vph/array", 0, 1, &
98  mgll_v%vph_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
99 
100  ! vs mesh
101  call adios_schedule_read(file_handle_adios, sel, "reg1/vsv/array", 0, 1, &
102  mgll_v%vsv_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
103  call adios_schedule_read(file_handle_adios, sel, "reg1/vsh/array", 0, 1, &
104  mgll_v%vsh_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
105 
106  ! eta mesh
107  call adios_schedule_read(file_handle_adios, sel, "reg1/eta/array", 0, 1, &
108  mgll_v%eta_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
109  endif
110 
111  ! rho mesh
112  call adios_schedule_read(file_handle_adios, sel, "reg1/rho/array", 0, 1, &
113  mgll_v%rho_new(:,:,:,1:nspec(iregion_crust_mantle)), adios_err)
114 
115  call adios_perform_reads(file_handle_adios, adios_err)
116  call check_adios_err(myrank,adios_err)
117 
118  call adios_read_close(file_handle_adios, adios_err)
119  call check_adios_err(myrank,adios_err)
120 
121 end subroutine read_gll_model_adios
