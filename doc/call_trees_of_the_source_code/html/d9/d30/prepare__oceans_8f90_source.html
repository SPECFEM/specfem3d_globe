29  subroutine prepare_oceans()
30 
31 ! prepares arrays for coupling with oceans
32 !
33 ! note: handling of coupling on GPU needs to avoid using a mutex to update acceleration;
34 ! tests so far have shown, that with a simple mutex implementation
35 ! the results differ between successive runs (probably still due to some race conditions?)
36 !
37 ! here we now totally avoid mutex usage and still update each global point only once,
38 ! also facilitating vectorization of the updating loop
39 
40  use specfem_par
41  use specfem_par_crustmantle
42 
43  implicit none
44  ! local parameters
45  integer :: ipoin,ispec,ispec2D,i,j,k,iglob,ier
46  ! flag to mask ocean-bottom degrees of freedom for ocean load
47  logical, dimension(:), allocatable :: updated_dof_ocean_load
48 
49  ! checks if anything to do
50  if (.not. oceans_val) return
51 
52  ! user info
53  if (myrank == 0) then
54  write(imain,*) "preparing oceans arrays"
55  call flush_imain()
56  endif
57 
58  allocate(updated_dof_ocean_load(nglob_crust_mantle_oceans),stat=ier)
59  if (ier /= 0) stop 'Error allocating arrays updated_dof_ocean_load'
60  updated_dof_ocean_load(:) = .false.
61 
62  ! counts global points on surface to oceans
63  ipoin = 0
64  do ispec2d = 1,nspec_top
65  ispec = ibelm_top_crust_mantle(ispec2d)
66  k = ngllz
67  do j = 1,nglly
68  do i = 1,ngllx
69  ! get global point number
70  iglob = ibool_crust_mantle(i,j,k,ispec)
71  ! updates once
72  if (.not. updated_dof_ocean_load(iglob)) then
73  ipoin = ipoin + 1
74  updated_dof_ocean_load(iglob) = .true.
75  endif
76  enddo
77  enddo
78  enddo
79  npoin_oceans = ipoin
80 
81  ! user info
82  if (myrank == 0) then
83  write(imain,*) " number of global points on oceans = ",npoin_oceans
84  call flush_imain()
85  endif
86 
87  ! allocates arrays with all global points on ocean surface
88  allocate(ibool_ocean_load(npoin_oceans), &
89  normal_ocean_load(ndim,npoin_oceans), &
90  rmass_ocean_load_selected(npoin_oceans), &
91  stat=ier)
92  if (ier /= 0 ) stop 'Error allocating oceans arrays'
93  ibool_ocean_load(:) = 0
94  normal_ocean_load(:,:) = 0._custom_real
95  rmass_ocean_load_selected(:) = 0._custom_real
96 
97  ! fills arrays for coupling surface at oceans
98  updated_dof_ocean_load(:) = .false.
99  ipoin = 0
100  do ispec2d = 1,nspec_top
101  ispec = ibelm_top_crust_mantle(ispec2d)
102  k = ngllz
103  do j = 1,nglly
104  do i = 1,ngllx
105  ! get global point number
106  iglob = ibool_crust_mantle(i,j,k,ispec)
107  ! updates once
108  if (.not. updated_dof_ocean_load(iglob)) then
109  ipoin = ipoin + 1
110  ! fills arrays
111  ibool_ocean_load(ipoin) = iglob
112  rmass_ocean_load_selected(ipoin) = rmass_ocean_load(iglob)
113  normal_ocean_load(:,ipoin) = normal_top_crust_mantle(:,i,j,ispec2d)
114  ! masks this global point
115  updated_dof_ocean_load(iglob) = .true.
116  endif
117  enddo
118  enddo
119  enddo
120 
121  ! frees memory
122  deallocate(updated_dof_ocean_load)
123  deallocate(rmass_ocean_load)
124 
125  ! synchronizes processes
126  call synchronize_all()
127 
128  end subroutine prepare_oceans
