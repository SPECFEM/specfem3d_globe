28  subroutine add_topography_icb(xelm,yelm,zelm)
29 
30 ! this is only a placeholder routine, which is not used yet... user must supply the sutopo_icb() routine
31 
32  use constants
33  use meshfem3d_par, only: ricb,rcmb
34 
35  implicit none
36 
37  double precision,intent(inout) :: xelm(ngnod)
38  double precision,intent(inout) :: yelm(ngnod)
39  double precision,intent(inout) :: zelm(ngnod)
40 
41  ! local parameters
42  integer :: ia
43 
44  double precision :: topoicb
45 
46  double precision :: r,lat,lon
47  double precision :: x,y,z
48  double precision :: gamma
49 
50 ! we loop on all the points of the element
51  do ia = 1,ngnod
52 
53  x = xelm(ia)
54  y = yelm(ia)
55  z = zelm(ia)
56 
57  ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
58  call xyz_2_rlatlon_dble(x,y,z,r,lat,lon)
59 
60  ! compute topography on ICB; the routine subtopo_icb needs to be supplied by the user
61  ! (see for example routine subtopo_sh_cmb() in model_full_sh.f90 for CMB topography)
62  ! call subtopo_icb(lat,lon,topoicb)
63  ! until then, no topography pertubations...
64  topoicb = 0.0d0
65 
66  ! non-dimensionalize the topography, which is in km
67  ! positive for a depression, so change the sign for a perturbation in radius
68  topoicb = -topoicb / r_earth_km
69 
70  gamma = 0.0d0
71  if (r > 0.0d0 .and. r <= ricb/r_earth) then
72  ! stretching between center and RICB
73  gamma = r/(ricb/r_earth)
74  else if (r >= ricb/r_earth .and. r <= rcmb/r_earth) then
75  ! stretching between RICB and RCMB
76  gamma = (r - rcmb/r_earth) / (ricb/r_earth - rcmb/r_earth)
77  endif
78  if (gamma < -0.0001 .or. gamma > 1.0001) call exit_mpi(myrank,'incorrect value of gamma for CMB topography')
79 
80  xelm(ia) = x*(one + gamma * topoicb / r)
81  yelm(ia) = y*(one + gamma * topoicb / r)
82  zelm(ia) = z*(one + gamma * topoicb / r)
83 
84  enddo
85 
86  end subroutine add_topography_icb
