95 
96  case (iregion_outer_core)

131  case (iregion_outer_core)
132  ! check that the number of points in this slice is correct
133  if (minval(ibool(:,:,:,:)) /= 1 .or. maxval(ibool(:,:,:,:)) /= nglob_outer_core) &
134  call exit_mpi(myrank,'incorrect global numbering: iboolmax does not equal nglob in outer core')
135 
136  case (iregion_inner_core)
137  ! check that the number of points in this slice is correct
138  if (minval(ibool(:,:,:,:)) /= 1 .or. maxval(ibool(:,:,:,:)) /= nglob_inner_core) &
139  call exit_mpi(myrank,'incorrect global numbering: iboolmax does not equal nglob in inner core')
140 
141  end select
142 
143  ! allocates arrays
144  allocate(buffer_send_chunkcorn_scalar(nglob1d_radial_cm), &
145  buffer_recv_chunkcorn_scalar(nglob1d_radial_cm),stat=ier)
146  if (ier /= 0) stop 'Error allocating buffer buffer_send_chunkcorn_scalar,.. arrays'
147 
148  allocate(buffer_send_chunkcorn_vector(ndim,nglob1d_radial_cm + nglob1d_radial_ic), &
149  buffer_recv_chunkcorn_vector(ndim,nglob1d_radial_cm + nglob1d_radial_ic),stat=ier)
150  if (ier /= 0) stop 'Error allocating buffer buffer_send_chunkcorn_vector,.. arrays'
151 
152  select case (iregion_code)
153  case (iregion_crust_mantle)
154  ! crust mantle
155  allocate(iboolcorner_crust_mantle(nglob1d_radial_cm,numcorners_shared))
156  allocate(iboolleft_xi_crust_mantle(nglob2dmax_xmin_xmax_cm), &
157  iboolright_xi_crust_mantle(nglob2dmax_xmin_xmax_cm))
158  allocate(iboolleft_eta_crust_mantle(nglob2dmax_ymin_ymax_cm), &
159  iboolright_eta_crust_mantle(nglob2dmax_ymin_ymax_cm))
160  allocate(iboolfaces_crust_mantle(nglob2dmax_xy,numfaces_shared))
161 
162  case (iregion_outer_core)
163  ! outer core
164  allocate(iboolcorner_outer_core(nglob1d_radial_oc,numcorners_shared))
165  allocate(iboolleft_xi_outer_core(nglob2dmax_xmin_xmax_oc), &
166  iboolright_xi_outer_core(nglob2dmax_xmin_xmax_oc))
167  allocate(iboolleft_eta_outer_core(nglob2dmax_ymin_ymax_oc), &
168  iboolright_eta_outer_core(nglob2dmax_ymin_ymax_oc))
169  allocate(iboolfaces_outer_core(nglob2dmax_xy,numfaces_shared))
170 
171  case (iregion_inner_core)
172  ! inner core
173  allocate(iboolcorner_inner_core(nglob1d_radial_ic,numcorners_shared))
174  allocate(iboolleft_xi_inner_core(nglob2dmax_xmin_xmax_ic), &
175  iboolright_xi_inner_core(nglob2dmax_xmin_xmax_ic))
176  allocate(iboolleft_eta_inner_core(nglob2dmax_ymin_ymax_ic), &
177  iboolright_eta_inner_core(nglob2dmax_ymin_ymax_ic))
178  allocate(iboolfaces_inner_core(nglob2dmax_xy,numfaces_shared))
179 
180  end select
181 
182  ! synchronize processes
183  call synchronize_all()
184 
185  end subroutine cmi_allocate_addressing

237  write(imain,*)

246  iboolleft_eta_crust_mantle,iboolright_eta_crust_mantle, &
247  npoin2d_xi_crust_mantle,npoin2d_eta_crust_mantle, &
248  iboolfaces_crust_mantle,npoin2d_faces_crust_mantle, &
249  iboolcorner_crust_mantle)

251  ! note: fix_... routines below update is_on_a_slice_edge_.. arrays:

253  ! thus, they include elements on top and bottom not shared with other MPI partitions

256  ! use these arrays for now as initial guess for the search for elements which share a global point
257  ! between different MPI processes
258  call fix_non_blocking_slices(is_on_a_slice_edge, &
259  iboolright_xi_crust_mantle,iboolleft_xi_crust_mantle, &
260  iboolright_eta_crust_mantle,iboolleft_eta_crust_mantle, &
261  npoin2d_xi_crust_mantle,npoin2d_eta_crust_mantle, &
262  ibool, &
263  nspec_crust_mantle,nglob_crust_mantle,nglob2dmax_xmin_xmax_cm,nglob2dmax_ymin_ymax_cm)
264 
265  ! debug: saves element flags
266  if (debug) then
267  write(filename,'(a,i6.6)') trim(output_files)//'/MPI_is_on_a_slice_edge_crust_mantle_proc',myrank
268  call write_vtk_data_elem_l(nspec_crust_mantle,nglob_crust_mantle, &
269  xstore_glob,ystore_glob,zstore_glob, &
270  ibool,is_on_a_slice_edge,filename)
271  endif
272 
273  ! added this to reduce the size of the buffers
274  ! size of buffers is the sum of two sizes because we handle two regions in the same MPI call
275  !npoin2D_max_all_CM_IC = max(maxval(npoin2D_xi_crust_mantle(:) + npoin2D_xi_inner_core(:)), &
276  ! maxval(npoin2D_eta_crust_mantle(:) + npoin2D_eta_inner_core(:)))

278  maxval(npoin2d_eta_crust_mantle(:)))
279 
280  case (iregion_outer_core)
281  ! outer core
282  if (myrank == 0) then
283  write(imain,*)
284  write(imain,*) 'outer core region:'
285  call flush_imain()
286  endif
287  call cmi_read_buffer_data(iregion_outer_core, &
288  nglob2dmax_xmin_xmax(iregion_outer_core), &

290  nglob1d_radial(iregion_outer_core), &
291  iboolleft_xi_outer_core,iboolright_xi_outer_core, &
292  iboolleft_eta_outer_core,iboolright_eta_outer_core, &
293  npoin2d_xi_outer_core,npoin2d_eta_outer_core, &
294  iboolfaces_outer_core,npoin2d_faces_outer_core, &

296 
297  ! note: fix_... routines below update is_on_a_slice_edge_.. arrays:

306  iboolright_eta_outer_core,iboolleft_eta_outer_core, &
307  npoin2d_xi_outer_core,npoin2d_eta_outer_core, &
308  ibool, &
309  nspec_outer_core,nglob_outer_core,nglob2dmax_xmin_xmax_oc,nglob2dmax_ymin_ymax_oc)
