29 ! This subroutine recomputes the 3D Jacobian for one element
30 ! based upon 125 GLL points

36 ! ispec,nspec ----- element number

43  subroutine recalc_jacobian_gll3d(xstore,ystore,zstore,xigll,yigll,zigll, &
44  ispec,nspec, &
45  xixstore,xiystore,xizstore, &

48 
49  use constants, only: myrank,ngllx,nglly,ngllz,custom_real,size_real, &
50  zero,one,tinyval,verysmallval,r_earth_km,radians_to_degrees
51 
52  implicit none
53

56

85  logical, parameter :: DEBUG = .false.
86 
87  double precision:: xmesh,ymesh,zmesh
88  double precision:: sumshape,sumdershapexi,sumdershapeeta,sumdershapegamma
89 
90  ! first go over all 125 GLL points
91  do k = 1,ngllz
92  do j = 1,nglly
93  do i = 1,ngllx
94 
95  xi = xigll(i)
96  eta = yigll(j)
97  gamma = zigll(k)
98 
99  ! calculate Lagrange polynomial and its derivative
100  call lagrange_any(xi,ngllx,xigll,hxir,hpxir)
101  call lagrange_any(eta,nglly,yigll,hetar,hpetar)
102  call lagrange_any(gamma,ngllz,zigll,hgammar,hpgammar)
103 
104  xxi = zero
105  xeta = zero
106  xgamma = zero
107  yxi = zero
108  yeta = zero
109  ygamma = zero
110  zxi = zero
111  zeta = zero
112  zgamma = zero
113 
114  ! test parameters
115  if (debug) then
116  sumshape = zero
117  sumdershapexi = zero
118  sumdershapeeta = zero
119  sumdershapegamma = zero
120 
121  xmesh = zero
122  ymesh = zero
123  zmesh = zero
124  endif
125 
126  do k1 = 1,ngllz
127  do j1 = 1,nglly
128  do i1 = 1,ngllx
129 
130  hlagrange = hxir(i1) * hetar(j1) * hgammar(k1)
131  hlagrange_xi = hpxir(i1) * hetar(j1) * hgammar(k1)
132  hlagrange_eta = hxir(i1) * hpetar(j1) * hgammar(k1)
133  hlagrange_gamma = hxir(i1) * hetar(j1) * hpgammar(k1)
134 
135  x = xstore(i1,j1,k1,ispec)
136  y = ystore(i1,j1,k1,ispec)
137  z = zstore(i1,j1,k1,ispec)
138 
139  xxi = xxi + x * hlagrange_xi
140  xeta = xeta + x * hlagrange_eta
141  xgamma = xgamma + x * hlagrange_gamma
142 
143  yxi = yxi + y * hlagrange_xi
144  yeta = yeta + y * hlagrange_eta
145  ygamma = ygamma + y * hlagrange_gamma
146 
147  zxi = zxi + z * hlagrange_xi
148  zeta = zeta + z * hlagrange_eta
149  zgamma = zgamma + z * hlagrange_gamma
150 
151  ! test the Lagrange polynomial and its derivative
152  if (debug) then
153  xmesh = xmesh + x * hlagrange
154  ymesh = ymesh + y * hlagrange
155  zmesh = zmesh + z * hlagrange
156 
157  sumshape = sumshape + hlagrange
158  sumdershapexi = sumdershapexi + hlagrange_xi
159  sumdershapeeta = sumdershapeeta + hlagrange_eta
160  sumdershapegamma = sumdershapegamma + hlagrange_gamma
161  endif
162 
163  enddo
164  enddo
165  enddo
166 
167  ! Check the Lagrange polynomial and its derivative
168  if (debug) then
169  if (abs(xmesh - xstore(i,j,k,ispec)) > tinyval &
170  .or. abs(ymesh - ystore(i,j,k,ispec)) > tinyval &
171  .or. abs(zmesh - zstore(i,j,k,ispec)) > tinyval) then
172  call exit_mpi(myrank,'Error new mesh is wrong in recalc_jacobian_gll3D.f90')
173  endif
174  if (abs(sumshape-one) > tinyval) then
175  call exit_mpi(myrank,'Error shape functions in recalc_jacobian_gll3D.f90')
176  endif
177  if (abs(sumdershapexi) > tinyval) then
178  call exit_mpi(myrank,'Error derivative xi in recalc_jacobian_gll3D.f90')
179  endif
180  if (abs(sumdershapeeta) > tinyval) then
181  call exit_mpi(myrank,'Error derivative eta in recalc_jacobian_gll3D.f90')
182  endif
183  if (abs(sumdershapegamma) > tinyval) then
184  call exit_mpi(myrank,'Error derivative gamma in recalc_jacobian_gll3D.f90')
185  endif
186  endif
187 
188  ! Jacobian calculation
189  jacobian = xxi*(yeta*zgamma-ygamma*zeta) - &
190  xeta*(yxi*zgamma-ygamma*zxi) + &
191  xgamma*(yxi*zeta-yeta*zxi)
192 
193  ! Check the Jacobian
194  ! note: when honoring the moho, we squeeze and stretch elements
195  ! thus, it can happen that with a coarse mesh resolution, the Jacobian encounters problems
196  if (jacobian <= verysmallval) then
197  ! note: the mesh can have ellipticity, thus the geocentric colatitude might differ from the geographic one
198  !
199  ! converts position to geocentric coordinates
200  call xyz_2_rthetaphi_dble(xmesh,ymesh,zmesh,r,theta,phi)
201  print *,'Error Jacobian rank:',myrank
202  print *,' location r/lat/lon: ',r*r_earth_km, &
203  90.0-(theta*radians_to_degrees),phi*radians_to_degrees
204  print *,' Jacobian: ',jacobian
205  call exit_mpi(myrank,'3D Jacobian undefined in recalc_jacobian_gll3D.f90')
206  endif
207 
208  ! invert the relation (Fletcher p. 50 vol. 2)
209  jacobian_inv = one / jacobian
210 
211  xix = (yeta*zgamma-ygamma*zeta) * jacobian_inv
212  xiy = (xgamma*zeta-xeta*zgamma) * jacobian_inv
213  xiz = (xeta*ygamma-xgamma*yeta) * jacobian_inv
214  etax = (ygamma*zxi-yxi*zgamma) * jacobian_inv
215  etay = (xxi*zgamma-xgamma*zxi) * jacobian_inv
216  etaz = (xgamma*yxi-xxi*ygamma) * jacobian_inv
217  gammax = (yxi*zeta-yeta*zxi) * jacobian_inv
218  gammay = (xeta*zxi-xxi*zeta) * jacobian_inv
219  gammaz = (xxi*yeta-xeta*yxi) * jacobian_inv
220 
221  ! resave the derivatives and the Jacobian
222  ! distinguish between single and double precision for reals
223  xixstore(i,j,k,ispec) = real(xix, kind=custom_real)
224  xiystore(i,j,k,ispec) = real(xiy, kind=custom_real)
225  xizstore(i,j,k,ispec) = real(xiz, kind=custom_real)
226  etaxstore(i,j,k,ispec) = real(etax, kind=custom_real)
227  etaystore(i,j,k,ispec) = real(etay, kind=custom_real)
228  etazstore(i,j,k,ispec) = real(etaz, kind=custom_real)
229  gammaxstore(i,j,k,ispec) = real(gammax, kind=custom_real)
230  gammaystore(i,j,k,ispec) = real(gammay, kind=custom_real)
231  gammazstore(i,j,k,ispec) = real(gammaz, kind=custom_real)
232 
233  enddo
234  enddo
235  enddo
236 
237  end subroutine recalc_jacobian_gll3d

297  ymesh = zero
298  zmesh = zero

310  xeta = xeta + xelm2d(i1,j1)*hlagrange_eta
311

317 
318  xmesh = xmesh + xelm2d(i1,j1)*hlagrange
319  ymesh = ymesh + yelm2d(i1,j1)*hlagrange
320  zmesh = zmesh + zelm2d(i1,j1)*hlagrange
321  sumshape = sumshape + hlagrange
322  sumdershapexi = sumdershapexi + hlagrange_xi
323  sumdershapeeta = sumdershapeeta + hlagrange_eta
324  enddo

331  .or. abs(zmesh - zelm2d(i,j)) > tinyval) then

333  endif
334

336  call exit_mpi(myrank,'Error shape functions in recalc_jacobian_gll2D')

344

347  uny = zxi*xeta - zeta*xxi
348  unz = xxi*yeta - xeta*yxi

359  normal(1,i,j,ispecb) = real(unx * jacobian_inv, kind=custom_real)
360  normal(2,i,j,ispecb) = real(uny * jacobian_inv, kind=custom_real)
361  normal(3,i,j,ispecb) = real(unz * jacobian_inv, kind=custom_real)
362  enddo
363  enddo
