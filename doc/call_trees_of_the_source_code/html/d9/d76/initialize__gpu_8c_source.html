893 static char *trim_and_default(char *s, int max_string_length)
894 {
895  // trim before
896  while (*s != '\0' && isspace(*s)) { s++; }
897 
898  if (*s == '\0') {
899  return s;

901

908

910  if (len > max_string_length) len = max_string_length;
911 
912  // trim after
913  char *back = s + len;
914  while (isspace(*--back));
915  *(back + 1) = '\0';
916 
917  // replace * by empty string
918  if (strlen(s) == 1 && *s == '*') {
919  *s = '\0';
920  }
921 
922  // debug
923  //printf("new string: %s has length %i \n",s,strlen(s));
924
