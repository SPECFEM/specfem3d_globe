46  subroutine prepare_movie_surface()
47 
48  use specfem_par
49  use specfem_par_movie
50  implicit none
51 
52  ! local parameters
53  integer :: ier
54 
55  ! checks if anything to do
56  if (.not. movie_surface) return
57 
58  ! user output
59  if (myrank == 0) then
60  write(imain,*) "preparing movie surface"
61  call flush_imain()
62  endif
63 
64  ! only output corners
65  ! note: for noise tomography, must NOT be coarse (have to be saved on all GLL points)
66  if (movie_coarse) then
67  ! checks setup
68  if (ngllx /= nglly) &
69  call exit_mpi(myrank,'MOVIE_COARSE together with MOVIE_SURFACE requires NGLLX=NGLLY')
70  ! number of points
71  nmovie_points = 2 * 2 * nspec2d_top(iregion_crust_mantle)
72  nit = ngllx - 1
73  else
74  ! number of points
75  nmovie_points = ngllx * nglly * nspec2d_top(iregion_crust_mantle)
76  nit = 1
77  endif
78 
79  ! checks exact number of points nmovie_points
80  call movie_surface_count_points()
81 
82  ! those arrays are not necessary for noise tomography, so only allocate them in MOVIE_SURFACE case
83  ! writes out movie point locations to file
84  call write_movie_surface_mesh()
85 
86  ! allocates movie surface arrays for wavefield values
87  allocate(store_val_ux(nmovie_points), &
88  store_val_uy(nmovie_points), &
89  store_val_uz(nmovie_points),stat=ier)
90  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating movie surface arrays')
91 
92  ! allocates arrays for gathering wavefield values
93  if (myrank == 0) then
94  ! only master needs full arrays
95  allocate(store_val_ux_all(nmovie_points,0:nproctot_val-1), &
96  store_val_uy_all(nmovie_points,0:nproctot_val-1), &
97  store_val_uz_all(nmovie_points,0:nproctot_val-1),stat=ier)
98  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating movie surface all arrays')
99  else
100  ! slave processes only need dummy arrays
101  allocate(store_val_ux_all(1,1), &
102  store_val_uy_all(1,1), &
103  store_val_uz_all(1,1),stat=ier)
104  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating movie surface all arrays')
105  endif
106 
107  ! user output
108  if (myrank == 0) then
109  write(imain,*) ' Movie surface:'
110  write(imain,*) ' Writing to moviedata*** files in output directory'
111  if (movie_volume_type == 5) then
112  write(imain,*) ' movie output: displacement'
113  else
114  write(imain,*) ' movie output: velocity'
115  endif
116  write(imain,*) ' time steps every: ',ntstep_between_frames
117  call flush_imain()
118  endif
119 
120  call synchronize_all()
121 
122  end subroutine prepare_movie_surface

128  subroutine prepare_movie_volume()
129 
130  use specfem_par
131  use specfem_par_crustmantle
132  use specfem_par_movie
133  implicit none
134 
135  ! local parameters
136  integer :: ier
137 
138  ! checks if anything to do
139  if (.not. movie_volume) return
140 
141  ! user output
142  if (myrank == 0) then
143  write(imain,*) "preparing movie volume"
144  call flush_imain()
145  endif
146 
147  ! checks
148  ! the following has to be true for the the array dimensions of eps to match with those of rstore etc..
149  ! note that epsilondev and eps_trace_over_3 don't have the same dimensions.. could cause trouble
150  if (nspec_crust_mantle_str_or_att /= nspec_crust_mantle) &
151  stop 'NSPEC_CRUST_MANTLE_STRAINS_ATT /= NSPEC_CRUST_MANTLE'
152  if (nspec_crust_mantle_strain_only /= nspec_crust_mantle) &
153  stop 'NSPEC_CRUST_MANTLE_STRAIN_ONLY /= NSPEC_CRUST_MANTLE'
154  ! checks movie type
155  if (movie_volume_type < 1 .or. movie_volume_type > 9) &
156  stop 'MOVIE_VOLUME_TYPE has to be in range from 1 to 9'
157 
158  ! counts total number of points for movie file output
159  call movie_volume_count_points()
160 
161  allocate(nu_3dmovie(3,3,npoints_3dmovie),stat=ier)
162  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating nu for 3D movie')
163 
164  allocate(mask_ibool(nglob_crust_mantle_3dmovie), &
165  num_ibool_3dmovie(nglob_crust_mantle_3dmovie), &
166  mask_3dmovie(ngllx,nglly,ngllz,nspec_crust_mantle_3dmovie), &
167  muvstore_crust_mantle_3dmovie(ngllx,nglly,ngllz,nspec_crust_mantle_3dmovie),stat=ier)
168  if (ier /= 0 ) stop 'Error allocating arrays muvstore_crust_mantle_3dmovie,..'
169 
170  call write_movie_volume_mesh(nu_3dmovie,num_ibool_3dmovie,mask_3dmovie,mask_ibool, &
171  muvstore_crust_mantle_3dmovie,npoints_3dmovie)
172 
173  if (myrank == 0) then
174  write(imain,*) ' Movie volume:'
175  write(imain,*) ' Writing to movie3D*** files on local disk databases directory'
176  select case (movie_volume_type)
177  case (1)
178  write(imain,*) ' movie output: strains'
179  case (2)
180  write(imain,*) ' movie output: time integral of strains'
181  case (3)
182  write(imain,*) ' movie output: potency or integral of strain'
183  case (4)
184  write(imain,*) ' movie output: divergence and curl'
185  case (5)
186  write(imain,*) ' movie output: displacement'
187  case (6)
188  write(imain,*) ' movie output: velocity'
189  case (7)
190  write(imain,*) ' movie output: norm of displacement'
191  case (8)
192  write(imain,*) ' movie output: norm of velocity'
193  case (9)
194  write(imain,*) ' movie output: norm of acceleration'
195  case default
196  call exit_mpi(myrank, 'MOVIE_VOLUME_TYPE has to be in range from 1 to 9')
197  end select
198  write(imain,*)
199  write(imain,*) ' depth(T,B):',movie_top,movie_bottom
200  write(imain,*) ' lon(W,E) :',movie_west,movie_east
201  write(imain,*) ' lat(S,N) :',movie_south,movie_north
202  write(imain,*) ' Starting at time step:',movie_start, 'ending at:',movie_stop,'every: ',ntstep_between_frames
203  call flush_imain()
204  endif
205 
206  call synchronize_all()
207 
208  end subroutine prepare_movie_volume
