162  subroutine mantle_s40rts(radius,theta,phi,dvs,dvp,drho)
163 
164  use constants
165  use model_s40rts_par
166 
167  implicit none
168 
169  double precision :: radius,theta,phi,dvs,dvp,drho
170 
171  ! local parameters
172  ! factor to convert perturbations in shear speed to perturbations in density
173  double precision, parameter :: SCALE_RHO = 0.40d0
174 
175  double precision, parameter :: R_EARTH_ = r_earth ! 6371000.d0
176  double precision, parameter :: RMOHO_ = r_earth - 24400.0 ! 6346600.d0
177  double precision, parameter :: RCMB_ = 3480000.d0
178  double precision, parameter :: ZERO_ = 0.d0
179 
180  integer :: l,m,k
181  double precision :: r_moho,r_cmb,xr
182  double precision :: dvs_alm,dvs_blm
183  double precision :: dvp_alm,dvp_blm
184  double precision :: s40rts_rsple,radial_basis(0:nk_20)
185  double precision :: sint,cost,x(2*ns_40+1),dx(2*ns_40+1)
186 
187  dvs = zero_
188  dvp = zero_
189  drho = zero_
190 
191  r_moho = rmoho_ / r_earth_
192  r_cmb = rcmb_ / r_earth_
193  if (radius >= r_moho .or. radius <= r_cmb) return
194 
195  xr=-1.0d0+2.0d0*(radius-r_cmb)/(r_moho-r_cmb)
196  if (xr > 1.0) print *,'xr > 1.0'
197  if (xr < -1.0) print *,'xr < -1.0'
198  do k = 0,nk_20
199  radial_basis(k)=s40rts_rsple(1,nk_20+1,s40rts_v_spknt(1),s40rts_v_qq0(1,nk_20+1-k),s40rts_v_qq(1,1,nk_20+1-k),xr)
200  enddo
201 
202  do l = 0,ns_40
203  sint=dsin(theta)
204  cost=dcos(theta)
205  call lgndr(l,cost,sint,x,dx)
206 
207  dvs_alm=0.0d0
208  dvp_alm=0.0d0
209  do k = 0,nk_20
210  dvs_alm=dvs_alm+radial_basis(k)*s40rts_v_dvs_a(k,l,0)
211  dvp_alm=dvp_alm+radial_basis(k)*s40rts_v_dvp_a(k,l,0)
212  enddo
213  dvs=dvs+dvs_alm*x(1)
214  dvp=dvp+dvp_alm*x(1)
215 
216  do m = 1,l
217  dvs_alm=0.0d0
218  dvp_alm=0.0d0
219  dvs_blm=0.0d0
220  dvp_blm=0.0d0
221  do k = 0,nk_20
222  dvs_alm=dvs_alm+radial_basis(k)*s40rts_v_dvs_a(k,l,m)
223  dvp_alm=dvp_alm+radial_basis(k)*s40rts_v_dvp_a(k,l,m)
224  dvs_blm=dvs_blm+radial_basis(k)*s40rts_v_dvs_b(k,l,m)
225  dvp_blm=dvp_blm+radial_basis(k)*s40rts_v_dvp_b(k,l,m)
226  enddo
227  dvs=dvs+(dvs_alm*dcos(dble(m)*phi)+dvs_blm*dsin(dble(m)*phi))*x(m+1)
228  dvp=dvp+(dvp_alm*dcos(dble(m)*phi)+dvp_blm*dsin(dble(m)*phi))*x(m+1)
229  enddo
230 
231  enddo
232 
233  drho = scale_rho*dvs
234 
235  end subroutine mantle_s40rts

292  double precision function s40rts_rsple(I1,I2,X,Y,Q,S)
293 
294  implicit none
295 
296 ! rsple returns the value of the function y(x) evaluated at point S
297 ! using the cubic spline coefficients computed by rspln and saved in Q.
298 ! If S is outside the interval (x(i1),x(i2)) rsple extrapolates
299 ! using the first or last interpolation polynomial. The arrays must
300 ! be dimensioned at least - x(i2), y(i2), and q(3,i2).
301 
302  integer i1,i2
303  double precision X(*),Y(*),Q(3,*),s
304 
305  integer i,ii
306  double precision h
307 
308  i = 1
309  ii=i2-1
310 
311 ! GUARANTEE I WITHIN BOUNDS.
312  i=max0(i,i1)
313  i=min0(i,ii)
314 
315 ! SEE IF X IS INCREASING OR DECREASING.
316  if (x(i2)-x(i1) < 0) goto 1
317  if (x(i2)-x(i1) >= 0) goto 2
318 
319 ! X IS DECREASING. CHANGE I AS NECESSARY.
320  1 if (s-x(i) <= 0) goto 3
321  if (s-x(i) > 0) goto 4
322 
323  4 i=i-1
324 
325  if (i-i1 < 0) goto 11
326  if (i-i1 == 0) goto 6
327  if (i-i1 > 0) goto 1
328 
329  3 if (s-x(i+1) < 0) goto 5
330  if (s-x(i+1) >= 0) goto 6
331 
332  5 i=i+1
333 
334  if (i-ii < 0) goto 3
335  if (i-ii == 0) goto 6
336  if (i-ii > 0) goto 7
337 
338 ! X IS INCREASING. CHANGE I AS NECESSARY.
339  2 if (s-x(i+1) <= 0) goto 8
340  if (s-x(i+1) > 0) goto 9
341 
342  9 i=i+1
343 
344  if (i-ii < 0) goto 2
345  if (i-ii == 0) goto 6
346  if (i-ii > 0) goto 7
347 
348  8 if (s-x(i) < 0) goto 10
349  if (s-x(i) >= 0) goto 6
350 
351  10 i=i-1
352  if (i-i1 < 0) goto 11
353  if (i-i1 == 0) goto 6
354  if (i-i1 > 0) goto 8
355 
356  7 i=ii
357  goto 6
358  11 i=i1
359 
360 ! CALCULATE RSPLE USING SPLINE COEFFICIENTS IN Y AND Q.
361  6 h=s-x(i)
362  s40rts_rsple=y(i)+h*(q(1,i)+h*(q(2,i)+h*q(3,i)))
363 
364  end function s40rts_rsple

368  subroutine s40rts_rspln(I1,I2,X,Y,Q,F)
369 
370  implicit none
371 
372 ! The subroutine rspln computes cubic spline interpolation coefficients
373 ! for y(x) between grid points i1 and i2 saving them in q.The
374 ! interpolation is continuous with continuous first and second
375 ! derivatives. It agrees exactly with y at grid points and with the
376 ! three point first derivatives at both end points (i1 and i2).
377 ! X must be monotonic but if two successive values of x are equal
378 ! a discontinuity is assumed and separate interpolation is done on
379 ! each strictly monotonic segment. The arrays must be dimensioned at
380 ! least - x(i2), y(i2), q(3,i2), and f(3,i2).
381 ! F is working storage for rspln.
382 
383  integer i1,i2
384  double precision X(*),Y(*),Q(3,*),F(3,*)
385 
386  integer i,j,k,j1,j2
387  double precision y0,a0,b0,b1,h,h2,ha,h2a,h3a,h2b
388  double precision YY(3),small
389  equivalence(yy(1),y0)
390  data small/1.0d-08/,yy/0.0d0,0.0d0,0.0d0/
391 
392  j1=i1+1
393  y0=0.0d0
394 
395 ! BAIL OUT IF THERE ARE LESS THAN TWO POINTS TOTAL
396  if (i2-i1 < 0) return
397  if (i2-i1 == 0) goto 17
398  if (i2-i1 > 0) goto 8
399 
400  8 a0=x(j1-1)
401 ! SEARCH FOR DISCONTINUITIES.
402  DO 3 i=j1,i2
403  b0=a0
404  a0=x(i)
405  if (dabs((a0-b0)/dmax1(a0,b0)) < small) goto 4
406  3 continue
407  17 j1=j1-1
408  j2=i2-2
409  goto 5
410  4 j1=j1-1
411  j2=i-3
412 ! SEE IF THERE ARE ENOUGH POINTS TO INTERPOLATE (AT LEAST THREE).
413  5 if (j2+1-j1 < 0) goto 9
414  if (j2+1-j1 == 0) goto 10
415  if (j2+1-j1 > 0) goto 11
416 
417 ! ONLY TWO POINTS. USE LINEAR INTERPOLATION.
418  10 j2=j2+2
419  y0=(y(j2)-y(j1))/(x(j2)-x(j1))
420  DO j=1,3
421  q(j,j1)=yy(j)
422  q(j,j2)=yy(j)
423  enddo
424  goto 12
425 
426 ! MORE THAN TWO POINTS. DO SPLINE INTERPOLATION.
427  11 a0 = 0.
428  h=x(j1+1)-x(j1)
429  h2=x(j1+2)-x(j1)
430  y0=h*h2*(h2-h)
431  h=h*h
432  h2=h2*h2
433 ! CALCULATE DERIVATIVE AT NEAR END.
434  b0=(y(j1)*(h-h2)+y(j1+1)*h2-y(j1+2)*h)/y0
435  b1=b0
436 
437 ! EXPLICITLY REDUCE BANDED MATRIX TO AN UPPER BANDED MATRIX.
438  DO i=j1,j2
439  h=x(i+1)-x(i)
440  y0=y(i+1)-y(i)
441  h2=h*h
442  ha=h-a0
443  h2a=h-2.0d0*a0
444  h3a=2.0d0*h-3.0d0*a0
445  h2b=h2*b0
446  q(1,i)=h2/ha
447  q(2,i)=-ha/(h2a*h2)
448  q(3,i)=-h*h2a/h3a
449  f(1,i)=(y0-h*b0)/(h*ha)
450  f(2,i)=(h2b-y0*(2.0d0*h-a0))/(h*h2*h2a)
451  f(3,i)=-(h2b-3.0d0*y0*ha)/(h*h3a)
452  a0=q(3,i)
453  b0=f(3,i)
454  enddo
455 
456 ! TAKE CARE OF LAST TWO ROWS.
457  i=j2+1
458  h=x(i+1)-x(i)
459  y0=y(i+1)-y(i)
460  h2=h*h
461  ha=h-a0
462  h2a=h*ha
463  h2b=h2*b0-y0*(2.0d0*h-a0)
464  q(1,i)=h2/ha
465  f(1,i)=(y0-h*b0)/h2a
466  ha=x(j2)-x(i+1)
467  y0=-h*ha*(ha+h)
468  ha=ha*ha
469 
470 ! CALCULATE DERIVATIVE AT FAR END.
471  y0=(y(i+1)*(h2-ha)+y(i)*ha-y(j2)*h2)/y0
472  q(3,i)=(y0*h2a+h2b)/(h*h2*(h-2.0d0*a0))
473  q(2,i)=f(1,i)-q(1,i)*q(3,i)
474 
475 ! SOLVE UPPER BANDED MATRIX BY REVERSE ITERATION.
476  DO j=j1,j2
477  k=i-1
478  q(1,i)=f(3,k)-q(3,k)*q(2,i)
479  q(3,k)=f(2,k)-q(2,k)*q(1,i)
480  q(2,k)=f(1,k)-q(1,k)*q(3,k)
481  i=k
482  enddo
483  q(1,i)=b1
484 ! FILL IN THE LAST POINT WITH A LINEAR EXTRAPOLATION.
485  9 j2=j2+2
486  DO j=1,3
487  q(j,j2)=yy(j)
488  enddo
489 
490 ! SEE IF THIS DISCONTINUITY IS THE LAST.
491  12 if (j2-i2 < 0) then
492  goto 6
493  else
494  return
495  endif
496 
497 ! NO. GO BACK FOR MORE.
498  6 j1=j2+2
499  if (j1-i2 <= 0) goto 8
500  if (j1-i2 > 0) goto 7
501 
502 ! THERE IS ONLY ONE POINT LEFT AFTER THE LATEST DISCONTINUITY.
503  7 DO j=1,3
504  q(j,i2)=yy(j)
505  enddo
506 
507  end subroutine s40rts_rspln
