95  subroutine gravity_observation_surface()
96 
97 ! create observation grid (surface) on which we compute the gravity integrals.
98 ! each processor has a copy of the whole grid and sums its contribution there,
99 ! and at the end of the process an MPI reduction is performed to compute the global sum
100 
101  use constants
102 
103  use meshfem3d_par, only: x_observation,y_observation,z_observation, &
104  lon_observation,lat_observation,ellipticity,topography,output_files
105 
106  use meshfem3d_models_par, only: nspl,rspl,espl,espl2,ibathy_topo
107 
108  implicit none
109 
110 ! local variables
111  integer :: ix,iy,ichunk,ier
112 
113  double precision :: gamma,x,y,rgt
114  double precision :: x_top,y_top,z_top
115  double precision :: ratio_xi, ratio_eta
116 
117  double precision :: r,lat,lon,elevation
118 
119 ! the observation surface is always above the whole Earth, thus each mesh chunk has a size of PI / 2 in each direction
120  double precision, parameter :: ANGULAR_WIDTH_XI_RAD = pi_over_two, angular_width_eta_rad = pi_over_two
121 
122 ! reuse an existing observation surface created in another run and stored to disk,
123 ! so that we are sure that they are exactly the same (for instance when comparing results for a reference ellipsoidal Earth
124 ! and results for a 3D Earth with topography)
125  if (reuse_existing_observation_surf) then
126 
127  if (myrank == 0) then
128  open(unit=iin,file=trim(output_files)//'/saved_observation_grid_real_x_y_z_used_by_the_code.txt', &
129  status='old',action='read',iostat=ier)
130  if (ier /= 0 ) call exit_mpi(myrank,'Error opening file for REUSE_EXISTING_OBSERVATION_SURF')
131 
132 ! loop on all the chunks and then on all the observation nodes in each chunk
133  do ichunk = 1,nchunks_max
134  do iy = 1,ny_observation
135  do ix = 1,nx_observation
136  ! read the saved values
137  read(iin,*) x_observation(ix,iy,ichunk),y_observation(ix,iy,ichunk),z_observation(ix,iy,ichunk)
138  enddo
139  enddo
140  enddo
141  close(unit=iin)
142 
143  endif
144 
145 ! implicitly converts the 3D to 1D arrays
146  call bcast_all_dp(x_observation, nx_observation * ny_observation * nchunks_max)
147  call bcast_all_dp(y_observation, nx_observation * ny_observation * nchunks_max)
148  call bcast_all_dp(z_observation, nx_observation * ny_observation * nchunks_max)
149 
150 ! loop on all the chunks and then on all the observation nodes in each chunk
151  do ichunk = 1,nchunks_max
152  do iy = 1,ny_observation
153  do ix = 1,nx_observation
154 
155  x_top = x_observation(ix,iy,ichunk)
156  y_top = y_observation(ix,iy,ichunk)
157  z_top = z_observation(ix,iy,ichunk)
158 
159  ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
160  call xyz_2_rlatlon_dble(x_top,y_top,z_top,r,lat,lon)
161 
162  ! store the values obtained for future display with GMT
163  if (lon > 180.0d0 ) lon = lon - 360.0d0
164  lon_observation(ix,iy,ichunk) = lon
165  lat_observation(ix,iy,ichunk) = lat
166 
167  enddo
168  enddo
169  enddo
170 
171  else ! of if (REUSE_EXISTING_OBSERVATION_SURF)
172 
173  ! for future GMT display
174  if (myrank == 0) then
175  open(unit=iout,file=trim(output_files)//'/observation_grid_long_lat_topo_for_GMT.txt',status='unknown',action='write')
176  open(unit=9965,file=trim(output_files)//'/saved_observation_grid_real_x_y_z_used_by_the_code.txt',status='unknown', &
177  action='write')
178  endif
179 
180 ! loop on all the chunks and then on all the observation nodes in each chunk
181  do ichunk = 1,nchunks_max
182  do iy = 1,ny_observation
183  do ix = 1,nx_observation
184 
185  ratio_xi = dble(ix - 1) / dble(nx_observation - 1)
186  x = 2.d0*ratio_xi-1
187 
188  ratio_eta = dble(iy - 1) / dble(ny_observation - 1)
189  y = 2.d0*ratio_eta-1
190 
191  x = tan((angular_width_xi_rad/2.d0) * x)
192  y = tan((angular_width_eta_rad/2.d0) * y)
193 
194  gamma = one / sqrt(one + x*x + y*y)
195 
196 ! first compute the position of the points exactly at the free surface of the Earth (without the oceans)
197 ! keeping in mind that the code non-dimensionalizes the radius of the spherical Earth to one
198  rgt = r_unit_sphere*gamma
199 
200  ! define the mesh points on the top and the bottom in the six regions of the cubed sphere
201  select case (ichunk)
202 
203  case (chunk_ab)
204 
205  x_top = -y*rgt
206  y_top = x*rgt
207  z_top = rgt
208 
209  case (chunk_ab_antipode)
210 
211  x_top = -y*rgt
212  y_top = -x*rgt
213  z_top = -rgt
214 
215  case (chunk_ac)
216 
217  x_top = -y*rgt
218  y_top = -rgt
219  z_top = x*rgt
220 
221  case (chunk_ac_antipode)
222 
223  x_top = -y*rgt
224  y_top = rgt
225  z_top = -x*rgt
226 
227  case (chunk_bc)
228 
229  x_top = -rgt
230  y_top = y*rgt
231  z_top = x*rgt
232 
233  case (chunk_bc_antipode)
234 
235  x_top = rgt
236  y_top = -y*rgt
237  z_top = x*rgt
238 
239  case default
240  stop 'incorrect chunk number in compute_coord_main_mesh'
241 
242  end select
243 
244  ! add ellipticity
245  if (ellipticity) call get_ellipticity_single_point(x_top,y_top,z_top,nspl,rspl,espl,espl2)
246 
247  ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
248  call xyz_2_rlatlon_dble(x_top,y_top,z_top,r,lat,lon)
249 
250  ! compute elevation at current point
251  if (topography) then
252  call get_topo_bathy(lat,lon,elevation,ibathy_topo)
253  else
254  elevation = zero
255  endif
256 
257  ! store the values obtained for future display with GMT
258  if (myrank == 0) then
259  if (lon > 180.0d0 ) lon = lon - 360.0d0
260  write(iout,*) lon,lat,elevation
261  lon_observation(ix,iy,ichunk) = lon
262  lat_observation(ix,iy,ichunk) = lat
263  endif
264 
265  ! add topography
266  if (topography) then
267  ! non-dimensionalize the elevation, which is in meters
268  elevation = elevation / r_earth
269 
270  x_top = x_top*(one + elevation)
271  y_top = y_top*(one + elevation)
272  z_top = z_top*(one + elevation)
273  endif
274 
275  ! compute the position of the point
276  x_observation(ix,iy,ichunk) = x_top * observation_elevation_ratio
277  y_observation(ix,iy,ichunk) = y_top * observation_elevation_ratio
278  z_observation(ix,iy,ichunk) = z_top * observation_elevation_ratio
279 
280  ! store the values obtained so that they can be reused from other runs
281  if (myrank == 0) write(9965,*) x_observation(ix,iy,ichunk),y_observation(ix,iy,ichunk),z_observation(ix,iy,ichunk)
282 
283  enddo
284  enddo
285  enddo
286 
287  ! for future GMT display
288  if (myrank == 0) then
289  close(unit=iout)
290  close(unit=9965)
291  endif
292 
293  endif ! of if (REUSE_EXISTING_OBSERVATION_SURF)
294 
295  end subroutine gravity_observation_surface

301  ! compute gravity integrals of that part of the slice, and then total integrals for the whole Earth
302 
303  subroutine gravity_compute_integrals(iregion_code,nspec,wxgll,wygll,wzgll,xstore,ystore,zstore, &
304  xixstore,xiystore,xizstore, &
305  etaxstore,etaystore,etazstore, &
306  gammaxstore,gammaystore,gammazstore, &
307  rhostore,idoubling)
308 
309  use constants
310 
311  use meshfem3d_par, only: output_files,nchunks, &
312  x_observation,y_observation,z_observation,g_x,g_y,g_z,g_xx,g_yy,g_zz,g_xy,g_xz,g_yz
313 
314  implicit none
315 
316  integer :: iregion_code,nspec
317  double precision :: wxgll(ngllx),wygll(nglly),wzgll(ngllz)
318 
319  integer,dimension(nspec) :: idoubling
320 
321  double precision,dimension(NGLLX,NGLLY,NGLLZ,nspec) :: xstore,ystore,zstore
322 
323  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLY,NGLLZ,nspec) :: &
324  xixstore,xiystore,xizstore,etaxstore,etaystore,etazstore,gammaxstore,gammaystore,gammazstore
325 
326  real(kind=CUSTOM_REAL),dimension(NGLLX,NGLLY,NGLLZ,nspec) :: rhostore
327 
328  ! local parameters
329  double precision :: weight
330  real(kind=CUSTOM_REAL) :: xixl,xiyl,xizl,etaxl,etayl,etazl,gammaxl,gammayl,gammazl
331  double precision :: jacobianl
332  integer :: i,j,k,ispec
333  double precision :: xval,yval,zval
334  double precision :: xval_squared,yval_squared,zval_squared
335  double precision :: x_meshpoint,y_meshpoint,z_meshpoint
336  double precision :: distance_squared,distance_cubed, &
337  three_over_distance_squared,one_over_distance_cubed,three_over_distance_fifth_power
338  double precision :: common_multiplying_factor,common_mult_times_one_over,common_mult_times_three_over
339 
340  ! name of the timestamp files
341  character(len=MAX_STRING_LEN) :: outputname
342 
343 #ifdef FORCE_VECTORIZATION
344  integer :: ix_iy_ichunk
345 #else
346  integer :: ix,iy,ichunk
347 #endif
348 
349  ! if we do not want to compute the gravity integrals, only the center of mass (computed before)
350  if (only_compute_center_of_mass) return
351 
352  ! safety check, needs global simulation
353  if (nchunks /= nchunks_max) stop 'Error calling compute_gravity_integrals(), only supported for global simulation'
354 
355  ! calculates volume of all elements in mesh
356  do ispec = 1,nspec
357 
358  ! print information about number of elements done so far
359  if (myrank == 0 .and. (mod(ispec,nspec_display_interval) == 0 .or. ispec == 1 .or. ispec == nspec)) then
360  write(imain,*) 'for gravity integrals ',ispec,' elements computed out of ',nspec
361  ! write time stamp file to give information about progression of simulation
362  write(outputname,"('/timestamp_reg',i1.1,'_ispec',i7.7,'_out_of_',i7.7)") iregion_code,ispec,nspec
363  ! timestamp file output
364  open(unit=iout,file=trim(output_files)//outputname,status='unknown',action='write')
365  write(iout,*) ispec,' elements done out of ',nspec,' in region ',iregion_code
366  close(unit=iout)
367  endif
368 
369  ! suppress fictitious elements in central cube
370  if (idoubling(ispec) == iflag_in_fictitious_cube) cycle
371 
372  ! see if we compute the contribution of the crust only
373  if (compute_crust_contrib_only .and. idoubling(ispec) /= iflag_crust) cycle
374 
375  do k = 1,ngllz
376  do j = 1,nglly
377  do i = 1,ngllx
378 
379  weight = wxgll(i)*wygll(j)*wzgll(k)
380 
381  ! compute the Jacobian
382  xixl = xixstore(i,j,k,ispec)
383  xiyl = xiystore(i,j,k,ispec)
384  xizl = xizstore(i,j,k,ispec)
385  etaxl = etaxstore(i,j,k,ispec)
386  etayl = etaystore(i,j,k,ispec)
387  etazl = etazstore(i,j,k,ispec)
388  gammaxl = gammaxstore(i,j,k,ispec)
389  gammayl = gammaystore(i,j,k,ispec)
390  gammazl = gammazstore(i,j,k,ispec)
391 
392  ! do this in double precision for accuracy
393  jacobianl = 1.d0 / dble(xixl*(etayl*gammazl-etazl*gammayl) &
394  - xiyl*(etaxl*gammazl-etazl*gammaxl) &
395  + xizl*(etaxl*gammayl-etayl*gammaxl))
396 
397  if (check_for_negative_jacobians .and. jacobianl <= zero) stop 'Error: negative Jacobian found in integral calculation'
398 
399  x_meshpoint = xstore(i,j,k,ispec)
400  y_meshpoint = ystore(i,j,k,ispec)
401  z_meshpoint = zstore(i,j,k,ispec)
402 
403 !! DK DK for gravity integral calculations we may want to shift the reference frame to a pre-computed center of mass
404  if (shift_to_this_center_of_mass) then
405  x_meshpoint = x_meshpoint - x_shift
406  y_meshpoint = y_meshpoint - y_shift
407  z_meshpoint = z_meshpoint - z_shift
408  endif
409 
410  common_multiplying_factor = jacobianl * weight * rhostore(i,j,k,ispec)
411 
412 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
413 !!!!!!!!!!! beginning of loop on all the data to create
414 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
415 
416 ! loop on all the chunks and then on all the observation nodes in each chunk
417 
418 #ifdef FORCE_VECTORIZATION
419 ! this works only if the arrays are contiguous in memory (which is always the case for static arrays, as used in the code)
420 ! but the code produced is extremely fast because we get a single and fully-vectorized loop on the whole array
421  do ix_iy_ichunk = 1,nx_observation * ny_observation * nchunks_max
422 
423  xval = x_meshpoint - x_observation(ix_iy_ichunk,1,1)
424  yval = y_meshpoint - y_observation(ix_iy_ichunk,1,1)
425  zval = z_meshpoint - z_observation(ix_iy_ichunk,1,1)
426 
427  xval_squared = xval**2
428  yval_squared = yval**2
429  zval_squared = zval**2
430 
431  distance_squared = xval_squared + yval_squared + zval_squared
432  distance_cubed = distance_squared * sqrt(distance_squared)
433 
434  three_over_distance_squared = 3.d0 / distance_squared
435  one_over_distance_cubed = 1.d0 / distance_cubed
436  three_over_distance_fifth_power = three_over_distance_squared * one_over_distance_cubed
437 
438  common_mult_times_one_over = common_multiplying_factor * one_over_distance_cubed
439  common_mult_times_three_over = common_multiplying_factor * three_over_distance_fifth_power
440 
441  g_x(ix_iy_ichunk,1,1) = g_x(ix_iy_ichunk,1,1) + common_mult_times_one_over * xval
442  g_y(ix_iy_ichunk,1,1) = g_y(ix_iy_ichunk,1,1) + common_mult_times_one_over * yval
443  g_z(ix_iy_ichunk,1,1) = g_z(ix_iy_ichunk,1,1) + common_mult_times_one_over * zval
444 
445  g_xx(ix_iy_ichunk,1,1) = g_xx(ix_iy_ichunk,1,1) &
446  + common_mult_times_one_over * (xval_squared * three_over_distance_squared - 1.d0)
447  g_yy(ix_iy_ichunk,1,1) = g_yy(ix_iy_ichunk,1,1) &
448  + common_mult_times_one_over * (yval_squared * three_over_distance_squared - 1.d0)
449  g_zz(ix_iy_ichunk,1,1) = g_zz(ix_iy_ichunk,1,1) &
450  + common_mult_times_one_over * (zval_squared * three_over_distance_squared - 1.d0)
451 
452  g_xy(ix_iy_ichunk,1,1) = g_xy(ix_iy_ichunk,1,1) + common_mult_times_three_over * xval*yval
453  g_xz(ix_iy_ichunk,1,1) = g_xz(ix_iy_ichunk,1,1) + common_mult_times_three_over * xval*zval
454  g_yz(ix_iy_ichunk,1,1) = g_yz(ix_iy_ichunk,1,1) + common_mult_times_three_over * yval*zval
455 
456  enddo
457 #else
458  ! non-vectorized
459  do ichunk = 1,nchunks_max
460  do iy = 1,ny_observation
461  do ix = 1,nx_observation
462 
463  xval = x_meshpoint - x_observation(ix,iy,ichunk)
464  yval = y_meshpoint - y_observation(ix,iy,ichunk)
465  zval = z_meshpoint - z_observation(ix,iy,ichunk)
466 
467  xval_squared = xval**2
468  yval_squared = yval**2
469  zval_squared = zval**2
470 
471  distance_squared = xval_squared + yval_squared + zval_squared
472  distance_cubed = distance_squared * sqrt(distance_squared)
473 
474  three_over_distance_squared = 3.d0 / distance_squared
475  one_over_distance_cubed = 1.d0 / distance_cubed
476  three_over_distance_fifth_power = three_over_distance_squared * one_over_distance_cubed
477 
478  common_mult_times_one_over = common_multiplying_factor * one_over_distance_cubed
479  common_mult_times_three_over = common_multiplying_factor * three_over_distance_fifth_power
480 
481  g_x(ix,iy,ichunk) = g_x(ix,iy,ichunk) + common_mult_times_one_over * xval
482  g_y(ix,iy,ichunk) = g_y(ix,iy,ichunk) + common_mult_times_one_over * yval
483  g_z(ix,iy,ichunk) = g_z(ix,iy,ichunk) + common_mult_times_one_over * zval
484 
485  g_xx(ix,iy,ichunk) = g_xx(ix,iy,ichunk) &
486  + common_mult_times_one_over * (xval_squared * three_over_distance_squared - 1.d0)
487  g_yy(ix,iy,ichunk) = g_yy(ix,iy,ichunk) &
488  + common_mult_times_one_over * (yval_squared * three_over_distance_squared - 1.d0)
489  g_zz(ix,iy,ichunk) = g_zz(ix,iy,ichunk) &
490  + common_mult_times_one_over * (zval_squared * three_over_distance_squared - 1.d0)
491 
492  g_xy(ix,iy,ichunk) = g_xy(ix,iy,ichunk) + common_mult_times_three_over * xval*yval
493  g_xz(ix,iy,ichunk) = g_xz(ix,iy,ichunk) + common_mult_times_three_over * xval*zval
494  g_yz(ix,iy,ichunk) = g_yz(ix,iy,ichunk) + common_mult_times_three_over * yval*zval
495 
496  enddo
497  enddo
498  enddo
499 #endif
500 
501 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
502 !!!!!!!!!!! end of loop on all the data to create
503 !!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
504 
505  enddo
506  enddo
507  enddo
508  enddo
509 
510  end subroutine gravity_compute_integrals

517  subroutine finalize_gravity_integrals()
518 
519  use meshfem3d_par
520  use meshfem3d_models_par
521 
522  implicit none
523 
524  ! for gravity integrals
525  ! take into account the fact that the density and the radius of the Earth have previously been non-dimensionalized
526  ! for the gravity vector force, a distance is involved in the dimensions
527  double precision, parameter :: nondimensionalizing_factor_gi = rhoav * r_earth
528  ! for the second-order gravity tensor, no distance is involved in the dimensions
529  double precision, parameter :: nondimensionalizing_factor_Gij = rhoav
530 
531  double precision, parameter :: scaling_factor_gi = grav * nondimensionalizing_factor_gi
532  double precision, parameter :: scaling_factor_Gij_Eotvos = grav * nondimensionalizing_factor_gij * si_units_to_eotvos
533 
534  double precision :: real_altitude_of_observ_point
535 
536  integer :: ixval,iyval,ichunkval
537 
538  ! multiply by the gravitational constant in S.I. units i.e. in m3 kg-1 s-2
539  ! and also take into account the fact that the density and the radius of the Earth have previously been non-dimensionalized
540  ! the final result is in m.s-2 i.e. in S.I. units
541  g_x(:,:,:) = g_x(:,:,:) * scaling_factor_gi
