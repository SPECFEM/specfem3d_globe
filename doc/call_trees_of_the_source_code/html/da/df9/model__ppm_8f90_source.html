301  call flush_imain()
302  endif
303 
304  ppm_num_latperlon = int( (ppm_maxlat - ppm_minlat) / ppm_dlat) + 1

314  subroutine model_ppm(radius,theta,phi,dvs,dvp,drho)
315 
316 ! returns dvs,dvp and drho for given radius,theta,phi location
317 
318  use constants
319  use model_ppm_par
320 
321  implicit none
322 
323  double precision :: radius,theta,phi,dvs,dvp,drho
324 
325  ! local parameters
326  integer:: i,j,k
327  double precision:: lat,lon,r_depth
328  double precision:: min_dvs,max_dvs
329 
330  double precision:: g_dvs,g_depth,g_lat,g_lon,x,g_weight,weight_sum,weight_prod
331 
332  double precision,parameter:: const_a = sigma_v/3.0
333  double precision,parameter:: const_b = sigma_h/3.0/(r_earth_km*degrees_to_radians)
334 
335  ! initialize
336  dvs = 0.0d0
337  dvp = 0.0d0
338  drho = 0.0d0
339 
340  ! depth of given radius (in km)
341  r_depth = r_earth_km*(1.0 - radius) ! radius is normalized between [0,1]
342  if (r_depth > ppm_maxdepth .or. r_depth < ppm_mindepth) return
343 
344  lat=(pi_over_two-theta)*radians_to_degrees
345  if (lat < ppm_minlat .or. lat > ppm_maxlat ) return
346 
347  lon=phi*radians_to_degrees
348  if (lon > 180.0d0) lon=lon-360.0d0
349  if (lon < ppm_minlon .or. lon > ppm_maxlon ) return
350 
351  ! search location value
352  if (.not. gauss_smoothing) then
353  call get_ppmmodel_value(lat,lon,r_depth,dvs)
354  return
355  endif
356 
357  ! loop over neighboring points
358  dvs = 0.0
359  weight_sum = 0.0
360  do i=-num_gausspoints,num_gausspoints
361  g_depth = r_depth + i*const_a
362  do j=-num_gausspoints,num_gausspoints
363  g_lon = lon + j*const_b
364  do k=-num_gausspoints,num_gausspoints
365  g_lat = lat + k*const_b
366 
367  call get_ppmmodel_value(g_lat,g_lon,g_depth,g_dvs)
368 
369  ! horizontal weighting
370  x = (g_lat-lat)*degrees_to_radians*r_earth_km
371  call get_gaussianweight(x,sigma_h,g_weight)
372  g_dvs = g_dvs*g_weight
373  weight_prod = g_weight
374 
375  x = (g_lon-lon)*degrees_to_radians*r_earth_km
376  call get_gaussianweight(x,sigma_h,g_weight)
377  g_dvs = g_dvs*g_weight
378  weight_prod = weight_prod * g_weight
379 
380  !vertical weighting
381  x = g_depth-r_depth
382  call get_gaussianweight(x,sigma_v,g_weight)
383  g_dvs = g_dvs*g_weight
384  weight_prod = weight_prod * g_weight
385 
386  ! averaging
387  weight_sum = weight_sum + weight_prod
388  dvs = dvs + g_dvs
389  enddo
390  enddo
391  enddo
392 
393  if (weight_sum > 1.e-15) dvs = dvs / weight_sum
394 
395  ! store min/max
396  max_dvs = ppm_max_dvs
397  min_dvs = ppm_min_dvs
398 
399  if (dvs > max_dvs ) max_dvs = dvs
400  if (dvs < min_dvs ) min_dvs = dvs
401 
402  ppm_max_dvs = max_dvs
403  ppm_min_dvs = min_dvs
404 
405  if (scale_model) then
406  ! scale density and shear velocity
407  drho = scale_rho*dvs
408  ! scale vp and shear velocity
409  dvp = scale_vp*dvs
410  endif
411 
412  end subroutine model_ppm
