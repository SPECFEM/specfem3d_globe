699  subroutine get_time(starttime, start_time_string, pde_start_time_string, cmt_start_time_string, end_time_string)
700 
701  use specfem_par, only: &
702  nstep,dt,t_shift_sac,hdur, &
703  yr_sac,ho_sac,mi_sac,sec_sac,jda_sac
704 
705  implicit none
706  character(len=*) :: start_time_string
707  character(len=*) :: end_time_string
708  character(len=*) :: cmt_start_time_string
709  character(len=*) :: pde_start_time_string
710  double precision, intent(inout) :: starttime
711  double precision :: trace_length_in_sec
712  integer :: year
713  double precision :: pdetime, cmttime, endtime
714  integer,dimension(8) :: values
715 
716  ! Calculates the start time since the epoch in seconds
717  ! Reference:
718  ! http://pubs.opengroup.org/onlinepubs/009695399/basedefs/xbd_chap04.html#tag_04_14
719  year = yr_sac-1900
720  pdetime = (year-70)*31536000.0d0+((year-69)/4)*86400.0d0 -((year-1)/100)*86400.0d0+&
721  ((year+299)/400)*86400.0d0+(jda_sac-1)*86400.0d0+ho_sac*(3600.0d0)+&
722  mi_sac*60.0d0+sec_sac
723  call convert_systime_to_string(pdetime, pde_start_time_string)
724 
725  ! cmt centroid time
726  cmttime = pdetime + t_shift_sac
727  call convert_systime_to_string(cmttime, cmt_start_time_string)
728 
729  ! trace start time
730  starttime = cmttime - 1.5*hdur(1)
731  call convert_systime_to_string(starttime, start_time_string)
732 
733  ! Calculates the number of seconds to add to the start_time
734  trace_length_in_sec = dt*nstep
735  endtime = starttime + trace_length_in_sec
736  call convert_systime_to_string(endtime, end_time_string)
737 
738  ! Calculates time in seconds since the epoch
739  call date_and_time(values=values)
740  !time8()
741 
742  !write(*,*) fmtdate(values,'The CPU time used by this program is now %c seconds')
743 
744  end subroutine get_time
