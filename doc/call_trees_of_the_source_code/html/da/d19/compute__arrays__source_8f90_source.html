68 ! compute Lagrange polynomials at the source location
69 ! the source does not necessarily correspond to a Gauss-Lobatto point
70  call lagrange_any(xi_source,ngllx,xigll,hxis,hpxis)
71  call lagrange_any(eta_source,nglly,yigll,hetas,hpetas)
72  call lagrange_any(gamma_source,ngllz,zigll,hgammas,hpgammas)
73 
74  dxis_dx = zero
75  dxis_dy = zero
76  dxis_dz = zero
77  detas_dx = zero
78  detas_dy = zero
79  detas_dz = zero
80  dgammas_dx = zero
81  dgammas_dy = zero
82  dgammas_dz = zero
83 
84  do m = 1,ngllz
85  do l = 1,nglly
86  do k = 1,ngllx
87 
88  xixd = dble(xix(k,l,m))
89  xiyd = dble(xiy(k,l,m))
90  xizd = dble(xiz(k,l,m))
91  etaxd = dble(etax(k,l,m))
92  etayd = dble(etay(k,l,m))
93  etazd = dble(etaz(k,l,m))
94  gammaxd = dble(gammax(k,l,m))
95  gammayd = dble(gammay(k,l,m))
96  gammazd = dble(gammaz(k,l,m))
97 
98  hlagrange = hxis(k) * hetas(l) * hgammas(m)
99 
100  dxis_dx = dxis_dx + hlagrange * xixd
101  dxis_dy = dxis_dy + hlagrange * xiyd
102  dxis_dz = dxis_dz + hlagrange * xizd
103 
104  detas_dx = detas_dx + hlagrange * etaxd
105  detas_dy = detas_dy + hlagrange * etayd
106  detas_dz = detas_dz + hlagrange * etazd
107 
108  dgammas_dx = dgammas_dx + hlagrange * gammaxd
109  dgammas_dy = dgammas_dy + hlagrange * gammayd
110  dgammas_dz = dgammas_dz + hlagrange * gammazd
111 
112  enddo
113  enddo
114  enddo
115 
116 ! calculate source array
117  sourcearrayd(:,:,:,:) = zero
118  do m = 1,ngllz
119  do l = 1,nglly
120  do k = 1,ngllx
121 
122  dsrc_dx = (hpxis(k)*dxis_dx)*hetas(l)*hgammas(m) + hxis(k)*(hpetas(l)*detas_dx)*hgammas(m) + &
123  hxis(k)*hetas(l)*(hpgammas(m)*dgammas_dx)
124  dsrc_dy = (hpxis(k)*dxis_dy)*hetas(l)*hgammas(m) + hxis(k)*(hpetas(l)*detas_dy)*hgammas(m) + &
125  hxis(k)*hetas(l)*(hpgammas(m)*dgammas_dy)
126  dsrc_dz = (hpxis(k)*dxis_dz)*hetas(l)*hgammas(m) + hxis(k)*(hpetas(l)*detas_dz)*hgammas(m) + &
127  hxis(k)*hetas(l)*(hpgammas(m)*dgammas_dz)
128 
129  sourcearrayd(1,k,l,m) = sourcearrayd(1,k,l,m) + (mxx*dsrc_dx + mxy*dsrc_dy + mxz*dsrc_dz)
130  sourcearrayd(2,k,l,m) = sourcearrayd(2,k,l,m) + (mxy*dsrc_dx + myy*dsrc_dy + myz*dsrc_dz)
131  sourcearrayd(3,k,l,m) = sourcearrayd(3,k,l,m) + (mxz*dsrc_dx + myz*dsrc_dy + mzz*dsrc_dz)
132 
133  enddo
134  enddo
135  enddo
136 
137  ! distinguish between single and double precision for reals
138  sourcearray(:,:,:,:) = real(sourcearrayd(:,:,:,:), kind=custom_real)
139 
140  end subroutine compute_arrays_source
