63  subroutine write_movie_surface_mesh()
64 
65 ! writes out movie point locations to file
66 
67  use specfem_par
68  use specfem_par_crustmantle
69  use specfem_par_movie
70 
71  implicit none
72 
73  ! local parameters
74  real(kind=CUSTOM_REAL), dimension(:), allocatable :: store_val_x,store_val_y,store_val_z
75  real(kind=CUSTOM_REAL), dimension(:,:), allocatable :: store_val_x_all,store_val_y_all,store_val_z_all
76  integer :: ipoin,ispec2D,ispec,i,j,k,ier,iglob,npoin
77  character(len=MAX_STRING_LEN) :: outputname
78 
79  ! allocates movie surface arrays
80  allocate(store_val_x(nmovie_points), &
81  store_val_y(nmovie_points), &
82  store_val_z(nmovie_points),stat=ier)
83  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating movie surface location arrays')
84 
85  ! allocates arrays for gathering movie point locations
86  if (myrank == 0) then
87  ! only master needs full arrays
88  allocate(store_val_x_all(nmovie_points,0:nproctot_val-1), &
89  store_val_y_all(nmovie_points,0:nproctot_val-1), &
90  store_val_z_all(nmovie_points,0:nproctot_val-1),stat=ier)
91  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating movie surface all arrays')
92  else
93  ! slave processes only need dummy arrays
94  allocate(store_val_x_all(1,1), &
95  store_val_y_all(1,1), &
96  store_val_z_all(1,1),stat=ier)
97  if (ier /= 0 ) call exit_mpi(myrank,'Error allocating movie surface all arrays')
98  endif
99 
100  ! gets coordinates of surface mesh
101  ipoin = 0
102  do ispec2d = 1, nspec_top ! NSPEC2D_TOP(IREGION_CRUST_MANTLE)
103  ispec = ibelm_top_crust_mantle(ispec2d)
104  ! in case of global, NCHUNKS_VAL == 6 simulations, be aware that for
105  ! the cubed sphere, the mapping changes for different chunks,
106  ! i.e. e.g. x(1,1) and x(5,5) flip left and right sides of the elements in geographical coordinates.
107  ! for future consideration, like in create_movie_GMT_global.f90 ...
108  k = ngllz
109  ! loop on all the points inside the element
110  do j = 1,nglly,nit
111  do i = 1,ngllx,nit
112  ipoin = ipoin + 1
113  ! stores values
114  iglob = ibool_crust_mantle(i,j,k,ispec)
115  store_val_x(ipoin) = rstore_crust_mantle(1,iglob) ! radius r (normalized)
116  store_val_y(ipoin) = rstore_crust_mantle(2,iglob) ! colatitude theta (in radian)
117  store_val_z(ipoin) = rstore_crust_mantle(3,iglob) ! longitude phi (in radian)
118  enddo
119  enddo
120  enddo
121  npoin = ipoin
122  if (npoin /= nmovie_points ) call exit_mpi(myrank,'Error number of movie points not equal to nmovie_points')
123 
124  ! gather info on master proc
125  call gather_all_cr(store_val_x,nmovie_points,store_val_x_all,nmovie_points,nproctot_val)
126  call gather_all_cr(store_val_y,nmovie_points,store_val_y_all,nmovie_points,nproctot_val)
127  call gather_all_cr(store_val_z,nmovie_points,store_val_z_all,nmovie_points,nproctot_val)
128 
129  ! save movie data locations to disk in home directory
130  if (myrank == 0) then
131 
132  ! outputs movie point locations to moviedata_xyz.bin file
133  outputname = "/moviedata_xyz.bin"
134  open(unit=iout,file=trim(output_files)//trim(outputname), &
135  status='unknown',form='unformatted',action='write',iostat=ier)
136  if (ier /= 0 ) call exit_mpi(myrank,'Error opening moviedata_xyz.bin file')
137 
138  ! point coordinates
139  ! (given as r theta phi for geocentric coordinate system)
140  write(iout) store_val_x_all
141  write(iout) store_val_y_all
142  write(iout) store_val_z_all
143  close(iout)
144  endif
145 
146  deallocate(store_val_x_all,store_val_y_all,store_val_z_all)
147  deallocate(store_val_x,store_val_y,store_val_z)
148 
149  end subroutine write_movie_surface_mesh

155  subroutine write_movie_surface()
156 
157  use specfem_par
158  use specfem_par_crustmantle
159  use specfem_par_movie
160 
161  implicit none
162 
163  ! local parameters
164  character(len=MAX_STRING_LEN) :: outputname
165  integer :: ipoin,ispec2D,ispec,i,j,k,ier,iglob
166 
167  ! by default: save velocity here to avoid static offset on displacement for movies
168 
169  ! gets coordinates of surface mesh and surface displacement
170  ipoin = 0
171  do ispec2d = 1, nspec_top ! NSPEC2D_TOP(IREGION_CRUST_MANTLE)
172  ispec = ibelm_top_crust_mantle(ispec2d)
173 
174  ! in case of global, NCHUNKS_VAL == 6 simulations, be aware that for
175  ! the cubed sphere, the mapping changes for different chunks,
176  ! i.e. e.g. x(1,1) and x(5,5) flip left and right sides of the elements in geographical coordinates.
177  ! for future consideration, like in create_movie_GMT_global.f90 ...
178  k = ngllz
179 
180  ! loop on all the points inside the element
181  do j = 1,nglly,nit
182  do i = 1,ngllx,nit
183  ipoin = ipoin + 1
184  iglob = ibool_crust_mantle(i,j,k,ispec)
185 
186  ! wavefield values
187  if (movie_volume_type == 5) then
188  ! stores displacement
189  store_val_ux(ipoin) = displ_crust_mantle(1,iglob)*scale_displ
190  store_val_uy(ipoin) = displ_crust_mantle(2,iglob)*scale_displ
191  store_val_uz(ipoin) = displ_crust_mantle(3,iglob)*scale_displ
192  else
193  ! stores velocity
194  store_val_ux(ipoin) = veloc_crust_mantle(1,iglob)*scale_veloc
195  store_val_uy(ipoin) = veloc_crust_mantle(2,iglob)*scale_veloc
196  store_val_uz(ipoin) = veloc_crust_mantle(3,iglob)*scale_veloc
197  endif
198 
199  enddo
200  enddo
201  enddo
202 
203  ! gather info on master proc
204  ! wavefield
205  call gather_all_cr(store_val_ux,nmovie_points,store_val_ux_all,nmovie_points,nproctot_val)
206  call gather_all_cr(store_val_uy,nmovie_points,store_val_uy_all,nmovie_points,nproctot_val)
207  call gather_all_cr(store_val_uz,nmovie_points,store_val_uz_all,nmovie_points,nproctot_val)
208 
209  ! save movie data to disk in home directory
210  if (myrank == 0) then
211  write(outputname,"('/moviedata',i6.6)") it
212  open(unit=iout,file=trim(output_files)//outputname, &
213  status='unknown',form='unformatted',action='write',iostat=ier)
214  if (ier /= 0 ) call exit_mpi(myrank,'Error opening moviedata file')
215 
216  ! -> find movie point locations stored in file moviedata_xyz.bin
217 
218  ! wavefield values
219  write(iout) store_val_ux_all
220  write(iout) store_val_uy_all
221  write(iout) store_val_uz_all
222 
223  close(iout)
224  endif
225 
226  end subroutine write_movie_surface
