56 ! if one wants to detect slow nodes compared to a reference time on normal nodes on a given cluster
57 ! and exclude them from the runs in order not to slow down all the others when NUMBER_OF_SIMULTANEOUS_RUNS > 1.

70  ! maximum of the norm of the displacement and of the potential in the fluid
71  real(kind=CUSTOM_REAL) Usolidnorm,Usolidnorm_all,Ufluidnorm,Ufluidnorm_all

138  b_displ_inner_core(2,:)**2 + &

142  else

144  call check_norm_elastic_from_device(b_usolidnorm,mesh_pointer,3)
145  call check_norm_acoustic_from_device(b_ufluidnorm,mesh_pointer,3)

161  ! on CPU
162  strain_norm = maxval(abs(eps_trace_over_3_crust_mantle))
163  strain2_norm= max( maxval(abs(epsilondev_xx_crust_mantle)), &

195  else

197  endif

199  ihours_remain = int_t_remain / 3600

201  iseconds_remain = int_t_remain - 3600*ihours_remain - 60*iminutes_remain
202 
203  ! compute estimated total simulation time
204  t_total = t_remain + tcpu
205  int_t_total = int(t_total)
206  ihours_total = int_t_total / 3600
207  iminutes_total = (int_t_total - 3600*ihours_total) / 60
208  iseconds_total = int_t_total - 3600*ihours_total - 60*iminutes_total
209 
210  ! current time (in seconds)
211  timeval = dble(it-1)*dt - t0
212 
213  ! user output
214  write(imain,*) 'Time step # ',it
215  write(imain,*) 'Time: ',sngl((timeval)/60.d0),' minutes'
216 
217  ! rescale maximum displacement to correct dimensions
218  usolidnorm_all = usolidnorm_all * sngl(scale_displ)
219  if (simulation_type == 1) then
220  write(imain,*) 'Max norm displacement vector U in solid in all slices for forward prop. (m) = ',usolidnorm_all
221  write(imain,*) 'Max non-dimensional potential Ufluid in fluid in all slices for forward prop. = ',ufluidnorm_all
222  else
223  write(imain,*) 'Max norm displacement vector U in solid in all slices for adjoint prop. (m)= ',usolidnorm_all
224  write(imain,*) 'Max non-dimensional potential Ufluid in fluid in all slices for adjoint prop. = ',ufluidnorm_all
225  endif
226 
227  if (simulation_type == 3) then
228  b_usolidnorm_all = b_usolidnorm_all * sngl(scale_displ)
229  if (.not. undo_attenuation) then
230  write(imain,*) 'Max norm displacement vector U in solid in all slices for back prop.(m) = ',b_usolidnorm_all
231  write(imain,*) 'Max non-dimensional potential Ufluid in fluid in all slices for back prop.= ',b_ufluidnorm_all
232  endif
233  endif
234 
235  if (compute_and_store_strain) then
236  write(imain,*) 'Max of strain, eps_trace_over_3_crust_mantle =',strain_norm_all
237  write(imain,*) 'Max of strain, epsilondev_crust_mantle =',strain2_norm_all
238  endif
239 
240  write(imain,*) 'Elapsed time in seconds = ',tcpu
241  write(imain,"(' Elapsed time in hh:mm:ss = ',i6,' h ',i2.2,' m ',i2.2,' s')") ihours,iminutes,iseconds
242  write(imain,*) 'Mean elapsed time per time step in seconds = ',tcpu/dble(it)
243 
244 ! do not check before MIN_TIME_STEPS_FOR_SLOW_NODES time steps
245 ! because the time step estimate (which is an average) may then be unreliable
246  if (check_for_slow_nodes .and. number_of_simultaneous_runs > 1 .and. it >= min_time_steps_for_slow_nodes .and. &

248  i_am_running_on_a_slow_node = .true.
249 
250  if (show_separate_run_information) then
251  write(imain,*) 'Time steps done for this run = ',it_run,' out of ',nstep_run
252  write(imain,*) 'Time steps done in total = ',it,' out of ',nstep
253  write(imain,*) 'Time steps remaining for this run = ',it_end - it
254  write(imain,*) 'Time steps remaining for all runs = ',nstep - it
255  else
256  write(imain,*) 'Time steps done = ',it,' out of ',nstep
257  write(imain,*) 'Time steps remaining = ',nstep - it
258  endif
259 
260  write(imain,*) 'Estimated remaining time in seconds = ',t_remain
261  write(imain,"(' Estimated remaining time in hh:mm:ss = ',i6,' h ',i2.2,' m ',i2.2,' s')") &
262  ihours_remain,iminutes_remain,iseconds_remain
263 
264  write(imain,*) 'Estimated total run time in seconds = ',t_total
265  write(imain,"(' Estimated total run time in hh:mm:ss = ',i6,' h ',i2.2,' m ',i2.2,' s')") &
266  ihours_total,iminutes_total,iseconds_total
267  write(imain,*) 'We have done ',sngl(100.d0*dble(it)/dble(nstep)),'% of that'

271

273  call date_and_time(datein,timein,zone,time_values)
274  ! time_values(1): year
275  ! time_values(2): month of the year
276  ! time_values(3): day of the month
277  ! time_values(5): hour of the day
278  ! time_values(6): minutes of the hour

297

304

308  if (add_time_estimate_elsewhere .and. hours_time_difference * 60 + minutes_time_difference /= 0) then
309 
310  ! add time difference with that remote location (can be negative)
311  timestamp_remote = timestamp + hours_time_difference * 60 + minutes_time_difference
312 
313  ! get date and time of that future timestamp in minutes
314  call invtime(timestamp_remote,year_remote,mon_remote,day_remote,hr_remote,minutes_remote)
315 
316  ! convert to Julian day to get day of the week
317  call calndr(day_remote,mon_remote,year_remote,julian_day_number)
318  day_of_week_remote = idaywk(julian_day_number)
319 
320  if (hours_time_difference * 60 + minutes_time_difference > 0) then
321  write(imain,*) 'Adding positive time difference of ',abs(hours_time_difference),' hours'
322  else
323  write(imain,*) 'Adding negative time difference of ',abs(hours_time_difference),' hours'
324  endif
325  write(imain,*) 'and ',abs(minutes_time_difference),' minutes to get estimate at a remote location'
326  write(imain, &
327  "(' The run will finish approximately on: ',a3,' ',a3,' ',i2.2,', ',i4.4,' ',i2.2,':',i2.2)") &
328  weekday_name(day_of_week_remote),month_name(mon_remote),day_remote,year_remote,hr_remote,minutes_remote
329  endif
330 
331  if (it_run < 100) then
332  write(imain,*) '************************************************************'
333  write(imain,*) '**** BEWARE: the above time estimates are not very reliable'
334  write(imain,*) '**** because fewer than 100 iterations have been performed'
335  write(imain,*) '************************************************************'
336  endif
337 
338  endif

343  call flush_imain()

346  call write_timestamp_file(usolidnorm_all,ufluidnorm_all,b_usolidnorm_all,b_ufluidnorm_all, &
347  tcpu,ihours,iminutes,iseconds, &
348  t_remain,ihours_remain,iminutes_remain,iseconds_remain, &
349  t_total,ihours_total,iminutes_total,iseconds_total, &
350  day_of_week,mon,day,year,hr,minutes, &
351  day_of_week_remote,mon_remote,day_remote,year_remote,hr_remote,minutes_remote, &
352  check_for_slow_nodes,number_of_simultaneous_runs,tolerance_factor_for_slow_nodes, &
353  reference_time_per_time_step_on_normal_nodes,i_am_running_on_a_slow_node,myrank,mygroup, &
354  min_time_steps_for_slow_nodes)
355 
356  ! check stability of the code, exit if unstable
357  ! negative values can occur with some compilers when the unstable value is greater
358  ! than the greatest possible floating-point number of the machine
359 ! this trick checks for NaN (Not a Number), which is not even equal to itself
360  if (usolidnorm_all > stability_threshold .or. usolidnorm_all < 0 .or. usolidnorm_all /= usolidnorm_all) &
361  call exit_mpi(myrank,'forward simulation became unstable and blew up in the solid')
362  if (ufluidnorm_all > stability_threshold .or. ufluidnorm_all < 0 .or. ufluidnorm_all /= ufluidnorm_all) &
363  call exit_mpi(myrank,'forward simulation became unstable and blew up in the fluid')
364 
365  if (simulation_type == 3 .and. .not. undo_attenuation) then
366 ! this trick checks for NaN (Not a Number), which is not even equal to itself
367  if (b_usolidnorm_all > stability_threshold .or. b_usolidnorm_all < 0 .or. b_usolidnorm_all /= b_usolidnorm_all) &
368  call exit_mpi(myrank,'backward simulation became unstable and blew up in the solid')
369  if (b_ufluidnorm_all > stability_threshold .or. b_ufluidnorm_all < 0 .or. b_ufluidnorm_all /= b_ufluidnorm_all) &
370  call exit_mpi(myrank,'backward simulation became unstable and blew up in the fluid')
371  endif
372 
373  endif
