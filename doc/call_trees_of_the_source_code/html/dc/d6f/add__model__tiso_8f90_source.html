28 ! add_model_globe_tiso
29 !
30 ! this program can be used to update TRANSVERSE ISOTROPIC model files
31 ! based on smoothed event kernels.
32 ! the kernels are given for tranverse isotropic parameters (bulk_c,bulk_betav,bulk_betah,eta).
33 !
34 ! the algorithm uses a steepest descent method with a step length
35 ! determined by the given maximum update percentage.
36 !
37 ! input:
38 ! - step_fac : step length to update the models, f.e. 0.03 for plusminus 3%
39 !
40 ! setup:
41 !
42 !- INPUT_MODEL/ contains:
43 ! proc000***_reg1_vsv.bin &
44 ! proc000***_reg1_vsh.bin &
45 ! proc000***_reg1_vpv.bin &
46 ! proc000***_reg1_vph.bin &
47 ! proc000***_reg1_eta.bin &
48 ! proc000***_reg1_rho.bin
49 !
50 !- INPUT_GRADIENT/ contains:
51 ! proc000***_reg1_bulk_c_kernel_smooth.bin &
52 ! proc000***_reg1_bulk_betav_kernel_smooth.bin &
53 ! proc000***_reg1_bulk_betah_kernel_smooth.bin &
54 ! proc000***_reg1_eta_kernel_smooth.bin
55 !
56 !- topo/ contains:
57 ! proc000***_reg1_solver_data_1.bin
58 !
59 ! new models are stored in
60 !- OUTPUT_MODEL/ as
61 ! proc000***_reg1_vpv_new.bin &
62 ! proc000***_reg1_vph_new.bin &
63 ! proc000***_reg1_vsv_new.bin &
64 ! proc000***_reg1_vsh_new.bin &
65 ! proc000***_reg1_eta_new.bin &
66 ! proc000***_reg1_rho_new.bin
67 !
68 ! USAGE: ./add_model_globe_tiso 0.3
69 
70 
71 program add_model
72 
73  use tomography_model_tiso
74  use tomography_kernels_tiso
75 
76  implicit none
77  integer :: i,j,k,ispec,ier
78  real(kind=CUSTOM_REAL) :: betav1,betah1,betav0,betah0,rho1,rho0, &
79  betaiso1,betaiso0,eta1,eta0,alphav1,alphav0,alphah1,alphah0
80  real(kind=CUSTOM_REAL) :: dbetaiso,dbulk
81 
82  ! ============ program starts here =====================
83 
84  ! initializes arrays
85  call initialize()
86 
87  ! reads in parameters needed
88  call read_parameters_tomo()
89 
90  ! user output
91  if (myrank == 0) then
92  print *
93  print *,'***********'
94  print *,'program add_model_tiso: '
95  print *,' NPROC_XI , NPROC_ETA: ',nproc_xi_val,nproc_eta_val
96  print *,' NCHUNKS: ',nchunks_val
97  print *
98  print *,'model update for vsv,vsh,vpv,vph,eta,rho:'
99  print *,' step_fac = ',step_fac
100  print *
101  print *,'***********'
102  print *
103  endif
104 
105  ! reads in current transverse isotropic model files: vpv.. & vsv.. & eta & rho
106  call read_model_tiso()
107 
108  ! reads in smoothed kernels: bulk, betav, betah, eta
109  call read_kernels_tiso()
110 
111  ! calculates gradient
112  ! steepest descent method
113  call get_gradient_steepest_tiso()
114 
115  ! compute new model in terms of alpha, beta, eta and rho
116  ! (see also Carl's Latex notes)
117  ! allocate new model arrays
118  allocate(model_vpv_new(ngllx,nglly,ngllz,nspec), &
119  model_vph_new(ngllx,nglly,ngllz,nspec), &
120  model_vsv_new(ngllx,nglly,ngllz,nspec), &
121  model_vsh_new(ngllx,nglly,ngllz,nspec), &
122  model_eta_new(ngllx,nglly,ngllz,nspec), &
123  model_rho_new(ngllx,nglly,ngllz,nspec),stat=ier)
124  if (ier /= 0) stop 'Error allocating model arrays'
125 
126  ! initializes arrays
127  model_vpv_new = 0.0_custom_real
128  model_vph_new = 0.0_custom_real
129  model_vsv_new = 0.0_custom_real
130  model_vsh_new = 0.0_custom_real
131  model_eta_new = 0.0_custom_real
132  model_rho_new = 0.0_custom_real
133 
134  ! model update:
135  ! transverse isotropic update only in layer Moho to 220 (where SPECFEM3D_GLOBE considers TISO)
136  ! everywhere else uses an isotropic update
137  do ispec = 1, nspec
138  do k = 1, ngllz
139  do j = 1, nglly
140  do i = 1, ngllx
141 
142  ! initial model values
143  eta0 = model_eta(i,j,k,ispec)
144  betav0 = model_vsv(i,j,k,ispec)
145  betah0 = model_vsh(i,j,k,ispec)
146  rho0 = model_rho(i,j,k,ispec)
147  alphav0 = model_vpv(i,j,k,ispec)
148  alphah0 = model_vph(i,j,k,ispec)
149 
150  eta1 = 0._custom_real
151  betav1 = 0._custom_real
152  betah1 = 0._custom_real
153  rho1 = 0._custom_real
154  alphav1 = 0._custom_real
155  alphah1 = 0._custom_real
156 
157  ! do not use transverse isotropy except if element is between d220 and Moho
158 ! if (.not. ( idoubling(ispec)==IFLAG_220_80 .or. idoubling(ispec)==IFLAG_80_MOHO)) then
159  if (.not. ispec_is_tiso(ispec)) then
160  ! isotropic model update
161 
162  ! no eta perturbation, since eta = 1 in isotropic media
163  eta1 = eta0
164 
165  ! shear values
166  ! isotropic kernel K_beta = K_betav + K_betah
167  ! with same scaling step_length the model update dbeta_iso = dbetav + dbetah
168  ! note:
169  ! this step length can be twice as big as that given by the input
170  dbetaiso = model_dbetav(i,j,k,ispec) + model_dbetah(i,j,k,ispec)
171  betav1 = betav0 * exp( dbetaiso )
172  betah1 = betah0 * exp( dbetaiso )
173  ! note: betah is probably not really used in isotropic layers
174  ! (see SPECFEM3D_GLOBE/get_model.f90)
175 
176  ! density: uses scaling relation with isotropic shear perturbations
177  ! dln rho = RHO_SCALING * dln betaiso
178  rho1 = rho0 * exp( rho_scaling * dbetaiso )
179 
180  ! alpha values
181  dbulk = model_dbulk(i,j,k,ispec)
182  alphav1 = sqrt( alphav0**2 * exp(2.0*dbulk) + four_thirds * betav0**2 * ( &
183  exp(2.0*dbetaiso) - exp(2.0*dbulk) ) )
184  alphah1 = sqrt( alphah0**2 * exp(2.0*dbulk) + four_thirds * betah0**2 * ( &
185  exp(2.0*dbetaiso) - exp(2.0*dbulk) ) )
186  ! note: alphah probably not used in SPECFEM3D_GLOBE
187 
188  else
189 
190  ! transverse isotropic model update
191 
192  ! eta value : limits updated values for eta range constraint
193  eta1 = eta0 * exp( model_deta(i,j,k,ispec) )
194  if (eta1 < limit_eta_min ) eta1 = limit_eta_min
195  if (eta1 > limit_eta_max ) eta1 = limit_eta_max
196 
197  ! shear values
198  betav1 = betav0 * exp( model_dbetav(i,j,k,ispec) )
199  betah1 = betah0 * exp( model_dbetah(i,j,k,ispec) )
200 
201  ! density: uses scaling relation with Voigt average of shear perturbations
202  betaiso0 = sqrt( ( 2.0 * betav0**2 + betah0**2 ) / 3.0 )
203  betaiso1 = sqrt( ( 2.0 * betav1**2 + betah1**2 ) / 3.0 )
204  dbetaiso = log( betaiso1 / betaiso0 )
205  rho1 = rho0 * exp( rho_scaling * dbetaiso )
206 
207  ! alpha values
208  dbulk = model_dbulk(i,j,k,ispec)
209  alphav1 = sqrt( alphav0**2 * exp(2.0*dbulk) &
210  + four_thirds * betav0**2 * ( &
211  exp(2.0*model_dbetav(i,j,k,ispec)) - exp(2.0*dbulk) ) )
212  alphah1 = sqrt( alphah0**2 * exp(2.0*dbulk) &
213  + four_thirds * betah0**2 * ( &
214  exp(2.0*model_dbetah(i,j,k,ispec)) - exp(2.0*dbulk) ) )
215 
216  endif
217 
218 
219  ! stores new model values
220  model_vpv_new(i,j,k,ispec) = alphav1
221  model_vph_new(i,j,k,ispec) = alphah1
222  model_vsv_new(i,j,k,ispec) = betav1
223  model_vsh_new(i,j,k,ispec) = betah1
224  model_eta_new(i,j,k,ispec) = eta1
225  model_rho_new(i,j,k,ispec) = rho1
226 
227  enddo
228  enddo
229  enddo
230  enddo
231 
232  ! stores new model in files
233  call write_new_model_tiso()
234 
235  ! stores relative model perturbations
236  call write_new_model_perturbations_tiso()
237 
238  ! computes volume element associated with points, calculates kernel integral for statistics
239  call compute_kernel_integral_tiso()
240 
241  ! stop all the MPI processes, and exit
242  call finalize_mpi()
243 
244 end program add_model
