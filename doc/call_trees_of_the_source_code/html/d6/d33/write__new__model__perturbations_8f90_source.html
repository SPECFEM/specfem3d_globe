29 subroutine write_new_model_perturbations_iso()
30 
31 ! file output for new model perturbations
32 
33  use tomography_model_iso
34 
35  implicit none
36  real(kind=CUSTOM_REAL) :: min_vp,min_vs,max_vp,max_vs,min_rho,max_rho
37  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLY,NGLLZ,NSPEC) :: total_model
38  character(len=MAX_STRING_LEN) :: m_file
39 
40  ! user output
41  if (myrank == 0) print *,'writing out model perturbations...'
42 
43  ! vp relative perturbations
44  ! logarithmic perturbation: log( v_new) - log( v_old) = log( v_new / v_old )
45  total_model = 0.0_custom_real
46  where( model_vp /= 0.0 ) total_model = log( model_vp_new / model_vp)
47  ! or
48  ! linear approximation: (v_new - v_old) / v_old
49  !where( model_vp /= 0.0 ) total_model = ( model_vp_new - model_vp) / model_vp
50 
51  write(m_file,'(a,i6.6,a)') trim(output_model_dir)//'proc',myrank,trim(reg)//'dvpvp.bin'
52  if (myrank == 0) print *,' ',trim(output_model_dir)//'proc**'//trim(reg)//'dvpvp.bin'
53 
54  open(iout,file=trim(m_file),form='unformatted',action='write')
55  write(iout) total_model
56  close(iout)
57  call max_all_cr(maxval(total_model),max_vp)
58  call min_all_cr(minval(total_model),min_vp)
59 
60  ! vs relative perturbations
61  total_model = 0.0_custom_real
62  where( model_vs /= 0.0 ) total_model = log( model_vs_new / model_vs)
63 
64  write(m_file,'(a,i6.6,a)') trim(output_model_dir)//'proc',myrank,trim(reg)//'dvsvs.bin'
65  if (myrank == 0) print *,' ',trim(output_model_dir)//'proc**'//trim(reg)//'dvsvs.bin'
66 
67  open(iout,file=trim(m_file),form='unformatted',action='write')
68  write(iout) total_model
69  close(iout)
70  call max_all_cr(maxval(total_model),max_vs)
71  call min_all_cr(minval(total_model),min_vs)
72 
73  ! rho relative model perturbations
74  total_model = 0.0_custom_real
75  where( model_rho /= 0.0 ) total_model = log( model_rho_new / model_rho)
76 
77  write(m_file,'(a,i6.6,a)') trim(output_model_dir)//'proc',myrank,trim(reg)//'drhorho.bin'
78  if (myrank == 0) print *,' ',trim(output_model_dir)//'proc**'//trim(reg)//'drhorho.bin'
79 
80  open(iout,file=trim(m_file),form='unformatted',action='write')
81  write(iout) total_model
82  close(iout)
83  call max_all_cr(maxval(total_model),max_rho)
84  call min_all_cr(minval(total_model),min_rho)
85 
86  if (myrank == 0) then
87  print *
88  print *,'relative update:'
89  print *,' dvp/vp min/max : ',min_vp,max_vp
90  print *,' dvs/vs min/max : ',min_vs,max_vs
91  print *,' drho/rho min/max: ',min_rho,max_rho
92  print *
93  endif
94  call synchronize_all()
95 
96  ! statistics output
97  if (print_statistics_files .and. myrank == 0) then
98  open(iout,file=trim(output_statistics_dir)//'statistics_relative_vs_vp_rho',status='unknown',action='write')
99  write(iout,*) '#min_vs #max_vs #min_vp #max_vp #min_rho #max_rho'
100  write(iout,'(6e24.12)') min_vs,max_vs,min_vp,max_vp,min_rho,max_rho
101  close(iout)
102  endif
103 
104 end subroutine write_new_model_perturbations_iso
