86  call flush_imain()
87  endif
88  call read_mesh_databases_ic()
89

91  if (sync_reading ) call synchronize_all()
92  if (myrank == 0) then
93  write(imain,*) ' reading in coupling surface databases...'
94  call flush_imain()
95  endif
96  call read_mesh_databases_coupling()
97 
98  ! reads "addressing.txt" 2-D addressing (needed for Stacey boundaries and
99  ! regular grid kernels)
100  if (sync_reading ) call synchronize_all()
101  if (myrank == 0) then
102  write(imain,*) ' reading in addressing...'
103  call flush_imain()
104  endif
105  allocate(addressing(nchunks_val,0:nproc_xi_val-1,0:nproc_eta_val-1))
106  call read_mesh_databases_addressing()
107  if (.not. save_regular_kl) then
108  deallocate(addressing)
109  endif
110 
111  ! sets up MPI interfaces, inner/outer elements and mesh coloring
112  if (sync_reading ) call synchronize_all()
113  if (myrank == 0) then
114  write(imain,*) ' reading in MPI databases...'
115  call flush_imain()
116  endif
117  call read_mesh_databases_mpi()
118 
119  ! absorbing boundaries
120  if (absorbing_conditions) then
121  ! reads "stacey.bin" files
122  if (sync_reading ) call synchronize_all()
123  if (myrank == 0) then
124  write(imain,*) ' reading in Stacey databases...'
125  call flush_imain()
126  endif
127  call read_mesh_databases_stacey()
128  endif
129 
130  ! kernels on regular grids
131  if (save_regular_kl) then
132  if (sync_reading ) call synchronize_all()
133  if (myrank == 0) then
134  write(imain,*) ' reading in regular kernel databases...'
135  call flush_imain()
136  endif
137  call read_mesh_databases_regular_kl()
138  deallocate(addressing)
139  endif
140 
141 #ifdef USE_SERIAL_CASCADE_FOR_IOs
142  ! serial i/o
143  you_can_start_doing_ios = .true.
144  if (myrank < nproc_xi_val*nproc_eta_val-1) call send_singlel(you_can_start_doing_ios,myrank+1,itag)
145 #endif
146 
147  ! user output
148  call synchronize_all()
149  if (myrank == 0) then
150  ! elapsed time since beginning of mesh generation
151  tcpu = wtime() - time_start
152  write(imain,*)
153  write(imain,*) 'Elapsed time for reading mesh in seconds = ',sngl(tcpu)
154  write(imain,*)
155  call flush_imain()
156  endif
157 
158  end subroutine read_mesh_databases
