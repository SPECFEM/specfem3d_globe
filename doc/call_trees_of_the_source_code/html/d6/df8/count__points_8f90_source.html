111 ! in the inner core (no doubling region + eventually central cube)
112  if (include_central_cube) then
113  nglob(iregion_inner_core) = ((nex_per_proc_xi/ratio_divide_central_cube) &
114  *(ngllx-1)+1)*((nex_per_proc_eta/ratio_divide_central_cube) &

122 ! in the crust-mantle and outercore
123  do iter_region = iregion_crust_mantle,iregion_outer_core

132  endif
133  tmp_sum = 0;
134  do iter_layer = ifirst_region, ilast_region
135  nglob_int_surf_eta = 0

138  nglob_center_edge = 0
139  nglob_corner_edge = 0
140  nglob_border_edge = 0
141  if (this_region_has_a_doubling(iter_layer)) then
142  if (iter_region == iregion_outer_core .and. iter_layer == last_doubling_layer .and. &
143  (cut_superbrick_xi .or. cut_superbrick_eta)) then
144  doubling = 1
145  normal_doubling = 0
146  cut_doubling = 1
147  nb_lay_sb = 2
148  nglob_edge = 0
149  nglob_surf = 0
150  nglob_vol = 8*ngllx**3 - 12*ngllx**2 + 6*ngllx - 1
151  nglob_int_surf_eta = 6*ngllx**2 - 7*ngllx + 2
152  nglob_int_surf_xi = 5*ngllx**2 - 5*ngllx + 1
153  nglob_ext_surf = 4*ngllx**2-4*ngllx+1
154  nglob_center_edge = 4*(ngllx-1)+1
155  nglob_corner_edge = 2*(ngllx-1)+1
156  nglob_border_edge = 3*(ngllx-1)+1
157  else
158  if (ner(iter_layer) == 1) then
159  nb_lay_sb = 1
160  nglob_vol = 28*ngllx**3 - 62*ngllx**2 + 47*ngllx - 12
161  nglob_surf = 6*ngllx**2-8*ngllx+3
162  nglob_edge = ngllx
163  else
164  nb_lay_sb = 2
165  nglob_vol = 32*ngllx**3 - 70*ngllx**2 + 52*ngllx - 13
166  nglob_surf = 8*ngllx**2-11*ngllx+4
167  nglob_edge = 2*ngllx-1
168  endif
169  doubling = 1
170  normal_doubling = 1
171  cut_doubling = 0
172  endif
173  padding = -1
174  else
175  doubling = 0
176  normal_doubling = 0
177  cut_doubling = 0
178  padding = 0
179  nb_lay_sb = 0
180  nglob_vol = 0
181  nglob_surf = 0
182  nglob_edge = 0
183  endif
184  if (iter_layer == ilast_region) padding = padding +1
185  nblocks_xi = nex_per_proc_xi / ratio_sampling_array(iter_layer)
186  nblocks_eta = nex_per_proc_eta / ratio_sampling_array(iter_layer)
187 
188  tmp_sum = tmp_sum + &
189  ((nblocks_xi)*(ngllx-1)+1) * ((nblocks_eta)*(ngllx-1)+1) * ((ner(iter_layer) - doubling*nb_lay_sb)*(ngllx-1)+padding)+&
190  normal_doubling * ((((nblocks_xi*nblocks_eta)/4)*nglob_vol) - &
191  (((nblocks_eta/2-1)*nblocks_xi/2+(nblocks_xi/2-1)*nblocks_eta/2)*nglob_surf) + &
192  ((nblocks_eta/2-1)*(nblocks_xi/2-1)*nglob_edge)) + &
193  cut_doubling*(nglob_vol*(nblocks_xi*nblocks_eta) - &
194  ( nblocks_eta*(int(nblocks_xi/2)*nglob_int_surf_xi + int((nblocks_xi-1)/2)*nglob_ext_surf) + &
195  nblocks_xi*(int(nblocks_eta/2)*nglob_int_surf_eta + int((nblocks_eta-1)/2)*nglob_ext_surf)&
196  ) + &
197  ( int(nblocks_xi/2)*int(nblocks_eta/2)*nglob_center_edge + &
198  int((nblocks_xi-1)/2)*int((nblocks_eta-1)/2)*nglob_corner_edge + &
199  ((int(nblocks_eta/2)*int((nblocks_xi-1)/2))+(int((nblocks_eta-1)/2)*int(nblocks_xi/2)))*nglob_border_edge&
200  ))
201  enddo
202  nglob(iter_region) = tmp_sum
203  enddo

222 !!!
223 !!! 32*NGLLX**3 - 70*NGLLX**2 + 52*NGLLX - 13 -> nb GLL points in a superbrick (Volume)
224 !!! 8*NGLLX**2-11*NGLLX+4 -> nb GLL points on a superbrick side (Surface)
225 !!! 2*NGLLX-1 -> nb GLL points on a corner edge of a superbrick (Edge)
226 
227 !!! for the one layer superbrick :
228 !!! NGLOB = 28.NGLL^3 - 62.NGLL^2 + 47.NGLL - 12 (Volume)
229 !!! NGLOB = 6.NGLL^2 - 8.NGLL + 3 (Surface)
230 !!! NGLOB = NGLL (Edge)
231 !!!
232 !!! those results were obtained by using the script UTILS/doubling_brick/count_nglob_analytical.pl
233 !!! with an OpenDX file of the superbrick's geometry

235 !!! for the basic doubling bricks (two layers)
236 !!! NGLOB = 8.NGLL^3 - 12.NGLL^2 + 6.NGLL - 1 (VOLUME)
237 !!! NGLOB = 5.NGLL^2 - 5.NGLL + 1 (SURFACE 1)
