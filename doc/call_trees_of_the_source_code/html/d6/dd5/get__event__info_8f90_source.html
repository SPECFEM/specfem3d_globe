122 ! get information about event name and location for SAC seismograms: MPI version by Bernhard Schuberth
123 ! This subroutine reads the first line of the DATA/CMTSOLUTION file
124 ! and extracts event information needed for SAC or PITSA headers
125 
126 ! This subroutine has been modified to read full CMTSOLUTION file particularly for multiple-source cases.
127 ! Time-shifts of all sources can be read and the minimum t_shift is taken to be written in sac headers!
128 ! by Ebru
129 
130  subroutine get_event_info_serial(yr,jda,mo,da,ho,mi,sec,event_name,tshift_cmt,t_shift, &
131  elat_pde,elon_pde,depth_pde,mb,ms, &
132  cmt_lat,cmt_lon,cmt_depth,cmt_hdur,nsources, &
133  mrr,mtt,mpp,mrt,mrp,mtp)
134 
135  use constants
136  use shared_parameters, only: number_of_simultaneous_runs
137 
138  implicit none
139 
140 !--- arguments of the subroutine below
141 
142  integer, intent(in) :: NSOURCES
143 
144  integer, intent(out) :: yr,jda,mo,da,ho,mi
145  double precision, intent(out) :: sec
146  double precision, intent(out) :: tshift_cmt,t_shift
147  double precision, intent(out) :: elat_pde,elon_pde,depth_pde
148  real, intent(out) :: mb, ms
149  double precision, intent(out) :: cmt_lat,cmt_lon,cmt_depth,cmt_hdur
150  double precision, intent(out) :: Mrr, Mtt, Mpp, Mrt, Mrp, Mtp
151 
152  character(len=20), intent(out) :: event_name ! event name for SAC header
153 
154  ! local parameters
155  integer :: ios,julian_day
156  integer :: isource
157  double precision, dimension(NSOURCES) :: t_s,hdur,lat,lon,depth
158  character(len=20), dimension(NSOURCES) :: e_n
159  character(len=5) :: datasource
160  character(len=256) :: string
161 
162  character(len=MAX_STRING_LEN) :: CMTSOLUTION_FILE, path_to_add
163 
164 !
165 !---- read hypocenter info
166  cmtsolution_file = 'DATA/CMTSOLUTION'
167 
168  if (number_of_simultaneous_runs > 1 .and. mygroup >= 0) then
169  write(path_to_add,"('run',i4.4,'/')") mygroup + 1
170  cmtsolution_file = path_to_add(1:len_trim(path_to_add))//cmtsolution_file(1:len_trim(cmtsolution_file))
171  endif
172 !
173  open(unit=iin,file=trim(cmtsolution_file),status='old',action='read',iostat=ios)
174  if (ios /= 0) stop 'Error opening DATA/CMTSOLUTION file (in get_event_info_serial)'
175 
176  ! example header line of CMTSOLUTION file
177  !PDE 2003 09 25 19 50 08.93 41.78 144.08 18.0 7.9 8.0 Hokkaido, Japan
178  ! which is: event_id, date,origin time,latitude,longitude,depth, mb, MS, region
179 
180  ! read source number isource
181  do isource = 1,nsources
182 
183  ! read header with event information
184  read(iin,*) datasource,yr,mo,da,ho,mi,sec,elat_pde,elon_pde,depth_pde,mb,ms
185  jda=julian_day(yr,mo,da)
186 
187  ! read line with event name
188  read(iin,"(a)") string
189  read(string(12:len_trim(string)),*) e_n(isource)
190 
191  ! read time shift
192  read(iin,"(a)") string
193  read(string(12:len_trim(string)),*) t_s(isource)
194 
195  ! read half duration
196  read(iin,"(a)") string
197  read(string(15:len_trim(string)),*) hdur(isource)
198 
199  ! read latitude
200  read(iin,"(a)") string
201  read(string(10:len_trim(string)),*) lat(isource)
202 
203  ! read longitude
204  read(iin,"(a)") string
205  read(string(11:len_trim(string)),*) lon(isource)
206 
207  ! read depth
208  read(iin,"(a)") string
209  read(string(7:len_trim(string)),*) depth(isource)
210 
211  ! read the last 6 lines with moment tensor info
212  read(iin,"(a)") string
213  read(string(5:len_trim(string)),*) mrr
214  read(iin,"(a)") string
215  read(string(5:len_trim(string)),*) mtt
216  read(iin,"(a)") string
217  read(string(5:len_trim(string)),*) mpp
218  read(iin,"(a)") string
219  read(string(5:len_trim(string)),*) mrt
220  read(iin,"(a)") string
221  read(string(5:len_trim(string)),*) mrp
222  read(iin,"(a)") string
223  read(string(5:len_trim(string)),*) mtp
224 
225  enddo
226 
227  ! sets tshift_cmt to zero
228  tshift_cmt = 0.
229 
230  ! takes first event id as event_name
231  event_name = e_n(1)
232 
233  ! sets cmt info
234  if (nsources == 1) then
235  cmt_lat = lat(1)
236  cmt_lon = lon(1)
237  cmt_depth = depth(1)
238  cmt_hdur = hdur(1)
239  t_shift = t_s(1)
240  else
241  cmt_lat = -1e8
242  cmt_lon = -1e8
243  cmt_depth = -1e8
244  cmt_hdur = -1e8
245  ! takes minimum time shift of all given sources
246  t_shift = minval(t_s(1:nsources))
247  endif
248 
249  close(iin)
250 
251  end subroutine get_event_info_serial
