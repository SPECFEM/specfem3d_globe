169  subroutine prepare_avs_dx_surfaces_data_adios(nspec,iboun, &

202  real(kind=CUSTOM_REAL) muvstore(ngllx,nglly,ngllz,nspec)

237  iglobval(7)=ibool(ngllx,nglly,ngllz,ispec)

243  mask_ibool(iglobval(6)) = .true.
244  mask_ibool(iglobval(7)) = .true.
245  mask_ibool(iglobval(8)) = .true.
246 
247  endif
248  enddo
249

251  npoin = count(mask_ibool(:))
252 
253  ! erase the logical mask used to mark points already found
254  mask_ibool(:) = .false.
255 
256  ! output global AVS or DX points
257  numpoin = 0
258  do ispec = 1,nspec
259  ! only if at the surface
260  if (iboun(6,ispec)) then
261 
262  iglobval(5)=ibool(1,1,ngllz,ispec)
263  iglobval(6)=ibool(ngllx,1,ngllz,ispec)
264  iglobval(7)=ibool(ngllx,nglly,ngllz,ispec)
265  iglobval(8)=ibool(1,nglly,ngllz,ispec)
266 
267  ! top face
268  if (iboun(6,ispec)) then
269 
270  if (.not. mask_ibool(iglobval(5))) then
271  numpoin = numpoin + 1
272  num_ibool_avs_dx(iglobval(5)) = numpoin
273  avs_dx_adios%x_adios(numpoin) = sngl(xstore(1,1,ngllz,ispec))
274  avs_dx_adios%y_adios(numpoin) = sngl(ystore(1,1,ngllz,ispec))
275  avs_dx_adios%z_adios(numpoin) = sngl(zstore(1,1,ngllz,ispec))
276  endif
277 
278  if (.not. mask_ibool(iglobval(6))) then
279  numpoin = numpoin + 1
280  num_ibool_avs_dx(iglobval(6)) = numpoin
281  avs_dx_adios%x_adios(numpoin) = sngl(xstore(ngllx,1,ngllz,ispec))
282  avs_dx_adios%y_adios(numpoin) = sngl(ystore(ngllx,1,ngllz,ispec))
283  avs_dx_adios%z_adios(numpoin) = sngl(zstore(ngllx,1,ngllz,ispec))
284  endif
285 
286  if (.not. mask_ibool(iglobval(7))) then
287  numpoin = numpoin + 1
288  num_ibool_avs_dx(iglobval(7)) = numpoin
289  avs_dx_adios%x_adios(numpoin) = sngl(xstore(ngllx,nglly,ngllz,ispec))
290  avs_dx_adios%y_adios(numpoin) = sngl(ystore(ngllx,nglly,ngllz,ispec))
291  avs_dx_adios%z_adios(numpoin) = sngl(zstore(ngllx,nglly,ngllz,ispec))
292  endif
293 
294  if (.not. mask_ibool(iglobval(8))) then
295  numpoin = numpoin + 1
296  num_ibool_avs_dx(iglobval(8)) = numpoin
297  avs_dx_adios%x_adios(numpoin) = sngl(xstore(1,nglly,ngllz,ispec))
298  avs_dx_adios%y_adios(numpoin) = sngl(ystore(1,nglly,ngllz,ispec))
299  avs_dx_adios%z_adios(numpoin) = sngl(zstore(1,nglly,ngllz,ispec))
300  endif
301 
302  mask_ibool(iglobval(5)) = .true.
303  mask_ibool(iglobval(6)) = .true.
304  mask_ibool(iglobval(7)) = .true.
305  mask_ibool(iglobval(8)) = .true.
306  endif
307 
308  endif
309  enddo
310 
311  ! check that number of global points output is okay
312  if (numpoin /= npoin) &
313  call exit_mpi(myrank, &
314  'incorrect number of global points in AVS or DX file creation')
315 
316  ! output global AVS or DX elements
317  ispecface = 0
318  do ispec = 1,nspec
319  ! only if at the surface
320  if (iboun(6,ispec)) then
321 
322  iglobval(5)=ibool(1,1,ngllz,ispec)
323  iglobval(6)=ibool(ngllx,1,ngllz,ispec)
324  iglobval(7)=ibool(ngllx,nglly,ngllz,ispec)
325  iglobval(8)=ibool(1,nglly,ngllz,ispec)
326 
327  if (isotropic_3d_mantle) then
328  ! pick a point within the element and get its radius
329  r=dsqrt(xstore(2,2,2,ispec)**2 &
330  + ystore(2,2,2,ispec)**2+zstore(2,2,2,ispec)**2)
331 
332  if (r > rcmb/r_earth .and. r < r_unit_sphere) then
333  ! average over the element
334  dvp = 0.0
335  dvs = 0.0
336  np = 0
337  do k=2,ngllz-1
338  do j=2,nglly-1
339  do i=2,ngllx-1
340  np=np+1
341  x=xstore(i,j,k,ispec)
342  y=ystore(i,j,k,ispec)
343  z=zstore(i,j,k,ispec)
344  r=dsqrt(x*x+y*y+z*z)
345  ! take out ellipticity
346  if (ellipticity) then
347  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi_dummy)
348  cost=dcos(theta)
349 ! this is the Legendre polynomial of degree two, P2(cos(theta)), see the discussion above eq (14.4) in Dahlen and Tromp (1998)
350  p20=0.5d0*(3.0d0*cost*cost-1.0d0)
351 ! get ellipticity using spline evaluation
352  call spline_evaluation(rspl,espl,espl2,nspl,r,ell)
353 ! this is eq (14.4) in Dahlen and Tromp (1998)
354  factor=one-(two/3.0d0)*ell*p20
355  r=r/factor
356  endif
357 
358  ! gets reference model values: rho,vpv,vph,vsv,vsh and eta_aniso
359  call meshfem3d_models_get1d_val(iregion_code, &
360  idoubling(ispec), &
361  r,rho,vpv,vph,vsv,vsh,eta_aniso, &
362  qkappa,qmu,ricb,rcmb, &
363  rtopddoubleprime,r80,r120,r220,r400,r600,r670,r771, &
364  rmoho,rmiddle_crust,rocean)
365 
366  ! calculates isotropic values
367  vp = sqrt(((8.d0+4.d0*eta_aniso)*vph*vph + 3.d0*vpv*vpv &
368  + (8.d0 - 8.d0*eta_aniso)*vsv*vsv)/15.d0)
369  vs = sqrt(((1.d0-2.d0*eta_aniso)*vph*vph + vpv*vpv &
370  + 5.d0*vsh*vsh + (6.d0+4.d0*eta_aniso)*vsv*vsv)/15.d0)
371 
372  if (abs(rhostore(i,j,k,ispec)) < 1.e-20) then
373  print *,' attention: rhostore close to zero', &
374  rhostore(i,j,k,ispec),r,i,j,k,ispec
375  dvp = 0.0
376  dvs = 0.0
377  else if (abs(sngl(vp)) < 1.e-20) then
378  print *,' attention: vp close to zero',sngl(vp),r,i,j,k,ispec
379  dvp = 0.0
380  else if (abs(sngl(vs)) < 1.e-20) then
381  print *,' attention: vs close to zero',sngl(vs),r,i,j,k,ispec
382  dvs = 0.0
383  else
384  dvp = dvp + (sqrt((kappavstore(i,j,k,ispec) &
385  + 4.*muvstore(i,j,k,ispec)/3.) &
386  / rhostore(i,j,k,ispec)) - sngl(vp))/sngl(vp)
387  dvs = dvs &
388  + (sqrt(muvstore(i,j,k,ispec)/rhostore(i,j,k,ispec)) &
389  - sngl(vs))/sngl(vs)
390  endif
391 
392  enddo
393  enddo
394  enddo
395  dvp = dvp / np
396  dvs = dvs / np
397  else
398  dvp = 0.0
399  dvs = 0.0
400  endif
401  endif
402 
403  ! top face
404  ispecface = ispecface + 1
405  avs_dx_adios%idoubling(ispecface) = idoubling(ispec)
406  avs_dx_adios%iglob1(ispecface) = num_ibool_avs_dx(iglobval(5))
407  avs_dx_adios%iglob2(ispecface) = num_ibool_avs_dx(iglobval(6))
408  avs_dx_adios%iglob3(ispecface) = num_ibool_avs_dx(iglobval(7))
409  avs_dx_adios%iglob4(ispecface) = num_ibool_avs_dx(iglobval(8))
410  if (isotropic_3d_mantle) then
411  avs_dx_adios%dvp(ispecface) = dvp
412  avs_dx_adios%dvs(ispecface) = dvs
413  endif
414 
415  endif
416  enddo
417 
418  ! check that number of surface elements output is okay
419  if (ispecface /= nspecface) &
420  call exit_mpi(myrank,'incorrect number of surface elements in AVS or DX file creation')
421 
422 end subroutine prepare_avs_dx_surfaces_data_adios
