28  subroutine add_topography_410_650(myrank,xelm,yelm,zelm)
29 
30  use constants
31  use meshfem3d_par, only: r220,r400,r670,r771
32 
33  implicit none
34 
35  integer :: myrank
36 
37  double precision :: xelm(ngnod)
38  double precision :: yelm(ngnod)
39  double precision :: zelm(ngnod)
40 
41  integer :: ia
42 
43  real(kind=4) :: xcolat,xlon
44  real(kind=4) :: topo410out,topo650out
45  double precision :: topo410,topo650
46 
47  double precision :: r,lat,lon,theta,phi
48  double precision :: gamma
49  double precision :: x,y,z
50 
51  !statistics
52  logical,parameter :: DEBUG_STATISTICS = .false.
53  real(kind=CUSTOM_REAL),save :: min_410 = hugeval,max_410 = - hugeval
54  real(kind=CUSTOM_REAL),save :: min_650 = hugeval,max_650 = - hugeval
55  real(kind=CUSTOM_REAL) :: min_410_all,max_410_all
56  real(kind=CUSTOM_REAL) :: min_650_all,max_650_all
57 
58 ! note: adding topography to 410 and 660 strongly affects PcP, PKiKP, etc. phases,
59 ! we leave it in and check whether the stretching makes simulation unstable
60 !
61 ! topography perturbations
62 ! 410-km: minimum / maximum = -13.48 km / + 13.24 km
63 ! 650-km: minimum / maximum = -14.34 km / + 19.19 km
64 
65 ! we loop on all the points of the element
66  do ia = 1,ngnod
67 
68  x = xelm(ia)
69  y = yelm(ia)
70  z = zelm(ia)
71 
72  if (use_old_version_5_1_5_format) then
73  ! convert to r theta phi
74  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
75  call reduce(theta,phi)
76  ! get colatitude and longitude in degrees
77  xcolat = sngl(theta*radians_to_degrees)
78  xlon = sngl(phi*radians_to_degrees)
79  else
80  ! note: the topography on 410 and 650 is given in geographic colat/lon,
81  ! thus we need to convert geocentric colatitude to geographic colatitudes
82  !
83  ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
84  call xyz_2_rlatlon_dble(x,y,z,r,lat,lon)
85 
86  ! converts lat/lon to (real) colat/lon in degrees
87  xcolat = sngl(90.d0 - lat) ! colatitude
88  xlon = sngl(lon)
89  endif
90 
91  ! stretching occurs between 220 and 770
92  if (r > r220/r_earth .or. r < r771/r_earth) cycle
93 
94  ! compute topography on 410 and 650 at current point
95  call model_s362ani_subtopo(xcolat,xlon,topo410out,topo650out)
96 
97  ! min/max statistics
98  if (debug_statistics) then
99  if ( topo410out < min_410 ) min_410 = topo410out
100  if ( topo410out > max_410 ) max_410 = topo410out
101  if ( topo650out < min_650 ) min_650 = topo650out
102  if ( topo650out > max_650 ) max_650 = topo650out
103  ! debug
104  !print *,'topo410 / topo650: ',r,xcolat,xlon,topo410out,topo650out
105  endif
106 
107  ! non-dimensionalize the topography, which is in km
108  ! positive for a depression, so change the sign for a perturbation in radius
109  topo410 = -dble(topo410out) / r_earth_km
110  topo650 = -dble(topo650out) / r_earth_km
111 
112  gamma = 0.d0
113  if (r >= r400/r_earth .and. r <= r220/r_earth) then
114  ! stretching between R220 and R400
115  gamma = (r220/r_earth - r) / (r220/r_earth - r400/r_earth)
116  xelm(ia) = x*(one + gamma * topo410 / r)
117  yelm(ia) = y*(one + gamma * topo410 / r)
118  zelm(ia) = z*(one + gamma * topo410 / r)
119  else if (r >= r771/r_earth .and. r <= r670/r_earth) then
120  ! stretching between R771 and R670
121  gamma = (r - r771/r_earth) / (r670/r_earth - r771/r_earth)
122  xelm(ia) = x*(one + gamma * topo650 / r)
123  yelm(ia) = y*(one + gamma * topo650 / r)
124  zelm(ia) = z*(one + gamma * topo650 / r)
125  else if (r > r670/r_earth .and. r < r400/r_earth) then
126  ! stretching between R670 and R400
127  gamma = (r400/r_earth - r) / (r400/r_earth - r670/r_earth)
128  xelm(ia) = x*(one + (topo410 + gamma * (topo650 - topo410)) / r)
129  yelm(ia) = y*(one + (topo410 + gamma * (topo650 - topo410)) / r)
130  zelm(ia) = z*(one + (topo410 + gamma * (topo650 - topo410)) / r)
131  endif
132  if (gamma < -0.0001 .or. gamma > 1.0001) call exit_mpi(myrank,'incorrect value of gamma for 410-650 topography')
133 
134  enddo
135 
136  ! debug
137  if (debug_statistics) then
138  ! collects min/max on master
139  call min_all_cr(min_410,min_410_all)
140  call max_all_cr(max_410,max_410_all)
141  call min_all_cr(min_650,min_650_all)
142  call max_all_cr(max_650,max_650_all)
143  if (myrank == 0) then
144  if (r <= r220/r_earth .and. r >= r771/r_earth) then
145  print *,'add_topography_410_650: min/max_410 = ',min_410_all,max_410_all,'min/max_650 = ',min_650_all,max_650_all
146  endif
147  endif
148  !if (r <= R220/R_EARTH .and. r >= R771/R_EARTH) then
149  ! print *,myrank,'add_topography_410_650: min/max_410 = ',min_410,max_410,'min/max_650 = ',min_650,max_650
150  ! print *,myrank,'add_topography_410_650: depth = ',(1.d0 - r)*R_EARTH_KM,' 410-km = ',topo410out,' 650-km = ',topo650out
151  !endif
152  endif
153 
154  end subroutine add_topography_410_650
