28  subroutine add_topography_410_650(xelm,yelm,zelm)
29 
30  use constants
31  use meshfem3d_par, only: r220,r400,r670,r771
32 
33  implicit none
34 
35  double precision :: xelm(ngnod)
36  double precision :: yelm(ngnod)
37  double precision :: zelm(ngnod)
38 
39  integer :: ia
40 
41  real(kind=4) :: xcolat,xlon
42  real(kind=4) :: topo410out,topo650out
43  double precision :: topo410,topo650
44 
45  double precision :: r,lat,lon,theta,phi
46  double precision :: gamma
47  double precision :: x,y,z
48 
49  !statistics
50  logical,parameter :: DEBUG_STATISTICS = .false.
51  real(kind=CUSTOM_REAL),save :: min_410 = hugeval,max_410 = - hugeval
52  real(kind=CUSTOM_REAL),save :: min_650 = hugeval,max_650 = - hugeval
53  real(kind=CUSTOM_REAL) :: min_410_all,max_410_all
54  real(kind=CUSTOM_REAL) :: min_650_all,max_650_all
55 
56 ! note: adding topography to 410 and 660 strongly affects PcP, PKiKP, etc. phases,
57 ! we leave it in and check whether the stretching makes simulation unstable
58 !
59 ! topography perturbations
60 ! 410-km: minimum / maximum = -13.48 km / + 13.24 km
61 ! 650-km: minimum / maximum = -14.34 km / + 19.19 km
62 
63 ! we loop on all the points of the element
64  do ia = 1,ngnod
65 
66  x = xelm(ia)
67  y = yelm(ia)
68  z = zelm(ia)
69 
70  if (use_old_version_5_1_5_format) then
71  ! convert to r theta phi
72  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
73  call reduce(theta,phi)
74  ! get colatitude and longitude in degrees
75  xcolat = sngl(theta*radians_to_degrees)
76  xlon = sngl(phi*radians_to_degrees)
77  else
78  ! note: the topography on 410 and 650 is given in geographic colat/lon,
79  ! thus we need to convert geocentric colatitude to geographic colatitudes
80  !
81  ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
82  call xyz_2_rlatlon_dble(x,y,z,r,lat,lon)
83 
84  ! converts lat/lon to (real) colat/lon in degrees
85  xcolat = sngl(90.d0 - lat) ! colatitude
86  xlon = sngl(lon)
87  endif
88 
89  ! stretching occurs between 220 and 770
90  if (r > r220/r_earth .or. r < r771/r_earth) cycle
91 
92  ! compute topography on 410 and 650 at current point
93  call model_s362ani_subtopo(xcolat,xlon,topo410out,topo650out)
94 
95  ! min/max statistics
96  if (debug_statistics) then
97  if ( topo410out < min_410 ) min_410 = topo410out
98  if ( topo410out > max_410 ) max_410 = topo410out
99  if ( topo650out < min_650 ) min_650 = topo650out
100  if ( topo650out > max_650 ) max_650 = topo650out
101  ! debug
102  !print *,'topo410 / topo650: ',r,xcolat,xlon,topo410out,topo650out
103  endif
104 
105  ! non-dimensionalize the topography, which is in km
106  ! positive for a depression, so change the sign for a perturbation in radius
107  topo410 = -dble(topo410out) / r_earth_km
108  topo650 = -dble(topo650out) / r_earth_km
109 
110  gamma = 0.d0
111  if (r >= r400/r_earth .and. r <= r220/r_earth) then
112  ! stretching between R220 and R400
113  gamma = (r220/r_earth - r) / (r220/r_earth - r400/r_earth)
114  xelm(ia) = x*(one + gamma * topo410 / r)
115  yelm(ia) = y*(one + gamma * topo410 / r)
116  zelm(ia) = z*(one + gamma * topo410 / r)
117  else if (r >= r771/r_earth .and. r <= r670/r_earth) then
118  ! stretching between R771 and R670
119  gamma = (r - r771/r_earth) / (r670/r_earth - r771/r_earth)
120  xelm(ia) = x*(one + gamma * topo650 / r)
121  yelm(ia) = y*(one + gamma * topo650 / r)
122  zelm(ia) = z*(one + gamma * topo650 / r)
123  else if (r > r670/r_earth .and. r < r400/r_earth) then
124  ! stretching between R670 and R400
125  gamma = (r400/r_earth - r) / (r400/r_earth - r670/r_earth)
126  xelm(ia) = x*(one + (topo410 + gamma * (topo650 - topo410)) / r)
127  yelm(ia) = y*(one + (topo410 + gamma * (topo650 - topo410)) / r)
128  zelm(ia) = z*(one + (topo410 + gamma * (topo650 - topo410)) / r)
129  endif
130  if (gamma < -0.0001 .or. gamma > 1.0001) call exit_mpi(myrank,'incorrect value of gamma for 410-650 topography')
131 
132  enddo
133 
134  ! debug
135  if (debug_statistics) then
136  ! collects min/max on master
137  call min_all_cr(min_410,min_410_all)
138  call max_all_cr(max_410,max_410_all)
139  call min_all_cr(min_650,min_650_all)
140  call max_all_cr(max_650,max_650_all)
141  if (myrank == 0) then
142  if (r <= r220/r_earth .and. r >= r771/r_earth) then
143  print *,'add_topography_410_650: min/max_410 = ',min_410_all,max_410_all,'min/max_650 = ',min_650_all,max_650_all
144  endif
145  endif
146  !if (r <= R220/R_EARTH .and. r >= R771/R_EARTH) then
147  ! print *,myrank,'add_topography_410_650: min/max_410 = ',min_410,max_410,'min/max_650 = ',min_650,max_650
148  ! print *,myrank,'add_topography_410_650: depth = ',(1.d0 - r)*R_EARTH_KM,' 410-km = ',topo410out,' 650-km = ',topo650out
149  !endif
150  endif
151 
152  end subroutine add_topography_410_650
