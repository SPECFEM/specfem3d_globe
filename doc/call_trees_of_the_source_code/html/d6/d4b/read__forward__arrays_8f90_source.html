30 ! reads in saved wavefields
31 
32  use specfem_par

37  implicit none

39  ! local parameters
40  integer :: ier
41  character(len=MAX_STRING_LEN) outputname
42 
43  ! checks run/checkpoint number
44  if (number_of_runs < 1 .or. number_of_runs > nstep) &
45  stop 'number of restart runs can not be less than 1 or greater than NSTEP'
46  if (number_of_this_run < 1 .or. number_of_this_run > number_of_runs) &
47  stop 'incorrect run number'
48  if (simulation_type /= 1 .and. number_of_runs /= 1) &
49  stop 'Only 1 run for SIMULATION_TYPE = 2/3'
50 
51  ! define correct time steps if restart files
52  ! set start/end steps for time iteration loop
53  it_begin = (number_of_this_run - 1) * (nstep / number_of_runs) + 1
54  if (number_of_this_run < number_of_runs) then
55  it_end = number_of_this_run * (nstep / number_of_runs)
56  else
57  ! Last run may be a bit larger
58  it_end = nstep
59  endif
60 
61  ! checks if anything to do
62  ! undoing attenuation doesn't support the following checkpointing
63  if (undo_attenuation ) return
64 
65  ! read files back from local disk or MT tape system if restart file
66  if (number_of_this_run > 1) then
67  ! user output
68  if (myrank == 0) then
69  write(imain,*) 'reading startup file for run = ',number_of_this_run
70  call flush_imain()
71  endif
72 
73  if (adios_for_forward_arrays) then
74  call read_intermediate_forward_arrays_adios()
75  else
76  write(outputname,"('dump_all_arrays',i6.6)") myrank
77  outputname = trim(local_tmp_path) // '/' // outputname(1:len_trim(outputname))
78 
79  open(unit=iin,file=trim(outputname),status='old',action='read',form='unformatted',iostat=ier)
80  if (ier /= 0 ) call exit_mpi(myrank,'Error opening file dump_all_arrays*** for reading')
81

190  read(iin) b_b_array_rotation

208 
209  ! transfers fields onto GPU
210  if (gpu_mode) then
211  ! transfers initialized wavefields to GPU device
212  call transfer_b_fields_cm_to_device(ndim*nglob_crust_mantle,b_displ_crust_mantle,b_veloc_crust_mantle,b_accel_crust_mantle, &
213  mesh_pointer)
214 
215  call transfer_b_fields_ic_to_device(ndim*nglob_inner_core,b_displ_inner_core,b_veloc_inner_core,b_accel_inner_core, &
216  mesh_pointer)
217 
218  call transfer_b_fields_oc_to_device(nglob_outer_core,b_displ_outer_core,b_veloc_outer_core,b_accel_outer_core, &
219  mesh_pointer)
220  ! strain
221  if (.not. undo_attenuation) then
222  call transfer_b_strain_cm_to_device(mesh_pointer, &
223  b_epsilondev_xx_crust_mantle,b_epsilondev_yy_crust_mantle, &
224  b_epsilondev_xy_crust_mantle,b_epsilondev_xz_crust_mantle, &
225  b_epsilondev_yz_crust_mantle)
226 
227  call transfer_b_strain_ic_to_device(mesh_pointer, &
228  b_epsilondev_xx_inner_core,b_epsilondev_yy_inner_core, &
229  b_epsilondev_xy_inner_core,b_epsilondev_xz_inner_core, &
230  b_epsilondev_yz_inner_core)
231  endif
232  ! rotation
233  if (rotation_val) then
234  call transfer_b_rotation_to_device(mesh_pointer,b_a_array_rotation,b_b_array_rotation)
235  endif
236  ! attenuation memory variables
237  if (attenuation_val) then
238  call transfer_b_rmemory_cm_to_device(mesh_pointer, &
239  b_r_xx_crust_mantle,b_r_yy_crust_mantle, &
240  b_r_xy_crust_mantle,b_r_xz_crust_mantle, &
241  b_r_yz_crust_mantle)
242  call transfer_b_rmemory_ic_to_device(mesh_pointer, &
243  b_r_xx_inner_core,b_r_yy_inner_core, &
244  b_r_xy_inner_core,b_r_xz_inner_core, &
245  b_r_yz_inner_core)
246  endif
247  endif
