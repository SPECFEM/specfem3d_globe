29 subroutine compute_kernel_integral_iso()
30 
31 ! computes volume element associated with points and calculates kernel integral
32 
33  use tomography_kernels_iso
34  use tomography_model_iso
35  implicit none
36 
37  ! jacobian
38  real(kind=CUSTOM_REAL), dimension(:,:,:,:),allocatable :: jacobian
39  real(kind=CUSTOM_REAL) :: volumel
40  ! integration values
41  real(kind=CUSTOM_REAL) :: kernel_integral_alpha,kernel_integral_beta,kernel_integral_rho
42  real(kind=CUSTOM_REAL) :: integral_alpha_sum,integral_beta_sum,integral_rho_sum
43 
44  real(kind=CUSTOM_REAL) :: volume_glob,volume_glob_sum
45 
46  ! gradient vector norm ( v^T * v )
47  real(kind=CUSTOM_REAL) :: norm_bulk,norm_beta,norm_rho
48  real(kind=CUSTOM_REAL) :: norm_bulk_sum,norm_beta_sum,norm_rho_sum
49 
50  ! root-mean square values
51  real(kind=CUSTOM_REAL) :: rms_vp,rms_vs,rms_rho
52  real(kind=CUSTOM_REAL) :: rms_vp_sum,rms_vs_sum,rms_rho_sum
53  real(kind=CUSTOM_REAL) :: dvp,dvs,drho
54 
55  ! Gauss-Lobatto-Legendre points of integration and weights
56  double precision, dimension(NGLLX) :: xigll, wxgll
57  double precision, dimension(NGLLY) :: yigll, wygll
58  double precision, dimension(NGLLZ) :: zigll, wzgll
59  ! array with all the weights in the cube
60  real(kind=CUSTOM_REAL), dimension(NGLLX,NGLLY,NGLLZ) :: wgll_cube
61 
62  integer :: iglob
63  integer :: i,j,k,ispec,ier
64 
65  ! user output
66  if (myrank == 0) then
67  print *
68  print *,'***********'
69  print *,'statistics:'
70  print *,'***********'
71  print *
72  endif
73 
74  ! allocates array
75  allocate(jacobian(ngllx,nglly,ngllz,nspec),stat=ier)
76  if (ier /= 0) stop 'Error allocating jacobian array'
77 
78  ! GLL points
79  wgll_cube = 0.0
80  call zwgljd(xigll,wxgll,ngllx,gaussalpha,gaussbeta)
81  call zwgljd(yigll,wygll,nglly,gaussalpha,gaussbeta)
82  call zwgljd(zigll,wzgll,ngllz,gaussalpha,gaussbeta)
83  do k=1,ngllz
84  do j=1,nglly
85  do i=1,ngllx
86  wgll_cube(i,j,k) = wxgll(i)*wygll(j)*wzgll(k)
87  enddo
88  enddo
89  enddo
90 
91  ! builds jacobian
92  call compute_jacobian(jacobian)
93 
94  ! volume associated with global point
95  volume_glob = 0._custom_real
96  kernel_integral_alpha = 0._custom_real
97  kernel_integral_beta = 0._custom_real
98  kernel_integral_rho = 0._custom_real
99  norm_bulk = 0._custom_real
100  norm_beta = 0._custom_real
101  norm_rho = 0._custom_real
102  rms_vp = 0._custom_real
103  rms_vs = 0._custom_real
104  rms_rho = 0._custom_real
105 
106  do ispec = 1, nspec
107  do k = 1, ngllz
108  do j = 1, nglly
109  do i = 1, ngllx
110  iglob = ibool(i,j,k,ispec)
111  if (iglob == 0) then
112  print *,'iglob zero',i,j,k,ispec
113  print *
114  print *,'ibool:',ispec
115  print *,ibool(:,:,:,ispec)
116  print *
117  call exit_mpi(myrank,'Error ibool')
118  endif
119 
120  ! volume associated with GLL point
121  volumel = jacobian(i,j,k,ispec) * wgll_cube(i,j,k)
122  volume_glob = volume_glob + volumel
123 
124  ! kernel integration: for each element
125  kernel_integral_alpha = kernel_integral_alpha &
126  + volumel * kernel_bulk(i,j,k,ispec)
127 
128  kernel_integral_beta = kernel_integral_beta &
129  + volumel * kernel_beta(i,j,k,ispec)
130 
131  kernel_integral_rho = kernel_integral_rho &
132  + volumel * kernel_rho(i,j,k,ispec)
133 
134  ! gradient vector norm sqrt( v^T * v )
135  norm_bulk = norm_bulk + kernel_bulk(i,j,k,ispec)**2
136  norm_beta = norm_beta + kernel_beta(i,j,k,ispec)**2
137  norm_rho = norm_rho + kernel_rho(i,j,k,ispec)**2
138 
139  ! checks number (isNaN)
140  if (kernel_integral_alpha /= kernel_integral_alpha) then
141  print *,'Error NaN: ',kernel_integral_alpha
142  print *,'rank:',myrank
143  print *,'i,j,k,ispec:',i,j,k,ispec
144  print *,'volumel: ',volumel,'kernel_bulk:',kernel_bulk(i,j,k,ispec)
145  call exit_mpi(myrank,'Error NaN')
146  endif
147 
148  ! root-mean square
149  ! integrates relative perturbations ( dv / v using logarithm ) squared
150  dvp = log( model_vp_new(i,j,k,ispec) / model_vp(i,j,k,ispec) ) ! alphav
151  rms_vp = rms_vp + volumel * dvp*dvp
152 
153  dvs = log( model_vs_new(i,j,k,ispec) / model_vs(i,j,k,ispec) ) ! betav
154  rms_vs = rms_vs + volumel * dvs*dvs
155 
156  drho = log( model_rho_new(i,j,k,ispec) / model_rho(i,j,k,ispec) ) ! rho
157  rms_rho = rms_rho + volumel * drho*drho
158 
159  enddo
160  enddo
161  enddo
162  enddo
163 
164  ! statistics
165  ! (note: sum_all_cr() will only return valid results to master process)
166  ! kernel integration: for whole volume
167  call sum_all_cr(kernel_integral_alpha,integral_alpha_sum)
168  call sum_all_cr(kernel_integral_beta,integral_beta_sum)
169  call sum_all_cr(kernel_integral_rho,integral_rho_sum)
170  call sum_all_cr(volume_glob,volume_glob_sum)
171 
172  if (myrank == 0) then
173  print *,'integral kernels:'
174  print *,' a : ',integral_alpha_sum
175  print *,' beta: ',integral_beta_sum
176  print *,' rho : ',integral_rho_sum
177  print *
178  print *,' total volume:',volume_glob_sum
179  print *
180  if (volume_glob_sum < 1.e-25) stop 'Error zero total volume'
181  endif
182 
183  ! norms: for whole volume
184  call sum_all_cr(norm_bulk,norm_bulk_sum)
185  call sum_all_cr(norm_beta,norm_beta_sum)
186  call sum_all_cr(norm_rho,norm_rho_sum)
187 
188  if (myrank == 0) then
189  norm_bulk = sqrt(norm_bulk_sum)
190  norm_beta = sqrt(norm_beta_sum)
191  norm_rho = sqrt(norm_rho_sum)
192 
193  print *,'norm kernels:'
194  print *,' a : ',norm_bulk
195  print *,' beta: ',norm_beta
196  print *,' rho : ',norm_rho
197  print *
198  endif
199 
200  ! root-mean square
201  call sum_all_cr(rms_vp,rms_vp_sum)
202  call sum_all_cr(rms_vs,rms_vs_sum)
203  call sum_all_cr(rms_rho,rms_rho_sum)
204 
205  if (myrank == 0) then
206  rms_vp = sqrt( rms_vp_sum / volume_glob_sum )
207  rms_vs = sqrt( rms_vs_sum / volume_glob_sum )
208  rms_rho = sqrt( rms_rho_sum / volume_glob_sum )
209 
210  print *,'root-mean square of perturbations:'
211  print *,' vp : ',rms_vp
212  print *,' vs : ',rms_vs
213  print *,' rho: ',rms_rho
214  print *
215  endif
216  call synchronize_all()
217 
218  ! frees memory
219  deallocate(jacobian)
220 
221 end subroutine compute_kernel_integral_iso

565 
566  integral_rho = integral_rho &
567  + volumel * kernel_rho(i,j,k,ispec)
568 
569  ! gradient vector norm sqrt( v^T * v )
570  norm_bulk = norm_bulk + kernel_bulk(i,j,k,ispec)**2
571  norm_beta = norm_beta + kernel_beta(i,j,k,ispec)**2
572  norm_rho = norm_rho + kernel_rho(i,j,k,ispec)**2
573 
574  ! checks number
575  if (integral_bulk /= integral_bulk) then
576  print *,'Error NaN: ',integral_bulk
577  print *,'rank:',myrank
578  print *,'i,j,k,ispec:',i,j,k,ispec
579  print *,'volumel: ',volumel,'kernel_bulk:',kernel_bulk(i,j,k,ispec)
580  call exit_mpi(myrank,'Error NaN')
581  endif
582 
583  ! root-mean square
584  ! integrates relative perturbations ( dv / v using logarithm ) squared
585  dvpv = log( model_vpv_new(i,j,k,ispec) / model_vpv(i,j,k,ispec) ) ! alphav
586  rms_vpv = rms_vpv + volumel * dvpv*dvpv
587 
588  dvph = log( model_vph_new(i,j,k,ispec) / model_vph(i,j,k,ispec) ) ! alphah
589  rms_vph = rms_vph + volumel * dvph*dvph
590 
591  dvsv = log( model_vsv_new(i,j,k,ispec) / model_vsv(i,j,k,ispec) ) ! betav
592  rms_vsv = rms_vsv + volumel * dvsv*dvsv
593 
594  dvsh = log( model_vsh_new(i,j,k,ispec) / model_vsh(i,j,k,ispec) ) ! betah
595  rms_vsh = rms_vsh + volumel * dvsh*dvsh
596 
597  deta = log( model_eta_new(i,j,k,ispec) / model_eta(i,j,k,ispec) ) ! eta
598  rms_eta = rms_eta + volumel * deta*deta
599 
600  drho = log( model_rho_new(i,j,k,ispec) / model_rho(i,j,k,ispec) ) ! rho
601  rms_rho = rms_rho + volumel * drho*drho
602 
603  enddo
604  enddo
605  enddo
606  enddo
607 
608  ! statistics
609  ! (note: sum_all_cr() will only return valid results to master process)
610  ! kernel integration: for whole volume
611  call sum_all_cr(integral_bulk,integral_bulk_sum)
612  call sum_all_cr(integral_beta,integral_beta_sum)
613  call sum_all_cr(integral_rho,integral_rho_sum)
614  call sum_all_cr(volume_glob,volume_glob_sum)
615 
616  if (myrank == 0) then
617  print *,'integral kernels:'
618  print *,' a : ',integral_bulk_sum
619  print *,' beta: ',integral_beta_sum
620  print *,' rho : ',integral_rho_sum
621  print *
622  print *,' total volume:',volume_glob_sum
623  print *
624  if (volume_glob_sum < 1.e-25) stop 'Error zero total volume'
625  endif
626 
627  ! norms: for whole volume
628  call sum_all_cr(norm_bulk,norm_bulk_sum)
629  call sum_all_cr(norm_beta,norm_beta_sum)
630  call sum_all_cr(norm_rho,norm_rho_sum)
631 
632  if (myrank == 0) then
633  norm_bulk = sqrt(norm_bulk_sum)
634  norm_beta = sqrt(norm_beta_sum)
635  norm_rho = sqrt(norm_rho_sum)
636 
637  print *,'norm kernels:'
638  print *,' a : ',norm_bulk
639  print *,' beta: ',norm_beta
640  print *,' rho : ',norm_rho
641  print *
642  endif
643 
644  ! root-mean square
645  call sum_all_cr(rms_vpv,rms_vpv_sum)
646  call sum_all_cr(rms_vph,rms_vph_sum)
647  call sum_all_cr(rms_vsv,rms_vsv_sum)
648  call sum_all_cr(rms_vsh,rms_vsh_sum)
649  call sum_all_cr(rms_eta,rms_eta_sum)
650  call sum_all_cr(rms_rho,rms_rho_sum)
651 
652  if (myrank == 0) then
653  rms_vpv = sqrt( rms_vpv_sum / volume_glob_sum )
654  rms_vph = sqrt( rms_vph_sum / volume_glob_sum )
655  rms_vsv = sqrt( rms_vsv_sum / volume_glob_sum )
656  rms_vsh = sqrt( rms_vsh_sum / volume_glob_sum )
657  rms_eta = sqrt( rms_eta_sum / volume_glob_sum )
658  rms_rho = sqrt( rms_rho_sum / volume_glob_sum )
659 
660  print *,'root-mean square of perturbations:'
661  print *,' vpv: ',rms_vpv
662  print *,' vph: ',rms_vph
663  print *,' vsv: ',rms_vsv
664  print *,' vsh: ',rms_vsh
665  print *,' eta: ',rms_eta
666  print *,' rho: ',rms_rho
667  print *
668  endif
669  call synchronize_all()
670 
671  ! frees memory
672  deallocate(jacobian)
673 
674 end subroutine compute_kernel_integral_tiso_iso
