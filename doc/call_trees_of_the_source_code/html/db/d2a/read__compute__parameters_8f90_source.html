56  ! reads in Par_file values
57  call read_parameter_file()
58 
59  ! count the total number of sources in the CMTSOLUTION file
60  call count_number_of_sources(nsources)
61 
62  ! converts values to radians
63  movie_east = movie_east_deg * degrees_to_radians
64  movie_west = movie_west_deg * degrees_to_radians
65  movie_north = (90.0d0 - movie_north_deg) * degrees_to_radians ! converting from latitude to colatitude
66  movie_south = (90.0d0 - movie_south_deg) * degrees_to_radians
67  ! converts movie top/bottom depths to radii
68  movie_top = (r_earth_km-movie_top_km)/r_earth_km
69  movie_bottom = (r_earth_km-movie_bottom_km)/r_earth_km
70 
71  ! include central cube or not
72  ! use regular cubed sphere instead of cube for large distances
73  if (nchunks == 6) then
74  include_central_cube = .true.
75  inflate_central_cube = .false.
76  else
77  include_central_cube = .false.
78  inflate_central_cube = .true.
79  endif
80 
81  if (.not. emulate_only) then
82  nex_xi = nex_xi_read
83  nex_eta = nex_eta_read
84  nproc_xi = nproc_xi_read
85  nproc_eta = nproc_eta_read
86  else
87  ! this is used in UTILS/estimate_best_values_runs.f90 only, to estimate memory use
88  nex_eta = nex_xi
89  nproc_eta = nproc_xi
90  endif
91 
92  ! turns on/off corresponding 1-D/3-D model flags
93  ! and sets radius for each discontinuity and ocean density values
94  call get_model_parameters(model,reference_1d_model,three_d_model, &
95  anisotropic_3d_mantle,anisotropic_inner_core,attenuation_3d, &
96  case_3d,crustal,heterogen_3d_mantle,honor_1d_spherical_moho, &
97  isotropic_3d_mantle,one_crust,transverse_isotropy, &
98  oceans,topography, &
99  rocean,rmiddle_crust,rmoho,r80,r120,r220,r400,r600,r670,r771, &
100  rtopddoubleprime,rcmb,ricb,rmoho_fictitious_in_mesher, &
101  r80_fictitious_in_mesher,rho_top_oc,rho_bottom_oc,rho_oceans, &
102  cem_request,cem_accept)
103 
104  ! checks parameters
105  call rcp_check_parameters()
106 
107  ! sets time step size and number of layers
108  ! right distribution is determined based upon maximum value of NEX
109  call get_timestep_and_layers()
110 
111  ! time steps: this is an initial estimate based on the record length.
112  ! we will need to add additional time steps for reaching the start time at -t0,
113  ! which is only known when reading in the CMT source(s).
114  ! (see also routine setup_timesteps() in setup_sources_receivers.f90)
115  !
116  ! initial guess : compute total number of time steps, rounded to next multiple of 100
117  if (record_length_in_minutes < tinyval) then
118  ! zero length, uses a minimum of 5 steps for testing
119  nstep = 5
120  else
121  nstep = 100 * (int(record_length_in_minutes * 60.d0 / (100.d0*dt)) + 1)
122  endif
123 
124  ! noise simulations
125  if (noise_tomography /= 0) then
126  ! time steps needs to be doubled, due to +/- branches (symmetric around zero)
127  nstep = 2 * nstep - 1
128  endif
129 
130  ! if doing benchmark runs to measure scaling of the code for a limited number of time steps only
131  if (do_benchmark_run_only) nstep = nstep_for_benchmark
132 
133  ! debug
134  !print *,'initial time steps = ',NSTEP,' record length = ',RECORD_LENGTH_IN_MINUTES,' DT = ',DT
135 
136  ! half-time duration
137  !
138  ! computes a default hdur_movie that creates nice looking movies.
139  ! Sets HDUR_MOVIE as the minimum period the mesh can resolve
140  if (hdur_movie <= tinyval) then
141  hdur_movie = 1.2d0*max(240.d0/nex_xi*18.d0*angular_width_xi_in_degrees/90.d0, &
142  240.d0/nex_eta*18.d0*angular_width_eta_in_degrees/90.d0)
143  endif
144  ! noise simulations require MOVIE_SURFACE flag to output wavefield at Earth's surface;
145  ! however they don't need to convolve the source time function with any HDUR_MOVIE
146  ! since they employ a separate noise-spectrum source S_squared
147  if (noise_tomography /= 0) hdur_movie = 0.d0
148 
149  ! check that mesh can be coarsened in depth three or four times
150  cut_superbrick_xi=.false.
151  cut_superbrick_eta=.false.
152 
153  if (suppress_crustal_mesh .and. .not. add_4th_doubling) then
154  if (mod(nex_xi,8) /= 0) stop 'NEX_XI must be a multiple of 8'
155  if (mod(nex_eta,8) /= 0) stop 'NEX_ETA must be a multiple of 8'
156  if (mod(nex_xi/4,nproc_xi) /= 0) stop 'NEX_XI must be a multiple of 4*NPROC_XI'
157  if (mod(nex_eta/4,nproc_eta) /= 0) stop 'NEX_ETA must be a multiple of 4*NPROC_ETA'
158  if (mod(nex_xi/8,nproc_xi) /= 0) cut_superbrick_xi = .true.
159  if (mod(nex_eta/8,nproc_eta) /= 0) cut_superbrick_eta = .true.
160  else if (suppress_crustal_mesh .or. .not. add_4th_doubling) then
161  if (mod(nex_xi,16) /= 0) stop 'NEX_XI must be a multiple of 16'
162  if (mod(nex_eta,16) /= 0) stop 'NEX_ETA must be a multiple of 16'
163  if (mod(nex_xi/8,nproc_xi) /= 0) stop 'NEX_XI must be a multiple of 8*NPROC_XI'
164  if (mod(nex_eta/8,nproc_eta) /= 0) stop 'NEX_ETA must be a multiple of 8*NPROC_ETA'
165  if (mod(nex_xi/16,nproc_xi) /= 0) cut_superbrick_xi = .true.
166  if (mod(nex_eta/16,nproc_eta) /= 0) cut_superbrick_eta = .true.
167  else
168  if (mod(nex_xi,32) /= 0) stop 'NEX_XI must be a multiple of 32'
169  if (mod(nex_eta,32) /= 0) stop 'NEX_ETA must be a multiple of 32'
170  if (mod(nex_xi/16,nproc_xi) /= 0) stop 'NEX_XI must be a multiple of 16*NPROC_XI'
171  if (mod(nex_eta/16,nproc_eta) /= 0) stop 'NEX_ETA must be a multiple of 16*NPROC_ETA'
172  if (mod(nex_xi/32,nproc_xi) /= 0) cut_superbrick_xi = .true.
173  if (mod(nex_eta/32,nproc_eta) /= 0) cut_superbrick_eta = .true.
174  endif
175 
176 !
177 !--- compute additional parameters
178 !
179 
180  ! number of elements horizontally in each slice (i.e. per processor)
181  ! these two values MUST be equal in all cases
182  nex_per_proc_xi = nex_xi / nproc_xi
183  nex_per_proc_eta = nex_eta / nproc_eta
184 
185  ! total number of processors in each of the six chunks
186  nproc = nproc_xi * nproc_eta
187 
188  ! total number of processors in the full Earth composed of the six chunks
189  nproctot = nchunks * nproc
190 
191  ! definition of general mesh parameters
192  call define_all_layers(ner_crust,ner_80_moho,ner_220_80, &
193  ner_400_220,ner_600_400,ner_670_600,ner_771_670, &
194  ner_topddoubleprime_771,ner_cmb_topddoubleprime,ner_outer_core, &
195  ner_top_central_cube_icb, &
196  rmiddle_crust,r220,r400,r600,r670,r771,rtopddoubleprime,rcmb,ricb, &
197  r_central_cube,rmoho_fictitious_in_mesher,r80_fictitious_in_mesher, &
198  one_crust,ner,ratio_sampling_array, &
199  number_of_mesh_layers,layer_offset,last_doubling_layer, &
200  r_bottom,r_top,this_region_has_a_doubling, &
201  ielem,elem_doubling_mantle,elem_doubling_middle_outer_core, &
202  elem_doubling_bottom_outer_core, &
203  depth_second_doubling_real,depth_third_doubling_real, &
204  depth_fourth_doubling_real,distance,distance_min,zval, &
205  doubling_index,rmins,rmaxs)
206 
207  ! calculates number of elements (NSPEC)
208  call count_elements(nex_xi,nex_eta,nex_per_proc_xi,nproc, &
209  nex_per_proc_eta,ratio_divide_central_cube, &
210  nspec,nspec2d_xi,nspec2d_eta, &
211  nspec2dmax_xmin_xmax,nspec2dmax_ymin_ymax,nspec2d_bottom,nspec2d_top, &
212  nspec1d_radial, &
213  nglob2dmax_xmin_xmax,nglob2dmax_ymin_ymax, &
214  ner,ratio_sampling_array,this_region_has_a_doubling, &
215  ifirst_region,ilast_region,iter_region,iter_layer, &
216  doubling,tmp_sum,tmp_sum_xi,tmp_sum_eta, &
217  number_of_mesh_layers,layer_offset,nspec2d_xi_sb,nspec2d_eta_sb, &
218  nb_lay_sb, nspec_sb, nglob_surf, &
219  cut_superbrick_xi,cut_superbrick_eta, include_central_cube, &
220  last_doubling_layer, &
221  diff_nspec1d_radial,diff_nspec2d_xi,diff_nspec2d_eta, &
222  tmp_sum_nglob2d_xi, tmp_sum_nglob2d_eta,divider,nglob_edges_h, &
223  nglob_edge_v,to_remove)
224 
225  ! calculates number of points (NGLOB)
226  call count_points(nex_per_proc_xi,nex_per_proc_eta,ratio_divide_central_cube, &
227  nspec1d_radial,nglob1d_radial, &
228  nglob2dmax_xmin_xmax,nglob2dmax_ymin_ymax,nglob, &
229  nblocks_xi,nblocks_eta,ner,ratio_sampling_array, &
230  this_region_has_a_doubling, &
231  ifirst_region, ilast_region, iter_region, iter_layer, &
232  doubling, padding, tmp_sum, &
233  include_central_cube,ner_top_central_cube_icb,nex_xi, &
234  number_of_mesh_layers,layer_offset, &
235  nb_lay_sb, nglob_vol, nglob_surf, nglob_edge, &
236  cut_superbrick_xi,cut_superbrick_eta, &
237  last_doubling_layer, cut_doubling, nglob_int_surf_xi, nglob_int_surf_eta,nglob_ext_surf, &
238  normal_doubling, nglob_center_edge, nglob_corner_edge, nglob_border_edge)
239 
240  if (attenuation) then
241 !! DK DK July 2013: to save a huge amount of memory, when 3D attenuation is off it is sufficient to save a single point
242 !! DK DK July 2013: per spectral element because the Q attenuation factor is then constant per layer of the geological model
243  if (attenuation_3d .or. attenuation_1d_with_3d_storage) then
244  att1 = ngllx
245  att2 = nglly
246  att3 = ngllz
247  else
248  att1 = 1
249  att2 = 1
250  att3 = 1
251  endif
252  att4 = nspec(iregion_crust_mantle)
253  att5 = nspec(iregion_inner_core)
254  else
255  att1 = 1
256  att2 = 1
257  att3 = 1
258  att4 = 1
259  att5 = 1
260  endif
261 
262  end subroutine read_compute_parameters

268  subroutine rcp_check_parameters()
269 
270  use constants, only: &
271  custom_real,size_real,size_double,numfaces_shared,numcorners_shared, &
272  n_sls,ngnod,ngnod2d,ngllx,nglly,gravity_integrals
273 
274  use shared_parameters
275 
276  implicit none
277  ! local parameter
278  integer :: nex_minimum
279 
280 ! checks parameters
281 
282  if (simulation_type /= 1 .and. simulation_type /= 2 .and. simulation_type /= 3) &
283  stop 'SIMULATION_TYPE must be either 1, 2 or 3'
284 
285  if (noise_tomography < 0 .or. noise_tomography > 3) &
286  stop 'NOISE_TOMOGRAPHY must be either 0, 1, 2 or 3'
287 
288  if (nchunks /= 1 .and. nchunks /= 2 .and. nchunks /= 3 .and. nchunks /= 6) &
289  stop 'NCHUNKS must be either 1, 2, 3 or 6'
290 
291  ! this MUST be 90 degrees for two chunks or more to match geometrically
292  if (nchunks > 1 .and. abs(angular_width_xi_in_degrees - 90.d0) > 0.00000001d0) &
293  stop 'ANGULAR_WIDTH_XI_IN_DEGREES must be 90 for more than one chunk'
294 
295  ! this can be any value in the case of two chunks
296  if (nchunks > 2 .and. abs(angular_width_eta_in_degrees - 90.d0) > 0.00000001d0) &
297  stop 'ANGULAR_WIDTH_ETA_IN_DEGREES must be 90 for more than two chunks'
298 
299  if (absorbing_conditions .and. nchunks == 6) &
300  stop 'cannot have absorbing conditions in the full Earth'
301 
302  if (absorbing_conditions .and. nchunks == 3) &
303  stop 'absorbing conditions not supported for three chunks yet'
304 
305  if (attenuation_3d .and. .not. attenuation) &
306  stop 'Please set ATTENUATION to .true. in Par_file to use ATTENUATION_3D'
307 
308  if (save_transverse_kl_only .and. .not. anisotropic_kl) &
309  stop 'Please set ANISOTROPIC_KL to .true. in Par_file to use SAVE_TRANSVERSE_KL_ONLY'
310 
311  if (partial_phys_dispersion_only .and. undo_attenuation) &
312  stop 'cannot have both PARTIAL_PHYS_DISPERSION_ONLY and UNDO_ATTENUATION, they are mutually exclusive'
313 
314  ! simulations with undoing attenuation
315  if (undo_attenuation .and. movie_volume .and. movie_volume_type == 4 ) &
316  stop 'MOVIE_VOLUME_TYPE == 4 is not implemented for UNDO_ATTENUATION in order to save memory'
317 
318  !! DK DK this should not be difficult to fix and test, but not done yet by lack of time
319  if (undo_attenuation .and. number_of_runs /= 1) &
320  stop 'NUMBER_OF_RUNS should be == 1 for now when using UNDO_ATTENUATION'
321 
322  !! DK DK this should not be difficult to fix and test, but not done yet by lack of time
323  if (undo_attenuation .and. number_of_this_run > 1) &
324  stop 'we currently do not support NUMBER_OF_THIS_RUN > 1 in the case of UNDO_ATTENUATION'
325 
326  ! check that reals are either 4 or 8 bytes
327  if (custom_real /= size_real .and. custom_real /= size_double) &
328  stop 'wrong size of CUSTOM_REAL for reals'
329 
330  ! check that the parameter file is correct
331  if (ngnod /= 27) &
332  stop 'number of control nodes must be 27'
333  if (ngnod == 27 .and. ngnod2d /= 9) &
334  stop 'elements with 27 points should have NGNOD2D = 9'
335 
336  ! for the number of standard linear solids for attenuation
337  if (n_sls /= 3) &
338  stop 'number of SLS must be 3'
339 
340  ! check number of slices in each direction
341  if (nchunks < 1) &
342  stop 'must have at least one chunk'
343  if (nproc_xi < 1) &
344  stop 'NPROC_XI must be at least 1'
345  if (nproc_eta < 1) &
346  stop 'NPROC_ETA must be at least 1'
347 
348  ! check number of chunks
349  if (nchunks /= 1 .and. nchunks /= 2 .and. nchunks /= 3 .and. nchunks /= 6) &
350  stop 'only one, two, three or six chunks can be meshed'
351 
352  ! check that the central cube can be included
353  if (include_central_cube .and. nchunks /= 6) &
354  stop 'need six chunks to include central cube'
355 
356  ! check that sphere can be cut into slices without getting negative Jacobian
357  if (nchunks == 6) then
358  ! sets minimum NEX allowed for simulation
359  if (topography) then
360  ! mesh with topography leads to negative Jacobian for NEX < 48, regardless if 1D or 3D model
361  nex_minimum = 48
362  else
363  ! for flat topography, NEX = 32 setting still okay
364  nex_minimum = 32
365  endif
366  ! checks nex
367  if (nex_xi < nex_minimum) &
368  stop 'NEX_XI must be greater to cut the sphere into slices with positive Jacobian'
369  if (nex_eta < nex_minimum) &
370  stop 'NEX_ETA must be greater to cut the sphere into slices with positive Jacobian'
371  endif
372 
373  ! check that topology is correct if more than two chunks
374  if (nchunks > 2 .and. nex_xi /= nex_eta) &
375  stop 'must have NEX_XI = NEX_ETA for more than two chunks'
376 
377  if (nchunks > 2 .and. nproc_xi /= nproc_eta) &
378  stop 'must have NPROC_XI = NPROC_ETA for more than two chunks'
379 
380  ! small meshes useful for testing, also for GPU version
381  if (nchunks > 1 .and. (nproc_xi == 1 .or. nproc_eta == 1)) then
382  if (numfaces_shared < 4 ) &
383  stop 'NPROC_XI,NPROC_ETA == 1: please set in constants.h NUMFACES_SHARED and NUMCORNERS_SHARED equal to 4 and recompile'
384  if (numcorners_shared < 4 ) &
385  stop 'NPROC_XI,NPROC_ETA == 1: please set in constants.h NUMFACES_SHARED and NUMCORNERS_SHARED equal to 4 and recompile'
386  endif
387 
388  ! checks movie setup
389  if (movie_surface) then
390  if (movie_coarse .and. ngllx /= nglly) &
391  stop 'MOVIE_COARSE together with MOVIE_SURFACE requires NGLLX == NGLLY'
392  endif
393  if (movie_volume) then
394  if (movie_volume_type < 1 .or. movie_volume_type > 9) &
395  stop 'MOVIE_VOLUME_TYPE has to be in range from 1 to 9'
396  endif
397 
398  ! noise simulations
399  if (noise_tomography /= 0) then
400  if ((noise_tomography == 1 .or. noise_tomography == 2) .and. simulation_type /= 1) &
401  stop 'Noise simulations with NOISE_TOMOGRAPHY == 1 / 2 must have SIMULATION_TYPE == 1'
402  if (noise_tomography == 3 .and. simulation_type /= 3) &
403  stop 'Noise simulations with NOISE_TOMOGRAPHY == 3 must have SIMULATION_TYPE == 3'
404  if (number_of_runs /= 1 .or. number_of_this_run /= 1) &
405  stop 'NUMBER_OF_RUNS and NUMBER_OF_THIS_RUN must be 1 for NOISE TOMOGRAPHY simulation'
406  if (rotate_seismograms_rt) &
407  stop 'Do NOT rotate seismograms in the code, change ROTATE_SEISMOGRAMS_RT in Par_file for noise simulation'
408  if (save_all_seismos_in_one_file .or. use_binary_for_large_file) &
409  stop 'Please set SAVE_ALL_SEISMOS_IN_ONE_FILE and USE_BINARY_FOR_LARGE_FILE to be .false. for noise simulation'
410  endif
411 
412 !! DK DK for gravity integrals
413  ! in the case of GRAVITY_INTEGRALS we should always use double precision
414  if (gravity_integrals .and. custom_real /= size_double) &
415  stop 'for GRAVITY_INTEGRALS use double precision i.e. configure the code with --enable-double-precision'
416 
417  end subroutine rcp_check_parameters
