28  subroutine moho_stretching_honor_crust(myrank,xelm,yelm,zelm, &
29  elem_in_crust,elem_in_mantle)
30 
31 ! stretching the moho according to the crust 2.0
32 ! input: myrank, xelm, yelm, zelm
33 ! Dec, 30, 2009
34 
35  use constants, only: &
36  ngnod,r_earth_km,r_earth,r_unit_sphere, &
37  pi_over_two,radians_to_degrees,tinyval,smallval,one,use_old_version_5_1_5_format, &
38  suppress_moho_stretching, &
39  itype_crustal_model,icrust_crust_sh
40 
41  use meshfem3d_par, only: &
42  rmoho_fictitious_in_mesher,r220,rmiddle_crust
43 
44  use meshfem3d_par, only: &
45  topography
46 
47  implicit none
48 
49  integer :: myrank
50  double precision,dimension(NGNOD) :: xelm,yelm,zelm
51  logical,intent(inout) :: elem_in_crust,elem_in_mantle
52 
53  ! local parameters
54  double precision :: r,theta,phi,lat,lon
55  double precision :: vpvc,vphc,vsvc,vshc,etac,rhoc
56  double precision :: moho,elevation,gamma
57  double precision :: x,y,z
58  double precision :: R_moho,R_middlecrust
59  integer :: ia,count_crust,count_mantle
60  logical :: found_crust
61 
62  ! minimum/maximum allowed moho depths (5km/90km non-dimensionalized)
63  double precision,parameter :: MOHO_MINIMUM_DEFAULT = 5.0 / r_earth_km
64  double precision,parameter :: MOHO_MAXIMUM_DEFAULT = 90.0 / r_earth_km
65 
66  double precision :: MOHO_MINIMUM,MOHO_MAXIMUM
67 
68  ! sets min/max allowed moho depth
69  moho_minimum = moho_minimum_default
70  moho_maximum = moho_maximum_default
71  if (itype_crustal_model == icrust_crust_sh) then
72  ! minimum moho < 3.9km
73  moho_minimum = 3.5d0 / r_earth_km
74  endif
75 
76  ! radii for stretching criteria
77  r_moho = rmoho_fictitious_in_mesher/r_earth
78  r_middlecrust = rmiddle_crust/r_earth
79 
80  ! loops over element's anchor points
81  count_crust = 0
82  count_mantle = 0
83  do ia = 1,ngnod
84  ! gets anchor point location
85  x = xelm(ia)
86  y = yelm(ia)
87  z = zelm(ia)
88 
89  ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
90  if (use_old_version_5_1_5_format) then
91  ! note: at this point, the mesh is still perfectly spherical, thus no need to
92  ! convert the geocentric colatitude to a geographic colatitude
93  call xyz_2_rthetaphi_dble(x,y,z,r,theta,phi)
94  call reduce(theta,phi)
95  lat = (pi_over_two - theta) * radians_to_degrees
96  lon = phi * radians_to_degrees
97  else
98  ! note: the moho information is given in geographic latitude/longitude (with respect to a reference ellipsoid).
99  ! we need to convert the geocentric mesh positions (x/y/z) to geographic ones (lat/lon),
100  ! thus correcting geocentric latitude for ellipticity
101  !
102  ! converts geocentric coordinates x/y/z to geographic radius/latitude/longitude (in degrees)
103  call xyz_2_rlatlon_dble(x,y,z,r,lat,lon)
104  endif
105 
106  ! sets longitude bounds [-180,180]
107  if (lon > 180.0d0 ) lon = lon - 360.0d0
108 
109  ! initializes
110  moho = 0.d0
111 
112  ! gets smoothed moho depth
113  call meshfem3d_model_crust(lat,lon,r,vpvc,vphc,vsvc,vshc,etac,rhoc,moho,found_crust,elem_in_crust)
114 
115  ! checks non-dimensionalized moho depth
116  !
117  ! note: flag found_crust returns .false. for points below moho,
118  ! nevertheless its moho depth should be set and will be used in linear stretching
119  if (moho < tinyval ) call exit_mpi(myrank,'Error moho depth to honor')
120 
121  if (.not. use_old_version_5_1_5_format) then
122  ! limits moho depth to a threshold value to avoid stretching problems
123  if (moho < moho_minimum) then
124  print *,'moho value exceeds minimum (in km): ',moho*r_earth_km,moho_minimum*r_earth_km,'lat/lon:',lat,lon
125  moho = moho_minimum
126  endif
127  if (moho > moho_maximum) then
128  print *,'moho value exceeds maximum (in km): ',moho*r_earth_km,moho_maximum*r_earth_km,'lat/lon:',lat,lon
129  moho = moho_maximum
130  endif
131  endif
132 
133  ! radius of moho depth (normalized)
134  moho = one - moho
135 
136  ! checks if moho will be honored by elements
137  !
138  ! note: we will honor the moho only, if the moho depth is below R_moho (~35km)
139  ! or above R_middlecrust (~15km). otherwise, the moho will be "interpolated"
140  ! within the element
141  if (.not. suppress_moho_stretching) then
142  ! globe surface must honor topography for elements to be stretched for moho
143  !
144  ! note: if no topography is honored, stretching may lead to distorted elements and invalid Jacobian
145  if (topography) then
146  if (moho < r_moho) then
147  ! actual moho below fictitious moho
148  ! elements in second layer will stretch down to honor moho topography
149 
150  elevation = moho - r_moho
151 
152  if (r >= r_moho) then
153  ! point above fictitious moho
154  ! gamma ranges from 0 (point at surface) to 1 (point at fictitious moho depth)
155  gamma = (( r_unit_sphere - r )/( r_unit_sphere - r_moho ))
156  else
157  ! point below fictitious moho
158  ! gamma ranges from 0 (point at R220) to 1 (point at fictitious moho depth)
159  gamma = (( r - r220/r_earth)/( r_moho - r220/r_earth))
160 
161  ! since not all GLL points are exactly at R220, use a small
162  ! tolerance for R220 detection, fix R220
163  if (abs(gamma) < smallval) then
164  gamma = 0.0d0
165  endif
166  endif
167 
168  if (gamma < -0.0001d0 .or. gamma > 1.0001d0) &
169  call exit_mpi(myrank,'incorrect value of gamma for moho from crust 2.0')
170 
171  call move_point(ia,xelm,yelm,zelm,x,y,z,gamma,elevation,r)
172 
173  else if (moho > r_middlecrust) then
174  ! moho above middle crust
175  ! elements in first layer will squeeze into crust above moho
176 
177  elevation = moho - r_middlecrust
178 
179  if (r > r_middlecrust) then
180  ! point above middle crust
181  ! gamma ranges from 0 (point at surface) to 1 (point at middle crust depth)
182  gamma = (r_unit_sphere-r)/(r_unit_sphere - r_middlecrust )
183  else
184  ! point below middle crust
185  ! gamma ranges from 0 (point at R220) to 1 (point at middle crust depth)
186  gamma = (r - r220/r_earth)/( r_middlecrust - r220/r_earth )
187 
188  ! since not all GLL points are exactly at R220, use a small
189  ! tolerance for R220 detection, fix R220
190  if (abs(gamma) < smallval) then
191  gamma = 0.0d0
192  endif
193  endif
194 
195  if (gamma < -0.0001d0 .or. gamma > 1.0001d0) &
196  call exit_mpi(myrank,'incorrect value of gamma for moho from crust 2.0')
197 
198  call move_point(ia,xelm,yelm,zelm,x,y,z,gamma,elevation,r)
199 
200  endif
201  endif ! TOPOGRAPHY
202  endif
203 
204  ! counts corners in above moho
205  ! note: uses a small tolerance
206  if (r >= 0.9999d0*moho) then
207  count_crust = count_crust + 1
208  endif
209  ! counts corners below moho
210  ! again within a small tolerance
211  if (r <= 1.0001d0*moho) then
212  count_mantle = count_mantle + 1
213  endif
214 
215  enddo
216 
217  ! sets flag when all corners are above moho
218  if (count_crust == ngnod) then
219  elem_in_crust = .true.
220  endif
221  ! sets flag when all corners are below moho
222  if (count_mantle == ngnod) then
223  elem_in_mantle = .true.
224  endif
225 
226  ! small stretch check: stretching should affect only points above R220
227  if (r*r_earth < r220) then
228  print *,'Error moho stretching: ',r*r_earth,r220,moho*r_earth
229  call exit_mpi(myrank,'incorrect moho stretching')
230  endif
231 
232  end subroutine moho_stretching_honor_crust

364  subroutine stretch_deep_moho(ia,xelm,yelm,zelm,x,y,z,r,moho)
365 
366 ! honors deep moho (below 60 km), otherwise keeps the mesh boundary at r60 fixed
367 
368  use constants
369  use meshfem3d_par, only: rmoho_fictitious_in_mesher,r220,rmiddle_crust
370 
371  implicit none
372 
373  integer ia
374 
375  double precision xelm(ngnod)
376  double precision yelm(ngnod)
377  double precision zelm(ngnod)
378 
379  double precision :: x,y,z
380 
381  double precision :: r,moho
382 
383  ! local parameters
384  double precision :: elevation,gamma
385  ! radii for stretching criteria
386  double precision,parameter :: R15=6356000.d0/r_earth
387  double precision,parameter :: R25=6346000.d0/r_earth
388  double precision,parameter :: R30=6341000.d0/r_earth
389  double precision,parameter :: R35=6336000.d0/r_earth
390  double precision,parameter :: R40=6331000.d0/r_earth
391  double precision,parameter :: R45=6326000.d0/r_earth
392  double precision,parameter :: R50=6321000.d0/r_earth
393  double precision,parameter :: R55=6316000.d0/r_earth
394  double precision,parameter :: R60=6311000.d0/r_earth
395 
396  ! checks moho position: supposed to be at 60 km
397  if (rmoho_stretch_adjustment /= -20000.d0 ) &
398  stop 'wrong moho stretch adjustment for stretch_deep_moho'
399  if (rmoho_fictitious_in_mesher/r_earth /= r60 ) &
400  stop 'wrong moho depth '
401  ! checks middle crust position: supposed to be bottom of first layer at 15 km
402  if (rmiddle_crust/r_earth /= r15 ) &
403  stop 'wrong middle crust depth'
404 
405  ! stretches mesh by moving point coordinates
406  if (moho < r25 .and. moho > r45) then
407  ! moho between r25 and r45
408 
409  ! stretches mesh at r35 to moho depth
410  elevation = moho - r35
411  if (r >= r35 .and. r < r15) then
412  gamma=((r15-r)/(r15-r35))
413  else if (r < r35 .and. r > r60) then
414  gamma = (( r - r60)/( r35 - r60)) ! keeps r60 fixed
415  if (abs(gamma) < smallval) then
416  gamma=0.0d0
417  endif
418  else
419  gamma=0.0d0
420  endif
421  if (gamma < -0.0001d0 .or. gamma > 1.0001d0) &
422  stop 'incorrect value of gamma for moho from crust 2.0'
423 
424  call move_point(ia,xelm,yelm,zelm,x,y,z,gamma,elevation,r)
425 
426  else if (moho < r45) then
427  ! moho below r45
428 
429  ! moves mesh at r35 down to r45
430  elevation = r45 - r35
431  if (r >= r35 .and. r < r15) then
432  gamma=((r15-r)/(r15-r35)) ! moves r35 down to r45
433  else if (r < r35 .and. r > r60) then
434  gamma=((r-r60)/(r35-r60)) ! keeps r60 fixed
435  if (abs(gamma) < smallval) then
436  gamma=0.0d0
437  endif
438  else
439  gamma=0.0d0
440  endif
441  if (gamma < -0.0001d0 .or. gamma > 1.0001d0) &
442  stop 'incorrect value of gamma for moho from crust 2.0'
443 
444  call move_point(ia,xelm,yelm,zelm,x,y,z,gamma,elevation,r)
445 
446  ! add deep moho here
447  if (moho < r60) then
448  ! moho below r60
449 
450  ! stretches mesh at r60 to moho
451  elevation = moho - r60
452  if (r < r45 .and. r >= r60) then
453  gamma=(r45-r)/(r45-r60)
454  else if (r < r60) then
455  gamma=(r-r220/r_earth)/(r60-r220/r_earth)
456  if (abs(gamma) < smallval) then
457  gamma=0.0d0
458  endif
459  else
460  gamma=0.0d0
461  endif
462 
463  call move_point(ia,xelm,yelm,zelm,x,y,z,gamma,elevation,r)
464  endif
465 
466  else if (moho > r25) then
467  ! moho above r25
468 
469  ! moves mesh at r35 up to r25
470  elevation = r25-r35
471  if (r >= r35 .and. r < r15) then
472  gamma=((r15-r)/(r15-r35)) ! stretches r35 up to r25
473  else if (r < r35 .and. r > r60) then
474  gamma=(r-r60)/(r35-r60) ! keeps r60 fixed
475  if (abs(gamma) < smallval) then
476  gamma=0.0d0
477  endif
478  else
479  gamma=0.0d0
480  endif
481  if (gamma < -0.0001d0 .or. gamma > 1.0001d0) &
482  stop 'incorrect value of gamma for moho from crust 2.0'
483 
484  call move_point(ia,xelm,yelm,zelm,x,y,z,gamma,elevation,r)
485 
486  ! add shallow moho here
487  if (moho > r15) then
488  ! moho above r15
489 
490  ! stretches mesh at r15 to moho depth
491  elevation = moho-r15
492  if (r >= r15) then
493  gamma=(r_unit_sphere-r)/(r_unit_sphere-r15)
494  else if (r < r15 .and. r > r25) then
495  gamma=(r-r25)/(r15-r25) ! keeps mesh at r25 fixed
496  if (abs(gamma) < smallval) then
497  gamma=0.0d0
498  endif
499  else
500  gamma=0.0d0
501  endif
502 
503  call move_point(ia,xelm,yelm,zelm,x,y,z,gamma,elevation,r)
504  endif
505  endif
506 
507  end subroutine stretch_deep_moho
