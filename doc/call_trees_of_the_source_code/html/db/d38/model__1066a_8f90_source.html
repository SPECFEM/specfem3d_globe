86  subroutine model_1066a(x,rho,vp,vs,Qkappa,Qmu,iregion_code)
87 
88  use constants
89  use model_1066a_par
90 
91  implicit none
92 
93 ! input:
94 ! radius r: meters
95 
96 ! output:
97 ! density rho: kg/m^3
98 ! compressional wave speed vp: km/s
99 ! shear wave speed vs: km/s
100 
101  double precision :: x,rho,vp,vs,Qmu,Qkappa
102  integer :: iregion_code
103 
104  ! local parameters
105  double precision :: r,frac,scaleval
106  integer :: i
107 
108  ! compute real physical radius in meters
109  r = x * r_earth
110 
111  i = 1
112  do while(r >= m1066a_v_radius_1066a(i) .and. i /= nr_1066a)
113  i = i + 1
114  enddo
115 
116 ! make sure we stay in the right region and never take a point above
117 ! and a point below the ICB or the CMB and interpolate between them,
118 ! which would lead to a wrong value (keeping in mind that we interpolate
119 ! between points i-1 and i below)
120  if (iregion_code == iregion_inner_core .and. i > 33) i = 33
121 
122  if (iregion_code == iregion_outer_core .and. i < 35) i = 35
123  if (iregion_code == iregion_outer_core .and. i > 66) i = 66
124 
125  if (iregion_code == iregion_crust_mantle .and. i < 68) i = 68
126 
127  if (i == 1) then
128  rho = m1066a_v_density_1066a(i)
129  vp = m1066a_v_vp_1066a(i)
130  vs = m1066a_v_vs_1066a(i)
131  qmu = m1066a_v_qmu_1066a(i)
132  qkappa = m1066a_v_qkappa_1066a(i)
133  else
134 
135 ! interpolate from radius_1066a(i-1) to r using the values at i-1 and i
136  frac = (r-m1066a_v_radius_1066a(i-1))/(m1066a_v_radius_1066a(i)-m1066a_v_radius_1066a(i-1))
137 
138  rho = m1066a_v_density_1066a(i-1) + frac * (m1066a_v_density_1066a(i)-m1066a_v_density_1066a(i-1))
139  vp = m1066a_v_vp_1066a(i-1) + frac * (m1066a_v_vp_1066a(i)-m1066a_v_vp_1066a(i-1))
140  vs = m1066a_v_vs_1066a(i-1) + frac * (m1066a_v_vs_1066a(i)-m1066a_v_vs_1066a(i-1))
141  qmu = m1066a_v_qmu_1066a(i-1) + frac * (m1066a_v_qmu_1066a(i)-m1066a_v_qmu_1066a(i-1))
142  qkappa = m1066a_v_qkappa_1066a(i-1) + frac * (m1066a_v_qkappa_1066a(i)-m1066a_v_qkappa_1066a(i-1))
143 
144  endif
145 
146 ! make sure Vs is zero in the outer core even if roundoff errors on depth
147 ! also set fictitious attenuation to a very high value (attenuation is not used in the fluid)
148  if (iregion_code == iregion_outer_core) then
149  vs = 0.d0
150  qkappa = 3000.d0
151  qmu = 3000.d0
152  endif
153 
154 ! non-dimensionalize
155 ! time scaling (s^{-1}) is done with scaleval
156  scaleval=dsqrt(pi*grav*rhoav)
157  rho=rho*1000.0d0/rhoav
158  vp=vp*1000.0d0/(r_earth*scaleval)
159  vs=vs*1000.0d0/(r_earth*scaleval)
160 
161  end subroutine model_1066a
