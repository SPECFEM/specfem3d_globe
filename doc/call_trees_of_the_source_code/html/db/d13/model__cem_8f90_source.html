116  subroutine request_cem (vsh, vsv, vph, vpv, rho, iregion_code, ispec, i, j, k)
117 
118  use cem_par
119  use constants
120 
121  use meshfem3d_par, only: ibool
122 
123  implicit none
124 
125  double precision, intent (out) :: vsh, vsv, vph, vpv, rho
126 
127  integer, intent (in) :: iregion_code, ispec, i, j, k
128  integer :: iglob
129 
130  if (iregion_code == iregion_crust_mantle) then
131 
132  iglob = ibool(i,j,k,ispec)
133 
134  rho = reg1bc%rho(iglob) * 1000.0d0 / (rhoav)
135  vpv = reg1bc%vpv(iglob) * 1000.0d0 / (r_earth * scaleval)
136  vsv = reg1bc%vsv(iglob) * 1000.0d0 / (r_earth * scaleval)
137  vsh = reg1bc%vsh(iglob) * 1000.0d0 / (r_earth * scaleval)
138  vph = reg1bc%vph(iglob) * 1000.0d0 / (r_earth * scaleval)
139 
140  else if (iregion_code == iregion_outer_core) then
141 
142  iglob = ibool(i,j,k,ispec)
143 
144  rho = reg2bc%rho(iglob) * 1000.0d0 / (rhoav)
145  vpv = reg2bc%vpv(iglob) * 1000.0d0 / (r_earth * scaleval)
146  vsv = reg2bc%vsv(iglob) * 1000.0d0 / (r_earth * scaleval)
147  vsh = reg2bc%vsh(iglob) * 1000.0d0 / (r_earth * scaleval)
148  vph = reg2bc%vph(iglob) * 1000.0d0 / (r_earth * scaleval)
149 
150  else if (iregion_code == iregion_inner_core) then
151 
152  iglob = ibool(i,j,k,ispec)
153 
154  rho = reg3bc%rho(iglob) * 1000.0d0 / (rhoav)
155  vpv = reg3bc%vpv(iglob) * 1000.0d0 / (r_earth * scaleval)
156  vsv = reg3bc%vsv(iglob) * 1000.0d0 / (r_earth * scaleval)
157  vsh = reg3bc%vsh(iglob) * 1000.0d0 / (r_earth * scaleval)
158  vph = reg3bc%vph(iglob) * 1000.0d0 / (r_earth * scaleval)
159 
160  endif
161 
162  end subroutine request_cem

215  subroutine write_cem_request (iregion_code)
216 
217  use netcdf
218  use constants
219  use cem_par
220 
221  implicit none
222 
223  integer, parameter :: NDIMS_WRITE=2
224  integer, dimension (NDIMS_WRITE) :: start, count, ids
225 
226  integer :: ncid, paramDimID, procDimID, varidX, varidY, varidZ
227  integer :: varidR, iregion_code, commWorldSize, Comm, info
228  character (len = MAX_STRING_LEN) :: fileName, fileNameTrim, formatString
229 
230  ! Get the total number of processors.
231  call world_size(commworldsize)
232  call world_duplicate(comm)
233  call world_get_info_null(info)
234 
235  ! Define filename.
236  formatstring = "(A,I0.2,A)"
237  write (filename, formatstring) "DATA/cemRequest/xyz_reg", iregion_code, ".nc"
238  filenametrim = trim(filename)
239 
240  ! Create parallel NetCDF file.
241  call checknc (nf90_create(filenametrim, ior(nf90_netcdf4, nf90_mpiio), ncid, comm = comm, info = info))
242 
243  ! Define the processor array.
244  call checknc (nf90_def_dim(ncid, 'glob', size (xyzout(:,1)), paramdimid))
245  call checknc (nf90_def_dim(ncid, 'proc', commworldsize, procdimid))
246 
247  ! Sort ids into array.
248  ids = (/ paramdimid, procdimid /)
249 
250  ! Define the kernel variable.
251  call checknc (nf90_def_var(ncid, 'x', nf90_float, ids, varidx))
252  call checknc (nf90_def_var(ncid, 'y', nf90_float, ids, varidy))
253  call checknc (nf90_def_var(ncid, 'z', nf90_float, ids, varidz))
254  call checknc (nf90_def_var(ncid, 'r', nf90_short, ids, varidr))
255 
256  ! End definitions.
257  call checknc (nf90_enddef(ncid))
258 
259  ! Each processor writes one row.
260  start = (/ 1, myrank + 1 /)
261  count = (/ size (xyzout(:,1)), 1 /)
262  call checknc (nf90_put_var(ncid, varidx, xyzout(:,1), start = start, count = count))
263  call checknc (nf90_put_var(ncid, varidy, xyzout(:,2), start = start, count = count))
264  call checknc (nf90_put_var(ncid, varidz, xyzout(:,3), start = start, count = count))
265  call checknc (nf90_put_var(ncid, varidr, regcode, start = start, count = count))
266 
267  ! Close the file.
268  call checknc (nf90_close(ncid))
269 
270  deallocate(xyzout)
271  deallocate(regcode)
272 
273  end subroutine write_cem_request

280  subroutine build_global_coordinates (nspec, nglob, iregion_code)
281 
282  use constants
283  use cem_par
284 
285  use meshfem3d_par, only: &
286  ibool,xstore,ystore,zstore
287 
288  implicit none
289 
290  integer, intent (in) :: nspec, nglob, iregion_code
291  integer :: i, j, k, iglob, ispec, region
292 
293  double precision, parameter :: R_020_KM=6351.0d0, r_052_km=6319.0d0
294  double precision, parameter :: R_100_KM=6271.0d0, r_400_km=5971.0d0
295  double precision, parameter :: R_670_KM=5701.0d0, r_cmb_km=3480.0d0
296  double precision, parameter :: R_ICB_KM=1221.0d0, r_tho_km=5371.0d0
297  double precision :: x, y, z, rad
298 
299  allocate(xyzout(nglob,ndims))
300  allocate(regcode(nglob))
301 
302  ! Global to local x co-ordinate numbering.
303  do ispec = 1,nspec
304  do k = 1,ngllz
305  do j = 1,nglly
306  do i = 1,ngllx
307 
308  x = sngl(xstore(i,j,k,ispec)) * r_earth_km
309  iglob = ibool(i,j,k,ispec)
310  xyzout(iglob,1) = sngl(x)
311 
312  enddo
313  enddo
314  enddo
315  enddo
316 
317  ! Global to local y co-ordinate numbering.
318  do ispec = 1,nspec
319  do k = 1,ngllz
320  do j = 1,nglly
321  do i = 1,ngllx
322 
323  y = sngl(ystore(i,j,k,ispec)) * r_earth_km
324  iglob = ibool(i,j,k,ispec)
325  xyzout(iglob,2) = sngl(y)
326 
327  enddo
328  enddo
329  enddo
330  enddo
331 
332  ! Global to local z co-ordinate numbering.
333  do ispec = 1,nspec
334  do k = 1,ngllz
335  do j = 1,nglly
336  do i = 1,ngllx
337 
338  z = sngl(zstore(i,j,k,ispec)) * r_earth_km
339  iglob = ibool(i,j,k,ispec)
340  xyzout(iglob,3) = sngl(z)
341 
342  enddo
343  enddo
344  enddo
345  enddo
346 
347  if (iregion_code == 1) then
348 
349  do ispec = 1,nspec
350  do k = 1,ngllz
351  do j = 1,nglly
352  do i = 1,ngllx
353 
354  x = sngl(xstore(i,j,k,ispec)) * r_earth_km
355  iglob = ibool(i,j,k,ispec)
356  xyzout(iglob,1) = sngl(x)
357 
358  y = sngl(ystore(i,j,k,ispec)) * r_earth_km
359  iglob = ibool(i,j,k,ispec)
360  xyzout(iglob,2) = sngl(y)
361 
362  z = sngl(zstore(i,j,k,ispec)) * r_earth_km
363  iglob = ibool(i,j,k,ispec)
364  xyzout(iglob,3) = sngl(z)
365 
366  rad = dsqrt( x * x + y * y + z * z )
367 
368  if (rad < r_670_km .and. rad >= r_cmb_km) then
369  region = 3
370  else if (rad < r_400_km .and. rad >= r_670_km) then
371  region = 2
372  else if (rad >= r_400_km) then
373  region = 1
374  endif
375 
376  iglob = ibool(i,j,k,ispec)
377  regcode(iglob) = region
378 
379  enddo
380  enddo
381  enddo
382  enddo
383 
384  else if (iregion_code == 2) then
385 
386  regcode(:) = 4
387 
388  else if (iregion_code == 3) then
389 
390  regcode(:) = 5
391 
392  endif
393 
394  end subroutine build_global_coordinates
