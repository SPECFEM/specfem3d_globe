185  ! only compute element which belong to current phase (inner or outer elements)
186 
187  do_loop_ijk
188  iglob = ibool(index_ijk,ispec)
189  ! gets "displacement"
190  dummyx_loc(index_ijk) = displfluid(iglob)

193  ! subroutines adapted from Deville, Fischer and Mund, High-order methods
194  ! for incompressible fluid flow, Cambridge University Press (2002),
195  ! pages 386 and 389 and Figure 8.3.1
196 
197  ! computes 1. matrix multiplication for tempx1,..
198  call mxm5_single(hprime_xx,m1,dummyx_loc,tempx1,m2)
199  ! computes 2. matrix multiplication for tempx2,..
200  call mxm5_3dmat_single(dummyx_loc,m1,hprime_xxt,m1,tempx2,ngllx)
201  ! computes 3. matrix multiplication for tempx1,..
202  call mxm5_single(dummyx_loc,m2,hprime_xxt,tempx3,m1)
203 
204  do_loop_ijk
205  ! get derivatives of velocity potential with respect to x, y and z
206  xixl = deriv(1,index_ijk,ispec)
207  xiyl = deriv(2,index_ijk,ispec)
208  xizl = deriv(3,index_ijk,ispec)
209  etaxl = deriv(4,index_ijk,ispec)
210  etayl = deriv(5,index_ijk,ispec)
211  etazl = deriv(6,index_ijk,ispec)
212  gammaxl = deriv(7,index_ijk,ispec)
213  gammayl = deriv(8,index_ijk,ispec)
214  gammazl = deriv(9,index_ijk,ispec)
215 
216  ! compute the Jacobian
217  jacobianl(index_ijk) = 1._custom_real / (xixl*(etayl*gammazl-etazl*gammayl) &
218  - xiyl*(etaxl*gammazl-etazl*gammaxl) &
219  + xizl*(etaxl*gammayl-etayl*gammaxl))
220 
221  dpotentialdxl(index_ijk) = xixl*tempx1(index_ijk) + etaxl*tempx2(index_ijk) + gammaxl*tempx3(index_ijk)
222  dpotentialdyl(index_ijk) = xiyl*tempx1(index_ijk) + etayl*tempx2(index_ijk) + gammayl*tempx3(index_ijk)
223  dpotentialdzl(index_ijk) = xizl*tempx1(index_ijk) + etazl*tempx2(index_ijk) + gammazl*tempx3(index_ijk)
224  ENDdo_loop_ijk
225 
226  ! compute contribution of rotation and add to gradient of potential
227  ! this term has no Z component
228  if (rotation_val) then
229  ! store the source for the Euler scheme for A_rotation and B_rotation
230  two_omega_deltat = deltat * two_omega_earth
231 
232  cos_two_omega_t = cos(two_omega_earth*timeval)
233  sin_two_omega_t = sin(two_omega_earth*timeval)
234 
235  ! update rotation term with Euler scheme
236  !
237  ! note: rotation with euler scheme is not exactly revertible;
238  ! backpropagation/reconstruction of wavefield will lead to small differences
239  if (use_lddrk) then
240  ! use the source saved above
241  do_loop_ijk
242  a_rotation(index_ijk) = a_array_rotation(index_ijk,ispec)
243  b_rotation(index_ijk) = b_array_rotation(index_ijk,ispec)
244 
245  ! time step deltat of Euler scheme is included in the source
246  source_euler_a = two_omega_deltat &
247  * (cos_two_omega_t * dpotentialdyl(index_ijk) + sin_two_omega_t * dpotentialdxl(index_ijk))
248  source_euler_b = two_omega_deltat &
249  * (sin_two_omega_t * dpotentialdyl(index_ijk) - cos_two_omega_t * dpotentialdxl(index_ijk))
250 
251  a_array_rotation_lddrk(index_ijk,ispec) = myalpha_lddrk(istage) * a_array_rotation_lddrk(index_ijk,ispec) &
252  + source_euler_a
253  a_array_rotation(index_ijk,ispec) = a_array_rotation(index_ijk,ispec) &
254  + mybeta_lddrk(istage) * a_array_rotation_lddrk(index_ijk,ispec)
255 
256  b_array_rotation_lddrk(index_ijk,ispec) = myalpha_lddrk(istage) * b_array_rotation_lddrk(index_ijk,ispec) &
257  + source_euler_b
258  b_array_rotation(index_ijk,ispec) = b_array_rotation(index_ijk,ispec) &
259  + mybeta_lddrk(istage) * b_array_rotation_lddrk(index_ijk,ispec)
260  ENDdo_loop_ijk
261  else
262  ! Euler scheme
263  do_loop_ijk
264  a_rotation(index_ijk) = a_array_rotation(index_ijk,ispec)
265  b_rotation(index_ijk) = b_array_rotation(index_ijk,ispec)
266 
267  ! time step deltat of Euler scheme is included in the source
268  source_euler_a = two_omega_deltat &
269  * (cos_two_omega_t * dpotentialdyl(index_ijk) + sin_two_omega_t * dpotentialdxl(index_ijk))
270  source_euler_b = two_omega_deltat &
271  * (sin_two_omega_t * dpotentialdyl(index_ijk) - cos_two_omega_t * dpotentialdxl(index_ijk))
272 
273  a_array_rotation(index_ijk,ispec) = a_array_rotation(index_ijk,ispec) + source_euler_a
274  b_array_rotation(index_ijk,ispec) = b_array_rotation(index_ijk,ispec) + source_euler_b
275  ENDdo_loop_ijk
276  endif
277 
278  ! adds rotation
279  do_loop_ijk
280  ux_rotation = a_rotation(index_ijk)*cos_two_omega_t + b_rotation(index_ijk)*sin_two_omega_t
281  uy_rotation = - a_rotation(index_ijk)*sin_two_omega_t + b_rotation(index_ijk)*cos_two_omega_t
282 
283  dpotentialdxl(index_ijk) = dpotentialdxl(index_ijk) + ux_rotation
284  dpotentialdyl(index_ijk) = dpotentialdyl(index_ijk) + uy_rotation
285  ENDdo_loop_ijk
286  endif ! end of section with rotation
287 
288  ! add (chi/rho)grad(rho) term in no gravity case
289  if (.not. gravity_val) then
290  ! With regards to the non-gravitating case: we cannot set N^2 = 0 *and* let g = 0.
291  ! We can *either* assume N^2 = 0 but keep gravity g, *or* we can assume that gravity
292  ! is negligible to begin with, as in our GJI 2002a, in which case N does not arise.
293  ! We get:
294  !
295  ! \ddot\chi = \rho^{-1}\kappa\bdel\cdot(\bdel\chi+\chi\bdel\ln\rho)
296  !
297  ! Then the displacement is
298  !
299  ! \bu = \bdel\chi+\chi\bdel\ln\rho = \rho^{-1}\bdel(\rho\chi)
300  !
301  ! and the pressure is
302  !
303  ! p = -\rho\ddot{\chi}
304  !
305  ! Thus in our 2002b GJI paper eqn (21) is wrong, and equation (41)
306  ! in our AGU monograph is incorrect; these equations should be replaced by
307  !
308  ! \ddot\chi = \rho^{-1}\kappa\bdel\cdot(\bdel\chi+\chi\bdel\ln\rho)
309  !
310  ! Note that the fluid potential we use in GJI 2002a differs from the one used here:
311  !
312  ! \chi_GJI2002a = \rho\partial\t\chi
313  !
314  ! such that
315  !
316  ! \bv = \partial_t\bu=\rho^{-1}\bdel\chi_GJI2002a (GJI 2002a eqn 20)
317  !
318  ! p = - \partial_t\chi_GJI2002a (GJI 2002a eqn 19)
319  do_loop_ijk
320  iglob = ibool(index_ijk,ispec)
321 
322  ! gradient of d ln(rho)/dr in Cartesian coordinates
323  vec_x = gravity_pre_store(1,iglob)
324  vec_y = gravity_pre_store(2,iglob)
325  vec_z = gravity_pre_store(3,iglob)
326 
327  ! grad(rho)/rho in Cartesian components
328  dpotentialdxl(index_ijk) = dpotentialdxl(index_ijk) + dummyx_loc(index_ijk) * vec_x
329  dpotentialdyl(index_ijk) = dpotentialdyl(index_ijk) + dummyx_loc(index_ijk) * vec_y
330  dpotentialdzl(index_ijk) = dpotentialdzl(index_ijk) + dummyx_loc(index_ijk) * vec_z
331  ENDdo_loop_ijk
332  endif
333 
334  do_loop_ijk
335  ! reloads derivatives of ux, uy and uz with respect to x, y and z
336  xixl = deriv(1,index_ijk,ispec)
337  xiyl = deriv(2,index_ijk,ispec)
338  xizl = deriv(3,index_ijk,ispec)
339  etaxl = deriv(4,index_ijk,ispec)
340  etayl = deriv(5,index_ijk,ispec)
341  etazl = deriv(6,index_ijk,ispec)
342  gammaxl = deriv(7,index_ijk,ispec)
343  gammayl = deriv(8,index_ijk,ispec)
344  gammazl = deriv(9,index_ijk,ispec)
345 
346  tempx1(index_ijk) = jacobianl(index_ijk)*(xixl*dpotentialdxl(index_ijk) &
347  + xiyl*dpotentialdyl(index_ijk) + xizl*dpotentialdzl(index_ijk))
348  tempx2(index_ijk) = jacobianl(index_ijk)*(etaxl*dpotentialdxl(index_ijk) &
349  + etayl*dpotentialdyl(index_ijk) + etazl*dpotentialdzl(index_ijk))
350  tempx3(index_ijk) = jacobianl(index_ijk)*(gammaxl*dpotentialdxl(index_ijk) &
351  + gammayl*dpotentialdyl(index_ijk) + gammazl*dpotentialdzl(index_ijk))
352 
353  ENDdo_loop_ijk
354 
355  ! subroutines adapted from Deville, Fischer and Mund, High-order methods
356  ! for incompressible fluid flow, Cambridge University Press (2002),
357  ! pages 386 and 389 and Figure 8.3.1
358 
359  ! computes 1. matrix multiplication for newtempx1,..
360  call mxm5_single(hprimewgll_xxt,m1,tempx1,newtempx1,m2)
361  ! computes 2. matrix multiplication for tempx2,..
362  call mxm5_3dmat_single(tempx2,m1,hprimewgll_xx,m1,newtempx2,ngllx)
363  ! computes 3. matrix multiplication for newtempx3,..
364  call mxm5_single(tempx3,m2,hprimewgll_xx,newtempx3,m1)
365 
366  ! sum contributions from each element to the global mesh and add gravity term
367  do_loop_ijk
368  sum_terms(index_ijk,ispec) = - ( wgllwgll_yz_3d(index_ijk)*newtempx1(index_ijk) &
369  + wgllwgll_xz_3d(index_ijk)*newtempx2(index_ijk) &
370  + wgllwgll_xy_3d(index_ijk)*newtempx3(index_ijk))
371  ENDdo_loop_ijk
372 
373  ! adds gravity term
374  if (gravity_val) then
375  do_loop_ijk
376  ! uses double precision calculations. needs testing if okay with single precision.
377  iglob = ibool(index_ijk,ispec)
378 
379  ! Cartesian components of the gravitational acceleration
380  ! gravitational acceleration (integrated and multiply by rho / Kappa)
381  vec_x = gravity_pre_store(1,iglob)
382  vec_y = gravity_pre_store(2,iglob)
383  vec_z = gravity_pre_store(3,iglob)
384 
385  ! compute divergence of displacement
386  ! distinguish between single and double precision for reals
387  gravity_term = jacobianl(index_ijk) * wgll_cube(index_ijk) &
388  * (dpotentialdxl(index_ijk) * vec_x &
389  + dpotentialdyl(index_ijk) * vec_y &
390  + dpotentialdzl(index_ijk) * vec_z)
391 
392  ! divergence of displacement field with gravity on
393  ! note: these calculations are only considered for SIMULATION_TYPE == 1 .and. SAVE_FORWARD
394  ! and one has set MOVIE_VOLUME_TYPE == 4 when MOVIE_VOLUME is .true.;
395  ! in case of SIMULATION_TYPE == 3, it gets overwritten by compute_kernels_outer_core()
396  if (movie_volume .and. nspec_outer_core_3dmovie /= 1) then
397  div_displfluid(index_ijk,ispec) = dpotentialdxl(index_ijk) * vec_x &
398  + dpotentialdyl(index_ijk) * vec_y &
399  + dpotentialdzl(index_ijk) * vec_z
400  endif
401 
402  ! adds gravity term
403  sum_terms(index_ijk,ispec) = sum_terms(index_ijk,ispec) + gravity_term
404  ENDdo_loop_ijk
405  endif
406 
407  enddo ! spectral element loop
408 !$OMP enddo
409 
410  ! updates acceleration
411 #ifdef FORCE_VECTORIZATION
412  ! updates acceleration
413 !$OMP DO
414  do iglob_p = 1,num_globs(iphase)
415  iglob = phase_iglob(iglob_p,iphase)
416  do ip = ibool_inv_st(iglob_p,iphase),ibool_inv_st(iglob_p+1,iphase)-1
417  ijk_spec = ibool_inv_tbl(ip,iphase)
418  accelfluid(iglob) = accelfluid(iglob) + sum_terms(ijk_spec,1,1,1)
419  enddo
420  enddo
421 !OMP enddo
422 #else
423  ! updates for non-vectorization case
424 !$OMP DO SCHEDULE(GUIDED)
425  do ispec_p = 1,num_elements
426  ispec = phase_ispec_inner(ispec_p,iphase)
427  do_loop_ijk
428  iglob = ibool(index_ijk,ispec)
429 !$OMP ATOMIC
430  accelfluid(iglob) = accelfluid(iglob) + sum_terms(index_ijk,ispec)
431  ENDdo_loop_ijk
432  enddo
433 !$OMP enddo
434 #endif

440 !
441 ! matrix-matrix multiplications
442 !
443 ! subroutines adapted from Deville, Fischer and Mund, High-order methods
444 ! for incompressible fluid flow, Cambridge University Press (2002),
445 ! pages 386 and 389 and Figure 8.3.1
446 !
447 !--------------------------------------------------------------------------------------------
448 !
449 ! note: the matrix-matrix multiplications are used for very small matrices ( 5 x 5 x 5 elements);
450 ! thus, calling external optimized libraries for these multiplications are in general slower
451 !
452 ! please leave the routines here to help compilers inlining the code
453 
454  subroutine mxm5_single(A,n1,B,C,n3)
455 
456 ! 2-dimensional arrays (25,5)/(5,25)
457 
458  use constants_solver, only: custom_real
459 
460 #ifdef XSMM
461  use my_libxsmm, only: libxsmm_smm_5_25_5,libxsmm_smm_25_5_5
462 #endif
463 
464  implicit none
465 
466  integer,intent(in) :: n1,n3
467  real(kind=CUSTOM_REAL),dimension(n1,5),intent(in) :: A
468  real(kind=CUSTOM_REAL),dimension(5,n3),intent(in) :: B
469  real(kind=CUSTOM_REAL),dimension(n1,n3),intent(out) :: C
470 
471  ! local parameters
472  integer :: i,j
473 
474 ! note: on CPUs like Haswell or Sandy Bridge, the following will slow down computations
475 ! however, on Intel Phi (KNC) it is still helpful
476 #if defined(XSMM_FORCE_EVEN_IF_SLOWER) || ( defined(XSMM) && defined(__MIC__) )
477  ! matrix-matrix multiplication C = alpha A * B + beta C
478  if (n1 == 5) then
479  ! with A(n1,n2) 5x5-matrix, B(n2,n3) 5x25-matrix and C(n1,n3) 5x25-matrix
