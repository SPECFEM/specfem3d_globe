152  subroutine compute_add_sources_adjoint()
153 
154  use specfem_par
155  use specfem_par_crustmantle, only: accel_crust_mantle,ibool_crust_mantle
156 
157  implicit none
158 
159  ! local parameters
160  integer :: irec,irec_local,i,j,k,iglob
161  integer :: ivec_index
162  logical :: ibool_read_adj_arrays
163 
164  ! note: we check if nadj_rec_local > 0 before calling this routine, but better be safe...
165  if (nadj_rec_local == 0) return
166 
167  ! figure out if we need to read in a chunk of the adjoint source at this timestep
168  ibool_read_adj_arrays = ( (it == it_begin) .or. (mod(it-1,ntstep_between_read_adjsrc) == 0) )
169 
170  ! needs to read in a new chunk/block of the adjoint source
171  if (ibool_read_adj_arrays) then
172  call read_adjoint_sources()
173  endif
174 
175  ! adds adjoint sources
176  if (.not. gpu_mode) then
177  ! on CPU
178  irec_local = 0
179  do irec = 1,nrec
180 
181  ! adds source (only if this proc carries the source)
182  if (myrank == islice_selected_rec(irec)) then
183  irec_local = irec_local + 1
184 
185  ! adjoint source array index
186  ivec_index = iadj_vec(it)
187 
188  ! adds source contributions
189  do k = 1,ngllz
190  do j = 1,nglly
191  do i = 1,ngllx
192  iglob = ibool_crust_mantle(i,j,k,ispec_selected_rec(irec))
193 
194  ! adds adjoint source acting at this time step (it):
195  !
196  ! note: we use index iadj_vec(it) which is the corresponding time step
197  ! for the adjoint source acting at this time step (it)
198  !
199  ! see routine: setup_sources_receivers_adjindx() how this adjoint index array is set up
200  !
201  ! e.g. total length NSTEP = 3000, chunk length NTSTEP_BETWEEN_READ_ADJSRC= 1000
202  ! then for it = 1,..1000, first block has iadjsrc(1,1) with start = 2001 and end = 3000;
203  ! corresponding iadj_vec(it) goes from
204  ! iadj_vec(1) = 1000, iadj_vec(2) = 999 to iadj_vec(1000) = 1,
205  ! that is, originally the idea was
206  ! adj_sourcearrays(.. iadj_vec(1) ) corresponds to adjoint source trace at time index 3000
207  ! adj_sourcearrays(.. iadj_vec(2) ) corresponds to adjoint source trace at time index 2999
208  ! ..
209  ! adj_sourcearrays(.. iadj_vec(1000) ) corresponds to adjoint source trace at time index 2001
210  ! then a new block will be read, etc, and it is going down till to adjoint source trace at time index 1
211  !
212  ! now comes the tricky part:
213  ! adjoint source traces are based on the seismograms from the forward run;
214  ! such seismograms have a time step index 1 which corresponds to time -t0
215  ! then time step index 2 which corresponds to -t0 + DT, and
216  ! the last time step in the file at time step NSTEP corresponds to time -t0 + (NSTEP-1)*DT
217  ! (see how we add the sources to the simulation in compute_add_sources() and
218  ! how we write/save the seismograms and wavefields at the end of the time loop).
219  !
220  ! then you use that seismogram and take e.g. the velocity of it for a traveltime adjoint source
221  !
222  ! now we read it in again, and remember the last time step in
223  ! the file at NSTEP corresponds to -t0 + (NSTEP-1)*DT
224  !
225  ! the same time step is saved for the forward wavefields to reconstruct them;
226  ! however, the Newmark time scheme acts at the very beginning of this time loop
227  ! such that we have the backward/reconstructed wavefield updated by
228  ! a single time step into the direction -DT and b_displ(it=1) would corresponds to -t0 + (NSTEP-1)*DT - DT
229  ! after the Newmark (predictor) time step update.
230  ! however, we will read the backward/reconstructed wavefield at the end of the first time loop,
231  ! such that b_displ(it=1) corresponds to -t0 + (NSTEP-1)*DT (which is the one saved in the files).
232  !
233  ! for the kernel calculations, we want:
234  ! adjoint wavefield at time t, starting from 0 to T
235  ! and forward wavefield at time T-t, starting from T down to 0
236  ! let's say time 0 corresponds to -t0 = -t0 + (it - 1)*DT at it=1
237  ! and time T corresponds to -t0 + (NSTEP-1)*DT at it = NSTEP
238  !
239  ! as seen before, the time for the forward wavefield b_displ(it=1) would then
240  ! correspond to time -t0 + (NSTEP-1)*DT - DT, which is T - DT.
241  ! the corresponding time for the adjoint wavefield thus would be 0 + DT
242  ! and the adjoint source index would be iadj_vec(it+1)
243  ! however, iadj_vec(it+1) which would go from 999 down to 0. 0 is out of bounds.
244  ! we thus would have to read in the adjoint source trace beginning from 2999 down to 0.
245  ! index 0 is not defined in the adjoint source trace, and would be set to zero.
246  !
247  ! however, since this complicates things, we read the backward/reconstructed
248  ! wavefield at the end of the first time loop, such that b_displ(it=1) corresponds to -t0 + (NSTEP-1)*DT.
249  ! assuming that until that end the backward/reconstructed wavefield and adjoint fields
250  ! have a zero contribution to adjoint kernels.
251  accel_crust_mantle(:,iglob) = accel_crust_mantle(:,iglob) &
252  + adj_sourcearrays(:,i,j,k,irec_local,ivec_index)
253 
254  enddo
255  enddo
256  enddo
257  endif
258 
259  enddo
260 
261  else
262 
263  ! on GPU
264  ! note: adjoint sourcearrays can become very big when used with many receiver stations
265  ! we overlap here the memory transfer to GPUs
266 
267  ! current time index
268  ivec_index = iadj_vec(it)
269 
270  if (gpu_async_copy) then
271  ! only synchronously transfers array at beginning or whenever new arrays were read in
272  if (ibool_read_adj_arrays) then
273  ! transfers adjoint arrays to GPU device memory
274  ! note: function call passes pointer to array adj_sourcearrays at corresponding time slice
275  call transfer_adj_to_device(mesh_pointer,nrec,adj_sourcearrays(1,1,1,1,1,ivec_index), &
276  islice_selected_rec)
277  endif
278  else
279  ! synchronously transfers adjoint arrays to GPU device memory before adding adjoint sources on GPU
280  call transfer_adj_to_device(mesh_pointer,nrec,adj_sourcearrays(1,1,1,1,1,ivec_index), &
281  islice_selected_rec)
282  endif
283 
284  ! adds adjoint source contributions
285  call compute_add_sources_adjoint_gpu(mesh_pointer,nrec)
286 
287  if (gpu_async_copy) then
288  ! starts asynchronously transfer of next adjoint arrays to GPU device memory
289  ! (making sure the next adj_sourcearrays values were already read in)
290  if ((.not. ibool_read_adj_arrays) .and. &
291  (.not. mod(it,ntstep_between_read_adjsrc) == 0) .and. &
292  (.not. it == it_end)) then
293  ! next time index
294  ivec_index = iadj_vec(it+1)
295 
296  ! checks next index
297  if (ivec_index < 1 .or. ivec_index > ntstep_between_read_adjsrc) then
298  print *,'Error iadj_vec bounds: rank',myrank,' it = ',it,' index = ',ivec_index, &
299  'out of bounds ',1,'to',ntstep_between_read_adjsrc
300  call exit_mpi(myrank,'Error iadj_vec index bounds')
301  endif
302 
303  ! asynchronously transfers next time slice
304  call transfer_adj_to_device_async(mesh_pointer,nrec,adj_sourcearrays(1,1,1,1,1,ivec_index), &
305  islice_selected_rec)
306  endif
307  endif
308 
309  endif
310 
311  end subroutine compute_add_sources_adjoint

367  ! to a time (NSTEP - 1)*DT - t0
368  ! (see sources for simulation_type 1 and seismograms)

371  ! the wavefields, that is b_displ( it=1) would correspond to time (NSTEP -1 - 1)*DT - t0.
372  ! however, we read in the backward/reconstructed wavefields at the end of the Newmark time scheme
373  ! in the first (it=1) time loop.
374  ! this leads to the timing (NSTEP-(it-1)-1)*DT-t0-tshift_cmt for the source time function here
375  !

377  if (use_lddrk) then
378  time_t = dble(nstep-it_tmp)*dt - dble(c_lddrk(istage))*dt - t0
379  else
380  time_t = dble(nstep-it_tmp)*dt - t0
381  endif

390

397  iglob = ibool_crust_mantle(nint(xi_source(isource)), &
398  nint(eta_source(isource)), &
399  nint(gamma_source(isource)), &
400  ispec_selected_source(isource))
401

403  f0 = hdur(isource) !! using hdur as a FREQUENCY just to avoid changing CMTSOLUTION file format
404 
405  !if (it == 1 .and. myrank == 0) then
406  ! write(IMAIN,*) 'using a source of dominant frequency ',f0
407  ! write(IMAIN,*) 'lambda_S at dominant frequency = ',3000./sqrt(3.)/f0
408  ! write(IMAIN,*) 'lambda_S at highest significant frequency = ',3000./sqrt(3.)/(2.5*f0)
409  !endif

415  ! note: time step is now at NSTEP-it
416  b_accel_crust_mantle(:,iglob) = b_accel_crust_mantle(:,iglob) &
417  + sngl( nu_source(component_force_source,:,isource) ) * stf_used
418

420

422  stf = comp_source_time_function(timeval,hdur_gaussian(isource))
423

429  do k = 1,ngllz
430  do j = 1,nglly
431  do i = 1,ngllx
