146  subroutine compute_add_sources_adjoint()
147 
148  use specfem_par

155  integer :: ivec_index
156  logical :: ibool_read_adj_arrays
157 
158  ! note: we check if nadj_rec_local > 0 before calling this routine, but better be safe...
159  if (nadj_rec_local == 0) return
160 
161  ! figure out if we need to read in a chunk of the adjoint source at this timestep
162  ibool_read_adj_arrays = ( (it == it_begin) .or. (mod(it-1,ntstep_between_read_adjsrc) == 0) )
163 
164  ! needs to read in a new chunk/block of the adjoint source
165  if (ibool_read_adj_arrays) then
166  call read_adjoint_sources()
167  endif
168 
169  ! adds adjoint sources
170  if (.not. gpu_mode) then
171  ! on CPU
172  irec_local = 0
173  do irec = 1,nrec
174 
175  ! adds source (only if this proc carries the source)
176  if (myrank == islice_selected_rec(irec)) then
177  irec_local = irec_local + 1
178 
179  ! adjoint source array index
180  ivec_index = iadj_vec(it)
181 
182  ! adds source contributions
183  do k = 1,ngllz
184  do j = 1,nglly
185  do i = 1,ngllx
186  iglob = ibool_crust_mantle(i,j,k,ispec_selected_rec(irec))
187 
188  ! adds adjoint source acting at this time step (it):
189  !
190  ! note: we use index iadj_vec(it) which is the corresponding time step
191  ! for the adjoint source acting at this time step (it)
192  !
193  ! see routine: setup_sources_receivers_adjindx() how this adjoint index array is set up
194  !
195  ! e.g. total length NSTEP = 3000, chunk length NTSTEP_BETWEEN_READ_ADJSRC= 1000
196  ! then for it = 1,..1000, first block has iadjsrc(1,1) with start = 2001 and end = 3000;
197  ! corresponding iadj_vec(it) goes from
198  ! iadj_vec(1) = 1000, iadj_vec(2) = 999 to iadj_vec(1000) = 1,
199  ! that is, originally the idea was
200  ! adj_sourcearrays(.. iadj_vec(1) ) corresponds to adjoint source trace at time index 3000
201  ! adj_sourcearrays(.. iadj_vec(2) ) corresponds to adjoint source trace at time index 2999
202  ! ..
203  ! adj_sourcearrays(.. iadj_vec(1000) ) corresponds to adjoint source trace at time index 2001
204  ! then a new block will be read, etc, and it is going down till to adjoint source trace at time index 1
205  !
206  ! now comes the tricky part:
207  ! adjoint source traces are based on the seismograms from the forward run;
208  ! such seismograms have a time step index 1 which corresponds to time -t0
209  ! then time step index 2 which corresponds to -t0 + DT, and
210  ! the last time step in the file at time step NSTEP corresponds to time -t0 + (NSTEP-1)*DT
211  ! (see how we add the sources to the simulation in compute_add_sources() and
212  ! how we write/save the seismograms and wavefields at the end of the time loop).
213  !
214  ! then you use that seismogram and take e.g. the velocity of it for a traveltime adjoint source
215  !
216  ! now we read it in again, and remember the last time step in
217  ! the file at NSTEP corresponds to -t0 + (NSTEP-1)*DT
218  !
219  ! the same time step is saved for the forward wavefields to reconstruct them;
220  ! however, the Newmark time scheme acts at the very beginning of this time loop
221  ! such that we have the backward/reconstructed wavefield updated by
222  ! a single time step into the direction -DT and b_displ(it=1) would corresponds to -t0 + (NSTEP-1)*DT - DT
223  ! after the Newmark (predictor) time step update.
224  ! however, we will read the backward/reconstructed wavefield at the end of the first time loop,
225  ! such that b_displ(it=1) corresponds to -t0 + (NSTEP-1)*DT (which is the one saved in the files).
226  !
227  ! for the kernel calculations, we want:
228  ! adjoint wavefield at time t, starting from 0 to T
229  ! and forward wavefield at time T-t, starting from T down to 0
230  ! let's say time 0 corresponds to -t0 = -t0 + (it - 1)*DT at it=1
231  ! and time T corresponds to -t0 + (NSTEP-1)*DT at it = NSTEP
232  !
233  ! as seen before, the time for the forward wavefield b_displ(it=1) would then
234  ! correspond to time -t0 + (NSTEP-1)*DT - DT, which is T - DT.
235  ! the corresponding time for the adjoint wavefield thus would be 0 + DT
236  ! and the adjoint source index would be iadj_vec(it+1)
237  ! however, iadj_vec(it+1) which would go from 999 down to 0. 0 is out of bounds.
238  ! we thus would have to read in the adjoint source trace beginning from 2999 down to 0.
239  ! index 0 is not defined in the adjoint source trace, and would be set to zero.
240  !
241  ! however, since this complicates things, we read the backward/reconstructed
242  ! wavefield at the end of the first time loop, such that b_displ(it=1) corresponds to -t0 + (NSTEP-1)*DT.
243  ! assuming that until that end the backward/reconstructed wavefield and adjoint fields
244  ! have a zero contribution to adjoint kernels.
245  accel_crust_mantle(:,iglob) = accel_crust_mantle(:,iglob) &
246  + adj_sourcearrays(:,i,j,k,irec_local,ivec_index)
247 
248  enddo
249  enddo
250  enddo
251  endif
252 
253  enddo
254 
255  else
256 
257  ! on GPU
258  ! note: adjoint sourcearrays can become very big when used with many receiver stations
259  ! we overlap here the memory transfer to GPUs
260 
261  ! current time index
262  ivec_index = iadj_vec(it)
263 
264  if (gpu_async_copy) then
265  ! only synchronously transfers array at beginning or whenever new arrays were read in
266  if (ibool_read_adj_arrays) then
267  ! transfers adjoint arrays to GPU device memory
268  ! note: function call passes pointer to array adj_sourcearrays at corresponding time slice
269  call transfer_adj_to_device(mesh_pointer,nrec,adj_sourcearrays(1,1,1,1,1,ivec_index), &
270  islice_selected_rec)
271  endif
272  else
273  ! synchronously transfers adjoint arrays to GPU device memory before adding adjoint sources on GPU
274  call transfer_adj_to_device(mesh_pointer,nrec,adj_sourcearrays(1,1,1,1,1,ivec_index), &
275  islice_selected_rec)
276  endif
277 
278  ! adds adjoint source contributions
279  call compute_add_sources_adjoint_gpu(mesh_pointer,nrec)
280 
281  if (gpu_async_copy) then
282  ! starts asynchronously transfer of next adjoint arrays to GPU device memory
283  ! (making sure the next adj_sourcearrays values were already read in)
284  if ((.not. ibool_read_adj_arrays) .and. &
285  (.not. mod(it,ntstep_between_read_adjsrc) == 0) .and. &
286  (.not. it == it_end)) then
287  ! next time index
288  ivec_index = iadj_vec(it+1)
289 
290  ! checks next index
291  if (ivec_index < 1 .or. ivec_index > ntstep_between_read_adjsrc) then
292  print *,'Error iadj_vec bounds: rank',myrank,' it = ',it,' index = ',ivec_index, &
293  'out of bounds ',1,'to',ntstep_between_read_adjsrc
294  call exit_mpi(myrank,'Error iadj_vec index bounds')
295  endif
296 
297  ! asynchronously transfers next time slice
298  call transfer_adj_to_device_async(mesh_pointer,nrec,adj_sourcearrays(1,1,1,1,1,ivec_index), &
299  islice_selected_rec)
300  endif
301  endif
302 
303  endif
304 
305  end subroutine compute_add_sources_adjoint
