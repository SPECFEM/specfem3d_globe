150  subroutine mantle_s20rts(radius,theta,phi,dvs,dvp,drho)
151 
152  use constants
153  use model_s20rts_par
154 
155  implicit none
156 
157  double precision :: radius,theta,phi,dvs,dvp,drho
158 
159  ! local parameters
160  ! factor to convert perturbations in shear speed to perturbations in density
161  double precision, parameter :: SCALE_RHO = 0.40d0
162 
163  double precision, parameter :: R_EARTH_ = r_earth ! 6371000.d0
164  double precision, parameter :: RMOHO_ = r_earth - 24400.0 ! 6346600.d0
165  double precision, parameter :: RCMB_ = 3480000.d0
166  double precision, parameter :: ZERO_ = 0.d0
167 
168  integer :: l,m,k
169  double precision :: r_moho,r_cmb,xr
170  double precision :: dvs_alm,dvs_blm
171  double precision :: dvp_alm,dvp_blm
172  double precision :: s20rts_rsple,radial_basis(0:nk_20)
173  double precision :: sint,cost,x(2*ns_20+1),dx(2*ns_20+1)
174 
175  dvs = zero_
176  dvp = zero_
177  drho = zero_
178 
179  r_moho = rmoho_ / r_earth_
180  r_cmb = rcmb_ / r_earth_
181  if (radius >= r_moho .or. radius <= r_cmb) return
182 
183  xr=-1.0d0+2.0d0*(radius-r_cmb)/(r_moho-r_cmb)
184  do k = 0,nk_20
185  radial_basis(k)=s20rts_rsple(1,nk_20+1,s20rts_v_spknt(1),s20rts_v_qq0(1,nk_20+1-k),s20rts_v_qq(1,1,nk_20+1-k),xr)
186  enddo
187 
188  do l = 0,ns_20
189  sint=dsin(theta)
190  cost=dcos(theta)
191  call lgndr(l,cost,sint,x,dx)
192 
193  dvs_alm=0.0d0
194  dvp_alm=0.0d0
195  do k = 0,nk_20
196  dvs_alm=dvs_alm+radial_basis(k)*s20rts_v_dvs_a(k,l,0)
197  dvp_alm=dvp_alm+radial_basis(k)*s20rts_v_dvp_a(k,l,0)
198  enddo
199  dvs=dvs+dvs_alm*x(1)
200  dvp=dvp+dvp_alm*x(1)
201 
202  do m = 1,l
203  dvs_alm=0.0d0
204  dvp_alm=0.0d0
205  dvs_blm=0.0d0
206  dvp_blm=0.0d0
207  do k = 0,nk_20
208  dvs_alm=dvs_alm+radial_basis(k)*s20rts_v_dvs_a(k,l,m)
209  dvp_alm=dvp_alm+radial_basis(k)*s20rts_v_dvp_a(k,l,m)
210  dvs_blm=dvs_blm+radial_basis(k)*s20rts_v_dvs_b(k,l,m)
211  dvp_blm=dvp_blm+radial_basis(k)*s20rts_v_dvp_b(k,l,m)
212  enddo
213  dvs=dvs+(dvs_alm*dcos(dble(m)*phi)+dvs_blm*dsin(dble(m)*phi))*x(m+1)
214  dvp=dvp+(dvp_alm*dcos(dble(m)*phi)+dvp_blm*dsin(dble(m)*phi))*x(m+1)
215  enddo
216 
217  enddo
218 
219  drho = scale_rho*dvs
220 
221  end subroutine mantle_s20rts

282  double precision function s20rts_rsple(I1,I2,X,Y,Q,S)
283 
284  implicit none
285 
286 ! rsple returns the value of the function y(x) evaluated at point S
287 ! using the cubic spline coefficients computed by rspln and saved in Q.
288 ! If S is outside the interval (x(i1),x(i2)) rsple extrapolates
289 ! using the first or last interpolation polynomial. The arrays must
290 ! be dimensioned at least - x(i2), y(i2), and q(3,i2).
291 
292  integer i1,i2
293  double precision X(*),Y(*),Q(3,*),s
294 
295  integer i,ii
296  double precision h
297 
298  i = 1
299  ii=i2-1
300 
301 ! GUARANTEE I WITHIN BOUNDS.
302  i=max0(i,i1)
303  i=min0(i,ii)
304 
305 ! SEE IF X IS INCREASING OR DECREASING.
306  if (x(i2)-x(i1) < 0) goto 1
307  if (x(i2)-x(i1) >= 0) goto 2
308 
309 ! X IS DECREASING. CHANGE I AS NECESSARY.
310  1 if (s-x(i) <= 0) goto 3
311  if (s-x(i) > 0) goto 4
312 
313  4 i=i-1
314 
315  if (i-i1 < 0) goto 11
316  if (i-i1 == 0) goto 6
317  if (i-i1 > 0) goto 1
318 
319  3 if (s-x(i+1) < 0) goto 5
320  if (s-x(i+1) >= 0) goto 6
321 
322  5 i=i+1
323 
324  if (i-ii < 0) goto 3
325  if (i-ii == 0) goto 6
326  if (i-ii > 0) goto 7
327 
328 ! X IS INCREASING. CHANGE I AS NECESSARY.
329  2 if (s-x(i+1) <= 0) goto 8
330  if (s-x(i+1) > 0) goto 9
331 
332  9 i=i+1
333 
334  if (i-ii < 0) goto 2
335  if (i-ii == 0) goto 6
336  if (i-ii > 0) goto 7
337 
338  8 if (s-x(i) < 0) goto 10
339  if (s-x(i) >= 0) goto 6
340 
341  10 i=i-1
342  if (i-i1 < 0) goto 11
343  if (i-i1 == 0) goto 6
344  if (i-i1 > 0) goto 8
345 
346  7 i=ii
347  goto 6
348  11 i=i1
349 
350 ! CALCULATE RSPLE USING SPLINE COEFFICIENTS IN Y AND Q.
351  6 h=s-x(i)
352  s20rts_rsple=y(i)+h*(q(1,i)+h*(q(2,i)+h*q(3,i)))
353 
354  end function s20rts_rsple

360  subroutine s20rts_rspln(I1,I2,X,Y,Q,F)
361 
362  implicit none
363 
364 ! The subroutine rspln computes cubic spline interpolation coefficients
365 ! for y(x) between grid points i1 and i2 saving them in q.The
366 ! interpolation is continuous with continuous first and second
367 ! derivatives. It agrees exactly with y at grid points and with the
368 ! three point first derivatives at both end points (i1 and i2).
369 ! X must be monotonic but if two successive values of x are equal
370 ! a discontinuity is assumed and separate interpolation is done on
371 ! each strictly monotonic segment. The arrays must be dimensioned at
372 ! least - x(i2), y(i2), q(3,i2), and f(3,i2).
373 ! F is working storage for rspln.
374 
375  integer i1,i2
376  double precision X(*),Y(*),Q(3,*),F(3,*)
377 
378  integer i,j,k,j1,j2
379  double precision y0,a0,b0,b1,h,h2,ha,h2a,h3a,h2b
380  double precision YY(3),small
381  equivalence(yy(1),y0)
382  data small/1.0d-08/,yy/0.0d0,0.0d0,0.0d0/
383 
384  j1=i1+1
385  y0=0.0d0
386 
387 ! BAIL OUT IF THERE ARE LESS THAN TWO POINTS TOTAL
388  if (i2-i1 < 0) return
389  if (i2-i1 == 0) goto 17
390  if (i2-i1 > 0) goto 8
391 
392  8 a0=x(j1-1)
393 ! SEARCH FOR DISCONTINUITIES.
394  DO 3 i=j1,i2
395  b0=a0
396  a0=x(i)
397  if (dabs((a0-b0)/dmax1(a0,b0)) < small) goto 4
398  3 continue
399  17 j1=j1-1
400  j2=i2-2
401  goto 5
402  4 j1=j1-1
403  j2=i-3
404 ! SEE IF THERE ARE ENOUGH POINTS TO INTERPOLATE (AT LEAST THREE).
405  5 if (j2+1-j1 < 0) goto 9
406  if (j2+1-j1 == 0) goto 10
407  if (j2+1-j1 > 0) goto 11
408 
409 ! ONLY TWO POINTS. USE LINEAR INTERPOLATION.
410  10 j2=j2+2
411  y0=(y(j2)-y(j1))/(x(j2)-x(j1))
412  DO j=1,3
413  q(j,j1)=yy(j)
414  q(j,j2)=yy(j)
415  enddo
416  goto 12
417 
418 ! MORE THAN TWO POINTS. DO SPLINE INTERPOLATION.
419  11 a0 = 0.
420  h=x(j1+1)-x(j1)
421  h2=x(j1+2)-x(j1)
422  y0=h*h2*(h2-h)
423  h=h*h
424  h2=h2*h2
425 ! CALCULATE DERIVATIVE AT NEAR END.
426  b0=(y(j1)*(h-h2)+y(j1+1)*h2-y(j1+2)*h)/y0
427  b1=b0
428 
429 ! EXPLICITLY REDUCE BANDED MATRIX TO AN UPPER BANDED MATRIX.
430  DO i=j1,j2
431  h=x(i+1)-x(i)
432  y0=y(i+1)-y(i)
433  h2=h*h
434  ha=h-a0
435  h2a=h-2.0d0*a0
436  h3a=2.0d0*h-3.0d0*a0
437  h2b=h2*b0
438  q(1,i)=h2/ha
439  q(2,i)=-ha/(h2a*h2)
440  q(3,i)=-h*h2a/h3a
441  f(1,i)=(y0-h*b0)/(h*ha)
442  f(2,i)=(h2b-y0*(2.0d0*h-a0))/(h*h2*h2a)
443  f(3,i)=-(h2b-3.0d0*y0*ha)/(h*h3a)
444  a0=q(3,i)
445  b0=f(3,i)
446  enddo
447 
448 ! TAKE CARE OF LAST TWO ROWS.
449  i=j2+1
450  h=x(i+1)-x(i)
451  y0=y(i+1)-y(i)
452  h2=h*h
453  ha=h-a0
454  h2a=h*ha
455  h2b=h2*b0-y0*(2.0d0*h-a0)
456  q(1,i)=h2/ha
457  f(1,i)=(y0-h*b0)/h2a
458  ha=x(j2)-x(i+1)
459  y0=-h*ha*(ha+h)
460  ha=ha*ha
461 
462 ! CALCULATE DERIVATIVE AT FAR END.
463  y0=(y(i+1)*(h2-ha)+y(i)*ha-y(j2)*h2)/y0
464  q(3,i)=(y0*h2a+h2b)/(h*h2*(h-2.0d0*a0))
465  q(2,i)=f(1,i)-q(1,i)*q(3,i)
466 
467 ! SOLVE UPPER BANDED MATRIX BY REVERSE ITERATION.
468  DO j=j1,j2
469  k=i-1
470  q(1,i)=f(3,k)-q(3,k)*q(2,i)
471  q(3,k)=f(2,k)-q(2,k)*q(1,i)
472  q(2,k)=f(1,k)-q(1,k)*q(3,k)
473  i=k
474  enddo
475  q(1,i)=b1
476 ! FILL IN THE LAST POINT WITH A LINEAR EXTRAPOLATION.
477  9 j2=j2+2
478  DO j=1,3
479  q(j,j2)=yy(j)
480  enddo
481 
482 ! SEE IF THIS DISCONTINUITY IS THE LAST.
483  12 if (j2-i2 < 0) then
484  goto 6
485  else
486  return
487  endif
488 
489 ! NO. GO BACK FOR MORE.
490  6 j1=j2+2
491  if (j1-i2 <= 0) goto 8
492  if (j1-i2 > 0) goto 7
493 
494 ! THERE IS ONLY ONE POINT LEFT AFTER THE LATEST DISCONTINUITY.
495  7 DO j=1,3
496  q(j,i2)=yy(j)
497  enddo
498 
499  end subroutine s20rts_rspln
